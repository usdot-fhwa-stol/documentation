31def generate_launch_description():
32 """
33 Launch CARMA Messenger System.
34 """
35
36 configuration_delay = LaunchConfiguration('configuration_delay')
37 declare_configuration_delay_arg = DeclareLaunchArgument(
38 name ='configuration_delay', default_value='4.0')
39
40 # Declare the route file folder launch argument
41 route_file_folder = LaunchConfiguration('route_file_folder')
42 declare_route_file_folder = DeclareLaunchArgument(
43 name = 'route_file_folder',
44 default_value='/opt/carma/routes/',
45 description = 'Path of folder on host PC containing route CSV file(s) that can be accessed by plugins'
46 )
47
48 transform_group = GroupAction(
49 actions=[
50 PushRosNamespace(EnvironmentVariable('CARMA_TF_NS', default_value='/')),
51 IncludeLaunchDescription(
52 PythonLaunchDescriptionSource([ThisLaunchFileDir(), '/transforms.launch.py']),
53 ),
54 ]
55 )
56
57 v2x_group = GroupAction(
58 actions=[
59 IncludeLaunchDescription(
60 PythonLaunchDescriptionSource([ThisLaunchFileDir(), '/message.launch.py']),
61 launch_arguments = {
62 'configuration_delay' : [configuration_delay]
63 }.items()
64 ),
65 ]
66 )
67
68 plugins_group = GroupAction(
69 actions=[
70 IncludeLaunchDescription(
71 PythonLaunchDescriptionSource([ThisLaunchFileDir(), '/plugins.launch.py']),
72 launch_arguments = {
73 'configuration_delay' : [configuration_delay],
74 'route_file_folder' : route_file_folder
75 }.items()
76 ),
77 ]
78 )
79
80 return LaunchDescription([
81 declare_configuration_delay_arg,
82 declare_route_file_folder,
83 transform_group,
84 v2x_group,
85 plugins_group
86 ])
