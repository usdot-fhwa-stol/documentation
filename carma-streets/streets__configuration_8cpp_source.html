78 void streets_configuration::parse_configurations_array( const rapidjson::GenericArray<true, rapidjson::Value> &arr) {
79 std::unique_lock<std::mutex> lck(config_lock);
80 for ( auto& cnf: arr ) {
81 std::string property_name = cnf.FindMember("name")->value.GetString();
82 auto val = configuration_map.find( property_name );
83 if ( val != configuration_map.end() && cnf.IsObject()) {
84 configuration config = val->second;
85 configuration file_config;
86 file_config.fromJson( cnf.GetObject());
87 if ( config != file_config ) {
88 SPDLOG_INFO("Updating configuration {0}!", property_name);
89 configuration_map.erase(property_name);
90 configuration_map.insert(std::make_pair(property_name, file_config));
91 }
92 }
93 else if ( val == configuration_map.end() && cnf.IsObject() ) {
94 configuration file_config;
95 file_config.fromJson(cnf.GetObject() );
96 configuration_map.insert(std::make_pair(property_name, file_config));
97 }
98 else {
99 throw streets_service::streets_configuration_exception("Configuration parameter " + property_name + " not properly formatted!");
100 }

184 void streets_configuration::create_default_logger(const std::string &service_name ) const{
185 try {
186 // Create logger thread pool
187 spdlog::init_thread_pool(8192, 1);
188 // Create logger file and stdout sinks
189 // Rotate files at 11:59 PM
190 auto file_sink = std::make_shared<spdlog::sinks::daily_file_sink_mt>("../logs/" + service_name, 23, 59, false, 3);
191 auto console_sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
192 // Link thread pool and sinks to async logger
193 auto logger = std::make_shared<spdlog::async_logger>("main", spdlog::sinks_init_list({console_sink, file_sink}),spdlog::thread_pool());
194 // Set pattern [2022-3-31 13:00:00.000] [loglevel] [file_name.cpp:123]
195 logger->set_pattern("[%Y-%m-%d %H:%M:%S.%e] [%^%l%$] [%s:%#] %v");
196 // Register logger and set as default
197 spdlog::set_default_logger(logger);
198 SPDLOG_INFO("Default Logger initialized!");
199 }
200 catch (const spdlog::spdlog_ex& ex)
201 {
202 SPDLOG_ERROR( "Log initialization failed: {0}!",ex.what());

212 void streets_configuration::check_update() {
213 try {
214 // Check last write time of manifest.json configuration file to see if updates have been made
215 std::time_t time = boost::filesystem::last_write_time(filepath);
216 SPDLOG_TRACE("Last Modified Time {0} vs Stored Last Modified Time {1}", time, last_modified);
217 if ( time > last_modified) {
218 // If updates have been made parse manifest.json into Document and update loglevel and
219 // any changed configuration parameters.
220 rapidjson::Document doc = parse_configuration_file();
221 update_log_level(doc);
222 update_configuration(doc);
223 // Set new update time to the last update read
