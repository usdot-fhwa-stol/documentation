25 double PIDController::calculate( double setpoint, double pv )
26 {
27 // Calculate error
28 double error = setpoint - pv;
29 ROS_DEBUG_STREAM("PID error: " << error);
30
31 // Proportional term
32 double Pout = config_.Kp * error;
33 ROS_DEBUG_STREAM("Proportional term: " << Pout);
34
35 // Integral term
36 _integral += error * config_.dt;
37 ROS_DEBUG_STREAM("Integral term: " << _integral);
38
39 if (_integral > config_.integratorMax){
40 _integral = config_.integratorMax;
41 }
42 else if (_integral < config_.integratorMin){
43 _integral = config_.integratorMin;
44 }
45 double Iout = config_.Ki * _integral;
46 ROS_DEBUG_STREAM("Iout: " << Iout);
47
48 // Derivative term
49 double derivative = (error - _pre_error) / config_.dt;
50 ROS_DEBUG_STREAM("derivative term: " << derivative);
51 double Dout = config_.Kd * derivative;
52 ROS_DEBUG_STREAM("Dout: " << Dout);
53
54 // Calculate total output
55 double output = Pout + Iout + Dout;
56 ROS_DEBUG_STREAM("total controller output: " << output);
57
58 // Restrict to max/min
59 if( output > config_.maxValue )
60 output = config_.maxValue;
61 else if( output < config_.minValue )
62 output = config_.minValue;
63 // Save error to previous error
64 _pre_error = error;
65
66 return output;
67
68 }
