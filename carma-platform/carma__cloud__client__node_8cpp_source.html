95 void CarmaCloudClient::XMLconversion(char* xml_str, carma_v2x_msgs::msg::TrafficControlRequest request_msg)
96 {
97 j2735_v2x_msgs::msg::TrafficControlRequest j2735_tcr;
98
99 j2735_convertor::geofence_request::convert(request_msg, j2735_tcr);
100
101 RCLCPP_DEBUG_STREAM( get_logger(), "converted: ");
102
103 size_t hexlen = 2; //size of each hex representation with a leading 0
104 char reqid[j2735_tcr.tcr_v01.reqid.id.size() * hexlen + 1];
105 for (int i = 0; i < j2735_tcr.tcr_v01.reqid.id.size(); i++)
106 {
107 sprintf(reqid+(i*hexlen), "%.2X", j2735_tcr.tcr_v01.reqid.id[i]);
108 }
109
110 RCLCPP_DEBUG_STREAM( get_logger(), "reqid: " << reqid);
111
112 long int reqseq = j2735_tcr.tcr_v01.reqseq;
113 RCLCPP_DEBUG_STREAM( get_logger(), "reqseq: " << reqseq);
114
115 long int scale = j2735_tcr.tcr_v01.scale;
116 RCLCPP_DEBUG_STREAM( get_logger(), "scale: " << scale);
117
118 int totBounds = j2735_tcr.tcr_v01.bounds.size();
119 int cnt=0;
120 char bounds_str[5000];
121 strcpy(bounds_str,"");
122
123 // get current time
124 std::time_t tm = this->now().seconds()/60 - config_.fetchtime*24*60; // T minus fetchtime*24 hours in min
125
126 while(cnt<totBounds)
127 {
128
129 uint32_t oldest = tm;
130 long lat = j2735_tcr.tcr_v01.bounds[cnt].reflat;
131 long longg = j2735_tcr.tcr_v01.bounds[cnt].reflon;
132
133
134 long dtx0 = j2735_tcr.tcr_v01.bounds[cnt].offsets[0].deltax;
135 long dty0 = j2735_tcr.tcr_v01.bounds[cnt].offsets[0].deltay;
136 long dtx1 = j2735_tcr.tcr_v01.bounds[cnt].offsets[1].deltax;
137 long dty1 = j2735_tcr.tcr_v01.bounds[cnt].offsets[1].deltay;
138 long dtx2 = j2735_tcr.tcr_v01.bounds[cnt].offsets[2].deltax;
139 long dty2 = j2735_tcr.tcr_v01.bounds[cnt].offsets[2].deltay;
140
141 int x = std::sprintf(bounds_str+strlen(bounds_str),"<bounds><TrafficControlBounds><oldest>%ld</oldest><reflon>%ld</reflon><reflat>%ld</reflat><offsets><OffsetPoint><deltax>%ld</deltax><deltay>%ld</deltay></OffsetPoint><OffsetPoint><deltax>%ld</deltax><deltay>%ld</deltay></OffsetPoint><OffsetPoint><deltax>%ld</deltax><deltay>%ld</deltay></OffsetPoint></offsets></TrafficControlBounds></bounds>",oldest,longg,lat,dtx0,dty0,dtx1,dty1,dtx2,dty2);

143 cnt++;

162 int CarmaCloudClient::CloudSend(const std::string &local_msg, const std::string& local_url, const std::string& local_base, const std::string& local_method)
163 {
164 CURL *req;
165 CURLcode res;
166 std::string urlfull = local_url + config_.port + local_base;
167 RCLCPP_DEBUG_STREAM( get_logger(), "full url: " << urlfull);
168 req = curl_easy_init();
169 if(req) {
170 curl_easy_setopt(req, CURLOPT_URL, urlfull.c_str());
171
172 if(strcmp(local_method.c_str(),"POST")==0)
173 {
174 curl_easy_setopt(req, CURLOPT_POSTFIELDS, local_msg.c_str());
175 curl_easy_setopt(req, CURLOPT_TIMEOUT_MS, 1000L); // Request operation complete within max millisecond timeout
176 res = curl_easy_perform(req);
177 if(res != CURLE_OK)
178 {
179 RCLCPP_ERROR_STREAM( get_logger(), "curl send failed: " << curl_easy_strerror(res));
180 return 1;
181 }
182 }
183 curl_easy_cleanup(req);
184 }
185
186 return 0;
187 }
