120void NS3Client::process(const std::shared_ptr<const std::vector<uint8_t>>& data)
121{
122 auto & entry = *data;
123 // Valid message should begin with 2 bytes message ID and 1 byte length.
124 for (size_t i = 0; i < entry.size() - 3; i++) { // leave 3 bytes after (for lsb of id, length byte 1, and either message body or length byte 2)
125 // Generate the 16-bit message id from two bytes, skip if it isn't a valid one
126 uint16_t msg_id = (static_cast<uint16_t>(entry[i]) << 8) | static_cast<uint16_t>(entry[i + 1]);
127
128 // Parse the length, check it doesn't run over
129 size_t len = 0;
130 size_t len_byte_1 = entry[i + 2];
131 int len_bytes = 0;
132 // length < 128 encoded by single byte with msb set to 0
133 if ((len_byte_1 & 0x80 ) == 0x00) {
134 len = static_cast<size_t>(len_byte_1);
135 len_bytes = 1;
136 // check for 0 length
137 if (len_byte_1 == 0x00) { continue; }
138 }
139 // length < 16384 encoded by 14 bits in 2 bytes (10xxxxxx xxxxxxxx)
140 else if ((len_byte_1 & 0x40) == 0x00) { //we know msb = 1, check that next bit is 0
141 if (i + 3 < entry.size()) {
142 size_t len_byte_2 = entry[i + 3];
143 len = ((len_byte_1 & 0x3f) << 8) | len_byte_2;
144 len_bytes = 2;
145 }
146 }
147 else {
148 // TODO lengths greater than 16383 (0x3FFF) are encoded by splitting up the message into discrete chunks, each with its own length
149 // marker. It doesn't look like we'll be receiving anything that long
150 std::cerr << "NS3Client::process() : received a message with length field longer than 16383." << std::endl;
151 continue;
152 }
153 if (len == -1) { continue; }
154 // If the length makes sense bsmPub(fits in the buffer), copy out the message bytes and pass to the Application class
155 if ((i + 1 + len + len_bytes) < entry.size()) {
156 // bool found_valid_msg = true;
157 size_t start_index = i;
158 size_t end_index = i + 2 + len + len_bytes; // includes 2 msgID bytes before message body
159 //this constructor has range [first, last) hence the + 1
160 std::vector<uint8_t> msg_vec(entry.begin() + start_index, entry.begin() + end_index);
161 onMessageReceived(msg_vec, msg_id);
162 break;
163 }
164 }
165}
