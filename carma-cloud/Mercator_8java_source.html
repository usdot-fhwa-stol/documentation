233 public static double eMercY(double lat) {
234 if (lat > 89.5) {
235 lat = 89.5;
236 }
237 if (lat < -89.5) {
238 lat = -89.5;
239 }
240 double phi = Math.toRadians(lat);
241 double con = ECC * Math.sin(phi);
242 con = Math.pow(((1.0-con)/(1.0+con)), ECC_OVER_TWO);
243 double ts = Math.tan(0.5 * ((PI_OVER_TWO) - phi))/con;
244 double y = 0 - R_MAJOR * Math.log(ts);
245 return y;
246 }

255 public static double eLat(double dY)
256 {
257 double ts = Math.exp(-dY / R_MAJOR);
258 double phi = PI_OVER_TWO - 2 * Math.atan(ts);
259 double dphi = 1.0;
260 for (int i = 0; Math.abs(dphi) > 0.000000001 && i < 15; i++)
261 {
262 double con = ECC * Math.sin(phi);
263 dphi = PI_OVER_TWO - 2 * Math.atan(ts * Math.pow((1.0 - con) / (1.0 + con), ECC_OVER_TWO)) - phi;
264 phi += dphi;
265 }
266
267 return Math.toDegrees(phi);
268 }
