37def create_ros2_tracing_action(context, *args, **kwargs):
38 """
39 Opaque Function for generating a 'Trace' ROS 2 launch action, which is dependent on the
40 'ROS_LOG_DIR' EnvironmentVariable and the 'is_ros2_tracing_enabled' LaunchConfiguration.
41
42 NOTE: This Opaque Function is required in order to evaluate the 'ROS_LOG_DIR' environment
43 variable as a string.
44 """
45 log_directory_string = launch.substitutions.EnvironmentVariable('ROS_LOG_DIR').perform(context)
46
47 trace = GroupAction(
48 condition=IfCondition(LaunchConfiguration('is_ros2_tracing_enabled')),
49 actions=[
50 Trace(
51 base_path = log_directory_string,
52 session_name='my-tracing-session-' + str(datetime.now().strftime('%Y-%m-%d_%H%M%S')),
53 events_kernel = [], # Empty since kernel tracing is not enabled for CARMA Platform
54 )
55 ]
56 )
57
58 return [trace]
