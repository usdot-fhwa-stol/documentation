116 signal_phase_and_timing::movement_event tsc_state::get_following_event(const signal_phase_and_timing::movement_event& current_event,
117 uint64_t current_event_end_time, const signal_group_state& phase_state) const
118 {
119 signal_phase_and_timing::movement_event next_event;
120 switch (current_event.event_state){
121 case signal_phase_and_timing::movement_phase_state::protected_movement_allowed: //Green
122 // Create next movement - yellow
123 next_event.event_state = signal_phase_and_timing::movement_phase_state::protected_clearance;
124 next_event.timing.set_start_time(current_event_end_time);
125 next_event.timing.set_min_end_time(current_event_end_time + phase_state.yellow_duration);
126 next_event.timing.set_max_end_time(current_event_end_time + phase_state.yellow_duration);
127 break;
128
129 case signal_phase_and_timing::movement_phase_state::protected_clearance: //Yellow
130 // Create next movement - red
131 next_event.event_state = signal_phase_and_timing::movement_phase_state::stop_and_remain;
132 next_event.timing.set_start_time(current_event_end_time);
133 next_event.timing.set_min_end_time(current_event_end_time + phase_state.red_duration);
134 next_event.timing.set_max_end_time(current_event_end_time + phase_state.red_duration);
135 break;
136
137 case signal_phase_and_timing::movement_phase_state::stop_and_remain: //Red
138 // Create next movement - green
139 next_event.event_state = signal_phase_and_timing::movement_phase_state::protected_movement_allowed;
140 next_event.timing.set_start_time(current_event_end_time);
141 next_event.timing.set_min_end_time(current_event_end_time + phase_state.green_duration);
142 next_event.timing.set_max_end_time(current_event_end_time + phase_state.green_duration);
143 break;
144
145 default:
146 SPDLOG_ERROR("This movement phase is not supported. Movement phase type: {0}", int(current_event.event_state));
147 throw snmp_client_exception("Failed request for unsupported movement phase type: " + std::to_string(int(current_event.event_state)));
148 }
149 return next_event;
150 }

166 // for phases 1-16 (.1 gets 1-8, .2 gets 9-16)
167 auto request_type = streets_snmp_cmd::REQUEST_TYPE::GET;
168 std::string vehicle_call_phases_1_8 = ntcip_oids::PHASE_STATUS_GROUP_VEH_CALLS + ".1";
169 std::string vehicle_call_phases_9_16 = ntcip_oids::PHASE_STATUS_GROUP_VEH_CALLS + ".2";
170 std::string pedestrian_call_phases_1_8 = ntcip_oids::PHASE_STATUS_GROUP_PED_CALLS + ".1";
171 std::string pedestrian_call_phases_9_16 = ntcip_oids::PHASE_STATUS_GROUP_PED_CALLS + ".2";
172
173 streets_snmp_cmd::snmp_response_obj veh_call_1_8;
174 veh_call_1_8.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
175 streets_snmp_cmd::snmp_response_obj veh_call_9_16;
176 veh_call_9_16.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
177 streets_snmp_cmd::snmp_response_obj ped_call_1_8;
178 ped_call_1_8.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
179 streets_snmp_cmd::snmp_response_obj ped_call_9_16;
180 ped_call_9_16.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;

183 snmp_client_worker_ ->process_snmp_request(vehicle_call_phases_9_16, request_type, veh_call_9_16);
184 snmp_client_worker_ ->process_snmp_request(pedestrian_call_phases_1_8, request_type, ped_call_1_8);
185 snmp_client_worker_ ->process_snmp_request(pedestrian_call_phases_9_16, request_type, ped_call_9_16);
186
187 SPDLOG_INFO("Response Veh {0}, {1} Response Ped {2}, {3}", veh_call_1_8.val_int, veh_call_9_16.val_int, ped_call_1_8.val_int, ped_call_9_16.val_int);

194
195 // Clear previously saved vehicle/pedestrian information and replace with new information
196
197 vehicle_calls.clear();
198 vehicle_calls.insert(vehicle_calls.end(), veh_resp_1_8.begin(), veh_resp_1_8.end());
199 vehicle_calls.insert(vehicle_calls.end(), veh_resp_9_16.begin(), veh_resp_9_16.end());
200
201 pedestrian_calls.clear();
202 pedestrian_calls.insert(pedestrian_calls.end(), ped_resp_1_8.begin(), ped_resp_1_8.end());
203 pedestrian_calls.insert(pedestrian_calls.end(), ped_resp_9_16.begin(), ped_resp_9_16.end());
204
205 SPDLOG_INFO("Pedestrian calls {0}, Vehicle calls, {1}", vector_to_string(pedestrian_calls), vector_to_string(vehicle_calls));

261 void tsc_state::add_future_movement_events(std::shared_ptr<signal_phase_and_timing::spat> spat_ptr)
262 {
263 // Modify spat according to phase configuration
264 // Note: Only first intersection is populated
265 auto intersection_state = spat_ptr->get_intersection();
266 for (auto movement : intersection_state.states)
267 {
268 int signal_group_id = movement.signal_group;
269
270 // Assumption here is that the movement states have only one event, since this method adds future events to the list.
271 // Throw exception is list size is great than 1
272 if (movement.state_time_speed.size() > 1)
273 {
274 throw monitor_states_exception("Event list has more than one events, not usable when adding future movement events. Associated with Signal Group:" + std::to_string(signal_group_id));
275 }
276
277 // Get movement_state by reference
278 auto& current_movement = intersection_state.get_movement(signal_group_id);
279
280 // Get start time as epoch time
281 uint64_t start_time = movement.state_time_speed.front().timing.get_epoch_start_time();
282
283 // Check if signal_group_id is associated with a vehicle phase : Only vehicle phases mapped to signal_group_states
284 signal_group_state phase_state;
285 if(signal_group_2tsc_state_map_.find(signal_group_id) == signal_group_2tsc_state_map_.end()){
286 continue;
287 }
288
289 phase_state = signal_group_2tsc_state_map_[signal_group_id];
290
291 uint64_t current_event_end_time_epoch = 0;
292 signal_phase_and_timing::movement_event current_event = current_movement.state_time_speed.front();
293
294 switch(current_event.event_state){
295 case signal_phase_and_timing::movement_phase_state::stop_and_remain : //Red
296 current_event_end_time_epoch = start_time + phase_state.red_duration;
297 break;
298
299 case signal_phase_and_timing::movement_phase_state::protected_movement_allowed : //Green
300 current_event_end_time_epoch = start_time + phase_state.green_duration;
301 break;
302
303 case signal_phase_and_timing::movement_phase_state::protected_clearance : //Yellow
304 current_event_end_time_epoch = start_time + phase_state.yellow_duration;
305 break;
306
307 default:
308 SPDLOG_DEBUG("This movement phase is not supported. Movement phase type: {0}", int(current_movement.state_time_speed.front().event_state));
309 throw monitor_states_exception("This movement phase is not supported. Movement phase type: " + std::to_string(int(current_movement.state_time_speed.front().event_state)));
310 }
311 // Update end_time for current_event
312 current_movement.state_time_speed.front().timing.set_min_end_time(current_event_end_time_epoch);
313 current_movement.state_time_speed.front().timing.set_max_end_time(current_event_end_time_epoch);
314
315 for(int i = 0; i < required_following_movements_; ++i)
316 {
317 signal_phase_and_timing::movement_event next_event = get_following_event(current_event, current_event_end_time_epoch, phase_state);
318 // Set next event to current event
319 current_event = next_event;
320 current_event_end_time_epoch = current_event.timing.get_epoch_min_end_time();
321 //Add events to list
322 current_movement.state_time_speed.push_back(next_event);
323 }
324
325 }
326 spat_ptr->set_intersection(intersection_state);
327
328 }

353 void tsc_state::map_phase_and_signalgroup(const std::vector<std::vector<int>>& active_ring_sequences,
354 std::unordered_map<int,int>& vehicle_phase_2signalgroup_map, std::unordered_map<int, int>& signal_group_2vehiclephase_map) const
355 {
356 // Loop through all ring sequences and remove vehicle phase entries in map
357 for(auto it = vehicle_phase_2signalgroup_map.begin(); it != vehicle_phase_2signalgroup_map.end();){
358 auto phase = it->first;
359 bool found_phase = false;
360 for(auto ring : active_ring_sequences)
361 {
362 if(std::find(ring.begin(), ring.end(), phase) !=ring.end()){
363 found_phase = true;
364 break;
365 }
366 }
367
368 if(!found_phase)
369 {
370 // if phase not found in active sequence, remove entry
371 it = vehicle_phase_2signalgroup_map.erase(it);
372 }
373 else{
374 // Add entry to signal to phase map
375 signal_group_2vehiclephase_map.insert(std::make_pair(it->second, it->first));
376 it++;
377 }

382 std::unordered_map<int,int> tsc_state::get_phases_associated_with_channel(const std::vector<int>& signal_group_ids) const {
383
384 std::unordered_map<int,int> phases_to_signal_group;
385 auto request_type= streets_snmp_cmd::REQUEST_TYPE::GET;
386
387 for(int signal_group : signal_group_ids)
388 {
389 streets_snmp_cmd::snmp_response_obj phase_num;
390 phase_num.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
391 // Control source returns the phase associated with the signal group. Channel and signal group id is synonymous according to the NTCIP documentation
392 std::string control_source_parameter_oid = ntcip_oids::CHANNEL_CONTROL_SOURCE_PARAMETER + "." + std::to_string(signal_group);
393 snmp_client_worker_->process_snmp_request(control_source_parameter_oid, request_type, phase_num);
394
395 // According to NTCIP 1202 v03 returned value of 0 here would mean a phase is not associated with the signal_group
396 if(phase_num.val_int != 0)
397 {
398 phases_to_signal_group.insert(std::make_pair(phase_num.val_int, signal_group));
399 }
400 else{
401 throw monitor_states_exception("No phase found associated with signal_group {0}" + std::to_string(signal_group));
402 }
403 SPDLOG_TRACE("Found mapping between signal group: {0} and phase num: {1}", signal_group , phase_num.val_int );
404
405 }
406
407 return phases_to_signal_group;

412
413 // Loop through channel control types and add channels with vehicle phase to list
414 streets_snmp_cmd::snmp_response_obj control_type;
415 control_type.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
416
417 auto request_type = streets_snmp_cmd::REQUEST_TYPE::GET;
418 for(int channel_num = 1; channel_num <= max_channels; ++channel_num)
419 {
420 std::string control_type_parameter_oid = ntcip_oids::CHANNEL_CONTROL_TYPE_PARAMETER + "." + std::to_string(channel_num);
421
422 if(!snmp_client_worker_->process_snmp_request(control_type_parameter_oid, request_type, control_type))
423 {
424 throw snmp_client_exception("Failed to get channel control type");
425 }
426
427 if(control_type.val_int == 2) // If 2, phase is a vehicle phase
428 {
429 vehicle_channels.push_back(channel_num);
430 }
431 else if(control_type.val_int == 3) //If 3, phase is a pedestrian phase
432 {
433 ped_channels.push_back(channel_num);
434 }
435
436 }
437
438 if(vehicle_channels.empty()){
439 SPDLOG_WARN("Found no vehicle phases");
440 }
441 if(ped_channels.empty()){
442 SPDLOG_DEBUG("Found no ped phases");
443 }
444
445 SPDLOG_DEBUG("Number of vehicle phase channels found: {0}", vehicle_channels.size());
446 SPDLOG_DEBUG("Number of ped phase channels found: {0}", ped_channels.size());

452 std::vector<std::vector<int>> tsc_state::get_active_ring_sequences(int max_rings, std::unordered_map<int,int>& vehicle_phase_2signalgroup_map, int sequence) const{
453
454 std::vector<std::vector<int>> active_ring_sequences;
455 // Read sequence data for rings
456 auto request_type= streets_snmp_cmd::REQUEST_TYPE::GET;
457
458 for(int i = 1 ; i <= max_rings; ++i){
459 int ring_num = i;
460 std::vector<int> phase_seq;
461
462 std::string phase_seq_oid= ntcip_oids::SEQUENCE_DATA + "." + std::to_string(sequence) + "." + std::to_string(ring_num);
463
464 streets_snmp_cmd::snmp_response_obj seq_data;
465 seq_data.type = streets_snmp_cmd::RESPONSE_TYPE::STRING;
466 snmp_client_worker_->process_snmp_request(phase_seq_oid, request_type, seq_data);
467
468 //extract phase numbers from strings
469 for(auto seq_val : seq_data.val_string)
470 {
471 auto phase = int(seq_val);
472 if(vehicle_phase_2signalgroup_map.find(phase)!= vehicle_phase_2signalgroup_map.end() && phase != 0 ){
473 phase_seq.push_back(phase);
474 }
475
476 }
477 if (!phase_seq.empty())
478 {
479 active_ring_sequences.push_back(phase_seq);
480 }
481 else{
482 SPDLOG_DEBUG("No phases found in sequence {0} for ring {1}", sequence, ring_num);
483 }
484 }
485
486 return active_ring_sequences;

609 int tsc_state::get_red_duration(int phase_num)
610 {
611 //red duration is calculated as red_clearance time for current phase + (green duration + yellow duration + red_clearance) for other phases in sequence
612
613 // Find signal state associated with phase num
614 int current_signal_group;
615 current_signal_group = get_vehicle_signal_group_id(phase_num);
616 auto current_signal_group_state = signal_group_2tsc_state_map_[current_signal_group];
617 // Only add clearance time for current phase
618 int red_duration = current_signal_group_state.red_clearance;
619
620 for(auto phase : current_signal_group_state.phase_seq)
621 {
622 // Find state params for each phase in seq
623 int seq_signal_group;
624 seq_signal_group = get_vehicle_signal_group_id(phase);
625 auto seq_signal_group_state = signal_group_2tsc_state_map_[seq_signal_group];
626 if(phase == phase_num)
627 {
628 continue;
629 }
630 else{
631 red_duration += seq_signal_group_state.green_duration + seq_signal_group_state.yellow_duration + seq_signal_group_state.red_clearance;
632 }
633 }
634
635 return red_duration;
636 }

638 std::vector<int> tsc_state::get_concurrent_signal_groups(int phase_num) const
639 {
640
641 std::vector<int> concurrent_signal_groups;
642 std::string concurrent_phases_oid = ntcip_oids::PHASE_CONCURRENCY + "." + std::to_string(phase_num);
643
644 streets_snmp_cmd::snmp_response_obj concurrent_phase_data;
645 auto request_type = streets_snmp_cmd::REQUEST_TYPE::GET;
646 concurrent_phase_data.type = streets_snmp_cmd::RESPONSE_TYPE::STRING;
647 snmp_client_worker_->process_snmp_request(concurrent_phases_oid, request_type, concurrent_phase_data);
648
649 //extract phase numbers from strings
650 for(auto con_phase : concurrent_phase_data.val_string)
651 { if(con_phase != 0){
652 concurrent_signal_groups.push_back(get_vehicle_signal_group_id(int(con_phase)));
653 }
654 }
655
656 if(concurrent_signal_groups.empty())
657 {
658 SPDLOG_WARN("No concurrent signal groups found for phase {0}", phase_num);
659 }
660
661 return concurrent_signal_groups;
662 }
