107 uint16_t time_change_details::convert_msepoch_to_hour_tenth_secs(uint64_t epoch_time_ms) const{
108 auto hours_since_epoch = std::floor(streets_service::streets_clock_singleton::time_in_ms()/ (HOUR_TO_SECONDS * SECOND_TO_MILLISECONDS) );
109 auto hours_since_epoch_ms = hours_since_epoch * HOUR_TO_SECONDS * SECOND_TO_MILLISECONDS;
110 uint64_t tenth_seconds_from_current_hour;
111 if ( hours_since_epoch_ms > epoch_time_ms ) {
112 SPDLOG_WARN("Epoch time provided {0} is smaller than the current UTC hour time {1}! SPaT cannot represent a time from a previous hour. Approximating time as beginning of current hour."
113 , epoch_time_ms, hours_since_epoch_ms);
114 tenth_seconds_from_current_hour = 0;
115 }
116 else {
117 tenth_seconds_from_current_hour = (epoch_time_ms - hours_since_epoch_ms )/100;
118 }
119 // TimeMark Max value is 36001 and 36001 is reserved for invalid unknown times (see J2737 TimeMark message documentation).
120 // Relevant section
121 // TimeMark ::= INTEGER (0..36001)
122 // -- Tenths of a second in the current or next hour
123 // -- In units of 1/10th second from UTC time
124 // -- A range of 0~36000 covers one hour
125 // -- The values 35991..35999 are used when a leap second occurs
126 // -- The value 36000 is used to indicate time >3600 seconds
127 // -- 36001 is to be used when value undefined or unknown
128 // -- Note that this is NOT expressed in GPS time
129 // -- or in local time
130 if ( tenth_seconds_from_current_hour >= 36000) {
131 SPDLOG_WARN("Approximating {0} as 36000! TimeMark maximum value is 36000!", tenth_seconds_from_current_hour);
132 tenth_seconds_from_current_hour = 36000;
133 }
134 return static_cast<uint16_t>(tenth_seconds_from_current_hour);
135 }
