82 public void setLaneIds(double dMaxStep)
83 {
84 double dTol = dMaxStep * 2;
85 ArrayList<Lane> oPrevLanes = new ArrayList();
86 get(0).getLanes(oPrevLanes);
87 int nNextId = Integer.MIN_VALUE;
88 for (Lane oLane : oPrevLanes)
89 {
90 oLane.m_nLaneIdByRoad = oLane.m_nLaneIndex;
91 int nComp = Math.abs(oLane.m_nLaneIndex);
92 if (nComp > nNextId)
93 nNextId = nComp;
94 }
95
96 for (int nIndex = 1; nIndex < size(); nIndex++)
97 {
98 ArrayList<Lane> oCurrLanes = new ArrayList();
99 get(nIndex).m_oPrevSect = get(nIndex - 1);
100 get(nIndex).getLanes(oCurrLanes);
101 for (Lane oCurr: oCurrLanes)
102 {
103 if (oCurr.m_nLaneIndex == 0)
104 continue;
105 double[] dCurr = oCurr.m_dCenter;
106 oCurr.m_nLaneIdByRoad = Integer.MIN_VALUE;
107 for (Lane oPrev: oPrevLanes)
108 {
109 if (oPrev.m_nLaneIndex == 0)
110 continue;
111 double[] dPrev = oPrev.m_dCenter;
112 int nSize = Arrays.size(dPrev);
113
114 double dDist = Geo.distance(dCurr[5], dCurr[6], dPrev[nSize - 3], dPrev[nSize - 2]);
115// if (dDist > dTol)
116// System.out.println(dDist);
117 if (dDist <= dTol && oCurr.m_nLaneType == oPrev.m_nLaneType)
118 {
119 oCurr.m_nLaneIdByRoad = oPrev.m_nLaneIdByRoad;
120 break;
121 }
122 }
123
124 if (oCurr.m_nLaneIdByRoad == Integer.MIN_VALUE)
125 {
126 ++nNextId;
127 if (oCurr.m_nLaneIndex < 0)
128 oCurr.m_nLaneIdByRoad = -nNextId;
129 else
130 oCurr.m_nLaneIdByRoad = nNextId;
131 }
132 }
133 oPrevLanes.clear();
134 oPrevLanes.addAll(oCurrLanes);
135 }
