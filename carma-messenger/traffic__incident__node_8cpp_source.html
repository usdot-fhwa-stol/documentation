25 void TrafficIncidentNode::initialize()
26 {
27
28 pnh_.getParam("sender_id", sender_id_);
29 pnh_.getParam("down_track", down_track_);
30 pnh_.getParam("up_track", up_track_);
31 pnh_.getParam("min_gap", min_gap_);
32 pnh_.getParam("event_reason", event_reason_);
33 pnh_.getParam("event_type", event_type_);
34
35 traffic_worker_.setSenderId(sender_id_);
36 traffic_worker_.setDownTrack(down_track_);
37 traffic_worker_.setUpTrack(up_track_);
38 traffic_worker_.setMinGap(min_gap_);
39 traffic_worker_.setEventReason(event_reason_);
40 traffic_worker_.setEventType(event_type_);
41
42 // Setup pub/sub
43 pinpoint_driver_sub_=nh_.subscribe("gps_common_fix",10,&TrafficIncidentWorker::pinpointDriverCallback,&traffic_worker_);
44 traffic_mobility_operation_pub_=nh_.advertise<cav_msgs::MobilityOperation>("outgoing_mobility_operation", 10);
45
46 //setup services
47 start_broadcast_request_service_server = nh_.advertiseService("start_broadcasting_traffic_event", &TrafficIncidentNode::startTrafficBroadcastCallback, this);
48 stop_broadcast_request_service_server = nh_.advertiseService("stop_broadcasting_traffic_event", &TrafficIncidentNode::stopTrafficBroadcastCallback, this);
49
50 //spin loop
51 ros::CARMANodeHandle::setSpinRate(10.0);
52 ros::CARMANodeHandle::setSpinCallback([this]() -> bool {
53 if(traffic_worker_.getDownTrack() >0 && traffic_worker_.getUpTrack()>0 && traffic_worker_.getMinGap() > 0 && traffic_worker_.getAdvisorySpeed() > 0)
54 {
55 //construct local mobilityOperation msg
56 cav_msgs::MobilityOperation traffic_mobility_msg = traffic_worker_.mobilityMessageGenerator(traffic_worker_.getPinPoint());
57
58 //start constantly broadcasting mobilityOperation msg
59 traffic_mobility_operation_pub_.publish(traffic_mobility_msg);
60 }
61 return true;
62 });
63 ROS_INFO_STREAM("Traffic Incident node is initialized...");
64 ros::CARMANodeHandle::spin();
65 }
