171 m_lTime = lTime;

174 m_nWidth = Mercator.mToCm(dLineArcs[7]);
175 int[] nPrevPt = new int[]{Mercator.mToCm(dLineArcs[5]), Mercator.mToCm(dLineArcs[6]), m_nWidth};
176 double dHdg = Geo.heading(nPrevPt[0], nPrevPt[1], dLineArcs[8], dLineArcs[9]);
177 dHdg = Math.toDegrees(dHdg);
178 m_nHeading = (int)(dHdg * 10 + 0.5);
179 Iterator<double[]> oIt = Arrays.iterator(dLineArcs, new double[3], 5, 3);
180 while (oIt.hasNext())

183 int nX = Mercator.mToCm(dPt[0]);
184 int nY = Mercator.mToCm(dPt[1]);

256 private void generateId(byte ySrc)
257 {
258 if (m_yId != null)
259 return;
260
261 Shake256 oMd = new Shake256();
262 try
263 (
264 DataOutputStream oAbsorb = new DataOutputStream(oMd.getAbsorbStream());
265 InputStream oSqueeze = oMd.getSqueezeStream();
266 )
267 {
268 oAbsorb.writeLong(m_lUpdated);
269 for (Integer nVal : m_nVTypes)
270 oAbsorb.writeByte(nVal); // unbox and force to byte
271
272 oAbsorb.writeLong(m_lStart); // include required schedule parameters
273 oAbsorb.writeLong(m_lEnd);
274 oAbsorb.writeByte(m_nDoW);
275
276 for (int[] nBegDur : m_nBetween) // auto-skip empty array
277 {
278 oAbsorb.writeShort(nBegDur[0]);
279 oAbsorb.writeShort(nBegDur[1]);
280 }
281
282 if (m_nOffset >= 0) // skip null schedule repeat parameters
283 {
284 oAbsorb.writeShort(m_nOffset);
285 oAbsorb.writeShort(m_nPeriod);
286 oAbsorb.writeShort(m_nSpan);
287 }
288
289 oAbsorb.writeBoolean(m_bRegulatory);
290 oAbsorb.writeByte(m_nControlType);
291 oAbsorb.write(m_yControlValue);
292
293 oAbsorb.writeLong(m_lTime);
294 oAbsorb.writeInt(m_nLon);
295 oAbsorb.writeInt(m_nLat);
296 oAbsorb.writeInt(m_nAlt);
297 oAbsorb.writeInt(m_nWidth);
298 oAbsorb.writeShort(m_nHeading);
299
300 for (TrafCtrlPt oPt : this) // include path points
301 {
302 oAbsorb.writeInt(oPt.m_nX);
303 oAbsorb.writeInt(oPt.m_nY);
304
305 if (oPt.m_nW >= 0)
306 oAbsorb.writeInt(oPt.m_nW);
307
308 if (oPt.m_nZ >= 0)
309 oAbsorb.writeInt(oPt.m_nZ);
310 }
311
312 byte[] yId = new byte[15];
313 oSqueeze.read(yId);
314 m_yId = new byte[16];
315 m_yId[0] = ySrc;
316 System.arraycopy(yId, 0, m_yId, 1, yId.length);
317 }
318 catch (Exception oEx)
319 {
320 }
321 }

424 protected void writeJson(StringBuilder sBuf)
425 throws Exception
426 {
427 sBuf.append(".json");
428 Path oPath = Paths.get(sBuf.toString());
429 Files.createDirectories(oPath.getParent(), FileUtil.DIRPERS);
430
431 sBuf.setLength(0); // reset JSON output buffer
432 sBuf.append("{\n"); // open ctrl object
433 sBuf.append("\t\"version\":\"").append(m_sVersion).append("\",\n");
434
435 sBuf.append("\t\"id\":\"");
436 Text.toHexString(m_yId, 0, m_yId.length, sBuf);
437 sBuf.append("\",\n");
438
439 SimpleDateFormat oFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
440 sBuf.append("\t\"updated\":\"").append(oFormat.format(new Date(m_lUpdated))).append("\",\n");
441
442 sBuf.append("\t\"vtypes\":\n\t["); // start vtype array
443 m_nVTypes.forEach((nVType) -> {sBuf.append("\n\t\t\"").append(TrafCtrlEnums.VTYPES[nVType]).append("\",");});
444 sBuf.setLength(sBuf.length() - 1); // remove trailing comma
445 sBuf.append("\n\t],\n"); // end vtype array
446
447 sBuf.append("\t\"schedule\":\n\t{\n"); // start schedule object
448 sBuf.append("\t\t\"start\":\"").append(oFormat.format(new Date(m_lStart))).append("\",\n");
449 sBuf.append("\t\t\"end\":\"").append(oFormat.format(new Date(m_lEnd))).append("\",\n");
450 sBuf.append("\t\t\"dow\":\"");
451 int nCount = 0;
452 int nDoW = m_nDoW; // bit-shift day-of-week characters
453 while (nDoW != 0)
454 {
455 if ((nDoW & 1) > 0)
456 sBuf.append(TrafCtrlEnums.DAYCHARS[nCount]);
457
458 ++nCount;
459 nDoW >>= 1;
460 }
461 sBuf.append("\",\n"); // end day-of-week
462
463 if (!m_nBetween.isEmpty()) // skip empty schedule between parameters
464 {
465 sBuf.append("\t\t\"between\":\n\t\t["); // start between array
466 for (int[] nBegDur : m_nBetween)

468 sBuf.append("\n\t\t\t{\"begin\":").append(nBegDur[0]);

472 sBuf.append("\n\t\t],\n"); // end between array

710 public void preparePoints(int nExplodeDist)
711 {
712 if (nExplodeDist == 0)
713 return;
714 double[] dC = m_oFullGeo.m_dC;
715 double[] dNT = m_oFullGeo.m_dNT;
716 int nLimit = Arrays.size(dC);
717 ArrayList<TrafCtrlPt> oNewPts = new ArrayList();
718 int[] nPrevPt = new int[]{Mercator.lonToCm(Geo.fromIntDeg(m_nLon)), Mercator.latToCm(Geo.fromIntDeg(m_nLat)), m_nWidth};
719 int nStep = (int)(nExplodeDist / (ProcCtrl.g_dExplodeStep * 100)) * 2;
720 for (int nIndex = 1; nIndex < nLimit; nIndex += nStep)
721 {
722 int nX = Mercator.mToCm(dC[nIndex]);
723 int nY = Mercator.mToCm(dC[nIndex + 1]);
724 int nW = Mercator.mToCm(Geo.distance(dC[nIndex], dC[nIndex + 1], dNT[nIndex], dNT[nIndex + 1]) * 2);
725 int nXd = nX - nPrevPt[0];
726 int nYd = nY - nPrevPt[1];
727 int nWd = nW - nPrevPt[2];
728 TrafCtrlPt oTemp = new TrafCtrlPt(nXd, nYd, nWd);
729 oNewPts.add(oTemp);
730 nPrevPt[0] = nX;
731 nPrevPt[1] = nY;
732 nPrevPt[2] = nW;
733 }
734
735 int nX = Mercator.mToCm(dC[nLimit - 2]);
736 int nY = Mercator.mToCm(dC[nLimit - 1]);
737 int nW = Mercator.mToCm(Geo.distance(dC[nLimit - 2], dC[nLimit - 1], dNT[nLimit - 2], dNT[nLimit - 1]) * 2);
738 int nXd = nX - nPrevPt[0];
739 int nYd = nY - nPrevPt[1];
740 int nWd = nW - nPrevPt[2];
741 if (nXd != 0 || nYd != 0 || nWd != 0)
742 oNewPts.add(new TrafCtrlPt(nXd, nYd, nWd));
743
744 clear();
745 addAll(oNewPts);
746 }
