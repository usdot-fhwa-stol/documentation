113 bool PlatoonControlPlugin::controlTimerCb()
114 {
115 ROS_DEBUG_STREAM("In control timer callback ");
116 // If it has been a long time since input data has arrived then reset the input counter and return
117 // Note: this quiets the controller after its input stream stops, which is necessary to allow
118 // the replacement controller to publish on the same output topic after this one is done.
119 long current_time = ros::Time::now().toNSec() / 1000000;
120 ROS_DEBUG_STREAM("current_time = " << current_time << ", prev_input_time_ = " << prev_input_time_ << ", input counter = " << consecutive_input_counter_);
121 if (current_time - prev_input_time_ > config_.shutdownTimeout)
122 {
123 ROS_DEBUG_STREAM("returning due to timeout.");
124 consecutive_input_counter_ = 0;
125 return false;
126 }
127
128 // If there have not been enough consecutive timely inputs then return (waiting for
129 // previous control plugin to time out and stop publishing, since it uses same output topic)
130 if (consecutive_input_counter_ <= config_.ignoreInitialInputs)
131 {
132 ROS_DEBUG_STREAM("returning due to first data input");
133 return false;
134 }
135
136 cav_msgs::TrajectoryPlanPoint second_trajectory_point = latest_trajectory_.trajectory_points[1];
137 cav_msgs::TrajectoryPlanPoint lookahead_point = getLookaheadTrajectoryPoint(latest_trajectory_);
138
139 trajectory_speed_ = getTrajectorySpeed(latest_trajectory_.trajectory_points);
140
141 generateControlSignals(second_trajectory_point, lookahead_point);
142
143 return true;
144 }

146 void PlatoonControlPlugin::trajectoryPlan_cb(const cav_msgs::TrajectoryPlan::ConstPtr& tp)
147 {
148 if (tp->trajectory_points.size() < 2) {
149 ROS_WARN_STREAM("PlatoonControlPlugin cannot execute trajectory as only 1 point was provided");
150 return;
151 }
152
153 latest_trajectory_ = *tp;
154 prev_input_time_ = ros::Time::now().toNSec() / 1000000;
155 ++consecutive_input_counter_;
156 ROS_DEBUG_STREAM("New trajectory plan #" << consecutive_input_counter_ << " at time " << prev_input_time_);
157 ROS_DEBUG_STREAM("tp header time = " << tp->header.stamp.toNSec() / 1000000);
158 }

160 cav_msgs::TrajectoryPlanPoint PlatoonControlPlugin::getLookaheadTrajectoryPoint(cav_msgs::TrajectoryPlan trajectory_plan)
161 {
162 cav_msgs::TrajectoryPlanPoint lookahead_point;
163
164 double lookahead_dist = config_.lookaheadRatio * current_speed_;
165 ROS_DEBUG_STREAM("lookahead based on speed: " << lookahead_dist);
166
167 lookahead_dist = std::max(config_.minLookaheadDist, lookahead_dist);
168 ROS_DEBUG_STREAM("final lookahead: " << lookahead_dist);
169
170 double traveled_dist = 0.0;
171 bool found_point = false;
172
173 for (size_t i = 1; i<trajectory_plan.trajectory_points.size() - 1; i++)
174 {
175 double dx = pose_msg_.pose.position.x - trajectory_plan.trajectory_points[i].x;
176 double dy = pose_msg_.pose.position.y - trajectory_plan.trajectory_points[i].y;
177
178 double dx1 = trajectory_plan.trajectory_points[i].x - trajectory_plan.trajectory_points[i-1].x;
179 double dy1 = trajectory_plan.trajectory_points[i].y - trajectory_plan.trajectory_points[i-1].y;
180 double dist1 = std::sqrt(dx1*dx1 + dy1*dy1);
181 ROS_DEBUG_STREAM("trajectory spacing: " << dist1);
182
183 double dist = std::sqrt(dx*dx + dy*dy);
184
185 traveled_dist = dist;
186
187 if ((lookahead_dist - traveled_dist) < 1.0)
188 {
189 lookahead_point = trajectory_plan.trajectory_points[i];
190 found_point = true;
191 ROS_DEBUG_STREAM("found lookahead point at index: " << i);
192 break;
193 }
194 }
195
196 if (!found_point)
197 {
198 lookahead_point = trajectory_plan.trajectory_points.back();
199 ROS_DEBUG_STREAM("lookahead point set as the last trajectory point");
200 }
201
202
203 return lookahead_point;
204 }

290 // extract maximum speed of trajectory
291 double PlatoonControlPlugin::getTrajectorySpeed(std::vector<cav_msgs::TrajectoryPlanPoint> trajectory_points)
292 {
293 double trajectory_speed = 0;
294
295 double dx1 = trajectory_points[trajectory_points.size()-1].x - trajectory_points[0].x;
296 double dy1 = trajectory_points[trajectory_points.size()-1].y - trajectory_points[0].y;
297 double d1 = sqrt(dx1*dx1 + dy1*dy1);
298 double t1 = (trajectory_points[trajectory_points.size()-1].target_time.toSec() - trajectory_points[0].target_time.toSec());
299
300 double avg_speed = d1/t1;
301 ROS_DEBUG_STREAM("trajectory_points size = " << trajectory_points.size() << ", d1 = " << d1 << ", t1 = " << t1 << ", avg_speed = " << avg_speed);
302
303 for(size_t i = 0; i < trajectory_points.size() - 2; i++ )
304 {
305 double dx = trajectory_points[i + 1].x - trajectory_points[i].x;
306 double dy = trajectory_points[i + 1].y - trajectory_points[i].y;
307 double d = sqrt(dx*dx + dy*dy);
308 double t = (trajectory_points[i + 1].target_time.toSec() - trajectory_points[i].target_time.toSec());
309 double v = d/t;
310 if(v > trajectory_speed)
311 {
312 trajectory_speed = v;
313 }
314 }
315
316 ROS_DEBUG_STREAM("trajectory speed: " << trajectory_speed);
317 ROS_DEBUG_STREAM("avg trajectory speed: " << avg_speed);
318
319 return avg_speed; //TODO: why are 2 speeds being calculated? Which should be returned?
320
321 }
