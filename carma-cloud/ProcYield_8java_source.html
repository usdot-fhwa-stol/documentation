60 @Override
61 public void parseMetadata(Path oSource)
62 throws Exception
63 {
64 m_oJunctions.clear();
65 m_oSignals.clear();
66 m_nInterPaths[0] = 1;
67 m_nSignalRoads[0] = 1;
68 m_oIncomingRoads.clear();
69 new XodrSignalParser(SIGNAL_TYPES).parseSignalData(oSource, m_oJunctions, m_oSignals);
70 for (Signal oSignal : m_oSignals)
71 {
72 Iterator<int[]> oIt = Arrays.iterator(oSignal.m_nRoads, new int[1], 1, 1);
73 while (oIt.hasNext())
74 {
75 int nRoad = oIt.next()[0];
76 int nIndex = Arrays.binarySearch(m_nSignalRoads, nRoad);
77 if (nIndex < 0)
78 m_nSignalRoads = Arrays.insert(m_nSignalRoads, nRoad, ~nIndex);
79 }
80 }
81
82 RoadPaths oSearch = new RoadPaths();
83 for (Junction oJunc : m_oJunctions)
84 {
85 for (Connection oConn : oJunc)
86 {
87 int nIndex = Arrays.binarySearch(m_nInterPaths, oConn.m_nConnRoad);
88 if (nIndex < 0)
89 m_nInterPaths = Arrays.insert(m_nInterPaths, oConn.m_nConnRoad, ~nIndex);
90
91 oSearch.m_nRoadId = oConn.m_nInRoad;
92 nIndex = Collections.binarySearch(m_oIncomingRoads, oSearch);
93 if (nIndex < 0)
94 {
95 nIndex = ~nIndex;
96 m_oIncomingRoads.add(nIndex, new RoadPaths(oConn.m_nInRoad));
97 }
98
99 if (Arrays.binarySearch(m_nSignalRoads, oConn.m_nConnRoad) >= 0)
100 {
101 RoadPaths oTemp = m_oIncomingRoads.get(nIndex);
102 oTemp.addConnection(oConn);
103 }
104 }
105 }

144 @Override
145 public ArrayList<CtrlLineArcs> combine(ArrayList<CtrlLineArcs> oLanes, double dTol)
146 {
147 ArrayList<CtrlLineArcs> oLanesByRoads = ProcCtrl.combineLaneByRoad(oLanes, dTol);
148 Collections.sort(oLanesByRoads, CtrlLineArcs.CMPBYLANE);
149 ArrayList<CtrlLineArcs> oCombined = new ArrayList();
150 ArrayList<CtrlLineArcs> oInRoadsToSignals = new ArrayList();
151 int nIndex = oLanesByRoads.size();
152 RoadPaths oRPSearch = new RoadPaths();
153 ArrayList<Signal> oSignals = new ArrayList();
154 while (nIndex-- > 0)
155 {
156 int nRoadId = XodrUtil.getRoadId(oLanesByRoads.get(nIndex).m_nLaneId);
157 oRPSearch.m_nRoadId = nRoadId;
158 int nSearchIndex;
159 if (Arrays.binarySearch(m_nInterPaths, nRoadId) >= 0)
160 {
161 oLanesByRoads.remove(nIndex);
162 }
163 else if ((nSearchIndex = Collections.binarySearch(m_oIncomingRoads, oRPSearch)) >= 0)
164 {
165 oSignals.clear();
166 RoadPaths oRoadPaths = m_oIncomingRoads.get(nSearchIndex);
167 int nLaneIndex = XodrUtil.getLaneIndex(oLanesByRoads.get(nIndex).m_lLaneSectionId);
168 for (Signal oSignal : m_oSignals)
169 {
170 boolean bAdded = false;
171 Iterator<int[]> oIt = Arrays.iterator(oSignal.m_nRoads, new int[1], 1, 1);
172 while (oIt.hasNext())
173 {
174 int nConnPath = oIt.next()[0];
175 for (Connection oConn : oRoadPaths.m_oConns)
176 {
177 if (oConn.m_nConnRoad == nConnPath)
178 {
179 if (nLaneIndex == oConn.m_nFromLane)
180 {
181 oSignals.add(oSignal);
182 bAdded = true;
183 break;
184 }
185 }
186 }
187 if (bAdded)
188 break;
189 }
190 }
191
192 boolean bHasSignal = false;
193 boolean bPosLaneIndex = nLaneIndex > 0;
194 for (Signal oSignal : oSignals)
195 {
196 if (oSignal.m_nOri == Signal.BOTHORI)
197 {
198 bHasSignal = true;
199 break;
200 }
201 else if (oSignal.m_nOri == Signal.POSORI)
202 {
203 if (bPosLaneIndex)
204 {
205 bHasSignal = true;
206 break;
207 }
208 }
209 else if (oSignal.m_nOri == Signal.NEGORI)
210 {
211 if (!bPosLaneIndex)
212 {
213 bHasSignal = true;
214 break;
215 }
216 }
217 }
218
219 if (bHasSignal)
220 oInRoadsToSignals.add(oLanesByRoads.get(nIndex));
221
222 oLanesByRoads.remove(nIndex);
223 }
224 }
225
226 int nOuterLimit = oInRoadsToSignals.size();
227 int nInnerLimit = oLanesByRoads.size();
228 double dSqTol = dTol * dTol;
229
230 for (int nOuter = 0; nOuter < nOuterLimit; nOuter++)
231 {
232 oSignals.clear();
233 CtrlLineArcs oCur = oInRoadsToSignals.get(nOuter);
234 double dLen = CtrlLineArcs.getLen(oCur.m_dLineArcs);
235 boolean bAdded = false;
236 if (dLen >= MINLEN)
237 {
238 oCombined.add(oCur);
239 continue;
240 }
241
242
243 for (int nInner = 0; nInner < nInnerLimit; nInner++)
244 {
245 CtrlLineArcs oCmp = oLanesByRoads.get(nInner);
246
247 int nConnect = oCur.connects(oCmp, dSqTol);
248 if (nConnect == CtrlLineArcs.CON_TSTART_OEND) // the other linearc connects at the start of this linearc
249 {
250 double dOtherLen = CtrlLineArcs.getLen(oCmp.m_dLineArcs);
251 if (dLen + dOtherLen < MINLEN)
252 {
253 oCur.combine(oCmp, nConnect); // combine the points of oCmp into oCur's point array
254 oLanesByRoads.remove(nInner);
255 --nInnerLimit;
256 nInner = -1; // start inner loop over
257 }
258 else
259 {
260 double[] dLineArc = oCmp.m_dLineArcs;
261 double[] dPartial = Arrays.newDoubleArray();
262 double[] dCenter = new double[2];
263 nIndex = Arrays.size(dLineArc) - 9;
264 dPartial = Arrays.add(dPartial, dLineArc[nIndex + 6], dLineArc[nIndex + 7]);
265 dPartial = Arrays.add(dPartial, dLineArc[nIndex + 8]);
266
267 for (;nIndex >= 5; nIndex -= 6)
268 {
269 if (dLen >= MINLEN)
270 {
271 nConnect = oCur.connects(dLineArc, dSqTol);
272 oCur.combine(dLineArc, nConnect);
273 oCombined.add(oCur);
274 bAdded = true;
275 break;
276 }
277 double dX3 = dLineArc[nIndex];
278 double dY3 = dLineArc[nIndex + 1];
279 double dX2 = dLineArc[nIndex + 3];
280 double dY2 = dLineArc[nIndex + 4];
281 double dX1 = dLineArc[nIndex + 6];
282 double dY1 = dLineArc[nIndex + 7];
283 dPartial = Arrays.add(dPartial, dX2, dY2);
284 dPartial = Arrays.add(dPartial, dLineArc[nIndex + 5]);
285 dPartial = Arrays.add(dPartial, dX3, dY3);
286 dPartial = Arrays.add(dPartial, dLineArc[nIndex + 2]);
287 double dR = Geo.circle(dX1, dY1, dX2, dY2, dX3, dY3, dCenter);
288 if (!Double.isFinite(dR) || dR >= 10000) // expand line
289 {
290 dLen += Geo.distance(dX1, dY1, dX3, dY3);
291 }
292 else
293 {
294 int nRightHand = Geo.rightHand(dX2, dY2, dX1, dY1, dX3, dY3);
295 double dC = -nRightHand / dR;
296 double dCmAngleStep = dC / 100;
297 double dCmAngleStepMag = Math.abs(dCmAngleStep);
298 double dH = dCenter[0];
299 double dK = dCenter[1];
300 double dHdg = Geo.heading(dH, dK, dX1, dY1);
301 int nSteps = 0;
302 double dPrevX = dX1;
303 double dPrevY = dY1;
304 double dPrevAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dX3, dY3);
305 while (dPrevAngle > dCmAngleStepMag)
306 {
307 dPrevX = dH + dR * Math.cos(dHdg + dCmAngleStep * ++nSteps);
308 dPrevY = dK + dR * Math.sin(dHdg + dCmAngleStep * nSteps);
309 dLen += 0.01;
310 dPrevAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dX3, dY3);
311 }
312 double dLastAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dX3, dY3);
313 dLen += dLastAngle * dR;
314 }
315 }
316 }
317 }
318 }
319
320 if (!bAdded)
321 oCombined.add(oCur);
322 }
323 return oCombined;
324 }

327 public static void renderTiledData(ArrayList<TrafCtrl> oCtrls, ArrayList<int[]> nTiles) throws IOException
328 {
329 String[] sEmpty = new String[0];
330 TdLayer oOutWhite = new TdLayer("yield-ow", sEmpty, sEmpty, TdLayer.POLYGON);
331 TdLayer oRed = new TdLayer("yield-r", sEmpty, sEmpty, TdLayer.POLYGON);
332 TdLayer oInWhite = new TdLayer("yield-iw", sEmpty, sEmpty, TdLayer.POLYGON);
333 TdLayer oStopLine = new TdLayer("yield-sl", sEmpty, sEmpty, TdLayer.POLYGON);
334 int[] nTags = new int[0];
335 double[] dPt = new double[2];
336 double[][] dPolys = new double[YIELDSIGN.length][];
337 int[] nPts = new int[dPolys.length];
338 for (int nIndex = 0; nIndex < dPolys.length; nIndex++)
339 {
340 double[] dTemp = new double[YIELDSIGN[nIndex].length + 1];
341 dTemp[0] = dTemp.length;
342 nPts[nIndex] = YIELDSIGN[nIndex].length / 2;
343 dPolys[nIndex] = dTemp;
344 }
345
346 ArrayList<double[]> dStopLines = new ArrayList();
347 for (int[] nTile : nTiles)
348 {
349 int nX = nTile[0];
350 int nY = nTile[1];
351 Files.createDirectories(Paths.get(String.format(g_sTdFileFormat, nX, 0, 0, 0)).getParent(), FileUtil.DIRPERS);
352 writeIndexFile(oCtrls, nX, nY);
353 oOutWhite.clear();
354 oRed.clear();
355 oInWhite.clear();
356 oStopLine.clear();
357 dStopLines.clear();
358 double[] dClipBounds = TileUtil.getClippingBounds(g_nDefaultZoom, nX, nY);
359 for (TrafCtrl oCtrl : oCtrls)
360 {
361 if (Collections.binarySearch(oCtrl.m_oFullGeo.m_oTiles, nTile, Mercator.TILECOMP) < 0 || oCtrl.m_oFullGeo.m_dAverageWidth < WIDTHTH)
362 continue;
363
364 double[] dC = oCtrl.m_oFullGeo.m_dC;
365 int nSize = Arrays.size(dC);
366 int nIndex = getStopLine(oCtrl, dStopLines);
367 double[] dStopLine = dStopLines.get(dStopLines.size() - 1);
368 double[] dBoundingBox = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
369 Iterator<double[]> oIt = Arrays.iterator(dStopLine, new double[2], 1, 2);
370 while (oIt.hasNext())
371 {
372 double[] dPoint = oIt.next();
373 if (dPoint[0] < dBoundingBox[0])
374 dBoundingBox[0] = dPoint[0];
375 if (dPoint[1] < dBoundingBox[1])
376 dBoundingBox[1] = dPoint[1];
377 if (dPoint[0] > dBoundingBox[2])
378 dBoundingBox[2] = dPoint[0];
379 if (dPoint[1] > dBoundingBox[3])
380 dBoundingBox[3] = dPoint[1];
381 }
382 if (Geo.boundingBoxesIntersect(dBoundingBox, dClipBounds))
383 {
384 oStopLine.add(new TdFeature(dStopLine, nTags, oCtrl));
385 }
386 double dLen = 0.0;
387 while (dLen < 0.5 && nIndex > 0)
388 {
389 double dX1 = dC[nIndex];
390 double dY1 = dC[nIndex + 1];
391 double dX2 = dC[nIndex + 2];
392 double dY2 = dC[nIndex + 3];
393 dLen += Geo.distance(dX1, dY1, dX2, dY2);
394 nIndex -= 2;
395 }
396 double dX1 = dC[nIndex];
397 double dY1 = dC[nIndex + 1];
398 double dX2 = dC[nSize - 2];
399 double dY2 = dC[nSize - 1];
400 double dHdg = Geo.heading(dX1, dY1, dX2, dY2);
401 if (Double.isNaN(dHdg))
402 continue;
403
404 // offset from center line
405 dPt[0] = dX1 + Math.sin(dHdg) * 0.5; // cos(x - pi/2) = sin(x)
406 dPt[1] = dY1 - Math.cos(dHdg) * 0.5; // sin(x - pi/2) = -cos(x)
407 double[] dSignBb = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
408 AffineTransform oAt = new AffineTransform();
409 oAt.translate(dPt[0], dPt[1]);
410 oAt.rotate(dHdg, 0, 0);
411 oAt.transform(YIELDSIGN[0], 0, dPolys[0], 1, nPts[0]);
412
413 oIt = Arrays.iterator(dPolys[0], new double[2], 1, 2);
414 while (oIt.hasNext())
415 {
416 double[] dPoint = oIt.next();
417 if (dPoint[0] < dSignBb[0])
418 dSignBb[0] = dPoint[0];
419 if (dPoint[1] < dSignBb[1])
420 dSignBb[1] = dPoint[1];
421 if (dPoint[0] > dSignBb[2])
422 dSignBb[2] = dPoint[0];
423 if (dPoint[1] > dSignBb[3])
424 dSignBb[3] = dPoint[1];
425 }
426
427 if (!Geo.boundingBoxesIntersect(dSignBb, dClipBounds))
428 continue;
429
430
431 oAt.transform(YIELDSIGN[0], 0, dPolys[0], 1, nPts[0]);
432 oOutWhite.add(new TdFeature(dPolys[0], nTags, oCtrl));
433
434 oAt.transform(YIELDSIGN[1], 0, dPolys[1], 1, nPts[1]);
435 oRed.add(new TdFeature(dPolys[1], nTags, oCtrl));
436
437 oAt.transform(YIELDSIGN[2], 0, dPolys[2], 1, nPts[2]);
438 oInWhite.add(new TdFeature(dPolys[2], nTags, oCtrl));
439 }
440 if (!oOutWhite.isEmpty() || !oStopLine.isEmpty())
441 {
442 try (DataOutputStream oOut = new DataOutputStream(new BufferedOutputStream(FileUtil.newOutputStream(Paths.get(String.format(g_sTdFileFormat, nX, g_nDefaultZoom, nX, nY)), FileUtil.APPENDTO, FileUtil.FILEPERS))))
443 {
444 oOutWhite.write(oOut);
445 oRed.write(oOut);
446 oInWhite.write(oOut);
447 oStopLine.write(oOut);
448 }
449 }
450 }
451 }
