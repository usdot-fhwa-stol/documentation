41 double getManeuverEndSpeed(const carma_planning_msgs::msg::Maneuver& mvr)
42 {
43 switch(mvr.type)
44 {
45 case carma_planning_msgs::msg::Maneuver::LANE_FOLLOWING:
46 return mvr.lane_following_maneuver.end_speed;
47 case carma_planning_msgs::msg::Maneuver::LANE_CHANGE:
48 return mvr.lane_change_maneuver.end_speed;
49 case carma_planning_msgs::msg::Maneuver::STOP_AND_WAIT:
50 return 0;
51 default:
52 RCLCPP_ERROR_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "Requested end speed from unsupported maneuver type");
53 return 0;
54 }
55 }

132StopAndDwellStrategicPlugin::VehicleState StopAndDwellStrategicPlugin::extractInitialState(const carma_planning_msgs::srv::PlanManeuvers::Request& req) const
133{
134 VehicleState state;
135 if (!req.prior_plan.maneuvers.empty())
136 {
137 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "Provided with initial plan...");
138 state.stamp = GET_MANEUVER_PROPERTY(req.prior_plan.maneuvers.back(), end_time);
139 state.speed = getManeuverEndSpeed(req.prior_plan.maneuvers.back());
140 state.downtrack = GET_MANEUVER_PROPERTY(req.prior_plan.maneuvers.back(), end_dist);
141 state.lane_id = getLaneletsBetweenWithException(state.downtrack, state.downtrack, true).front().id();
142 }
143 else
144 {
145 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "No initial plan provided...");
146
147 state.stamp = req.header.stamp;
148 state.downtrack = req.veh_downtrack;
149 state.speed = req.veh_logitudinal_velocity;
150 state.lane_id = stoi(req.veh_lane_id);
151 }
152
153 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "state.stamp: " << std::to_string(state.stamp.seconds()));
154 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "state.downtrack : " << state.downtrack );
155 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "state.speed: " << state.speed);
156 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "state.lane_id: " << state.lane_id);
157
158 return state;
159}

179std::vector<lanelet::ConstLanelet> StopAndDwellStrategicPlugin::getLaneletsBetweenWithException(double start_downtrack,
180 double end_downtrack,
181 bool shortest_path_only,
182 bool bounds_inclusive) const
183{
184 std::vector<lanelet::ConstLanelet> crossed_lanelets =
185 wm_->getLaneletsBetween(start_downtrack, end_downtrack, shortest_path_only, bounds_inclusive);
186
187 if (crossed_lanelets.empty())
188 {
189 throw std::invalid_argument("getLaneletsBetweenWithException called but inputs do not cross any lanelets going "
190 "from: " +
191 std::to_string(start_downtrack) + " to: " + std::to_string(end_downtrack));
192 }
193
194 return crossed_lanelets;
195}

207carma_planning_msgs::msg::Maneuver StopAndDwellStrategicPlugin::composeLaneFollowingManeuverMessage(int case_num, double start_dist, double end_dist,
208 double start_speed, double target_speed,

210 std::vector<lanelet::Id> lane_ids)

214 maneuver_msg.type = carma_planning_msgs::msg::Maneuver::LANE_FOLLOWING;

220 maneuver_msg.lane_following_maneuver.start_dist = start_dist;
221 maneuver_msg.lane_following_maneuver.end_dist = end_dist;
222 maneuver_msg.lane_following_maneuver.start_speed = start_speed;

224 maneuver_msg.lane_following_maneuver.start_time = start_time;
225 maneuver_msg.lane_following_maneuver.end_time = start_time + rclcpp::Duration(time_to_stop*1e9);

247carma_planning_msgs::msg::Maneuver StopAndDwellStrategicPlugin::composeStopAndWaitManeuverMessage(double current_dist, double end_dist,
248 double start_speed,
249 const lanelet::Id& starting_lane_id,
250 const lanelet::Id& ending_lane_id, double stopping_accel,
251 rclcpp::Time start_time, rclcpp::Time end_time) const
252{
253 carma_planning_msgs::msg::Maneuver maneuver_msg;
254 maneuver_msg.type = carma_planning_msgs::msg::Maneuver::STOP_AND_WAIT;
255 maneuver_msg.stop_and_wait_maneuver.parameters.planning_strategic_plugin = config_.strategic_plugin_name;
256 maneuver_msg.stop_and_wait_maneuver.parameters.presence_vector =
257 carma_planning_msgs::msg::ManeuverParameters::HAS_TACTICAL_PLUGIN | carma_planning_msgs::msg::ManeuverParameters::HAS_FLOAT_META_DATA;
258 maneuver_msg.stop_and_wait_maneuver.parameters.negotiation_type = carma_planning_msgs::msg::ManeuverParameters::NO_NEGOTIATION;
259 maneuver_msg.stop_and_wait_maneuver.parameters.planning_tactical_plugin = config_.stop_and_wait_plugin_name;
260 maneuver_msg.stop_and_wait_maneuver.start_speed = start_speed;
261 maneuver_msg.stop_and_wait_maneuver.start_dist = current_dist;
262 maneuver_msg.stop_and_wait_maneuver.end_dist = end_dist;
263 maneuver_msg.stop_and_wait_maneuver.start_time = start_time;
264 maneuver_msg.stop_and_wait_maneuver.end_time = end_time;
265 maneuver_msg.stop_and_wait_maneuver.starting_lane_id = std::to_string(starting_lane_id);
266 maneuver_msg.stop_and_wait_maneuver.ending_lane_id = std::to_string(ending_lane_id);
267 // Set the meta data for the stop location buffer
268 maneuver_msg.stop_and_wait_maneuver.parameters.float_valued_meta_data.push_back(config_.stop_line_buffer);
269 maneuver_msg.stop_and_wait_maneuver.parameters.float_valued_meta_data.push_back(stopping_accel);
270
271 RCLCPP_INFO_STREAM(rclcpp::get_logger("stop_and_dwell_strategic_plugin"), "Creating stop and wait start dist: " << current_dist << " end dist: " << end_dist);
272
273 return maneuver_msg;
274}
