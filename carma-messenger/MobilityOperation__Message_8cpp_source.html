75 //get bsm id
76 //sender_bsm_id is meant to represent the vehicle BSM id in hex string (Ex: FFFFFFFF)
77 str_len=message->value.choice.TestMessage03.header.hostBSMId.size;
78 for(size_t i=0;i<str_len;i++){
79 sender_bsm_id +=message->value.choice.TestMessage03.header.hostBSMId.buf[i];
80 }
81 if(str_len<Header_constant.BSM_ID_LENGTH){
82 sender_bsm_id=std::string((Header_constant.BSM_ID_LENGTH-str_len),'0').append(sender_bsm_id);
83 }
84 else if(str_len>Header_constant.BSM_ID_LENGTH){
85 RCLCPP_WARN(node_logging_->get_logger(),"BSM ID -size greater than limit, changing to default");
86 sender_bsm_id=Header_constant.BSM_ID_DEFAULT;
87 }
88
89 header.sender_bsm_id=sender_bsm_id;
90
91 //get plan id
92 str_len=message->value.choice.TestMessage03.header.planId.size;
93 if(str_len==Header_constant.GUID_LENGTH)
94 {
95 for(size_t i=0;i<str_len;i++){
96 plan_id +=message->value.choice.TestMessage03.header.planId.buf[i];
97 }
98 }
99 else plan_id=Header_constant.GUID_DEFAULT;
100
101 header.plan_id=plan_id;
102
103 //recover uint64_t timestamp from string
104 str_len=message->value.choice.TestMessage03.header.timestamp.size;
105 timestamp=0;
106 char timestamp_ch[str_len];
107 for(size_t i=0;i<str_len;i++){
108 timestamp_ch[i]=message->value.choice.TestMessage03.header.timestamp.buf[i];
109 }
110 timestamp=atoll(timestamp_ch);
111 header.timestamp=timestamp;
112 output.m_header=header;
113
114 //get strategy
115 str_len=message->value.choice.TestMessage03.body.strategy.size;
116 if(str_len>=STRATEGY_MIN_LENGTH && str_len<=STRATEGY_MAX_LENGTH)
117 {
118 for(size_t i=0;i<str_len;i++){
119 strategy +=message->value.choice.TestMessage03.body.strategy.buf[i];
120 }
121 }
122 else strategy=Header_constant.STRING_DEFAULT;
123
124 output.strategy=strategy;
125
126 //get strategy params
127 str_len=message->value.choice.TestMessage03.body.operationParams.size;
128 if(str_len>=STRATEGY_PARAMS_MIN_LENGTH && str_len<=STRATEGY_PARAMS_MAX_LENGTH){
129 for(size_t i=0;i<str_len;i++){
130 strategy_params +=message->value.choice.TestMessage03.body.operationParams.buf[i];
131 }
132 boost::replace_all(strategy_params, "&quot;", "\"");
133 }
134 else strategy_params=STRATEGY_PARAMS_STRING_DEFAULT;
135
136 output.strategy_params=strategy_params;
137
138 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
139 return boost::optional<carma_v2x_msgs::msg::MobilityOperation>(output);
140 }
141 RCLCPP_WARN_STREAM(node_logging_->get_logger(), "mobility operation decoding failed");
142 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
143 return boost::optional<carma_v2x_msgs::msg::MobilityOperation>{};
