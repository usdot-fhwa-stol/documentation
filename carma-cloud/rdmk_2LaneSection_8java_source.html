72 boolean bCenterNone = m_oCenter.m_nType == XodrUtil.getLaneType("none");
73 Collections.sort(m_oLeft, LEFTLANES);
74 setInnerRoadMarks(m_oLeft);

78 m_oLeft.add(0, null);
79 else if (bCenterNone) // if center is none, use the first right lane
80 m_oLeft.add(0, m_oRight.get(0));

84 Collections.sort(m_oRight, RIGHTLANES);

91 m_oRight.add(0, m_oLeft.get(1));
