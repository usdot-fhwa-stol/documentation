319void NS3Adapter::loadWaveConfig(const std::string &fileName)
320{
321 ROS_DEBUG_STREAM("Loading wave config");
322 const char* schema = "{\n"
323 " \"$schema\":\"http://json-schema.org/draft-06/schema\",\n"
324 " \"title\":\"Wave Config Schema\",\n"
325 " \"description\":\"A simple schema to describe DSRC/Wave messages\",\n"
326 " \"type\": \"array\",\n"
327 " \"items\": {\n"
328 " \"type\": \"object\",\n"
329 " \"properties\": {\n"
330 " \"name\": {\n"
331 " \"description\": \"message type - abbreviated name\",\n"
332 " \"type\": \"string\"\n"
333 " },\n"
334 " \"psid\": {\n"
335 " \"description\": \"psid assigned to message type in decimal\",\n"
336 " \"type\": \"string\"\n"
337 " },\n"
338 " \"ns3_id\": {\n"
339 " \"description\": \"J2735 NS-3 id assigned to message type in decimal\",\n"
340 " \"type\": \"string\"\n"
341 " },\n"
342 " \"channel\": {\n"
343 " \"description\": \"DSRC radio channel assigned to message type in decimal\",\n"
344 " \"type\": \"string\"\n"
345 " },\n"
346 " \"priority\": {\n"
347 " \"description\": \"WSM Priotiy to use assigned to message type in decimal\",\n"
348 " \"type\":\"string\"\n"
349 " }\n"
350 " },\n"
351 " \"required\":[\"name\",\"psid\",\"ns3_id\",\"channel\",\"priority\"]"
352 " }\n"
353 "}\n";
354
355 std::ifstream file;
356 try
357 {
358 file.open(fileName);
359 }
360 catch (std::exception& e)
361 {
362 ROS_ERROR_STREAM("Unable to open file : " << fileName << ", exception: " << e.what());
363 return;
364 }
365
366 rapidjson::Document sd;
367 if(sd.Parse(schema).HasParseError())
368 {
369 ROS_ERROR_STREAM("Invalid Wave Config Schema");
370 return;
371 }
372
373 rapidjson::SchemaDocument schemaDocument(sd);
374 rapidjson::Document doc;
375 rapidjson::IStreamWrapper isw(file);
376 if(doc.ParseStream(isw).HasParseError())
377 {
378 ROS_ERROR_STREAM("Error Parsing Wave Config");
379 return;
380 }
381
382 rapidjson::SchemaValidator validator(schemaDocument);
383 if(!doc.Accept(validator))
384 {
385 ROS_ERROR_STREAM("Wave Config improperly formatted");
386 return;
387 }
388
389 for(auto& it : doc.GetArray())
390 {
391 auto entry = it.GetObject();
392 wave_cfg_items_.emplace_back(entry["name"].GetString(),
393 entry["psid"].GetString(),
394 entry["ns3_id"].GetString(),
395 entry["channel"].GetString(),
396 entry["priority"].GetString());
397
398 }

423std::string NS3Adapter::compose_handshake_msg(std::string veh_id, std::string role_id, std::string port, std::string ip)
424{
425 // document is the root of a json message
426 rapidjson::Document document;
427
428 // define the document as an object rather than an array
429 document.SetObject();
430
431 // create a rapidjson array type with similar syntax to std::vector
432 rapidjson::Value array(rapidjson::kArrayType);
433
434 // must pass an allocator when the object may need to allocate memory
435 rapidjson::Document::AllocatorType& allocator = document.GetAllocator();
436
437 rapidjson::Value idtextPart;
438 idtextPart.SetString(veh_id.c_str(), allocator);
439 document.AddMember("vehicle_id", idtextPart, allocator);
440
441 rapidjson::Value roletextPart;
442 roletextPart.SetString(role_id.c_str(), allocator);
443 document.AddMember("role_id", roletextPart, allocator);
444
445 rapidjson::Value porttextPart;
446 porttextPart.SetString(port.c_str(), allocator);
447 document.AddMember("port", porttextPart, allocator);
448
449 rapidjson::Value iptextPart;
450 iptextPart.SetString(ip.c_str(), allocator);
451 document.AddMember("host_ip", iptextPart, allocator);
452
453 rapidjson::StringBuffer strbuf;
454 rapidjson::Writer<rapidjson::StringBuffer> writer(strbuf);
455 document.Accept(writer);
456
457 std::string strbufstring = strbuf.GetString();
458
459 return strbufstring;
460}
