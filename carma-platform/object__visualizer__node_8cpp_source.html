86 void Node::external_objects_callback(carma_perception_msgs::msg::ExternalObjectList::UniquePtr msg)
87 {
88 RCLCPP_DEBUG_STREAM( get_logger(), "external_objects_callback called");
89
90 if (!config_.enable_external_objects_viz) {
91 RCLCPP_DEBUG_STREAM( get_logger(), "external_objects_callback called, but visualization is not enabled.");
92 return;
93 }
94
95 visualization_msgs::msg::MarkerArray viz_msg;
96 //delete all markers before adding new ones
97 visualization_msgs::msg::Marker marker;
98 viz_msg.markers.reserve(msg->objects.size() + 1); //+1 to account for delete all marker
99 marker.id = 0;
100 marker.action = visualization_msgs::msg::Marker::DELETEALL;
101 viz_msg.markers.push_back(marker);
102
103
104 size_t id = 0; // We always count the id from zero so we can delete markers later in a consistent manner
105
106 for (auto obj : msg->objects) {
107 visualization_msgs::msg::Marker marker;
108
109 marker.header = msg->header;
110
111 marker.ns = config_.external_objects_viz_ns;
112
113 // Pedestrian's will be represented as red box
114 if (obj.object_type == carma_perception_msgs::msg::ExternalObject::PEDESTRIAN) {
115
116 marker.color.r = 1.0;
117 marker.color.g = 0.0;
118 marker.color.b = 0.0;
119 marker.color.a = 1.0;
120
121 } // All other detected objects will be blue
122 else {
123
124 marker.color.r = 0.0;
125 marker.color.g = 0.0;
126 marker.color.b = 1.0;
127 marker.color.a = 1.0;
128 }
129
130 marker.id = id;
131 marker.type = visualization_msgs::msg::Marker::CUBE;
132 marker.action = visualization_msgs::msg::Marker::ADD;
133 marker.pose = obj.pose.pose;
134 marker.scale.x = obj.size.x * 2.0; // Size in carma is half the length/width/height
135 marker.scale.y = obj.size.y * 2.0;
136 marker.scale.z = obj.size.z * 2.0;
137
138 viz_msg.markers.push_back(marker);
139
140 id++;
141 }
142
143 clear_and_update_old_objects(viz_msg, prev_external_objects_size_);
144
145 external_objects_viz_pub_->publish(viz_msg);
146
147 }

149 void Node::clear_and_update_old_objects(visualization_msgs::msg::MarkerArray &viz_msg, size_t &old_size)
150 {
151 size_t size_swap = viz_msg.markers.size();
152
153 // Delete any markers from the previous message that are no longer in the current message
154 if (viz_msg.markers.size() < old_size) {
155
156 for (size_t i = viz_msg.markers.size(); i < old_size; ++i) {
157 visualization_msgs::msg::Marker marker;
158 marker.id = i;
159 marker.action = visualization_msgs::msg::Marker::DELETE;
160 viz_msg.markers.push_back(marker);
161 }
162
163 }
164
165 old_size = size_swap; // Assign the new size of added objects to the old size
166 }
