59 @Override
60 public void proc(String sLineArcsFile, double dTol)
61 throws Exception
62 {
63 try
64 {
65 parseMetadata(Paths.get(m_sXodrDir + sLineArcsFile.substring(sLineArcsFile.lastIndexOf("/")).replace(".bin", ".xodr")));
66 ArrayList<TrafCtrl> oCtrls = new ArrayList();
67 ArrayList<CtrlLineArcs> oLineArcs = new ArrayList();
68 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(Paths.get(sLineArcsFile)))))

96 @Override
97 public ArrayList<CtrlLineArcs> combine(ArrayList<CtrlLineArcs> oLanes, double dTol)
98 {
99 ArrayList<CtrlLineArcs> oLanesByRoads = ProcCtrl.combineLaneByRoad(oLanes, dTol);
100 Collections.sort(oLanesByRoads, CtrlLineArcs.CMPBYLANE);
101 ArrayList<CtrlLineArcs> oCombined = new ArrayList();
102 int nIndex = oLanesByRoads.size();
103 while (nIndex-- > 0)
104 {
105 CtrlLineArcs oCla = oLanesByRoads.get(nIndex);
106 String sRoadId = Integer.toString(XodrUtil.getRoadId(oCla.m_nLaneId));
107 if (m_oJunctions.containsKey(sRoadId))
108 {
109 oLanesByRoads.remove(nIndex);
110 oCombined.add(oCla);
111 }
112 }
113 int nLimit = oLanesByRoads.size();
114 double dSqTol = dTol * dTol;
115 for (int nOuter = 0; nOuter < nLimit; nOuter++)
116 {
117 CtrlLineArcs oCur = oLanesByRoads.get(nOuter);
118
119 for (int nInner = 0; nInner < nLimit; nInner++)
120 {
121 if (nOuter == nInner)
122 continue;
123 CtrlLineArcs oCmp = oLanesByRoads.get(nInner);
124 if (oCmp.m_nLaneType != oCur.m_nLaneType)
125 continue;
126
127 int nConnect = oCur.connects(oCmp, dSqTol);
128 if (nConnect == CtrlLineArcs.CON_TEND_OSTART || nConnect == CtrlLineArcs.CON_TSTART_OEND) // the pts of both lines are in the same direction
129 {
130 oCur.combine(oCmp, nConnect); // combine the points of oCmp into oCur's point array
131 oLanesByRoads.remove(nInner);
132 --nLimit;
133 nInner = -1; // start inner loop over
134 }
135 }
136
137 oCombined.add(oCur);
138
139 }
140 return oCombined;
141 }

144 public static void renderTiledData(ArrayList<TrafCtrl> oCtrls, ArrayList<int[]> nTiles) throws IOException
145 {
146 String[] sEmpty = new String[0];
147 TdLayer oLayer = new TdLayer("direction", sEmpty, sEmpty, TdLayer.LINESTRING);
148 int[] nTags = new int[0];
149 double[] dPt = new double[2];
150 double[] dPoint = new double[2];
151 double[] dLine = new double[ARROW[0].length + 1];
152 double[] dHead = new double[ARROW[1].length + 1];
153 dLine[0] = dLine.length;
154 dHead[0] = dHead.length;
155 int nLinePts = ARROW[0].length / 2;
156 int nHeadPts = ARROW[1].length / 2;
157 for (int[] nTile : nTiles)
158 {
159 int nX = nTile[0];
160 int nY = nTile[1];
161 Files.createDirectories(Paths.get(String.format(g_sTdFileFormat, nX, 0, 0, 0)).getParent(), FileUtil.DIRPERS);
162 writeIndexFile(oCtrls, nX, nY);
163 oLayer.clear();
164 double[] dClipBounds = TileUtil.getClippingBounds(g_nDefaultZoom, nX, nY);
165 for (TrafCtrl oCtrl : oCtrls)
166 {
167 CtrlGeo oFullGeo = oCtrl.m_oFullGeo;
168 if (Collections.binarySearch(oFullGeo.m_oTiles, nTile, Mercator.TILECOMP) < 0 || oFullGeo.m_dLength < LENLOWTH || oFullGeo.m_dAverageWidth < WIDTHTH)
169 continue;
170
171 int nArrows = 2 + ((int)(oFullGeo.m_dLength / LENLOWTH) - 1) * 2;
172 double dPercentStep = 1.0 / (nArrows + 1);
173
174 for (int nIndex = 1; nIndex <= nArrows; nIndex++)
175 {
176 boolean bForward = TrafCtrlEnums.getCtrlVal("direction", "forward") == MathUtil.bytesToInt(oCtrl.m_yControlValue);
177 double dHdg = getCoordAndHeadingAtLength(oFullGeo.m_dC, oFullGeo.m_dLength * dPercentStep * nIndex, !bForward, dPt);
178 if (Double.isNaN(dHdg))
179 continue;
180
181 AffineTransform oAt = new AffineTransform();
182 oAt.translate(dPt[0], dPt[1]);
183 oAt.rotate(dHdg, 0, 0);
184 oAt.transform(ARROW[0], 0, dLine, 1, nLinePts);
185 oAt.transform(ARROW[1], 0, dHead, 1, nHeadPts);
186
187 double[] dArrowBounds = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
188 Iterator<double[]> oIt = Arrays.iterator(dLine, dPoint, 1, 2);
189 while (oIt.hasNext())
190 {
191 oIt.next();
192 Geo.updateBounds(dPoint[0], dPoint[1], dArrowBounds);
193 }
194 oIt = Arrays.iterator(dHead, dPoint, 1, 2);
195 while (oIt.hasNext())
196 {
197 oIt.next();
198 Geo.updateBounds(dPoint[0], dPoint[1], dArrowBounds);
199 }
200
201 if (Geo.boundingBoxesIntersect(dClipBounds[0], dClipBounds[1], dClipBounds[2], dClipBounds[3], dArrowBounds[0], dArrowBounds[1], dArrowBounds[2], dArrowBounds[3]))
202 {
203 oLayer.add(new TdFeature(dLine, nTags, oCtrl));
204 oLayer.add(new TdFeature(dHead, nTags, oCtrl));
205 }
206 }
207
208
209 if (!oLayer.isEmpty())
210 {
211 try (DataOutputStream oOut = new DataOutputStream(new BufferedOutputStream(FileUtil.newOutputStream(Paths.get(String.format(g_sTdFileFormat, nX, g_nDefaultZoom, nX, nY)), FileUtil.APPENDTO, FileUtil.FILEPERS))))
212 {
213 oLayer.write(oOut);
214 }
215 }
216 }
217 }
