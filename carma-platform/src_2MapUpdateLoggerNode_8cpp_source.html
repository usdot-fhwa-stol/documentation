54carma_debug_msgs::LaneletIdRegulatoryElementPair pairToDebugMessage(const std::pair<lanelet::Id, lanelet::RegulatoryElementPtr>& id_reg_pair) {
55 carma_debug_msgs::LaneletIdRegulatoryElementPair pair;
56 pair.lanelet_id = std::get<0>(id_reg_pair);
57 auto element = std::get<1>(id_reg_pair);
58 std::string rule_name = element->attribute(lanelet::AttributeName::Subtype).value();
59 pair.element.rule_name = rule_name;
60
61 if (rule_name.compare("digital_minimum_gap") == 0) {
62
63 auto cast = std::static_pointer_cast<lanelet::DigitalMinimumGap>(element);
64 pair.element.min_gap = cast->getMinimumGap();
65 pair.element.participants.insert(pair.element.participants.begin(), cast->participants_.begin(), cast->participants_.end());
66
67 } else if (rule_name.compare("digital_speed_limit") == 0) {
68
69 auto cast = std::static_pointer_cast<lanelet::DigitalSpeedLimit>(element);
70 pair.element.speed_limit = cast->getSpeedLimit().value();
71 pair.element.participants.insert(pair.element.participants.begin(), cast->participants_.begin(), cast->participants_.end());
72
73 } else if (rule_name.compare("direction_of_travel") == 0) {
74
75 auto cast = std::static_pointer_cast<lanelet::DirectionOfTravel>(element);
76 pair.element.direction = cast->direction_;
77 pair.element.participants.insert(pair.element.participants.begin(), cast->participants_.begin(), cast->participants_.end());
78
79 } else if (rule_name.compare("passing_control_line") == 0) {
80
81 auto cast = std::static_pointer_cast<lanelet::PassingControlLine>(element);
82 pair.element.left_participants.insert(pair.element.left_participants.begin(), cast->left_participants_.begin(), cast->left_participants_.end());
83 pair.element.right_participants.insert(pair.element.right_participants.begin(), cast->right_participants_.begin(), cast->right_participants_.end());
84
85 } else if (rule_name.compare("region_access_rule") == 0) {
86
87 auto cast = std::static_pointer_cast<lanelet::RegionAccessRule>(element);
88 pair.element.reason = cast->getReason();
89 pair.element.participants.insert(pair.element.participants.begin(), cast->participants_.begin(), cast->participants_.end());
90
91 } else if (rule_name.compare("stop_rule") == 0) {
92
93 auto cast = std::static_pointer_cast<lanelet::StopRule>(element);
94 pair.element.participants.insert(pair.element.participants.begin(), cast->participants_.begin(), cast->participants_.end());
95
96 } else {
97
98 ROS_WARN_STREAM("MapUpdateLogger recieved unsupported regulatory element in map update.");
99 pair.element.unsupported_type = true;
100
101 }
102 return pair;
103}

112carma_debug_msgs::MapUpdateReadable mapUpdateCallback(const autoware_lanelet2_msgs::MapBinConstPtr& update) {
113
114 ROS_INFO_STREAM("Recieved map update ");
115
116 auto control = std::make_shared<carma_wm::TrafficControl>(carma_wm::TrafficControl());
117 carma_wm::fromBinMsg(*update, control);
118
119 carma_debug_msgs::MapUpdateReadable msg;
120 msg.header = update->header;
121 msg.format_version = update->format_version;
122 msg.map_version = update->map_version;
123 msg.route_id = update->route_id;
124 msg.route_version = update->route_version;
125 msg.invalidates_route = update->invalidates_route;
126
127 msg.traffic_control_id = boost::lexical_cast<std::string>(control->id_);
128
129 for (auto id_reg_pair : control->update_list_) {
130
131 msg.update_list.push_back(pairToDebugMessage(id_reg_pair));
132 }
133
134 for (auto id_reg_pair : control->remove_list_) {
135
136 msg.remove_list.push_back(pairToDebugMessage(id_reg_pair));
137 }
138
139 return msg;
140}
