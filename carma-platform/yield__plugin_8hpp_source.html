267 std::vector<double> moving_average_filter(const std::vector<double> input, int window_size, bool ignore_first_point=true)
268 {
269 if (window_size % 2 == 0) {
270 throw std::invalid_argument("moving_average_filter window size must be odd");
271 }
272
273 std::vector<double> output;
274 output.reserve(input.size());
275
276 if (input.size() == 0) {
277 return output;
278 }
279
280 int start_index = 0;
281 if (ignore_first_point) {
282 start_index = 1;
283 output.push_back(input[0]);
284 }
285
286 for (int i = start_index; i<input.size(); i++) {
287
288 double total = 0;
289 int sample_min = std::max(0, i - window_size / 2);
290 int sample_max = std::min((int) input.size() - 1 , i + window_size / 2);
291
292 int count = sample_max - sample_min + 1;
293 std::vector<double> sample;
294 sample.reserve(count);
295 for (int j = sample_min; j <= sample_max; j++) {
296 total += input[j];
297 }
298 output.push_back(total / (double) count);
299
300 }
301
302 return output;
303 }
