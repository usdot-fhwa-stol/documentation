223void OAIDefaultApi::getConflictLanelets(const qint32 &link_lanelet_id) {
224 QString fullPath = QString(_serverConfigs["getConflictLanelets"][_serverIndices.value("getConflictLanelets")].URL()+"/get_conflict_lanelets/{link_lanelet_id}");
225
226
227 {
228 QString link_lanelet_idPathParam("{");
229 link_lanelet_idPathParam.append("link_lanelet_id").append("}");
230 QString pathPrefix, pathSuffix, pathDelimiter;
231 QString pathStyle = "simple";
232 if (pathStyle == "")
233 pathStyle = "simple";
234 pathPrefix = getParamStylePrefix(pathStyle);
235 pathSuffix = getParamStyleSuffix(pathStyle);
236 pathDelimiter = getParamStyleDelimiter(pathStyle, "link_lanelet_id", false);
237 QString paramString = (pathStyle == "matrix") ? pathPrefix+"link_lanelet_id"+pathSuffix : pathPrefix;
238 fullPath.replace(link_lanelet_idPathParam, paramString+QUrl::toPercentEncoding(::OpenAPI::toStringValue(link_lanelet_id)));
239 }
240 OAIHttpRequestWorker *worker = new OAIHttpRequestWorker(this, _manager);
241 worker->setTimeOut(_timeOut);
242 worker->setWorkingDirectory(_workingDirectory);
243 OAIHttpRequestInput input(fullPath, "GET");
244
245
246#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
247 for (auto keyValueIt = _defaultHeaders.keyValueBegin(); keyValueIt != _defaultHeaders.keyValueEnd(); keyValueIt++) {
248 input.headers.insert(keyValueIt->first, keyValueIt->second);
249 }
250#else
251 for (auto key : _defaultHeaders.keys()) {
252 input.headers.insert(key, _defaultHeaders[key]);
253 }
254#endif
255
256 connect(worker, &OAIHttpRequestWorker::on_execution_finished, this, &OAIDefaultApi::getConflictLaneletsCallback);
257 connect(this, &OAIDefaultApi::abortRequestsSignal, worker, &QObject::deleteLater);
258 connect(worker, &QObject::destroyed, this, [this]() {
259 if (findChildren<OAIHttpRequestWorker*>().count() == 0) {
260 emit allPendingRequestsCompleted();
261 }
262 });
263
264 worker->execute(&input);
