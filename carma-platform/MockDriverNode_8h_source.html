113 [&](PublisherWapper p) { return p.base_topic_name == topic; });
114 if (pub == publishers_.end())
115 throw std::invalid_argument("Attempted to publish to topic " + topic + " but no publisher was found");
116 pub->pub.publish(msg);
117 }
