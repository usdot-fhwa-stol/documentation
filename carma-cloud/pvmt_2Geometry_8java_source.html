52 private void addLanePoints(LaneSection oLaneSection, ArrayList<Lane> oLanes, double dTangent, double dXLaneZero, double dYLaneZero, double dLengthAlongRoad, Proj oProj, double[] dProjPt, Road oRoad)
53 {
54 double dXPrevLane = dXLaneZero;
55 double dYPrevLane = dYLaneZero;
56 oProj.cs2cs(dXPrevLane, dYPrevLane, dProjPt);
57 double dXProjPrevLane = dProjPt[0];
58 double dYProjPrevLane = dProjPt[1];
59 for (Lane oLane : oLanes)
60 {
61 double dPx = dXProjPrevLane;
62 double dPy = dYProjPrevLane;
63 double dLengthAlongSection = dLengthAlongRoad - oLaneSection.m_dS;
64 LaneWidth oLaneWidth = LaneWidth.getLaneWidth(dLengthAlongSection, oLane);
65 double dT = MathUtil.cubic(dLengthAlongSection - oLaneWidth.m_dS, oLaneWidth.m_dA, oLaneWidth.m_dB, oLaneWidth.m_dC, oLaneWidth.m_dD);
66 double dXNextLane = dXPrevLane + Math.cos(dTangent) * dT;
67 double dYNextLane = dYPrevLane + Math.sin(dTangent) * dT;
68 oLane.m_dOuter = addProjectedPoint(dXNextLane, dYNextLane, oLane.m_dOuter, oProj, dProjPt);
69 oLane.m_dOuterW = Arrays.addAndUpdate(oLane.m_dOuterW, dProjPt[0], dProjPt[1]);
70 oLane.m_dOuterW = Arrays.add(oLane.m_dOuterW, 0);
71 double dProjW = Geo.distance(dXProjPrevLane, dYProjPrevLane, dProjPt[0], dProjPt[1]);
72 dXProjPrevLane = dProjPt[0];
73 dYProjPrevLane = dProjPt[1];
74 double dXPath = (dXPrevLane + dXNextLane) / 2;
75 double dYPath = (dYPrevLane + dYNextLane) / 2;
76 oLane.m_dCenter = addProjectedPoint(dXPath, dYPath, oLane.m_dCenter, oProj, dProjPt);
77 if (this instanceof Arc)
78 {
79 oRoad.m_dPerps.add(new double[]{dPx, dPy, dXProjPrevLane, dYProjPrevLane});
80 }
81 if (oLaneWidth.m_dS != oLane.m_dLastLaneWidth)
82 {
83 oLane.m_dLastLaneWidth = oLaneWidth.m_dS;
84 oRoad.m_dLWPoints.add(new double[]{dProjPt[0], dProjPt[1]});
85 }
86
87 dXPrevLane = dXNextLane;
88 dYPrevLane = dYNextLane;
89
90 oLane.m_dCenter = Arrays.add(oLane.m_dCenter, dProjW);
91 }
