22autoware_msgs::Waypoint MPCFollowerWrapperWorker::TrajectoryPlanPointToWaypointConverter(const cav_msgs::TrajectoryPlanPoint& tpp, const cav_msgs::TrajectoryPlanPoint& tpp2) {
23
24 autoware_msgs::Waypoint waypoint;
25
26
27 waypoint.pose.pose.position.x = tpp.x;
28 waypoint.pose.pose.position.y = tpp.y;
29 waypoint.pose.pose.orientation = tf::createQuaternionMsgFromYaw(tpp.yaw);
30 double delta_x = tpp.x - tpp2.x;
31 double delta_y = tpp.y - tpp2.y;
32 double delta_pos = sqrt(delta_x * delta_x + delta_y * delta_y);
33 ros::Duration delta_t = tpp2.target_time - tpp.target_time;
34 double delta_t_second = fabs(delta_t.toSec());
35
36 if(delta_t_second != 0) {
37 waypoint.twist.twist.linear.x = delta_pos / delta_t_second;
38 }
39 else {
40 waypoint.twist.twist.linear.x = 0;
41 }
42
43 return waypoint;
44};
