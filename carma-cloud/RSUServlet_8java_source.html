204 private int identifyRSU(HttpServletRequest oReq, JSONObject oResponse) throws Exception {
205 StringBuilder sReq = new StringBuilder();
206 try (BufferedInputStream oIn = new BufferedInputStream(oReq.getInputStream())) {
207 int nByte = 0;
208 while ((nByte = oIn.read()) >= 0) {
209 sReq.append((char) nByte);
210 }
211 }
212
213 // Tracking list for BSM request
214 BSMRequestParser parser = new BSMRequestParser();
215 BSMRequest newBsmReq = parser
216 .parseRequest(new ByteArrayInputStream(sReq.toString().getBytes(StandardCharsets.UTF_8)));
217 ArrayList<BSMRequest> updatedBsmReqsList = new ArrayList<BSMRequest>();
218 Boolean isIgnore = false;
219 try {
220 // Already processed existing BSM list
221 ArrayList<BSMRequest> existingBsmReqList = (ArrayList<BSMRequest>) getServletContext()
222 .getAttribute(BSMREQLIST);
223 if (existingBsmReqList != null) {
224 // Check if the new BSM already processed. If processed, ignore the BSM request
225 // and update existing matching BSM request update time to now
226 for (BSMRequest bsmReq : existingBsmReqList) {
227 if (bsmReq.getId().equals(newBsmReq.getId())) {
228 LOGGER.debug("IGNORE already Processed BSM: " + newBsmReq.getId());
229 bsmReq.setLast_update_at(Instant.now());
230 isIgnore = true;
231 }
232 updatedBsmReqsList.add(bsmReq);
233 }
234 }
235 } catch (NullPointerException oEx) {
236 LOGGER.error("No existing BSM");
237 }
238
239 if (isIgnore) {
240 // Update the tracking list
241 getServletContext().setAttribute(BSMREQLIST, updatedBsmReqsList);
242 return HttpServletResponse.SC_CONFLICT;
243 }
244 // Processing BSM request
245 ExecutorService singleExector = Executors.newSingleThreadExecutor();
246 singleExector.submit(new RSUIdentificationTask(newBsmReq));
247 singleExector.shutdown();
248
249 // Add newly processed BSM request to tracking list
250 newBsmReq.setLast_update_at(Instant.now());
251 updatedBsmReqsList.add(newBsmReq);
252 getServletContext().setAttribute(BSMREQLIST, updatedBsmReqsList);
253 return HttpServletResponse.SC_OK;
254 }
