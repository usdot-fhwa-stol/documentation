87 std::vector<std::vector<double>> parse_covariance(const rapidjson::Value::ConstArray &val)
88 {
89 // Initialize 3x3 covariance matrix
90 std::vector<std::vector<double>> covariance;
91
92 for (rapidjson::SizeType i = 0; i < val.Size(); i++)
93 {
94 const rapidjson::Value& row = val[i];
95 std::vector<double> val_row;
96
97 for (rapidjson::SizeType j = 0; j < row.Size(); j++)
98 {
99 val_row.push_back(val[i][j].GetDouble());
100 }
101 covariance.push_back(val_row);
102 }
103 return covariance;
104 }
