38auto to_time_msg(const DDateTime & d_date_time) noexcept -> builtin_interfaces::msg::Time
39{
40 double seconds;
41 const auto fractional_secs{
42 std::modf(remove_units(d_date_time.second.value_or(units::time::second_t{0.0})), &seconds)};
43
44 builtin_interfaces::msg::Time msg;
45 msg.sec = static_cast<std::int32_t>(seconds);
46 msg.nanosec = static_cast<std::int32_t>(fractional_secs * 1e9);
47
48 return msg;
49}

59auto to_position_msg(const UtmCoordinate & position_utm) noexcept -> geometry_msgs::msg::Point
60{
61 geometry_msgs::msg::Point msg;
62
63 msg.x = remove_units(position_utm.easting);
64 msg.y = remove_units(position_utm.northing);
65 msg.z = remove_units(position_utm.elevation);
66

70auto heading_to_enu_yaw(const units::angle::degree_t & heading) noexcept -> units::angle::degree_t
71{
72 return units::angle::degree_t{std::fmod(-(remove_units(heading) - 90.0) + 360.0, 360.0)};

75auto to_detection_list_msg(const carma_v2x_msgs::msg::SensorDataSharingMessage & sdsm) noexcept
76 -> carma_cooperative_perception_interfaces::msg::DetectionList
77{
78 carma_cooperative_perception_interfaces::msg::DetectionList detection_list;
79
80 const auto ref_pos_3d{Position3D::from_msg(sdsm.ref_pos)};
81 const Wgs84Coordinate ref_pos_wgs84{
82 ref_pos_3d.latitude, ref_pos_3d.longitude, ref_pos_3d.elevation.value()};
83 const auto ref_pos_utm{project_to_utm(ref_pos_wgs84)};
84
85 for (const auto & object_data : sdsm.objects.detected_object_data) {
86 const auto common_data{object_data.detected_object_common_data};
87
88 carma_cooperative_perception_interfaces::msg::Detection detection;
89 detection.header.frame_id = to_string(ref_pos_utm.utm_zone);
90
91 const auto detection_time{calc_detection_time_stamp(
92 DDateTime::from_msg(sdsm.sdsm_time_stamp),
93 MeasurementTimeOffset::from_msg(common_data.measurement_time))};
94
95 detection.header.stamp = to_time_msg(detection_time);
96
97 detection.id = std::to_string(common_data.detected_id.object_id);
98
99 const auto pos_offset{PositionOffsetXYZ::from_msg(common_data.pos)};
100 const auto utm_displacement{
101 UtmDisplacement{pos_offset.offset_x, pos_offset.offset_y, pos_offset.offset_z.value()}};
102
103 const auto detection_pos_utm{ref_pos_utm + utm_displacement};
104 detection.pose.pose.position = to_position_msg(detection_pos_utm);
105
106 const auto true_heading{units::angle::degree_t{Heading::from_msg(common_data.heading).heading}};
107
108 // Note: This should really use the detection's WGS-84 position, so the
109 // convergence will be off slightly. TODO
110 const auto grid_convergence{calculate_grid_convergence(ref_pos_wgs84, ref_pos_utm.utm_zone)};
111
112 const auto grid_heading{true_heading - grid_convergence};
113 const auto enu_yaw{heading_to_enu_yaw(grid_heading)};
114
115 tf2::Quaternion quat_tf;
116 quat_tf.setRPY(0, 0, remove_units(units::angle::radian_t{enu_yaw}));
117 detection.pose.pose.orientation = tf2::toMsg(quat_tf);
118
119 const auto speed{Speed::from_msg(common_data.speed)};
120 detection.twist.twist.linear.x =
121 remove_units(units::velocity::meters_per_second_t{speed.speed});
122
123 const auto speed_z{Speed::from_msg(common_data.speed_z)};
124 detection.twist.twist.linear.z =
125 remove_units(units::velocity::meters_per_second_t{speed_z.speed});
126
127 const auto accel_set{AccelerationSet4Way::from_msg(common_data.accel_4_way)};
128 detection.accel.accel.linear.x =
129 remove_units(units::acceleration::meters_per_second_squared_t{accel_set.longitudinal});
130 detection.accel.accel.linear.y =
131 remove_units(units::acceleration::meters_per_second_squared_t{accel_set.lateral});
132 detection.accel.accel.linear.z =
133 remove_units(units::acceleration::meters_per_second_squared_t{accel_set.vert});
134
135 detection.twist.twist.angular.z =
136 remove_units(units::angular_velocity::degrees_per_second_t{accel_set.yaw_rate});
137
138 switch (common_data.obj_type.object_type) {
139 case common_data.obj_type.ANIMAL:
140 detection.motion_model = detection.MOTION_MODEL_CTRV;
141 break;
142 case common_data.obj_type.VRU:
143 detection.motion_model = detection.MOTION_MODEL_CTRV;
144 break;
145 case common_data.obj_type.VEHICLE:
146 detection.motion_model = detection.MOTION_MODEL_CTRV;
147 break;
148 default:
149 detection.motion_model = detection.MOTION_MODEL_CTRV;
150 }
151
152 detection_list.detections.push_back(std::move(detection));
153 }
154
155 return detection_list;
156}

158auto to_detection_msg(
159 const carma_perception_msgs::msg::ExternalObject & object,
160 const MotionModelMapping & motion_model_mapping) noexcept
161 -> carma_cooperative_perception_interfaces::msg::Detection
162{
163 carma_cooperative_perception_interfaces::msg::Detection detection;
164
165 detection.header = object.header;
166
167 if (object.presence_vector & object.BSM_ID_PRESENCE_VECTOR) {
168 detection.id = "";
169 std::transform(
170 std::cbegin(object.bsm_id), std::cend(object.bsm_id), std::back_inserter(detection.id),
171 [](const auto & i) { return i + '0'; });
172 }
173
174 if (object.presence_vector & object.ID_PRESENCE_VECTOR) {
175 detection.id += '-' + std::to_string(object.id);
176 }
177
178 if (object.presence_vector & object.POSE_PRESENCE_VECTOR) {
179 detection.pose = object.pose;
180 }
181
182 if (object.presence_vector & object.VELOCITY_INST_PRESENCE_VECTOR) {
183 detection.twist = object.velocity_inst;
184 }
185
186 if (object.presence_vector & object.OBJECT_TYPE_PRESENCE_VECTOR) {
187 switch (object.object_type) {
188 case object.SMALL_VEHICLE:
189 detection.motion_model = motion_model_mapping.small_vehicle_model;
190 break;
191 case object.LARGE_VEHICLE:
192 detection.motion_model = motion_model_mapping.large_vehicle_model;
193 break;
194 case object.MOTORCYCLE:
195 detection.motion_model = motion_model_mapping.motorcycle_model;
196 break;
197 case object.PEDESTRIAN:
198 detection.motion_model = motion_model_mapping.pedestrian_model;
199 break;
200 case object.UNKNOWN:
201 default:
202 detection.motion_model = motion_model_mapping.unknown_model;
203 }
204 }
205
206 return detection;
207}

209auto to_detection_list_msg(
210 const carma_perception_msgs::msg::ExternalObjectList & object_list,
211 const MotionModelMapping & motion_model_mapping) noexcept
212 -> carma_cooperative_perception_interfaces::msg::DetectionList
213{
214 carma_cooperative_perception_interfaces::msg::DetectionList detection_list;
215
216 std::transform(
217 std::cbegin(object_list.objects), std::cend(object_list.objects),
218 std::back_inserter(detection_list.detections),
219 [&motion_model_mapping = std::as_const(motion_model_mapping)](const auto & object) {
220 return to_detection_msg(object, motion_model_mapping);
221 });
222
223 return detection_list;
224}

226auto to_external_object_msg(
227 const carma_cooperative_perception_interfaces::msg::Track & track) noexcept
228 -> carma_perception_msgs::msg::ExternalObject
229{
230 carma_perception_msgs::msg::ExternalObject external_object;
231 external_object.header = track.header;
232 external_object.presence_vector = 0;
233
234 try {
235 if (const auto numeric_id{std::stol(track.id)};
236 numeric_id >= 0 && numeric_id <= std::numeric_limits<std::uint32_t>::max()) {
237 external_object.presence_vector |= external_object.ID_PRESENCE_VECTOR;
238 external_object.id = static_cast<std::uint32_t>(numeric_id);
239 }
240 } catch (const std::invalid_argument & /* exception */) {
241 external_object.presence_vector &= ~external_object.ID_PRESENCE_VECTOR;
242 } catch (const std::out_of_range & /* exception */) {
243 external_object.presence_vector &= ~external_object.ID_PRESENCE_VECTOR;
244 }
245
246 external_object.presence_vector |= external_object.POSE_PRESENCE_VECTOR;
247 external_object.pose = track.pose;
248
249 external_object.presence_vector |= external_object.VELOCITY_PRESENCE_VECTOR;
250 external_object.velocity = track.twist;
251
252 return external_object;
253}

255auto to_external_object_list_msg(
256 const carma_cooperative_perception_interfaces::msg::TrackList & track_list) noexcept
257 -> carma_perception_msgs::msg::ExternalObjectList
258{
259 carma_perception_msgs::msg::ExternalObjectList external_object_list;
260

262 external_object_list.objects.push_back(to_external_object_msg(track));
