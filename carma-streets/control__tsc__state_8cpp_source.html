107 void control_tsc_state::update_tsc_control_queue(std::shared_ptr<streets_phase_control_schedule::streets_phase_control_schedule> phase_control_schedule_ptr,
108 std::queue<streets_snmp_cmd::snmp_cmd_struct> &tsc_command_queue) const
109 {
110 if (!phase_control_schedule_ptr)

112 SPDLOG_ERROR("Phase control schedule is not initialized.");
113 return;
114 }
115
116 if (phase_control_schedule_ptr->is_clear_current_schedule)
117 {
118 //Clear all phase controls from the traffic signal controller
119 SPDLOG_DEBUG("Clear all phase controls from TSC!");
120 run_clear_all_snmp_commands();
121
122 SPDLOG_INFO("Clear SNMP command queue!");
123 // Clear all commands on the update command queue.
124 std::queue<streets_snmp_cmd::snmp_cmd_struct> empty_queue;
125 std::swap(tsc_command_queue, empty_queue);
126 }
127 else
128 {
129 std::stringstream ss;
130 ss << *phase_control_schedule_ptr;
131 SPDLOG_DEBUG("Update SNMP command queue with new phase control schedule commands: {0}", ss.str());
132 // Update command queue with the new phase control schedule commands.
133 streets_snmp_cmd::streets_snmp_cmd_converter converter;
134 auto snmp_cmd_queue = converter.create_snmp_cmds_by_phase_control_schedule(phase_control_schedule_ptr);
135 std::swap(tsc_command_queue, snmp_cmd_queue);
136 }
137 SPDLOG_INFO("Updated Queue Size: {0}", tsc_command_queue.size());
