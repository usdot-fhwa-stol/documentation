21 void PortDrayageStateMachine::processEvent(PortDrayageEvent event) {
22 switch (state_)
23 {
24 case PortDrayageState::INACTIVE:
25 if (event == PortDrayageEvent::DRAYAGE_START) {
26 state_ = PortDrayageState::EN_ROUTE_TO_INITIAL_DESTINATION;
27 }
28 break;
29 case PortDrayageState::EN_ROUTE_TO_INITIAL_DESTINATION:
30 if (event == PortDrayageEvent::ARRIVED_AT_DESTINATION) {
31 if (on_arrived_at_destination_) {
32 on_arrived_at_destination_();
33 }
34 state_ = PortDrayageState::AWAITING_DIRECTION;
35 }
36 break;
37 case PortDrayageState::EN_ROUTE_TO_RECEIVED_DESTINATION:
38 if (event == PortDrayageEvent::ARRIVED_AT_DESTINATION) {
39 if (on_arrived_at_destination_) {
40 on_arrived_at_destination_();
41 }
42 state_ = PortDrayageState::AWAITING_DIRECTION;
43 }
44 break;
45 case PortDrayageState::AWAITING_DIRECTION:
46 if (event == PortDrayageEvent::RECEIVED_NEW_DESTINATION) {
47 state_ = PortDrayageState::EN_ROUTE_TO_RECEIVED_DESTINATION;
48 if (on_received_new_destination_) {
49 on_received_new_destination_();
50 }
51 }
52 break;
53 default:
54 RCLCPP_ERROR_STREAM(logger_->get_logger(), "Unhandled port drayage state: " << state_ << "!");
55 throw std::invalid_argument("Unhandled port drayage state");
56 }
57 }
