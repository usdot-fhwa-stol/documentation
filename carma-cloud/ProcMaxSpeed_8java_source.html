108 @Override
109 public ArrayList<CtrlLineArcs> combine(ArrayList<CtrlLineArcs> oLanes, double dTol)
110 {
111 ArrayList<CtrlLineArcs> oLanesByRoads = ProcCtrl.combineLaneByRoad(oLanes, dTol);
112 Collections.sort(oLanesByRoads, CtrlLineArcs.CMPBYLANE);
113 ArrayList<CtrlLineArcs> oCombined = new ArrayList();
114 int nIndex = oLanesByRoads.size();
115 SpdMapping oSearch = new SpdMapping();
116 while (nIndex-- > 0)
117 {
118 CtrlLineArcs oCla = oLanesByRoads.get(nIndex);
119 String sRoadId = Integer.toString(XodrUtil.getRoadId(oCla.m_nLaneId));
120 if (m_oJunctions.containsKey(sRoadId))
121 {
122 oLanesByRoads.remove(nIndex);
123 oCombined.add(oCla);
124 }
125 }
126 int nLimit = oLanesByRoads.size();
127 double dSqTol = dTol * dTol;
128 for (int nOuter = 0; nOuter < nLimit; nOuter++)
129 {
130 CtrlLineArcs oCur = oLanesByRoads.get(nOuter);
131
132 for (int nInner = 0; nInner < nLimit; nInner++)
133 {
134 if (nOuter == nInner)
135 continue;
136 CtrlLineArcs oCmp = oLanesByRoads.get(nInner);
137 if (oCmp.m_nLaneType != oCur.m_nLaneType)
138 continue;
139
140 int nConnect = oCur.connects(oCmp, dSqTol);
141 if (nConnect == CtrlLineArcs.CON_TEND_OSTART || nConnect == CtrlLineArcs.CON_TSTART_OEND) // the pts of both lines are in the same direction
142 {
143 oSearch.m_nId = oCur.m_nLaneId;
144 int nSearchIndex = Collections.binarySearch(m_oSpds, oSearch);
145 int nCurSpeed = nSearchIndex >= 0 ? m_oSpds.get(nSearchIndex).m_nSpd : DEFAULTSPD;
146
147 oSearch.m_nId = oCmp.m_nLaneId;
148 nSearchIndex = Collections.binarySearch(m_oSpds, oSearch);
149 int nCmpSpeed = nSearchIndex >= 0 ? m_oSpds.get(nSearchIndex).m_nSpd : DEFAULTSPD;
150 if (nCurSpeed != nCmpSpeed)
151 continue;
152
153 oCur.combine(oCmp, nConnect); // combine the points of oCmp into oCur's point array
154 oLanesByRoads.remove(nInner);
155 --nLimit;
156 nInner = -1; // start inner loop over
157 }
158 }
159 oCombined.add(oCur);
160 }
161 return oCombined;
162 }
