86 void monitor_desired_phase_plan::fix_upcoming_green( const std::shared_ptr<signal_phase_and_timing::spat> spat_ptr,
87 const std::shared_ptr<tsc_state> tsc_state) {
88 // Find start time of next green
89 bool is_yellow_change = false;
90 // Start time for fixed up coming green
91 uint64_t start_time_epoch_ms = 0;
92 // Current time used for any calculations
93 uint64_t cur_time_since_epoch = streets_service::streets_clock_singleton::time_in_ms();
94 auto state = spat_ptr->get_intersection();
95 std::vector<signal_phase_and_timing::movement_state> yellow_phase_present;
96 // Check if current spat includes yellow clearance
97 for (const auto &movement : state.states) {
98 // Found yellow states
99 if ( movement.state_time_speed.front().event_state == signal_phase_and_timing::movement_phase_state::protected_clearance ) {
100 is_yellow_change = true;
101 auto yellow_phase_configuration =tsc_state->get_signal_group_state_map().find(movement.signal_group)->second;
102 // Start time of upcoming green will be start of yellow + yellow duration + red clearance
103 uint64_t local_start_time_epoch = movement.state_time_speed.front().timing.get_epoch_start_time() + yellow_phase_configuration.yellow_duration + yellow_phase_configuration.red_clearance;
104 if ( local_start_time_epoch > start_time_epoch_ms) {
105 // If there are multiple movement groups in yellow, start time will be the largest end of yellow + red clearance time
106 start_time_epoch_ms = local_start_time_epoch;
107 }
108 // If service starts on yellow change, set last green served to yellow change.
109 yellow_phase_present.push_back( movement);
110 }
111 }
112 if ( !yellow_phase_present.empty() ) {
113 update_last_green_served(yellow_phase_present);
114 }
115 // If it does not include a green and does not include a yellow assume it is in all red clearance
116 if ( !is_yellow_change ) {
117 for (const auto &movement : state.states) {
118 // If last green served is empty
119 // TSC is in YELLOW CHANGE or RED CLEAR and monitor desired phase plan
120 // has not yet encountered a green to store as last green served
121 if (last_green_served.empty() ) {
122 throw monitor_desired_phase_plan_exception("No information on previous green!");
123 }
124 // Found red clearance states
125 if ( (movement.signal_group == last_green_served.front().signal_group
126 || movement.signal_group == last_green_served.back().signal_group)
127 && movement.state_time_speed.front().event_state == signal_phase_and_timing::movement_phase_state::stop_and_remain ) {
128 auto red_phase_configuration =tsc_state->get_signal_group_state_map().find(movement.signal_group)->second;
129 // Start time is red clearance start + red clearance if start time is > current time
130 uint64_t local_start_time_epoch = movement.state_time_speed.front().timing.get_epoch_start_time() + red_phase_configuration.red_clearance;
131 // If red phase start + red clearance is greater than or equal to current time this phase is in red clearance
132 // and will get green next.
133 start_time_epoch_ms = local_start_time_epoch;
134
135 if ( local_start_time_epoch < cur_time_since_epoch ) {
136 SPDLOG_ERROR("Error state SPaT : {0}", spat_ptr->toJson());
137 SPDLOG_ERROR("Error state last green served : {0}, {1}", last_green_served.front().signal_group, last_green_served.back().signal_group);
138 throw monitor_desired_phase_plan_exception("Local start time is less than current time which means this is not an upcoming green!");
139 }
140
141 }
142 }
143 }
144 // Once start time is determine. We must determine next phase.
145 streets_snmp_cmd::snmp_response_obj response;
146 response.type = streets_snmp_cmd::RESPONSE_TYPE::INTEGER;
147 _snmp_client->process_snmp_request(ntcip_oids::PHASE_STATUS_GROUP_PHASE_NEXT, streets_snmp_cmd::REQUEST_TYPE::GET, response );
148
149 // Generate Desired Phase Plan with next green fixed.
150 streets_desired_phase_plan::streets_desired_phase_plan one_fixed_green;
151 one_fixed_green.timestamp = cur_time_since_epoch;
152 streets_desired_phase_plan::signal_group2green_phase_timing fixed_green;
153 fixed_green.start_time = start_time_epoch_ms;
154
170 for (uint i = 0; i < 8; ++i) {
171 if (((response.val_int >> i) & 1) == 1) {
172 // Add any signal group for phase that has bit as 1
173 auto signal_group_id = tsc_state->get_vehicle_signal_group_id(i+1);
174
175 fixed_green.signal_groups.push_back(signal_group_id);
176 }
177 }
178 if ( fixed_green.signal_groups.size() > 2 ) {
179 throw monitor_desired_phase_plan_exception( "More than two phases were found to be next!");
180 }
181 else if ( fixed_green.signal_groups.size() < 1) {
182 throw monitor_desired_phase_plan_exception("No signal groups were found to be next");
183 }
184 // Calculating the end of the fixed green by using min green configuration
185 // for all signal groups included in the movement group that is assigned the
186 // next green.
187 uint64_t fixed_green_end_time_epoch_ms = 0;
188 for ( const auto &s_g_id : fixed_green.signal_groups ) {
189 auto sig_group_config = tsc_state->get_signal_group_state_map().find(s_g_id)->second;
190 uint64_t local_end_time_epoch_ms = fixed_green.start_time + sig_group_config.min_green;
191 if ( local_end_time_epoch_ms > fixed_green_end_time_epoch_ms ) {
192 // find max of green start + min green to assign end time
193 fixed_green_end_time_epoch_ms = local_end_time_epoch_ms;
194 }
195 }
196 fixed_green.end_time = fixed_green_end_time_epoch_ms;
197 one_fixed_green.desired_phase_plan.push_back(fixed_green);
198 SPDLOG_DEBUG("Updating SPaT with fixed green : \n{0}", one_fixed_green.toJson());
199 spat_ptr->update_spat_with_candidate_dpp(one_fixed_green, tsc_state->get_tsc_config_state());
200

233 // Loop through the desired phase plan and remove an event if the end time is past current
234 bool no_expired_events = false;
235 uint64_t cur_time_since_epoch = streets_service::streets_clock_singleton::time_in_ms();
236 while( !no_expired_events && dpp != nullptr && !dpp->desired_phase_plan.empty()){
237 // Check first event in desired phase plan and remove if expired
238 if(cur_time_since_epoch < dpp->desired_phase_plan.front().end_time){
239 no_expired_events = true;
240 }
241 else{
242 // Warning for removing entries from dpp
243 SPDLOG_WARN("Current time is greater than event end time {0}, removing expired entry from desired phase plan", dpp->desired_phase_plan.front().end_time);
244 dpp->desired_phase_plan.erase(dpp->desired_phase_plan.begin());
245 }
