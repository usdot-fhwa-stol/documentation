47 public static void renderTiledData(ArrayList<TrafCtrl> oCtrls, ArrayList<int[]> nTiles) throws IOException
48 {
49 String[] sEmpty = new String[0];
50 TdLayer oLayer = new TdLayer("minhdwy", sEmpty, sEmpty, TdLayer.LINESTRING);
51 int[] nTags = new int[0];
52 double[] dPt = new double[2];
53 double[] dPoint = new double[2];
54 double[][] dNumbers = new double[NUMBERS.length][];
55 int[] nNumbersPts = new int[NUMBERS.length];

67 double[] dTemp = new double[oEntry.getValue().length + 1];
68 dTemp[0] = dTemp.length;
69 oChars.put(oEntry.getKey(), dTemp);
70 oCharPts.put(oEntry.getKey(), oEntry.getValue().length / 2);
71 }
72
73 double[] dSymbol1 = new double[SYMBOL.length + 1];
74 double[] dSymbol2 = new double[SYMBOL.length + 1];
75 dSymbol1[0] = dSymbol1.length;
76 dSymbol2[0] = dSymbol2.length;

78 Units oUnits = Units.getInstance();

82 int nX = nTile[0];
83 int nY = nTile[1];

85 writeIndexFile(oCtrls, nX, nY);
86 oLayer.clear();
87 double[] dClipBounds = TileUtil.getClippingBounds(g_nDefaultZoom, nX, nY);
88 DecimalFormat oDf = new DecimalFormat("#.#");
89 for (TrafCtrl oCtrl : oCtrls)
90 {
91 CtrlGeo oFullGeo = oCtrl.m_oFullGeo;
92 if (Collections.binarySearch(oFullGeo.m_oTiles, nTile, Mercator.TILECOMP) < 0 || oFullGeo.m_dLength < LENLOWTH || oFullGeo.m_dAverageWidth < WIDTHTH)
93 continue;
94
95 double dHdg = getCoordAndHeadingAtLength(oFullGeo.m_dC, oFullGeo.m_dLength / 2.0, false, dPt);
96 if (Double.isNaN(dHdg))
97 continue;
98
99
100 double dPTAngle = dHdg + Mercator.PI_OVER_TWO;
101 double dXc = dPt[0];
102 double dYc = dPt[1];
103
104 dPt[0] += Math.cos(dHdg) * 0.8;
105 dPt[1] += Math.sin(dHdg) * 0.8;
106
107 AffineTransform oAt = new AffineTransform();
108 oAt.translate(dPt[0], dPt[1]);
109 oAt.rotate(dHdg - Mercator.PI_OVER_TWO, 0, 0);
110 oAt.transform(SYMBOL, 0, dSymbol1, 1, nSymbolPts);
111
112 dPt[0] = dXc;
113 dPt[1] = dYc;
114 dPt[0] -= Math.cos(dHdg) * 0.8;
115 dPt[1] -= Math.sin(dHdg) * 0.8;
116 oAt = new AffineTransform();
117 oAt.translate(dPt[0], dPt[1]);
118 oAt.rotate(dHdg + Mercator.PI_OVER_TWO, 0, 0);
119 oAt.transform(SYMBOL, 0, dSymbol2, 1, nSymbolPts);
120
121
122 double[] dBounds = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
123 Iterator<double[]> oIt = Arrays.iterator(dSymbol1, dPoint, 1, 2);
124 while (oIt.hasNext())
125 {
126 oIt.next();
127 Geo.updateBounds(dPoint[0], dPoint[1], dBounds);
128 }
129 oIt = Arrays.iterator(dSymbol2, dPoint, 1, 2);
130 while (oIt.hasNext())
131 {
132 oIt.next();
133 Geo.updateBounds(dPoint[0], dPoint[1], dBounds);
134 }
135
136 if (Geo.boundingBoxesIntersect(dClipBounds[0], dClipBounds[1], dClipBounds[2], dClipBounds[3], dBounds[0], dBounds[1], dBounds[2], dBounds[3]))
137 {
138 oLayer.add(new TdFeature(dSymbol1, nTags, oCtrl));
139 oLayer.add(new TdFeature(dSymbol2, nTags, oCtrl));
140 int nVal = (int)Math.round(oUnits.convert(sUnits[0], sUnits[1], MathUtil.bytesToInt(oCtrl.m_yControlValue)));
141 String sVal = oDf.format(nVal) + "ft";
142 int nLimit = sVal.length();
143 double dStep = -0.3;
144 double dOffset = -0.45 + (nLimit * 0.3);
145 for (int nIndex = 0; nIndex < sVal.length(); nIndex++)
146 {
147 char cChar = sVal.charAt(nIndex);
148 double dMultiplier = dOffset + (dStep * nIndex);
149 dPt[0] = dXc;
150 dPt[1] = dYc;
151 dPt[0] += Math.cos(dPTAngle) * dMultiplier;
152 dPt[1] += Math.sin(dPTAngle) * dMultiplier;
153
154 oAt = new AffineTransform();
155 oAt.translate(dPt[0], dPt[1]);
156 oAt.rotate(dHdg - Mercator.PI_OVER_TWO, 0, 0);
157 if (Character.isDigit(cChar))
158 {
159 oAt.scale(0.45, 0.45);
160 int nNumber = Character.getNumericValue(cChar);
161 oAt.transform(NUMBERS[nNumber], 0, dNumbers[nNumber], 1, nNumbersPts[nNumber]);
162 oLayer.add(new TdFeature(dNumbers[nNumber], nTags, oCtrl));
163 }
164 else
165 {
166 oAt.scale(0.65, 0.65);
167 String sChar = Character.toString(cChar);
168 oAt.transform(CHARS.get(sChar), 0, oChars.get(sChar), 1, oCharPts.get(sChar));
169 oLayer.add(new TdFeature(oChars.get(sChar), nTags, oCtrl));
170 }
171 }
172 }
173 }
174 if (!oLayer.isEmpty())
175 {
176 try (DataOutputStream oOut = new DataOutputStream(new BufferedOutputStream(FileUtil.newOutputStream(Paths.get(String.format(g_sTdFileFormat, nX, g_nDefaultZoom, nX, nY)), FileUtil.APPENDTO, FileUtil.FILEPERS))))
177 {
178 oLayer.write(oOut);
179 }
180 }
181 }
