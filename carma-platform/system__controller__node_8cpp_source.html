36 void SystemControllerNode::initialize()
37 {
38 try
39 {
40
41 RCLCPP_INFO(get_logger(), "Initializing SystemControllerNode");
42
43 // Create subscriptions. History size of 100 is used here to ensure no missed alerts
44 system_alert_sub_ = create_subscription<carma_msgs::msg::SystemAlert>(
45 system_alert_topic_, 100,
46 std::bind(&SystemControllerNode::on_system_alert, this, std::placeholders::_1));
47
48 system_alert_pub_ = create_publisher<carma_msgs::msg::SystemAlert>(
49 system_alert_topic_, 10);
50
51 config_.signal_configure_delay = this->declare_parameter<double>("signal_configure_delay", config_.signal_configure_delay);
52 config_.service_timeout_ms = this->declare_parameter<int64_t>("service_timeout_ms", config_.service_timeout_ms);
53 config_.call_timeout_ms = this->declare_parameter<int64_t>("call_timeout_ms", config_.call_timeout_ms);
54 config_.required_subsystem_nodes = this->declare_parameter<std::vector<std::string>>("required_subsystem_nodes", config_.required_subsystem_nodes);
55
56 RCLCPP_INFO_STREAM(get_logger(), "Config: " << config_);
57
58 lifecycle_mgr_.set_managed_nodes(config_.required_subsystem_nodes);
59
60 // Create startup timer
61 startup_timer_ = rclcpp::create_timer(
62 this,
63 get_clock(),
64 std::chrono::milliseconds(static_cast<long>(config_.signal_configure_delay * 1000)),
65 std::bind(&SystemControllerNode::startup_delay_callback, this));
66
67
68 }
69 catch (const std::exception &e)
70 {
71 on_error(e);
72 }
73 }

80 void SystemControllerNode::on_error(const std::exception &e)
81 {
82
83 if (startup_timer_)
84 startup_timer_->cancel(); // Cancel the timer to ensure we are not interrupted
85
86 std::string reason = "Uncaught exception: " + std::string(e.what());
87
88 RCLCPP_ERROR_STREAM(
89 get_logger(), reason); // Log error
90
91 lifecycle_mgr_.shutdown(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms), false); // Trigger shutdown when internal error occurs
92
93 rclcpp::shutdown(nullptr, reason); // Fully shutdown this node
94 }

96 void SystemControllerNode::startup_delay_callback()
97 {
98 try
99 {
100 RCLCPP_INFO(get_logger(), "Attempting to configure system...");
101
102 if (startup_timer_)
103 startup_timer_->cancel(); // Cancel the timer as this callback should only trigger once
104
105 // Walk the managed nodes through their lifecycle
106 // First we configure the nodes
107 if (!lifecycle_mgr_.configure(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms)).empty())
108 {
109 // If some nodes failed to configure then we will shutdown the system
110 RCLCPP_ERROR_STREAM(
111 get_logger(), "System could not be configured on startup. Shutting down.");
112
113 lifecycle_mgr_.shutdown(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms), false);
114 rclcpp::shutdown(nullptr, "System could not be configured on startup. Shutting down.");
115 return;
116 }
117
118 // Second we activate the nodes
119 if (!lifecycle_mgr_.activate(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms)).empty())
120 {
121
122 RCLCPP_ERROR_STREAM(
123 get_logger(), "System could not be activated. Shutting down.");
124
125 lifecycle_mgr_.shutdown(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms), false);
126 rclcpp::shutdown(nullptr, "System could not be activated. Shutting down.");
127 return;
128 }
129
130 RCLCPP_INFO(get_logger(), "System configured successfully");
131 }
132 catch (const std::exception &e)
133 {
134 on_error(e);
135 }
136 }

138 void SystemControllerNode::publish_system_alert(carma_msgs::msg::SystemAlert msg)
139 {
140
141 if (!get_node_base_interface())
142 throw std::runtime_error("get_node_base_interface() returned null pointer. May indicate ROS2 bug.");
143
144 msg.source_node = get_node_base_interface()->get_fully_qualified_name(); // The the source name for the message
145
146 system_alert_pub_->publish(msg);

149 void SystemControllerNode::on_system_alert(const carma_msgs::msg::SystemAlert::UniquePtr msg)
150 {
151
152 try
153 {
154
155 RCLCPP_INFO_STREAM(
156 get_logger(), "Received SystemAlert message of type: " << static_cast<int>(msg->type) << " with message: " << msg->description);
157
158
159 if (msg->type == carma_msgs::msg::SystemAlert::SHUTDOWN // If a SHUTDOWN signal was received
160 || ( msg->type == carma_msgs::msg::SystemAlert::FATAL // OR if a required node notified FATAL
161 && std::find(config_.required_subsystem_nodes.begin(), config_.required_subsystem_nodes.end(), msg->source_node)
162 != config_.required_subsystem_nodes.end() ) )
163 {
164 // TODO might make more sense for external shutdown to be a service call and remove SHUTDOWN from system alert entirely
165 RCLCPP_INFO_STREAM( get_logger(), "Shutting down");
166
167 lifecycle_mgr_.shutdown(std_msec(config_.service_timeout_ms), std_msec(config_.call_timeout_ms), false);
168
169 carma_msgs::msg::SystemAlert shutdown_msg; // Notify ros1 nodes of shutdown
170
171 shutdown_msg.type = carma_msgs::msg::SystemAlert::SHUTDOWN;
172 shutdown_msg.description = "SHUTDOWN " + msg->description;
173
174 publish_system_alert(shutdown_msg);
175
176 std::this_thread::sleep_for(std::chrono::milliseconds(5)); // Add a bit of sleep to allow message publication
177
178 rclcpp::shutdown(nullptr, "System Alert requested shutdown"); // Fully shutdown this node
179 }
180 }
181 catch (const std::exception &e)
182 {
183 on_error(e);
184 }
185 }
