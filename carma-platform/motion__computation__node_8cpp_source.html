45 rcl_interfaces::msg::SetParametersResult MotionComputationNode::parameter_update_callback(const std::vector<rclcpp::Parameter> &parameters)
46 {
47 auto error = update_params<double>({
48 {"prediction_time_step", config_.prediction_time_step},
49 {"prediction_period", config_.prediction_period},
50 {"cv_x_accel_noise", config_.cv_x_accel_noise},
51 {"cv_y_accel_noise", config_.cv_y_accel_noise},
52 {"prediction_process_noise_max", config_.prediction_process_noise_max},
53 {"prediction_confidence_drop_rate", config_.prediction_confidence_drop_rate}
54 }, parameters);
55
56 auto error_2 = update_params<bool>({
57 {"enable_bsm_processing", config_.enable_bsm_processing},
58 {"enable_psm_processing", config_.enable_psm_processing},
59 {"enable_mobility_path_processing", config_.enable_mobility_path_processing},
60 {"enable_sensor_processing", config_.enable_sensor_processing}
61 }, parameters);
62
63 rcl_interfaces::msg::SetParametersResult result;
64
65 result.successful = !error && !error_2;
66
67 if (result.successful){
68 // Set motion_worker_'s prediction parameters
69 motion_worker_.setPredictionTimeStep(config_.prediction_time_step);
70 motion_worker_.setPredictionPeriod(config_.prediction_period);
71 motion_worker_.setXAccelerationNoise(config_.cv_x_accel_noise);
72 motion_worker_.setYAccelerationNoise(config_.cv_y_accel_noise);
73 motion_worker_.setProcessNoiseMax(config_.prediction_process_noise_max);
74 motion_worker_.setConfidenceDropRate(config_.prediction_confidence_drop_rate);
75 motion_worker_.setDetectionInputFlags(
76 config_.enable_sensor_processing,
77 config_.enable_bsm_processing,
78 config_.enable_psm_processing,
79 config_.enable_mobility_path_processing
80 );
81 }
82
83 return result;
84 }

86 carma_ros2_utils::CallbackReturn MotionComputationNode::handle_on_configure(const rclcpp_lifecycle::State &)
87 {
88 RCLCPP_INFO_STREAM(get_logger(), "MotionComputationNode trying to configure");
89
90 // Reset config
91 config_ = Config();
92
93 // Load parameters
94 get_parameter<double>("prediction_time_step", config_.prediction_time_step);
95 get_parameter<double>("prediction_period", config_.prediction_period);
96 get_parameter<double>("cv_x_accel_noise", config_.cv_x_accel_noise);
97 get_parameter<double>("cv_y_accel_noise", config_.cv_y_accel_noise);
98 get_parameter<double>("prediction_process_noise_max", config_.prediction_process_noise_max);
99 get_parameter<double>("prediction_confidence_drop_rate", config_.prediction_confidence_drop_rate);
100 get_parameter<bool>("enable_bsm_processing", config_.enable_bsm_processing);
101 get_parameter<bool>("enable_psm_processing", config_.enable_psm_processing);
102 get_parameter<bool>("enable_mobility_path_processing", config_.enable_mobility_path_processing);
103 get_parameter<bool>("enable_sensor_processing", config_.enable_sensor_processing);
104
105 RCLCPP_INFO_STREAM(get_logger(), "Loaded params: " << config_);
106
107 // Register runtime parameter update callback
108 add_on_set_parameters_callback(std::bind(&MotionComputationNode::parameter_update_callback, this, std_ph::_1));
109
110 // Setup subscribers
111 motion_comp_sub_ = create_subscription<carma_perception_msgs::msg::ExternalObjectList>("external_objects", 1,
112 std::bind(&MotionComputationWorker::predictionLogic, &motion_worker_, std_ph::_1));
113
114 mobility_path_sub_ = create_subscription<carma_v2x_msgs::msg::MobilityPath>("incoming_mobility_path", 100,
115 std::bind(&MotionComputationWorker::mobilityPathCallback, &motion_worker_, std_ph::_1));
116
117
118 bsm_sub_ = create_subscription<carma_v2x_msgs::msg::BSM>("incoming_bsm", 100,
119 std::bind(&MotionComputationWorker::bsmCallback, &motion_worker_, std_ph::_1));
120
121 psm_sub_ = create_subscription<carma_v2x_msgs::msg::PSM>("incoming_psm", 100,
122 std::bind(&MotionComputationWorker::psmCallback, &motion_worker_, std_ph::_1));
123
124 georeference_sub_ = create_subscription<std_msgs::msg::String>("georeference", 1,
125 std::bind(&MotionComputationWorker::georeferenceCallback, &motion_worker_, std_ph::_1));
126
127
128 // Setup publishers
129 carma_obj_pub_ = create_publisher<carma_perception_msgs::msg::ExternalObjectList>("external_object_predictions", 2);
130
131 // Set motion_worker_'s prediction parameters
132 motion_worker_.setPredictionTimeStep(config_.prediction_time_step);
133 motion_worker_.setPredictionPeriod(config_.prediction_period);
134 motion_worker_.setXAccelerationNoise(config_.cv_x_accel_noise);
135 motion_worker_.setYAccelerationNoise(config_.cv_y_accel_noise);
136 motion_worker_.setProcessNoiseMax(config_.prediction_process_noise_max);
137 motion_worker_.setConfidenceDropRate(config_.prediction_confidence_drop_rate);
138 motion_worker_.setDetectionInputFlags(
139 config_.enable_sensor_processing,
140 config_.enable_bsm_processing,
141 config_.enable_psm_processing,
142 config_.enable_mobility_path_processing
143 );
144
145 // Return success if everthing initialized successfully
146 return CallbackReturn::SUCCESS;
147 }
