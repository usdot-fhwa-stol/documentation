89 ros::CARMANodeHandle::setSystemAlertCallback([&](const cav_msgs::SystemAlertConstPtr& msg) -> void {
90
91 bool shutdown = false;
92
93
94 if (msg->type == cav_msgs::SystemAlert::FATAL)
95 {
96 // An empty source_node indicates that the alert was generated by a ROS1 node
97 // Therefore there is no complex error handling and the system should shutdown
98 if(msg->source_node.empty())
99 {
100 shutdown = true;
101 }
102 // If the ROS2 system controller failed we need to shutdown
103 else if (msg->source_node.compare("/system_controller") == 0)
104 {
105 shutdown = true;
106 }
107 else { // No need to shutdown if the alert was generated by a ROS2 node and it wasn't the system controller
108 shutdown = false;
109 }
110 }
111
112
113 if (shutdown)
114 {
115 ROS_ERROR_STREAM("Health Monitor shutting down due to system alert: " << msg->description);
116 std::string header = "health_monitor requesting shutdown due to: " + msg->description;
117
118 cav_msgs::SystemAlert new_msg;
119 new_msg.description = header;
120 new_msg.type = cav_msgs::SystemAlert::SHUTDOWN;
121
122 nh_->publishSystemAlert(new_msg);
123 }
124
125 });

155 bool HealthMonitor::spin_cb()
156 {
157 long time_now=(ros::Time::now().toNSec() / 1e6);
158 ros::Duration sd(startup_duration_);
159 long start_duration=sd.toNSec() / 1e6;
160
161 auto dm = driver_manager_.handleSpin(truck_,car_,time_now,start_up_timestamp_,start_duration);
162 if (!prev_alert) {
163 prev_alert = dm;
164 nh_->publishSystemAlert(dm);
165 } else if (prev_alert->type == dm.type && prev_alert->description.compare(dm.description) == 0) { // Do not publish duplicate alerts
166 ROS_DEBUG_STREAM("No change to alert status");
167 } else {
168 prev_alert = dm;
169 nh_->publishSystemAlert(dm);
170 }
171
172 return true;
173 }
