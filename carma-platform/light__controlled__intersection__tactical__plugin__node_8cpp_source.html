52 {

55 {"trajectory_time_length", config_.trajectory_time_length},
56 {"curve_resample_step_size", config_.curve_resample_step_size},

62 {"stop_line_buffer", config_.stop_line_buffer},
63 {"minimum_speed", config_.minimum_speed},
64 {"algorithm_evaluation_distance", config_.algorithm_evaluation_distance},
65 {"algorithm_evaluation_period", config_.algorithm_evaluation_period}}, parameters); // Global acceleration limits not allowed to dynamically update
66 auto error_2 = update_params<int>(
67 {{"default_downsample_ratio", config_.default_downsample_ratio},
68 {"turn_downsample_ratio", config_.turn_downsample_ratio},
69 {"curvature_moving_average_window_size", config_.curvature_moving_average_window_size},
70 {"speed_moving_average_window_size", config_.speed_moving_average_window_size}}, parameters);
71
72 if (worker_)
73 {
74 worker_->setConfig(config_);
75 }
76
77 rcl_interfaces::msg::SetParametersResult result;
78
79 result.successful = !error && !error_2;
80
81 return result;

84 carma_ros2_utils::CallbackReturn LightControlledIntersectionTransitPluginNode::on_configure_plugin()
85 {
86 RCLCPP_INFO_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "LightControlledIntersectionTransitPluginNode trying to configure");
87
88 // Reset config
89 config_ = Config();
90
91 // Load parameters
92 get_parameter<double>("centerline_sampling_spacing", config_.centerline_sampling_spacing);
93 get_parameter<double>("trajectory_time_length", config_.trajectory_time_length);
94 get_parameter<int>("default_downsample_ratio", config_.default_downsample_ratio);
95 get_parameter<int>("turn_downsample_ratio", config_.turn_downsample_ratio);
96 get_parameter<double>("curve_resample_step_size", config_.curve_resample_step_size);
97 get_parameter<int>("curvature_moving_average_window_size", config_.curvature_moving_average_window_size);
98 get_parameter<int>("speed_moving_average_window_size", config_.speed_moving_average_window_size);
99 get_parameter<double>("back_distance", config_.back_distance);
100 get_parameter<double>("buffer_ending_downtrack", config_.buffer_ending_downtrack);
101 get_parameter<double>("vehicle_decel_limit_multiplier", config_.vehicle_decel_limit_multiplier);
102 get_parameter<double>("vehicle_accel_limit_multiplier", config_.vehicle_accel_limit_multiplier);
103 get_parameter<double>("lat_accel_multiplier", config_.lat_accel_multiplier);
104 get_parameter<double>("stop_line_buffer", config_.stop_line_buffer);
105 get_parameter<double>("minimum_speed", config_.minimum_speed);
106 get_parameter<double>("algorithm_evaluation_distance", config_.algorithm_evaluation_distance);
107 get_parameter<double>("algorithm_evaluation_period", config_.algorithm_evaluation_period);
108 get_parameter<double>("vehicle_lateral_accel_limit", config_.lateral_accel_limit);
109 get_parameter<double>("vehicle_acceleration_limit", config_.vehicle_accel_limit);
110 get_parameter<double>("vehicle_deceleration_limit", config_.vehicle_decel_limit);
111
112 // Use the configured multipliers to update the accel limits
113 config_.lateral_accel_limit = config_.lateral_accel_limit * config_.lat_accel_multiplier;
114 config_.vehicle_accel_limit = config_.vehicle_accel_limit * config_.vehicle_accel_limit_multiplier;
115 config_.vehicle_decel_limit = config_.vehicle_decel_limit * config_.vehicle_decel_limit_multiplier;
116
117 // Register runtime parameter update callback
118 add_on_set_parameters_callback(std::bind(&LightControlledIntersectionTransitPluginNode::parameter_update_callback, this, std_ph::_1));
119
120 RCLCPP_INFO_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Loaded params: " << config_);
121
122 // Initialize worker object
123 worker_ = std::make_shared<LightControlledIntersectionTacticalPlugin>(get_world_model(), config_, get_plugin_name(), get_node_logging_interface());
124
125 // Return success if everything initialized successfully
126 return CallbackReturn::SUCCESS;
127 }
