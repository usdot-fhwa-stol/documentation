49 protected void proc(String sLineArcsFile, double dTol) throws Exception
50 {
51 try
52 {
53 ArrayList<TrafCtrl> oCtrls = new ArrayList();
54 ArrayList<CtrlLineArcs> oLineArcs = new ArrayList();
55 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(Paths.get(sLineArcsFile)))))
56 {
57 while (oIn.available() > 0)
58 {
59 oLineArcs.add(new CtrlLineArcs(oIn));
60 }
61 }
62 ArrayList<int[]> nTiles = new ArrayList();
63 for (CtrlLineArcs oCLA : oLineArcs)
64 {
65 String sType = XodrUtil.getLaneType(oCLA.m_nLaneType);
66 TrafCtrl oCtrl = null;
67 if (sType.compareTo("roadWorks") == 0)
68 {
69 oCtrl = new TrafCtrl("closed", "notopen", 0, oCLA.m_dLineArcs, "", true, CC);
70 }
71 else if (sType.contains("special1"))
72 {
73 oCtrl = new TrafCtrl("closed", sType.contains("1") ? "taperright" : "taperleft", 0, oCLA.m_dLineArcs, "", true, CC);
74 }
75 else if (sType.contains("special2"))
76 {
77 oCtrl = new TrafCtrl("closed", sType.contains("2") ? "openright" : "openleft", 0, oCLA.m_dLineArcs, "", true, CC);
78 }
79
80 if (oCtrl == null)
81 continue;
82 oCtrls.add(oCtrl);
83 oCtrl.write(g_sTrafCtrlDir, g_dExplodeStep, g_nDefaultZoom, CC);
84 updateTiles(nTiles, oCtrl.m_oFullGeo.m_oTiles);
85 }
86 renderTiledData(oCtrls, nTiles);
87 }
88 catch (Exception oEx)
89 {
90 oEx.printStackTrace();
91 }
92 }

101 ArrayList<TrafCtrl> oTaperCtrls = new ArrayList();
102 ArrayList<Integer> oTaperValues = new ArrayList();
103 ArrayList<double[]> dTapers = new ArrayList();
104 for (String sVal : new String[]{"taperleft", "taperright", "openleft", "openright"})
105 oTaperValues.add(TrafCtrlEnums.getCtrlVal("closed", sVal));

107 Collections.sort(oTaperValues);
108 int nOpen = TrafCtrlEnums.getCtrlVal("closed", "open");
109 int nCtrlIndex = oCtrls.size();

111 {
112 TrafCtrl oCtrl = oCtrls.get(nCtrlIndex);
113 int nCtrlVal = MathUtil.bytesToInt(oCtrl.m_yControlValue);
114 if (nCtrlVal == nOpen) // do not render not open
115 continue;
116 if (Collections.binarySearch(oTaperValues, nCtrlVal) >= 0)
117 {
118 oCtrls.remove(nCtrlIndex);
119 oTaperCtrls.add(oCtrl);
120 String sCtrlVal = TrafCtrlEnums.CTRLS[oCtrl.m_nControlType][nCtrlVal];

122 double[] dC = oCtrl.m_oFullGeo.m_dC;
123 int nPoints = Arrays.size(dC) / 2;
124 double dPercent = 1.0 / nPoints;
125 int nCount = nPoints;
126 int nIncrease = -1;
127 if (sCtrlVal.contains("taper"))
128 {
129 nCount = 0;
130 nIncrease = 1;
131 }
132
133
134 boolean bRight = sCtrlVal.contains(("right"));
135 double[] dStraight = bRight ? oCtrl.m_oFullGeo.m_dNT : oCtrl.m_oFullGeo.m_dPT;

142 double dX1 = dC[nIndex];
143 double dY1 = dC[nIndex + 1];

151 double dY3 = dStraight[nIndex + 1];
152 double dHeading = Geo.heading(dX3, dY3, dX1, dY1);
153 nCount += nIncrease;
154 double dOffset = Geo.distance(dX3, dY3, dX1, dY1) * 2 * dPercent * nCount;
155 dTaper = Arrays.add(dTaper, dX3 + Math.cos(dHeading) * dOffset, dY3 + Math.sin(dHeading) * dOffset);

157 }
158 if (bRight)

168 }
169 }
170 for (int[] nTile : nTiles)
171 {
172 int nX = nTile[0];
173 int nY = nTile[1];
174 Files.createDirectories(Paths.get(String.format(g_sTdFileFormat, nX, 0, 0, 0)).getParent(), FileUtil.DIRPERS);
175 writeIndexFile(oCtrls, nX, nY);

183 oClippedLanes.add(new ArrayList());

185 oClippedLanes.add(new ArrayList());

192 TileUtil.clipCtrlGeoForTile(oCtrl.m_oFullGeo, dClips, dClipBounds, oClippedLanes);

244 private static void renderLines(double[] dE1, double[] dE2, TrafCtrl oCtrl, TdLayer oLayer)
245 {
246 int[] nColor = new int[]{0, 0};
247 double dLength = 0.0;
248 Iterator<double[]> oIt = Arrays.iterator(dE1, new double[4], 1, 2);
249 while (oIt.hasNext())
250 {
251 double[] dSeg = oIt.next();
252 dLength += Geo.distance(dSeg[0], dSeg[1], dSeg[2], dSeg[3]);
253 }
254
255 int nStripes = (int)Math.ceil(dLength / 0.5); // 50 cm
256 if (nStripes % 2 == 0) // ensure we start and end with a white stripe
257 ++nStripes;
258 double dStep = dLength / nStripes;
259 nStripes = 0;
260 double[] dStripe = Arrays.newDoubleArray(4);
261 double dTemp = 0.0;
262 Iterator<double[]> oIt2 = Arrays.iterator(dE2, new double[4], 1, 2);
263 oIt = Arrays.iterator(dE1, new double[4], 1, 2);
264 while (oIt.hasNext())
265 {
266 double[] dSeg1 = oIt.next();
267 double[] dSeg2 = oIt2.next();
268 dTemp += Geo.distance(dSeg1[0], dSeg1[1], dSeg1[2], dSeg1[3]);
269 if ((int)(dTemp / dStep) != nStripes && dTemp < dLength - 0.25)
270 {
271 dStripe[0] = 1;
272 nColor[1] = nStripes++ % 2;
273 dStripe = Arrays.add(dStripe, dSeg1[2], dSeg1[3]);
274 dStripe = Arrays.add(dStripe, dSeg2[2], dSeg2[3]);
275 oLayer.add(new TdFeature(dStripe, nColor, oCtrl));
276 }
277 }
