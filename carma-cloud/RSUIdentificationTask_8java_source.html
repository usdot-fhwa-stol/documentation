51 }

57 //NOTE: For the carma-cloud registered RSUs, it needs additional consideration for those RSUs (RSUs, geofences, etc.) in a certain requested region.
58 if (registeredRSUs != null && registeredRSUs.size() > 0) {
59 ArrayList<RSUBoundingbox> identifiedRSUs = new ArrayList<>();
60 String identifiedRSUNames = "";
61 for (RSUBoundingbox rsuBoundingBox : registeredRSUs) {
62 BoundingBox bBox = rsuBoundingBox.getBoundingBox();
63 for (BoundingBox bsmBoxOther : boundingBoxes) {
64 if (bBox.intersects(bsmBoxOther)) {
65 identifiedRSUs.add(rsuBoundingBox);

69 }
