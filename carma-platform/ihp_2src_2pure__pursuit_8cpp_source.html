29 double PurePursuit::getLookaheadDist(const cav_msgs::TrajectoryPlanPoint& tp) const
30 {
31 double x_diff = (tp.x - current_pose_.position.x);
32 double y_diff = (tp.y - current_pose_.position.y);
33 double dist = std::sqrt(x_diff * x_diff + y_diff * y_diff);
34 ROS_DEBUG_STREAM("calculated lookahead: " << dist);
35 return dist;
36 }

64 double PurePursuit::calculateKappa(const cav_msgs::TrajectoryPlanPoint& tp)
65 {
66 double lookahead = getLookaheadDist(tp);
67 ROS_DEBUG_STREAM("used lookahead: " << lookahead);
68 double alpha = getAlphaSin(tp, current_pose_);
69 ROS_DEBUG_STREAM("calculated alpha: " << alpha);
70 double kappa = 2 * (alpha)/(lookahead);
71 ROS_DEBUG_STREAM("calculated kappa: " << kappa);
72 return kappa;
73 }

75 void PurePursuit::calculateSteer(const cav_msgs::TrajectoryPlanPoint& tp)
76 {
77
78 double kappa = calculateKappa(tp);
79
80 double steering = atan(config_.wheelBase * kappa);
81 ROS_DEBUG_STREAM("calculated steering angle: " << steering);
82 double filtered_steering = lowPassfilter(config_.lowpassGain, prev_steering, steering);
83 ROS_DEBUG_STREAM("filtered steering: " << filtered_steering);
84 if (std::isnan(filtered_steering)) filtered_steering = prev_steering;
85 prev_steering = filtered_steering;
86 steering_angle_ = filtered_steering;
87
88 double ang_vel = velocity_ * kappa;
89 ROS_DEBUG_STREAM("calculated angular velocity: " << ang_vel);
90 double filtered_ang_vel = lowPassfilter(config_.lowpassGain, prev_ang_vel, ang_vel);
91 ROS_DEBUG_STREAM("filtered angular velocity: " << filtered_ang_vel);
92 prev_ang_vel = filtered_ang_vel;
93 if (std::isnan(filtered_ang_vel)) filtered_ang_vel = prev_ang_vel;
94 angular_velocity_ = filtered_ang_vel;
95 }

97 double PurePursuit::getAlphaSin(cav_msgs::TrajectoryPlanPoint tp, geometry_msgs::Pose current_pose)
98 {
99 tf::Transform inverse;
100 tf::poseMsgToTF(current_pose, inverse);
101 tf::Transform transform = inverse.inverse();
102
103 geometry_msgs::Point point_msg;
104 point_msg.x = tp.x;
105 point_msg.y = tp.y;
106 point_msg.z = current_pose.position.z;
107 tf::Point p;
108 pointMsgToTF(point_msg, p);
109 tf::Point tf_p = transform * p;
110 geometry_msgs::Point tf_point_msg;
111 pointTFToMsg(tf_p, tf_point_msg);
112 ROS_DEBUG_STREAM("relative latitude: " << tf_point_msg.y);
113 ROS_DEBUG_STREAM("relative longitude: " << tf_point_msg.x);
114 ROS_DEBUG_STREAM("relative z: " << tf_point_msg.z);
115 double vec_mag = std::sqrt(tf_point_msg.y*tf_point_msg.y + tf_point_msg.x*tf_point_msg.x + tf_point_msg.z*tf_point_msg.z);
116 ROS_DEBUG_STREAM("relative vector mag: " << vec_mag);
117 double sin_alpha = tf_point_msg.y/vec_mag;
118 ROS_DEBUG_STREAM("alpha sin from transform: " << sin_alpha);
119
120 double angle_tp_map = atan2(tp.y, tp.x); // angle of vector to tp point in map frame
121 ROS_DEBUG_STREAM("angle_tp_map: " << angle_tp_map);
122 tf::Quaternion quat;
123 tf::quaternionMsgToTF(current_pose.orientation, quat);
124 double roll, pitch, yaw;
125 tf::Matrix3x3(quat).getRPY(roll, pitch, yaw);
126 ROS_DEBUG_STREAM("yaw: " << yaw);
127 double alpha = angle_tp_map - yaw;
128 double sin_alpha2 = sin(alpha);
129 ROS_DEBUG_STREAM("alpha from orientation: " << alpha);
130 ROS_DEBUG_STREAM("alpha sin from orientation: " << sin_alpha2);
131
132 return sin_alpha;
133 }

136 {
