55rcl_interfaces::msg::SetParametersResult MotionComputationNode::parameter_update_callback(
56 const std::vector<rclcpp::Parameter> & parameters)
57{
58 auto error = update_params<double>(
59 {{"prediction_time_step", config_.prediction_time_step},
60 {"prediction_period", config_.prediction_period},
61 {"cv_x_accel_noise", config_.cv_x_accel_noise},
62 {"cv_y_accel_noise", config_.cv_y_accel_noise},
63 {"prediction_process_noise_max", config_.prediction_process_noise_max},
64 {"prediction_confidence_drop_rate", config_.prediction_confidence_drop_rate}},
65 parameters);
66
67 auto error_2 = update_params<bool>(
68 {{"enable_bsm_processing", config_.enable_bsm_processing},
69 {"enable_psm_processing", config_.enable_psm_processing},
70 {"enable_mobility_path_processing", config_.enable_mobility_path_processing},
71 {"enable_sensor_processing", config_.enable_sensor_processing}},
72 parameters);
73
74 rcl_interfaces::msg::SetParametersResult result;
75
76 result.successful = !error && !error_2;
77
78 if (result.successful) {
79 // Set motion_worker_'s prediction parameters
80 motion_worker_.setPredictionTimeStep(config_.prediction_time_step);
81 motion_worker_.setPredictionPeriod(config_.prediction_period);
82 motion_worker_.setXAccelerationNoise(config_.cv_x_accel_noise);
83 motion_worker_.setYAccelerationNoise(config_.cv_y_accel_noise);
84 motion_worker_.setProcessNoiseMax(config_.prediction_process_noise_max);
85 motion_worker_.setConfidenceDropRate(config_.prediction_confidence_drop_rate);
86 motion_worker_.setDetectionInputFlags(
87 config_.enable_sensor_processing, config_.enable_bsm_processing,
88 config_.enable_psm_processing, config_.enable_mobility_path_processing);
89 }
90
91 return result;
92}

94carma_ros2_utils::CallbackReturn MotionComputationNode::handle_on_configure(
95 const rclcpp_lifecycle::State &)
96{
97 RCLCPP_INFO_STREAM(get_logger(), "MotionComputationNode trying to configure");
98
99 // Reset config
100 config_ = Config();
101
102 // Load parameters
103 get_parameter<double>("prediction_time_step", config_.prediction_time_step);
104 get_parameter<double>("prediction_period", config_.prediction_period);
105 get_parameter<double>("cv_x_accel_noise", config_.cv_x_accel_noise);
106 get_parameter<double>("cv_y_accel_noise", config_.cv_y_accel_noise);
107 get_parameter<double>("prediction_process_noise_max", config_.prediction_process_noise_max);
108 get_parameter<double>("prediction_confidence_drop_rate", config_.prediction_confidence_drop_rate);
109 get_parameter<bool>("enable_bsm_processing", config_.enable_bsm_processing);
110 get_parameter<bool>("enable_psm_processing", config_.enable_psm_processing);
111 get_parameter<bool>("enable_mobility_path_processing", config_.enable_mobility_path_processing);
112 get_parameter<bool>("enable_sensor_processing", config_.enable_sensor_processing);
113
114 RCLCPP_INFO_STREAM(get_logger(), "Loaded params: " << config_);
115
116 // Register runtime parameter update callback
117 add_on_set_parameters_callback(
118 std::bind(&MotionComputationNode::parameter_update_callback, this, std_ph::_1));
119
120 // Setup subscribers
121 motion_comp_sub_ = create_subscription<carma_perception_msgs::msg::ExternalObjectList>(
122 "external_objects", 1,
123 std::bind(&MotionComputationWorker::predictionLogic, &motion_worker_, std_ph::_1));
124
125 mobility_path_sub_ = create_subscription<carma_v2x_msgs::msg::MobilityPath>(
126 "incoming_mobility_path", 100,
127 std::bind(&MotionComputationWorker::mobilityPathCallback, &motion_worker_, std_ph::_1));
128
129 bsm_sub_ = create_subscription<carma_v2x_msgs::msg::BSM>(
130 "incoming_bsm", 100,
131 std::bind(&MotionComputationWorker::bsmCallback, &motion_worker_, std_ph::_1));
132
133 psm_sub_ = create_subscription<carma_v2x_msgs::msg::PSM>(
134 "incoming_psm", 100,
135 std::bind(&MotionComputationWorker::psmCallback, &motion_worker_, std_ph::_1));
136
137 georeference_sub_ = create_subscription<std_msgs::msg::String>(
138 "georeference", 1,
139 std::bind(&MotionComputationWorker::georeferenceCallback, &motion_worker_, std_ph::_1));
140
141 // Setup publishers
142 carma_obj_pub_ = create_publisher<carma_perception_msgs::msg::ExternalObjectList>(
143 "external_object_predictions", 2);
144
145 // Set motion_worker_'s prediction parameters
146 motion_worker_.setPredictionTimeStep(config_.prediction_time_step);
147 motion_worker_.setPredictionPeriod(config_.prediction_period);
148 motion_worker_.setXAccelerationNoise(config_.cv_x_accel_noise);
149 motion_worker_.setYAccelerationNoise(config_.cv_y_accel_noise);
150 motion_worker_.setProcessNoiseMax(config_.prediction_process_noise_max);
151 motion_worker_.setConfidenceDropRate(config_.prediction_confidence_drop_rate);
152 motion_worker_.setDetectionInputFlags(
153 config_.enable_sensor_processing, config_.enable_bsm_processing, config_.enable_psm_processing,
154 config_.enable_mobility_path_processing);
155
156 // Return success if everything initialized successfully
157 return CallbackReturn::SUCCESS;
158}
