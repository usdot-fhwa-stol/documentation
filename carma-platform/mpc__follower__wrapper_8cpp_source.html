62void MPCFollowerWrapper::TrajectoryPlanPoseHandler(const cav_msgs::TrajectoryPlan::ConstPtr& tp){
63 ROS_DEBUG_STREAM("Received TrajectoryPlanCurrentPosecallback message");
64 try {
65 autoware_msgs::Lane lane;
66 lane.header = tp->header;
67 std::vector <autoware_msgs::Waypoint> waypoints;
68
69 for(int i = 0; i < tp->trajectory_points.size() - 1; i++ ) {
70
71 cav_msgs::TrajectoryPlanPoint t1 = tp->trajectory_points[i];
72 cav_msgs::TrajectoryPlanPoint t2 = tp->trajectory_points[i + 1];
73 autoware_msgs::Waypoint waypoint = mpcww.TrajectoryPlanPointToWaypointConverter(t1, t2);
74
75 waypoints.push_back(waypoint);
76 }
77
78 lane.waypoints = waypoints;
79 PublisherForWayPoints(lane);
80 }
81 catch(const std::exception& e) {
82 ros::CARMANodeHandle::handleException(e);
83 }
84
85};
