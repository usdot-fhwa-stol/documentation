333 void tsc_service::log_spat_latency(int &count, uint64_t &spat_processing_time, uint64_t spat_time_stamp) const {
334 // Calculate and average spat processing time over 20 messages sent
335 if (count <= 20 ) {
336 uint64_t timestamp = streets_clock_singleton::time_in_ms();
337 spat_processing_time += timestamp - spat_time_stamp;
338 count++;
339 }
340 // Log result at 20 and clear counts
341 if (count >= 20 ) {
342 double total_processing_time = ((double)(spat_processing_time))/20.0;
343 SPDLOG_INFO("SPat average processing time over 20 messages is {0} ms and total processing time for 20 messages is {1} ms!", total_processing_time, spat_processing_time);
344 spat_processing_time = 0;
345 count = 0;
346 }
