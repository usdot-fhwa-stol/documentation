23 void TruckInspectionClient::initialize()
24 {
25 nh_.reset(new ros::CARMANodeHandle());
26 pnh_.reset(new ros::CARMANodeHandle("~"));
27 pnh_->getParam("/vin_number", vin_number_); //get vin from global params in UniqueVehicleParams.ymal
28 pnh_->getParam("license_plate", license_plate_);
29 pnh_->getParam("state_short_name", state_short_name_);
30 pnh_->getParam("carrier_name", carrier_name_);
31 pnh_->getParam("carrier_id", carrier_id_);
32 pnh_->getParam("weight", weight_);
33 pnh_->getParam("date_of_last_state_inspection", date_of_last_state_inspection_);
34 pnh_->getParam("date_of_last_ads_calibration", date_of_last_ads_calibration_);
35 pnh_->getParam("iss_score", iss_score_);
36 pnh_->getParam("permit_required", permit_required_);
37 pnh_->getParam("pre_trip_ads_health_check", pre_trip_ads_health_check_);
38 mo_pub_ = nh_->advertise<cav_msgs::MobilityOperation>("outgoing_mobility_operation", 5);
39 request_sub_ = nh_->subscribe("incoming_mobility_request", 1, &TruckInspectionClient::requestCallback, this);
40 ads_state_sub_ = nh_->subscribe("guidance/state", 1, &TruckInspectionClient::guidanceStatesCallback, this);
41 ads_system_alert_sub_ = nh_->subscribe("system_alert", 1, &TruckInspectionClient::systemAlertsCallback, this);
42 version_sub_ = nh_->subscribe("carma_system_version", 1, &TruckInspectionClient::versionCallback, this);
43 bsm_sub_ = nh_->subscribe("bsm_outbound", 1, &TruckInspectionClient::bsmCallback, this);
44 this->ads_engaged_ = false;
45 this->ads_system_alert_type_ = std::to_string(cav_msgs::SystemAlert::NOT_READY);
46 this->ads_software_version_ = "System Version Unknown";
47 //Checking: vin number does not exist yet in global parameter server
48 while((!pnh_->getParam("/vin_number", vin_number_)) && vin_retrive_count < MAX_RETRIEVE_VIN_COUNT)
49 {
50 //sleep for 0.1 second
51 boost::this_thread::sleep_for(boost::chrono::milliseconds(100));
52 vin_retrive_count++;
53
54 if(pnh_->getParam("/vin_number", vin_number_))
55 {
56 ROS_INFO("retrieved vin_number1: %s",vin_number_.c_str());
57 break;
58 }
59 ROS_DEBUG("retrieving vin_number: %s",vin_number_.c_str());
60 }
61 //if exceed maximum count, publish warning messages to system_alert
62 if(vin_retrive_count >= MAX_RETRIEVE_VIN_COUNT)
63 {
64 cav_msgs::SystemAlert alert_msg;
65 alert_msg.type = cav_msgs::SystemAlert::WARNING;
66 alert_msg.description = ros::this_node::getName() + ": vin_number does not exist in global parameter server.";
67 ROS_ERROR_STREAM(alert_msg.description);
68 nh_->publishSystemAlert(alert_msg);
69 }
70
71 ros::CARMANodeHandle::setSpinRate(20.0);
72 // set vin publisher
73 ros::CARMANodeHandle::setSpinCallback([this]() -> bool {
74 cav_msgs::MobilityOperation msg_out;
75 msg_out.strategy = this->INSPECTION_STRATEGY;
76 msg_out.strategy_params = "vin_number:" + vin_number_ + ",license_plate:" + license_plate_ +",state_short_name:"+ state_short_name_;
77 mo_pub_.publish(msg_out);
78 return true;
79 });
80 ROS_INFO_STREAM("Truck inspection plugin is initialized...");
81 }

104 void TruckInspectionClient::requestCallback(const cav_msgs::MobilityRequestConstPtr& msg)
105 {
106 if(msg->strategy == this->INSPECTION_STRATEGY) {
107 cav_msgs::MobilityOperation mo_msg;
108 mo_msg.m_header.sender_bsm_id = bsm_id_;
109 mo_msg.strategy = this->INSPECTION_STRATEGY;
110 std::string ads_auto_status = this->ads_engaged_ ? "Engaged" : "Not Engaged";
111 std::string ads_health_status = ads_system_alert_type_ ;
112 std::string params = boost::str(boost::format("vin_number:%s,license_plate:%s,carrier_name:%s,carrier_id:%s,weight:%d,ads_software_version:%s,date_of_last_state_inspection:%s,date_of_last_ads_calibration:%s,pre_trip_ads_health_check:%s,ads_health_status:%s,ads_auto_status:%s,iss_score:%d,permit_required:%s")
113 % vin_number_ % license_plate_ % carrier_name_ % carrier_id_ % weight_ % ads_software_version_ % date_of_last_state_inspection_ % date_of_last_ads_calibration_ % pre_trip_ads_health_check_ % ads_health_status % ads_auto_status % iss_score_ % permit_required_);
114 long time = (long)(ros::Time::now().toNSec() / pow(10, 6));
115 mo_msg.m_header.timestamp = time;
116 mo_msg.strategy_params = params;
117 mo_pub_.publish(mo_msg);
118 }
119 }
