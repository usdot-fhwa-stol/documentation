90
91 // Find the nearest lanelets with maximum number (=3) of return lanelets because a point in intersection may return maximum three link/bridge lanelets
92 auto nearest_lanelets = lanelet::geometry::findNearest(this->map_ptr->laneletLayer, subj_point2d, 3);

104
105 if (itr->first == 0)

116 std::vector<lanelet::Lanelet> result_lanelets;
117
118 /****
119 * Check the lanelet turn direction to help identify the current lanelet if current geo-loc return more than 1 current lanelets.
120 * If there is only one element in the current total lanelets, ignore the turn direction
121 * ****/
122 if (current_total_lanelets.size() > 1)
123 {
124 for (auto itr = current_total_lanelets.begin(); itr != current_total_lanelets.end(); itr++)
125 {
126 const lanelet::Lanelet cur_lanelet = *itr;
127 if (cur_lanelet.hasAttribute("turn_direction"))
128 {
129 const lanelet::Attribute attr = cur_lanelet.attribute("turn_direction");
130 if (attr.value() == turn_direction)
131 {
132 result_lanelets.push_back(cur_lanelet);
133 }
134 }
135 }
136
137 if (result_lanelets.size() == 1)
138 {
139 return result_lanelets.front();
140 }
141
142 // If there are more than two current lanelets return, check the trajectory
143 if (result_lanelets.size() > 1)
144 {
145 lanelet::BasicPoint3d basic_point3d_start = this->ecef_2_map_point(trajectory.location.ecef_x, trajectory.location.ecef_y, trajectory.location.ecef_z);
146 if (trajectory.offsets.empty())
147 {
148 SPDLOG_ERROR("Cannot determine current lanelet and ids with vehicle trajectory offset size = 0. ");
149 return lanelet::Lanelet();
150 }
151
152 std::int32_t dest_x = trajectory.location.ecef_x;
153 std::int32_t dest_y = trajectory.location.ecef_y;
154 std::int32_t dest_z = trajectory.location.ecef_z;
155 for (auto offset_itr = trajectory.offsets.begin(); offset_itr != trajectory.offsets.end(); offset_itr++)
156 {
157 dest_x += offset_itr->offset_x;
158 dest_y += offset_itr->offset_y;
159 dest_z += offset_itr->offset_z;
160 SPDLOG_TRACE("offset_x = {0},offset_y = {1},offset_z = {2}", offset_itr->offset_x, offset_itr->offset_y, offset_itr->offset_z);
161 SPDLOG_TRACE("dest_x = {0},dest_y = {1},dest_z = {2}", dest_x, dest_y, dest_z);
162 }
163
164 lanelet::BasicPoint3d basic_point3d_dest = ecef_2_map_point(dest_x, dest_y, dest_z);
165 lanelet::Lanelet result_lanelet = lanelet::Lanelet();
166 double smaller_distance_sum = 0;
167 for (auto itr = result_lanelets.begin(); itr != result_lanelets.end(); itr++)
168 {
169 const lanelet::Lanelet cur_lanelet = *itr;
170 double start_distance2_ctl = lanelet::geometry::distance2d(lanelet::utils::to2D(basic_point3d_start), lanelet::utils::toHybrid(cur_lanelet.centerline2d()));
171 SPDLOG_DEBUG("cur_lanelet = {0} to trajectory start point distance = {1}", cur_lanelet.id(), start_distance2_ctl);
172
173 double dest_distance2_ctl = lanelet::geometry::distance2d(lanelet::utils::to2D(basic_point3d_dest), lanelet::utils::toHybrid(cur_lanelet.centerline2d()));
174 SPDLOG_DEBUG("cur_lanelet = {0} to trajectory dest point distance = {1}", cur_lanelet.id(), dest_distance2_ctl);
175
176 double cur_distance_sum = start_distance2_ctl + dest_distance2_ctl;
177
178 if (smaller_distance_sum >= cur_distance_sum || smaller_distance_sum == 0)
179 {
180 smaller_distance_sum = cur_distance_sum;
181 result_lanelet = cur_lanelet;
182 }
183 }
184 return result_lanelet;
185 }
186 else {
187 SPDLOG_ERROR("Cannot determine the current lanelet with this turn direction : {0} for point = (x : {1} , y : {2}, z=:{3})", turn_direction, subj_point3d.x(), subj_point3d.y(), subj_point3d.z());
188 return lanelet::Lanelet();
189 }
190 }
191 else if (current_total_lanelets.size() == 1)
192 {
193 return current_total_lanelets.front();
194 }
195 else {
196 SPDLOG_ERROR("No current lanelets to the vehicle in map point: x = {0}, y = {1}, z = {2}, and turn direction = {3}", subj_point3d.x(), subj_point3d.y(), subj_point3d.z(), turn_direction);
197 return lanelet::Lanelet();

207 double message_lanelet2_translation::distance2_cur_lanelet_end(lanelet::BasicPoint3d subj_point3d, lanelet::Lanelet subj_lanelet,std::string turn_direction, models::trajectory &trajectory) const
208 {
209 double total_length = 0.0;
210 double p2p_d = -1;
211 bool start_accumulate = false;
212 if (subj_lanelet.id() == lanelet::InvalId)
213 {
214 SPDLOG_ERROR("Get invalid lanelet id = {0} from position: ({1}, {2} , {3}) and turn direction: {4}", subj_point3d.x(), subj_point3d.y(), subj_point3d.z(), turn_direction);
215 return -1;
216 }
217
218 auto sub_lanelet_centerline = subj_lanelet.centerline2d();
219 lanelet::ConstHybridLineString2d lsHybrid = lanelet::utils::toHybrid(sub_lanelet_centerline);
220 lanelet::BasicPoint2d subj_point2d = lanelet::utils::to2D(subj_point3d);
221
222 // Get the distance between the subject point to the centerline.
223 auto dP2Line2d = lanelet::geometry::distance(subj_point2d, lsHybrid);
224 for (size_t i = 0; i < sub_lanelet_centerline.numSegments(); i++)
225 {
226 // Project the subject point to a point on the centerline
227 if(!start_accumulate)
228 {
229 p2p_d = lanelet::geometry::distance2d(subj_point2d, sub_lanelet_centerline.segment(i).first);
230 }
231 // Start from the closest point on the centerline to the subject point, and accumulate the closest point to the end of centerline
232 if (p2p_d != -1 && p2p_d <= (dP2Line2d + 0.2))
233 {
234 start_accumulate = true;
235 }
236
237 if (start_accumulate)
238 {
239 double segment_distance = lanelet::geometry::distance2d(sub_lanelet_centerline.segment(i).first, sub_lanelet_centerline.segment(i).second);
240 total_length += segment_distance;
241 }
242 }
243 return total_length;
244 }

278 std::map<int64_t, models::intersection_lanelet_type> message_lanelet2_translation::get_lanelet_types_ids(lanelet::Lanelet subj_lanelet, std::string turn_direction) const
279 {
280 std::map<int64_t, models::intersection_lanelet_type> lanelet_id_type_m;
281 if (subj_lanelet.id() == 0 )
282 {
283 SPDLOG_ERROR("Invalid start or end lanelet id. ");
284 return lanelet_id_type_m;
285 }
286
287 lanelet::ConstLanelet entry_lanelet;
288 lanelet::ConstLanelet link_lanelet;
289 lanelet::ConstLanelet departure_lanelet;
290 lanelet::Lanelet* ll_itr = &subj_lanelet;
291 try
292 {
296 lanelet::ConstLanelet local_ll = vehicleGraph_ptr->previous(*ll_itr).front();
297 if (!local_ll.regulatoryElements().empty())
298 {
299 lanelet::RegulatoryElementConstPtrs reg_ptrs = local_ll.regulatoryElements();
300 for (auto reg_ptrs_itr = reg_ptrs.begin(); reg_ptrs_itr != reg_ptrs.end(); reg_ptrs_itr++)
301 {
302 const lanelet::RegulatoryElement *reg = reg_ptrs_itr->get();
303 if (reg->attribute(lanelet::AttributeName::Subtype).value() == lanelet::AttributeValueString::AllWayStop)
304 {
305 SPDLOG_DEBUG("Found link lanelet id :{0} ", ll_itr->id());
306 entry_lanelet = local_ll;
307 link_lanelet = *ll_itr;
308 departure_lanelet = vehicleGraph_ptr->following(link_lanelet).front();
309 break;
310 }
311 }
312 }
313
314 /***
315 * Checking whether the current lanelet is entry lanelet, and entry lanelet has the all_way_stop regulatory element
316 * **/
317 if (!ll_itr->regulatoryElements().empty())
318 {
319 lanelet::RegulatoryElementPtrs reg_ptrs = ll_itr->regulatoryElements();
320 for (auto reg_ptrs_itr = reg_ptrs.begin(); reg_ptrs_itr != reg_ptrs.end(); reg_ptrs_itr++)
321 {
322 const lanelet::RegulatoryElement *reg = reg_ptrs_itr->get();
323 if (reg->attribute(lanelet::AttributeName::Subtype).value() == lanelet::AttributeValueString::AllWayStop)
324 {
325 SPDLOG_DEBUG("Found entry lanelet id :{0} ", ll_itr->id());
326 entry_lanelet = *ll_itr;
327 lanelet::ConstLanelets possible_link_lanelets = vehicleGraph_ptr->following(*ll_itr);
328 // Check turn_direction to determine the link lanelet for subject vehicle
329 // If turn direction is "NA" or empty, it cannot determine which link lanelet inside the intersection
330 for (auto itr = possible_link_lanelets.begin(); itr != possible_link_lanelets.end(); itr++)
331 {
332 if (itr->hasAttribute("turn_direction") && itr->attribute("turn_direction").value() == turn_direction)
333 {
334 link_lanelet = *itr;
335 SPDLOG_DEBUG("Found link lanelet id :{0} ", itr->id());
336 departure_lanelet = vehicleGraph_ptr->following(link_lanelet).front();
337 break;
338 }
339 }
340 }
341 }
342 }
348 // Get possible link lanelets
349 lanelet::ConstLanelets possible_links = vehicleGraph_ptr->previous(*ll_itr);
350 lanelet::ConstLanelet possible_link;
351 for ( const auto &lane : possible_links ) {
352 if ( lane.hasAttribute("turn_direction") && lane.attribute("turn_direction").value() == turn_direction ) {
353 possible_link = lane;
354 break;
355 }
356 }
357 if ( possible_link.id() != lanelet::InvalId ) {
358 lanelet::ConstLanelet possible_entry = vehicleGraph_ptr->previous(possible_link).front();
359 if (!possible_entry.regulatoryElements().empty())
360 {
361 lanelet::RegulatoryElementConstPtrs reg_ptrs = possible_entry.regulatoryElements();
362 for (auto reg_ptrs_itr = reg_ptrs.begin(); reg_ptrs_itr != reg_ptrs.end(); reg_ptrs_itr++)
363 {
364 const lanelet::RegulatoryElement *reg = reg_ptrs_itr->get();
365 if (reg->attribute(lanelet::AttributeName::Subtype).value() == lanelet::AttributeValueString::AllWayStop)
366 {
367 entry_lanelet = possible_entry;
368 link_lanelet = possible_link;
369 departure_lanelet = *ll_itr;
370 break;
371 }
372 }
373 }
374 }
375
376
377 // insert the type for each lanelet id in the list of lanelet ids
378 if (entry_lanelet.id() != lanelet::InvalId)
379 {
380 lanelet_id_type_m.insert(std::make_pair(entry_lanelet.id(), models::intersection_lanelet_type::entry));
381 }
382
383 if (link_lanelet.id() != lanelet::InvalId)
384 {
385 lanelet_id_type_m.insert(std::make_pair(link_lanelet.id(), models::intersection_lanelet_type::link));
386 }
387
388 if (departure_lanelet.id() != lanelet::InvalId)
389 {
390 lanelet_id_type_m.insert(std::make_pair(departure_lanelet.id(), models::intersection_lanelet_type::departure));
391 }
392 return lanelet_id_type_m;
393 }
394 catch(const lanelet::LaneletError &e)
395 {
396 SPDLOG_ERROR("Cannot determine lanelet type and ids with vehicle current lanelet. \n {0}", e.what());
397 lanelet_id_type_m.clear();
398 return lanelet_id_type_m;
399 }
