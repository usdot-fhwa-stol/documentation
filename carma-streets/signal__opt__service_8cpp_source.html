84 {
85 std::thread tsc_config_t(&signal_opt_service::consume_tsc_config, this, this->_tsc_config_consumer, this->tsc_configuration_ptr);
86 // Block until TSC configuration information is received
87 tsc_config_t.join();

89 populate_movement_groups(movement_groups_ptr, tsc_configuration_ptr);
90 // After movement groups are created, process vehicle status and intent and spat messages to

92 std::thread vsi_t(&signal_opt_service::consume_vsi, this, this->_vsi_consumer, this->vehicle_list_ptr);

95 this->intersection_info_ptr,
96 this->vehicle_list_ptr,
97 this->spat_ptr,
98 this->tsc_configuration_ptr,
99 this->movement_groups_ptr,
100 this->dpp_config,
101 this->so_sleep_time);
102 // Running in parallel
103 spat_t.join();
104 vsi_t.join();
105 dpp_t.join();

152 const std::shared_ptr<OpenAPI::OAIIntersection_info> _intersection_info_ptr,
153 const std::shared_ptr<streets_vehicles::vehicle_list> _vehicle_list_ptr,

179
185 if (current_future_move_group_count != prev_future_move_group_count) {
186 SPDLOG_INFO("The number of future movement groups in the spat is updated from {0} to {1}.",
187 prev_future_move_group_count, current_future_move_group_count);
188 if (current_future_move_group_count <= _dpp_config.desired_future_move_group_count) {
189 streets_desired_phase_plan::streets_desired_phase_plan optimal_dpp =
190 _so_processing_worker_ptr->select_optimal_dpp(_intersection_info_ptr,

198 /* produce the optimal desired phase plan to kafka */
199 dpp_producer->send(msg_to_send);
200 }
201 }
202 }
203 else {
204 SPDLOG_INFO("The number of future movement groups in the spat did not change from the previous check.");
205 }
206 prev_future_move_group_count = current_future_move_group_count;

215 void signal_opt_service::populate_movement_groups(std::shared_ptr<streets_signal_optimization::movement_groups> _groups,
216 const std::shared_ptr<streets_tsc_configuration::tsc_configuration_state> tsc_config) const
217 {
218 if (tsc_config) {
219 for (const auto &phase_config : tsc_config->tsc_config_list) {
220 // If signal group has one or more concurrent phase
221 if ( !phase_config.concurrent_signal_groups.empty() ) {
222 // Loop through concurrent phases and create a movement group for each pair
223 for (const auto &concur : phase_config.concurrent_signal_groups) {
224 streets_signal_optimization::movement_group new_group;
225 new_group.signal_groups = {phase_config.signal_group_id, concur};
226 // Initialize a flag for if this pair already exists in the movement groups in reverse.
227 bool already_added = false;
228 // loop through movement groups and set flag to true if pair already exists in reverse order.
229 for ( const auto &add_group: _groups->groups ) {
230 if ( add_group.signal_groups.second == new_group.signal_groups.first &&
231 add_group.signal_groups.first == new_group.signal_groups.second) {
232 already_added = true;
233 }
234 }
235 // Only add a new pair if it does not already exist in movement groups in reverse.
236 if (!already_added) {
237 _groups->groups.push_back(new_group);
238 }
239 }
240 }
241 // If signal group does not have any concurrent phases add it as a movement group pair of between itself and 0 since
242 // 0 is an invalid signal group
243 else {
244 streets_signal_optimization::movement_group new_group;
245 // Add invalid signal group 0 as current signal group to indicate there is none.
246 new_group.signal_groups = { phase_config.signal_group_id, 0};
247 _groups->groups.push_back(new_group);
248 }
249 }
250 SPDLOG_DEBUG("Movement Groups: ");
251 for ( const auto &mg : _groups->groups) {
252 SPDLOG_DEBUG("Signal Group {0} and Signal Group {1}.", mg.signal_groups.first, mg.signal_groups.second );
253 }
254
255 }

259 bool signal_opt_service::update_intersection_info(unsigned long sleep_millisecs, unsigned long int_client_request_attempts) const
260 {
261 auto sleep_secs = static_cast<unsigned int>(sleep_millisecs / 1000);
262 SPDLOG_INFO("Send client request to update intersection inforamtion every {0} seconds for maximum {1} times.", sleep_secs, int_client_request_attempts);
263 int attempt_count = 0;
264 while (attempt_count < int_client_request_attempts)
265 {
266 // Send HTTP request, and update intersection information. If updated successfully, it returns true and exit the while loop.
267 if (_so_msgs_worker_ptr && _so_msgs_worker_ptr->request_intersection_info(intersection_info_ptr))
268 {
269 return true;
270 }
271 sleep(sleep_secs);
272 attempt_count++;
273 }
274 // If failed to update the intersection information after certain numbers of attempts
275 SPDLOG_ERROR("Updating Intersection information failed. ");
276 return false;
