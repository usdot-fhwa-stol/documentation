48 double PlatoonControlIHPWorker::getIHPTargetPositionFollower(double leaderCurrentPosition)
49 {
59 // 1. find the summation of "veh_len/veh_speed" for all predecessors
60 double tmp_time_hdw = current_predecessor_time_headway_sum_;
61
62 // 2. read host veh and front veh info
63 // Predecessor vehicle data.
64 double pred_spd = predecessor_speed_; // m/s
65 double pred_pos = predecessor_position_; // m
66
67 // host data.
68 double ego_spd = currentSpeed; // m/s
69 double ego_pos = leaderCurrentPosition - actual_gap_; // m
70
71 // platoon position index
72 int pos_idx = host_platoon_position_;
73
74 double desirePlatoonGap = ctrl_config_.intra_tau; //s
75
76 // IHP desired position calculation methods
77 double pos_g; // desired downtrack position calculated with time-gap, in m.
78 double pos_h; // desired downtrack position calculated with distance headaway, in m.
79
80 // 3. IHP gap regualtion
81 // intermediate variables
82 double timeGapAndStepRatio = desirePlatoonGap/ctrl_config_.time_step; // The ratio between desired platoon time gap and the current time_step.
83 double totalTimeGap = desirePlatoonGap*pos_idx; // The overall time gap from host vehicle to the platoon leader, in s.
84
85 // calcilate pos_gap and pos_headway
86 if ((pred_spd <= ego_spd) && (ego_spd <= ctrl_config_.ss_theta))
87 {
88 // pos_g
89 pos_g = (pred_pos - ctrl_config_.vehicleLength - ctrl_config_.standstill + ego_pos*timeGapAndStepRatio) / (1 + timeGapAndStepRatio);
90 // pos_h
91 double pos_h_nom = (pred_pos - ctrl_config_.standstill + ego_pos*(totalTimeGap + tmp_time_hdw)/ctrl_config_.time_step);
92 double pos_h_denom = (1 + ((totalTimeGap + tmp_time_hdw)/ctrl_config_.time_step));
93 pos_h = pos_h_nom/pos_h_denom;
94
95 }
96 else
97 {
98 // pos_g
99 pos_g = (pred_pos - ctrl_config_.vehicleLength + ego_pos*(timeGapAndStepRatio)) / (1 + timeGapAndStepRatio);
100 // pos_h
101 double pos_h_nom = (pred_pos + ego_pos*(totalTimeGap + tmp_time_hdw)/ctrl_config_.time_step);
102 double pos_h_denom = (1 + ((totalTimeGap + tmp_time_hdw)/ctrl_config_.time_step));
103 pos_h = pos_h_nom/pos_h_denom;
104 }
105
106 // desire speed and desire location
107 double pos_des = ctrl_config_.gap_weight*pos_g + (1.0 - ctrl_config_.gap_weight)*pos_h;
108 // double des_spd = (pos_des - ego_pos) / ctrl_config_.time_step;
109
110 // return IHP calculated desired location
111 return pos_des;
112 }

115 void PlatoonControlIHPWorker::generateSpeed(const cav_msgs::TrajectoryPlanPoint& point)
116 {
117 double speed_cmd = 0;
118
119 if (!last_cmd_set_)
120 {
121 // last speed command for smooth speed transition
122 lastCmdSpeed = currentSpeed;
123 last_cmd_set_ = true;
124 }
125
126 PlatoonLeaderInfo leader = platoon_leader;
127 if(leader.staticId != "" && leader.staticId != ctrl_config_.vehicleID)
128 {
129 double controllerOutput = 0.0;
130
131 // --------- Calculate desired gap ---------
132 double leaderCurrentPosition = leader.vehiclePosition;
133 ROS_DEBUG_STREAM("The current leader position is " << leaderCurrentPosition);
134
135 double desiredHostPosition = leaderCurrentPosition - desired_gap_;
136 ROS_DEBUG_STREAM("desiredHostPosition = " << desiredHostPosition);
137
138 // --------- conisder use IHP here instead to regualte gap ----------
139 // Call IHP gap regulation function to re-calculate desired Host position based on entire platoon's info
140 double desiredHostPosition_IHP = getIHPTargetPositionFollower(leaderCurrentPosition);
141 double hostVehiclePosition = leaderCurrentPosition - actual_gap_;
142 ROS_DEBUG_STREAM("hostVehiclePosition = " << hostVehiclePosition);
143
144 // UCLA: Replace desiredPosition with desiredPosition_IHP here.
145 controllerOutput = pid_ctrl_.calculate(desiredHostPosition_IHP, hostVehiclePosition);
146
147 double adjSpeedCmd = controllerOutput + leader.commandSpeed;
148 ROS_DEBUG_STREAM("Adjusted Speed Cmd = " << adjSpeedCmd << "; Controller Output = " << controllerOutput
149 << "; Leader CmdSpeed= " << leader.commandSpeed << "; Adjustment Cap " << ctrl_config_.adjustmentCap);
150 // After we get a adjSpeedCmd, we apply three filters on it if the filter is enabled
151 // First: we do not allow the difference between command speed of the host vehicle and the leader's commandSpeed higher than adjustmentCap
152
153 speed_cmd = adjSpeedCmd;
154 ROS_DEBUG_STREAM("A speed command is generated from command generator: " << speed_cmd << " m/s");
155
156 if(enableMaxAdjustmentFilter)
157 {
158 if(speed_cmd > leader.commandSpeed + ctrl_config_.adjustmentCap) {
159 speed_cmd = leader.commandSpeed + ctrl_config_.adjustmentCap;
160 } else if(speed_cmd < leader.commandSpeed - ctrl_config_.adjustmentCap) {
161 speed_cmd = leader.commandSpeed - ctrl_config_.adjustmentCap;
162 }
163 ROS_DEBUG_STREAM("The adjusted cmd speed after max adjustment cap is " << speed_cmd << " m/s");
164 }
165
166 }
167
168 else if (leader.staticId == ctrl_config_.vehicleID)
169 {
170 ROS_DEBUG_STREAM("Host vehicle is the leader");
171 speed_cmd = currentSpeed;
172
173 if(enableMaxAdjustmentFilter)
174 {
175 if(speed_cmd > ctrl_config_.adjustmentCap)
176 {
177 speed_cmd = ctrl_config_.adjustmentCap;
178 }
179
180 ROS_DEBUG_STREAM("The adjusted leader cmd speed after max adjustment cap is " << speed_cmd << " m/s");
181 }
182
183 pid_ctrl_.reset();
184 }
185
186 else
187 {
188 // If there is no leader available, the vehicle will stop. This means there is a mis-communication between platooning strategic and control plugins.
189 ROS_DEBUG_STREAM("There is no leader available");
190 speed_cmd = 0.0;
191 pid_ctrl_.reset();
192 }
193
194
195
196 // Third: we allow do not a large gap between two consecutive speed commands
197 if(enableMaxAccelFilter)
198 {
199 double max = lastCmdSpeed + (ctrl_config_.maxAccel * (ctrl_config_.cmdTmestamp / 1000.0));
200 double min = lastCmdSpeed - (ctrl_config_.maxAccel * (ctrl_config_.cmdTmestamp / 1000.0));
201 if(speed_cmd > max) {
202 speed_cmd = max;
203 } else if (speed_cmd < min) {
204 speed_cmd = min;
205 }
206 lastCmdSpeed = speed_cmd;
207 ROS_DEBUG_STREAM("The speed command after max accel cap is: " << speed_cmd << " m/s");
208 }
209
210 speedCmd_ = speed_cmd;
211
212 lastCmdSpeed = speedCmd_;
213
214 }
