240 double intersection_model::get_speed_limit_by_lanelet(const lanelet::ConstLanelet &subj_lanelet) const
241 {
242 double speed_limit_result = 0;
243 std::regex r("[0-9]+");
244 lanelet::RegulatoryElementConstPtrs reg_elements = subj_lanelet.regulatoryElements();
245
246 /***
247 * Identify the speed limit regulatory element by using an attribute with a key value that equals to 'limit'.
248 * The value of this attribute stores the speed limit value in unit of measure MPH
249 **/
250 for (auto itr = reg_elements.begin(); itr != reg_elements.end(); itr++)
251 {
252 lanelet::AttributeMap attr_map = itr->get()->attributes();
253 for (auto inner_itr = attr_map.begin(); inner_itr != attr_map.end(); inner_itr++)
254 {
255 if (inner_itr->first == lanelet::DigitalSpeedLimit::Limit)
256 {
257 std::string speed_str = inner_itr->second.value();
258 if (std::regex_search(speed_str, r))
259 {
260 speed_limit_result = std::stod(std::regex_replace(speed_str, std::regex("\\s"), std::string("")));
261 speed_limit_result *= MPH_TO_MS; //MPH to M/S conversion
262 };
263 }
264 }
265 }
266 return speed_limit_result;

274 const std::vector<lanelet_info_t> intersection_model::get_conflict_lanelets_info(int64_t sub_link_lanelet_id)
275 {
276 //Check the subject link_lanelet conflict by comparing it to all link lanelets at the intersection
277 lanelet::Id subject_link_lanelet_id{sub_link_lanelet_id};
278 lanelet::Points2d intersectionPts;
279 std::vector<lanelet_info_t> conflict_lanelets;
280
281 if ( !is_valid_link_lanelet_id(sub_link_lanelet_id) )
282 {
283 SPDLOG_ERROR("Not a valid input link lanelet id {0}.", sub_link_lanelet_id);
284 return conflict_lanelets;
285 }
286
287 //Get the vector of link lanelets at the intersection that share the same entry lanelet.
288 //The vector of link lanelets can be skipped.
289 std::vector<int> lanelet_ids_to_skip = get_shared_entry_link_lanelet_ids(sub_link_lanelet_id);
290
291 //get lanelets from map
292 lanelet::LaneletLayer &lanelets = this->map->laneletLayer;
293 try
294 {
295 //comparing centerline between subject link lanelet and all other link lanelets at the intersection
296 lanelet::ConstLineString2d subject_centerline_b = lanelets.get(subject_link_lanelet_id).centerline2d();
297 for (const auto& link_lanelet : this->int_info.link_lanelets_info)
298 {
299 if (std::find(lanelet_ids_to_skip.begin(), lanelet_ids_to_skip.end(), link_lanelet.id) != lanelet_ids_to_skip.end())
300 {
301 continue;
302 }
303 lanelet::ConstLineString2d centerline_link_lanelet_b = lanelets.get(link_lanelet.id).centerline2d();
304 lanelet::ConstHybridLineString2d c_sub(centerline_link_lanelet_b);
305 lanelet::ConstHybridLineString2d c(subject_centerline_b);
306
307 boost::geometry::intersection(c, c_sub, intersectionPts);
308 if (intersectionPts.size() > 0)
309 {
310 conflict_lanelets.push_back(link_lanelet);
311 }
312 intersectionPts.clear();
313 }
314 }
315 catch (const lanelet::LaneletError &ex)
316 {
317 SPDLOG_ERROR("Failed to lookup element with id {0} : {1} " , ex.what() ,sub_link_lanelet_id);
318 conflict_lanelets.clear();
319 }
320 return conflict_lanelets;

463 {

466 std::vector<lanelet::Id> fel_ids;
467 std::vector<lanelet::Id> pdl_ids;
468 std::vector<lanelet::Id> link_lanelet_id_intersect_v;
469 for(const auto& temp_lanelet: following_enter_lanelets)
470 {
471 fel_ids.push_back(temp_lanelet.id());
472 }
473 std::sort(fel_ids.begin(), fel_ids.end());
474
475 for(const auto& temp_lanelet: prev_depart_lanelets)
476 {
477 pdl_ids.push_back(temp_lanelet.id());
478 }
479 std::sort(pdl_ids.begin(), pdl_ids.end());

483 SPDLOG_DEBUG("Found matching link lanelet id = {0} for enter lanelet id = {1} and departure lanelet id = {2}", link_lanelet_id_intersect_v.front(),enter_lanelet_id, depart_lanelet_id);

490 void intersection_model::mapping_lane_id_2_lanelet_id(const map_referencepoint& ref_point,
491 const map_lane& lane,
492 const long lane_width,
493 const std::vector<lanelet::ConstLanelet>& subj_lanelets,
494 std::unordered_map<long ,
495 lanelet::ConstLanelet>& lane2lanelet_m,
496 const double lane_to_lanelet_corr_thres) const
497 {
498 std::vector<lanelet::BasicPoint3d> basic_points = convert_lane_path_2_basic_points(ref_point, lane);
499 std::unordered_map<lanelet::Id, double> lanelet2lane_path_distance_m;
500 for(const auto& subj_l: subj_lanelets)
501 {
502 double avg_distance = compute_points_2_lanelet_avg_distance(basic_points, subj_l);
503 SPDLOG_DEBUG("Lanelet id {0} to lane id = {1} path points average distance {2}", subj_l.id(), lane.lane_id, avg_distance);
504 lanelet2lane_path_distance_m.insert({subj_l.id(), avg_distance});
505 }
506 //Find the nearest lanelet from the lane path by shortest average distance
507 const auto& min_distance_pair = std::min_element(lanelet2lane_path_distance_m.begin(), lanelet2lane_path_distance_m.end(), [](const auto& l, const auto& r){return l.second < r.second; });
508
509 double max_lane_width = static_cast<double>(lane_width) * CM_TO_M;
510 //Check whether the average distance between points and lanelet is larger than the lane with (defined from MAP message) itself.
511 if(min_distance_pair->second < max_lane_width + lane_to_lanelet_corr_thres )
512 {
513 for(const auto& subj_l: subj_lanelets)
514 {
515 if(subj_l.id() == min_distance_pair->first)
516 {
517 lane2lanelet_m.insert({lane.lane_id, subj_l});
518 SPDLOG_DEBUG("min_distance_pair lanelet id = {0}, distance = {1}", min_distance_pair->first, min_distance_pair->second);
519 }
520 }
521 }else{
522 SPDLOG_CRITICAL("Mapping failure: min_distance_pair distance = {0} is greater than lane width = {1} defined in the MAP message.", min_distance_pair->second, max_lane_width);
523 }

526 std::vector<lanelet::BasicPoint3d> intersection_model::convert_lane_path_2_basic_points(const map_referencepoint& ref_point, const map_lane& lane) const
527 {
528 std::vector<lanelet::BasicPoint3d> basic_point_v;
529 double lat = static_cast<double>(ref_point.latitude) * TENTH_TO_ONE * MICRO_DEGREE_TO_DEGREE;
530 double lon = static_cast<double>(ref_point.longitude) * TENTH_TO_ONE * MICRO_DEGREE_TO_DEGREE;
531 double elev = static_cast<double>(ref_point.elevation) * DM_TO_M;
532 //ref_point3d is a the reference point from which subsequent data points are offset
533 const auto& ref_point3d = gps_2_map_point(lat, lon, elev);
534 double cur_x = ref_point3d.x();
535 double cur_y = ref_point3d.y();
536 double cur_z = ref_point3d.z();
537 //The lane nodes are offsets from the prior node
538 for(const auto& node : lane.nodes)
539 {
540 cur_x += static_cast<double>(node.x) * CM_TO_M;
541 cur_y += static_cast<double>(node.y) * CM_TO_M;
542
