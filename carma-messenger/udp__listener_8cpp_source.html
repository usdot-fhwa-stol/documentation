39 void UDPListener::handleReceivedUdp(const boost::system::error_code& error_code, std::size_t bytes_transferred)
40 {
41 if(!error_code)
42 {
43 // Get the number of bytes available for reading
44 size_t buffer_size = socket_.available();
45
46 // Create buffer based on number of bytes available for reading
47 std::shared_ptr<std::vector<uint8_t>> buffer = std::make_shared<std::vector<uint8_t>>(buffer_size);
48
49 boost::system::error_code receive_error_code;
50
51 // Receive data on UDP socket; data is stored in 'buffer'
52 socket_.receive(boost::asio::buffer(*buffer), 0, receive_error_code);
53
54 if(!receive_error_code)
55 {
56 // Trigger the onReceive() function with 'buffer' (containing the received data) as an argument
57 io_.post([this, buffer](){onReceive(buffer);});
58 }
59
60 startReceiveUdp();
61 }
62 else
63 {
64 RCLCPP_WARN_STREAM(rclcpp::get_logger(logger_name_), "Error occurred while reading data from UDP socket!");
65 }
