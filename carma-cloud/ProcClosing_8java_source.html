81 public static void renderTiledData(ArrayList<TrafCtrl> oCtrls, ArrayList<int[]> oTiles) throws IOException
82 {
83 String[] sEmpty = new String[0];
84 TdLayer oLayer = new TdLayer("closing-poly", sEmpty, sEmpty, TdLayer.POLYGON);
85 TdLayer oLayerOutline = new TdLayer("closing-outline", sEmpty, sEmpty, TdLayer.LINESTRING);
86 int[] nTags = new int[0];
87 ArrayList<double[]> dTapers = new ArrayList();
88 for (TrafCtrl oCtrl : oCtrls)
89 {
90 int nPoints = (int)(oCtrl.m_oFullGeo.m_dLength / 0.2) + 1;
91 double dPercent = 1.0 / nPoints;
92 int nCount = 0;
93 double[] dC = oCtrl.m_oFullGeo.m_dC;
94 int nCtrlVal = MathUtil.bytesToInt(oCtrl.m_yControlValue);
95 boolean bRight = TrafCtrlEnums.getCtrlVal("closing", "right") == nCtrlVal;
96 double[] dStraight = bRight ? oCtrl.m_oFullGeo.m_dNT : oCtrl.m_oFullGeo.m_dPT;
97 int nLimit = Arrays.size(dC) - 2;
98 double dDist = 0.0;
99 double[] dTaper = Arrays.newDoubleArray(nPoints * 2);
100 dTaper = Arrays.add(dTaper, dStraight[1], dStraight[2]);
101 for (int nIndex = 1; nIndex < nLimit; nIndex += 2)
102 {
103 double dX1 = dC[nIndex];
104 double dY1 = dC[nIndex + 1];
105 double dX2 = dC[nIndex + 2];
106 double dY2 = dC[nIndex + 3];
107 dDist += Geo.distance(dX1, dY1, dX2, dY2);
108 if (dDist >= 0.2)
109 {
110 dDist = 0.0;
111 double dX3 = dStraight[nIndex];
112 double dY3 = dStraight[nIndex + 1];
113 double dHeading = Geo.heading(dX3, dY3, dX1, dY1);
114 double dOffset = Geo.distance(dX3, dY3, dX1, dY1) * 2 * dPercent * ++nCount;
115 dTaper = Arrays.add(dTaper, dX3 + Math.cos(dHeading) * dOffset, dY3 + Math.sin(dHeading) * dOffset);
116 }
117 }
118 if (bRight)
119 {
120 dTapers.add(dTaper);
121 dTapers.add(dStraight);
122 }
123 else
124 {
125 dTapers.add(dStraight);
126 dTapers.add(dTaper);
127 }
128 }
129 for (int[] nTile : oTiles)
130 {
131 int nX = nTile[0];
132 int nY = nTile[1];
133 Files.createDirectories(Paths.get(String.format(g_sTdFileFormat, nX, 0, 0, 0)).getParent(), FileUtil.DIRPERS);
134 writeIndexFile(oCtrls, nX, nY);
135 oLayer.clear();
136 oLayerOutline.clear();
137 int nTaperCnt = 0;
138 for (TrafCtrl oCtrl : oCtrls)
139 {
140 if (Collections.binarySearch(oCtrl.m_oFullGeo.m_oTiles, nTile, Mercator.TILECOMP) < 0)
141 {
142 nTaperCnt += 2;
143 continue;
144 }
145 double[] dPoly = Geo.createPolygon(dTapers.get(nTaperCnt++), dTapers.get(nTaperCnt++));
146
147 oLayer.add(new TdFeature(dPoly, nTags, oCtrl));
148 dPoly = Arrays.add(dPoly, dPoly[1], dPoly[2]);
149 oLayerOutline.add(new TdFeature(dPoly, nTags, oCtrl));
150 }
151
152 if (!oLayer.isEmpty())
153 {
154 try (DataOutputStream oOut = new DataOutputStream(new BufferedOutputStream(FileUtil.newOutputStream(Paths.get(String.format(g_sTdFileFormat, nX, g_nDefaultZoom, nX, nY)), FileUtil.APPENDTO, FileUtil.FILEPERS))))
155 {
156 oLayer.write(oOut);
157 oLayerOutline.write(oOut);
158 }
159 }
160 }
161 }
