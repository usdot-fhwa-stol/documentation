6 bool intersection_client::update_intersection_info(const int sleep_millisecs, const int int_client_request_attempts)
7 {
8 auto sleep_secs = static_cast<unsigned int>(sleep_millisecs);
9 int attempt_count = 0;
10 while (attempt_count < int_client_request_attempts)
11 {
12 // Send HTTP request, and update intersection information. If updated successfully, it returns true and exit the while loop.
13 if (request_intersection_info())
14 {
15 SPDLOG_INFO("Intersection information is updated successfuly! ");
16 return true;
17 }
18 // sleep takes seconds
19 std::this_thread::sleep_for(std::chrono::milliseconds(sleep_secs));
20 attempt_count++;
21 }
