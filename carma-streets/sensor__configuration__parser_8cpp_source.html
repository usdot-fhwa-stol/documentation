18 lanelet::BasicPoint3d parse_sensor_location( const std::string &filepath , const std::string &sensor_id ){
19 // Parse JSON configuration file
20 auto doc = streets_utils::json_utils::parse_json_file(filepath);
21 if (!doc.IsArray()) {
22 throw streets_utils::json_utils::json_parse_exception("Invadid format for sensor configuration file " + filepath
23 + ". Sensor configuration file should contain an array of json sensor configurations!");
24 }
25 bool found = false;
26 lanelet::BasicPoint3d sensor_location;
27 for (auto itr = doc.Begin(); itr != doc.End(); ++itr) {
28 // Access the data in the object
29 auto sensor_config_id = streets_utils::json_utils::parse_string_member("sensorId", itr->GetObject(), true ).value();
30 if ( sensor_config_id == sensor_id ) {
31 found = true;
32 auto location = streets_utils::json_utils::parse_object_member("location", itr->GetObject(), true ).value();
33 sensor_location = lanelet::BasicPoint3d{
34 streets_utils::json_utils::parse_double_member("x", location, true ).value(),
35 streets_utils::json_utils::parse_double_member("y", location, true ).value(),
36 streets_utils::json_utils::parse_double_member("z", location, true ).value()
37 };
38 }
39 }
40 if (!found) {
41 throw streets_utils::json_utils::json_parse_exception("Did not find sensor with id " + sensor_id + " in sensor configuration file " + filepath + "!");
42 }
43 return sensor_location;
44 }
