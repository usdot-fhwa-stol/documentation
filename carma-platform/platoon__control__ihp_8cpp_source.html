109 bool PlatoonControlIHPPlugin::controlTimerCb()
110 {
111 ROS_DEBUG_STREAM("In control timer callback ");
112 // If it has been a long time since input data has arrived then reset the input counter and return
113 // Note: this quiets the controller after its input stream stops, which is necessary to allow
114 // the replacement controller to publish on the same output topic after this one is done.
115 long current_time = ros::Time::now().toNSec() / 1000000;
116 ROS_DEBUG_STREAM("current_time = " << current_time << ", prev_input_time_ = " << prev_input_time_ << ", input counter = " << consecutive_input_counter_);
117 if (current_time - prev_input_time_ > config_.shutdownTimeout)
118 {
119 ROS_DEBUG_STREAM("returning due to timeout.");
120 consecutive_input_counter_ = 0;
121 return false;
122 }
123
124 // If there have not been enough consecutive timely inputs then return (waiting for
125 // previous control plugin to time out and stop publishing, since it uses same output topic)
126 if (consecutive_input_counter_ <= config_.ignoreInitialInputs)
127 {
128 ROS_DEBUG_STREAM("returning due to first data input");
129 return false;
130 }
131
132 cav_msgs::TrajectoryPlanPoint second_trajectory_point = latest_trajectory_.trajectory_points[1];
133 cav_msgs::TrajectoryPlanPoint lookahead_point = getLookaheadTrajectoryPoint(latest_trajectory_);
134
135 trajectory_speed_ = getTrajectorySpeed(latest_trajectory_.trajectory_points);
136
137 generateControlSignals(second_trajectory_point, lookahead_point);
138
139 return true;
140 }

142 void PlatoonControlIHPPlugin::trajectoryPlan_cb(const cav_msgs::TrajectoryPlan::ConstPtr& tp)
143 {
144 if (tp->trajectory_points.size() < 2) {
145 ROS_WARN_STREAM("PlatoonControlIHPPlugin cannot execute trajectory as only 1 point was provided");
146 return;
147 }
148
149 latest_trajectory_ = *tp;
150 prev_input_time_ = ros::Time::now().toNSec() / 1000000;
151 ++consecutive_input_counter_;
152 ROS_DEBUG_STREAM("New trajectory plan #" << consecutive_input_counter_ << " at time " << prev_input_time_);
153 ROS_DEBUG_STREAM("tp header time = " << tp->header.stamp.toNSec() / 1000000);
154 }

156 cav_msgs::TrajectoryPlanPoint PlatoonControlIHPPlugin::getLookaheadTrajectoryPoint(cav_msgs::TrajectoryPlan trajectory_plan)
157 {
158 cav_msgs::TrajectoryPlanPoint lookahead_point;
159
160 double lookahead_dist = config_.lookaheadRatio * current_speed_;
161 ROS_DEBUG_STREAM("lookahead based on speed: " << lookahead_dist);
162
163 lookahead_dist = std::max(config_.minLookaheadDist, lookahead_dist);
164 ROS_DEBUG_STREAM("final lookahead: " << lookahead_dist);
165
166 double traveled_dist = 0.0;
167 bool found_point = false;
168
169 for (size_t i = 1; i<trajectory_plan.trajectory_points.size() - 1; i++)
170 {
171 double dx = pose_msg_.pose.position.x - trajectory_plan.trajectory_points[i].x;
172 double dy = pose_msg_.pose.position.y - trajectory_plan.trajectory_points[i].y;
173
174 double dx1 = trajectory_plan.trajectory_points[i].x - trajectory_plan.trajectory_points[i-1].x;
175 double dy1 = trajectory_plan.trajectory_points[i].y - trajectory_plan.trajectory_points[i-1].y;
176 double dist1 = std::sqrt(dx1*dx1 + dy1*dy1);
177 ROS_DEBUG_STREAM("trajectory spacing: " << dist1);
178
179 double dist = std::sqrt(dx*dx + dy*dy);
180
181 traveled_dist = dist;
182
183 if ((lookahead_dist - traveled_dist) < 1.0)
184 {
185 lookahead_point = trajectory_plan.trajectory_points[i];
186 found_point = true;
187 ROS_DEBUG_STREAM("found lookahead point at index: " << i);
188 break;
189 }
190 }
191
192 if (!found_point)
193 {
194 lookahead_point = trajectory_plan.trajectory_points.back();
195 ROS_DEBUG_STREAM("lookahead point set as the last trajectory point");
196 }
197
198
199 return lookahead_point;
200 }

288 // extract maximum speed of trajectory
289 double PlatoonControlIHPPlugin::getTrajectorySpeed(std::vector<cav_msgs::TrajectoryPlanPoint> trajectory_points)
290 {
291 double trajectory_speed = 0;
292
293 double dx1 = trajectory_points[trajectory_points.size()-1].x - trajectory_points[0].x;
294 double dy1 = trajectory_points[trajectory_points.size()-1].y - trajectory_points[0].y;
295 double d1 = sqrt(dx1*dx1 + dy1*dy1);
296 double t1 = (trajectory_points[trajectory_points.size()-1].target_time.toSec() - trajectory_points[0].target_time.toSec());
297
298 double avg_speed = d1/t1;
299 ROS_DEBUG_STREAM("trajectory_points size = " << trajectory_points.size() << ", d1 = " << d1 << ", t1 = " << t1 << ", avg_speed = " << avg_speed);
300
301 for(size_t i = 0; i < trajectory_points.size() - 2; i++ )
302 {
303 double dx = trajectory_points[i + 1].x - trajectory_points[i].x;
304 double dy = trajectory_points[i + 1].y - trajectory_points[i].y;
305 double d = sqrt(dx*dx + dy*dy);
306 double t = (trajectory_points[i + 1].target_time.toSec() - trajectory_points[i].target_time.toSec());
307 double v = d/t;
308 if(v > trajectory_speed)
309 {
310 trajectory_speed = v;
311 }
312 }
313
314 ROS_DEBUG_STREAM("trajectory speed: " << trajectory_speed);
315 ROS_DEBUG_STREAM("avg trajectory speed: " << avg_speed);
316
317 return avg_speed; //TODO: why are 2 speeds being calculated? Which should be returned?
318
319 }
