46 carma_ros2_utils::CallbackReturn TrajectoryFollowerWrapperNode::on_configure_plugin()
47 {
48 // Reset config
49 config_ = TrajectoryFollowerWrapperConfig();
50
51 // Load parameters
52 get_parameter<double>("vehicle_response_lag", config_.vehicle_response_lag);
53 get_parameter<double>("incoming_cmd_time_threshold", config_.incoming_cmd_time_threshold);
54
55 RCLCPP_INFO_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "Loaded Params: " << config_);
56 // Register runtime parameter update callback
57 add_on_set_parameters_callback(std::bind(&TrajectoryFollowerWrapperNode::parameter_update_callback, this, std_ph::_1));
58
59 // NOTE: Currently, intra-process comms must be disabled for the following subscriber that is transient_local: https://github.com/ros2/rclcpp/issues/1753
60
61 rclcpp::SubscriptionOptions intra_proc_disabled;
62 intra_proc_disabled.use_intra_process_comm = rclcpp::IntraProcessSetting::Disable;
63
64 auto sub_qos_transient_local = rclcpp::QoS(rclcpp::KeepAll()); // A subscriber with this QoS will store all messages that it has sent on the topic
65 sub_qos_transient_local.transient_local();
66 // Setup subscriber
67 control_cmd_sub_ = create_subscription<autoware_auto_msgs::msg::AckermannControlCommand>("output/control_cmd", sub_qos_transient_local,
68 std::bind(&TrajectoryFollowerWrapperNode::ackermann_control_cb, this, std::placeholders::_1),intra_proc_disabled);
69
70 // Setup publishers
71 autoware_traj_pub_ = create_publisher<autoware_auto_msgs::msg::Trajectory>("input/reference_trajectory", 10);
72 autoware_state_pub_ = create_publisher<autoware_auto_msgs::msg::VehicleKinematicState>("input/current_kinematic_state", 10);
73
74 // Setup timers to publish autoware compatible info (trajectory and state)
75 autoware_info_timer_ = create_timer(
76 get_clock(),
77 std::chrono::milliseconds(33), // Spin at 30 Hz per plugin API
78 std::bind(&TrajectoryFollowerWrapperNode::autoware_info_timer_callback, this));
79
80 // Return success if everthing initialized successfully
81 return CallbackReturn::SUCCESS;
82 }

85 void TrajectoryFollowerWrapperNode::autoware_info_timer_callback()
86 {
87 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "In autoware info timer callback");
88
89 if (current_trajectory_ && current_pose_ && current_twist_)
90 {
91 // generate and publish autoware kinematic state
92 auto autoware_state = convert_state(current_pose_.get(), current_twist_.get());
93 autoware_state_pub_->publish(autoware_state);
94
95 // generate and publish autoware trajectory
96 current_trajectory_.get().header.frame_id = autoware_state.header.frame_id;
97 auto autoware_traj_plan = basic_autonomy::waypoint_generation::process_trajectory_plan(current_trajectory_.get(), config_.vehicle_response_lag);
98 autoware_traj_pub_->publish(autoware_traj_plan);
99
100 }
101 }

110 autoware_auto_msgs::msg::VehicleKinematicState TrajectoryFollowerWrapperNode::convert_state(const geometry_msgs::msg::PoseStamped& pose, const geometry_msgs::msg::TwistStamped& twist) const
111 {
112 autoware_auto_msgs::msg::VehicleKinematicState state;
113 state.header = pose.header;
114 state.state.x = pose.pose.position.x;
115 state.state.y = pose.pose.position.y;
116 state.state.z = pose.pose.position.z;
117 state.state.heading.real = pose.pose.orientation.w;
118 state.state.heading.imag = pose.pose.orientation.z;
119
120 state.state.longitudinal_velocity_mps = twist.twist.linear.x;
121 return state;
122 }

124 autoware_msgs::msg::ControlCommandStamped TrajectoryFollowerWrapperNode::convert_cmd(const autoware_auto_msgs::msg::AckermannControlCommand& cmd) const
125 {
126 autoware_msgs::msg::ControlCommandStamped return_cmd;
127 return_cmd.header.stamp = cmd.stamp;
128
129 return_cmd.cmd.linear_acceleration = cmd.longitudinal.acceleration;
130 return_cmd.cmd.linear_velocity = cmd.longitudinal.speed;
131 return_cmd.cmd.steering_angle = cmd.lateral.steering_tire_angle;
132
133 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "generated command cmd.stamp: " << std::to_string(rclcpp::Time(cmd.stamp).seconds()));
134 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "generated command cmd.longitudinal.acceleration: " << cmd.longitudinal.acceleration);
135 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "generated command cmd.longitudinal.speed: " << cmd.longitudinal.speed);
136 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "generated command cmd.lateral.steering_tire_angle: " << cmd.lateral.steering_tire_angle);
137 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "generated command cmd.lateral.steering_tire_rotation_rate: " << cmd.lateral.steering_tire_rotation_rate);
138
139 return return_cmd;
140 }

142 autoware_msgs::msg::ControlCommandStamped TrajectoryFollowerWrapperNode::generate_command()
143 {
144 // process and save the trajectory
145 autoware_msgs::msg::ControlCommandStamped converted_cmd;
146
147 if (!current_trajectory_ || !current_pose_ || !current_twist_ || !received_ctrl_command_.has_value())
148 {
149 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "Insufficient data, empty control command generated");
150 return converted_cmd;
151 }
152
153
154 if (isControlCommandOld(received_ctrl_command_.value()))
155 {
156 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("trajectory_follower_wrapper"), "Control Command is old, empty control command generated");
157 return converted_cmd;
158 }
159
160 converted_cmd = convert_cmd(received_ctrl_command_.value());
161
162 return converted_cmd;
163 }

165 bool TrajectoryFollowerWrapperNode::isControlCommandOld(const autoware_auto_msgs::msg::AckermannControlCommand& cmd) const
166 {
167 double difference = std::abs(this->now().seconds() - rclcpp::Time(cmd.stamp).seconds());
168
169 if (difference >= config_.incoming_cmd_time_threshold)
170 {
171 return true;
172 }
173
174 return false;
175 }
