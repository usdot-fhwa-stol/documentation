36carma_ros2_utils::CallbackReturn TrafficIncidentNode::handle_on_configure(
37 const rclcpp_lifecycle::State &)
38{
39 get_parameter<std::string>("sender_id", sender_id_);
40 get_parameter<std::string>("event_reason", event_reason_);
41 get_parameter<std::string>("event_type", event_type_);
42 get_parameter<double>("down_track", down_track_);
43 get_parameter<double>("up_track", up_track_);
44 get_parameter<double>("min_gap", min_gap_);
45
46 traffic_worker_.setSenderId(sender_id_);
47 traffic_worker_.setDownTrack(down_track_);
48 traffic_worker_.setUpTrack(up_track_);
49 traffic_worker_.setMinGap(min_gap_);
50 traffic_worker_.setEventReason(event_reason_);
51 traffic_worker_.setEventType(event_type_);
52
53 // Setup pub/sub
54 pinpoint_driver_sub_ = create_subscription<gps_msgs::msg::GPSFix>(
55 "gps_common_fix",
56 10,
57 std::bind(&TrafficIncidentWorker::pinpointDriverCallback, &traffic_worker_, std_ph::_1));
58 traffic_mobility_operation_pub_ =
59 create_publisher<carma_v2x_msgs::msg::MobilityOperation>("outgoing_mobility_operation", 10);
60
61 // setup services
62 start_broadcast_request_service_server = create_service<carma_msgs::srv::SetTrafficEvent>(
63 "start_broadcasting_traffic_event",
64 std::bind(
65 &TrafficIncidentNode::startTrafficBroadcastCallback,
66 this,
67 std_ph::_1,
68 std_ph::_2,
69 std_ph::_3));
70 stop_broadcast_request_service_server = create_service<std_srvs::srv::Trigger>(
71 "stop_broadcasting_traffic_event",
72 std::bind(
73 &TrafficIncidentNode::stopTrafficBroadcastCallback,
74 this,
75 std_ph::_1,
76 std_ph::_2,
77 std_ph::_3));
78
79 return CallbackReturn::SUCCESS;
80}

101/*****
102 * Used by UI to start broadcasting traffic event (geofence)
103 * Msg: MobilityOperation
104 * **/
105bool TrafficIncidentNode::startTrafficBroadcastCallback(
106 const std::shared_ptr<rmw_request_id_t>,
107 const carma_msgs::srv::SetTrafficEvent::Request::SharedPtr req,
108 carma_msgs::srv::SetTrafficEvent::Response::SharedPtr resp)
109{

121/*****

125bool TrafficIncidentNode::stopTrafficBroadcastCallback(

127 const std_srvs::srv::Trigger::Request::SharedPtr req,

129{

134 traffic_worker_.setUpTrack(0);
135 traffic_worker_.setAdvisorySpeed(0);
136
137 resp->success = true;
138 resp->message = "stop broadcasting";
139
140 // return service response true
141 return true;
142
143 } catch (...) {
144 // in case any exception
145 resp->success = false;
146 return false;
147 }

150void TrafficIncidentNode::spin_callback(void)
151{
152 if (
153 traffic_worker_.getDownTrack() > 0 && traffic_worker_.getUpTrack() > 0 &&
154 traffic_worker_.getMinGap() > 0 && traffic_worker_.getAdvisorySpeed() > 0) {
155 // construct local mobilityOperation msg
156 carma_v2x_msgs::msg::MobilityOperation traffic_mobility_msg =
157 traffic_worker_.mobilityMessageGenerator(traffic_worker_.getPinPoint());
158
159 // start constantly broadcasting mobilityOperation msg
160 traffic_mobility_operation_pub_->publish(traffic_mobility_msg);
161 }
162}
