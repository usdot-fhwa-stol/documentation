57void MapConvertor::convertNodeOffsetPointXY(const j2735_v2x_msgs::msg::NodeOffsetPointXY& in_msg,
58 carma_v2x_msgs::msg::NodeOffsetPointXY& out_msg)
59{
60 out_msg.choice = in_msg.choice;
61 // Convert XY or Lat/Lon points based on choice field
62 switch (in_msg.choice)
63 {
64 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY1:
65 out_msg.x = (double)in_msg.node_xy1.x / units::CM_PER_M;
66 out_msg.y = (double)in_msg.node_xy1.y / units::CM_PER_M;
67 break;
68 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY2:
69 out_msg.x = (double)in_msg.node_xy2.x / units::CM_PER_M;
70 out_msg.y = (double)in_msg.node_xy2.y / units::CM_PER_M;
71 break;
72 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY3:
73 out_msg.x = (double)in_msg.node_xy3.x / units::CM_PER_M;
74 out_msg.y = (double)in_msg.node_xy3.y / units::CM_PER_M;
75 break;
76 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY4:
77 out_msg.x = (double)in_msg.node_xy4.x / units::CM_PER_M;
78 out_msg.y = (double)in_msg.node_xy4.y / units::CM_PER_M;
79 break;
80 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY5:
81 out_msg.x = (double)in_msg.node_xy5.x / units::CM_PER_M;
82 out_msg.y = (double)in_msg.node_xy5.y / units::CM_PER_M;
83 break;
84 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_XY6:
85 out_msg.x = (double)in_msg.node_xy6.x / units::CM_PER_M;
86 out_msg.y = (double)in_msg.node_xy6.y / units::CM_PER_M;
87 break;
88 case j2735_v2x_msgs::msg::NodeOffsetPointXY::NODE_LATLON:
89 out_msg.latitude = (double)in_msg.node_latlon.latitude / units::TENTH_MICRO_DEG_PER_DEG;
90 out_msg.longitude = (double)in_msg.node_latlon.longitude / units::TENTH_MICRO_DEG_PER_DEG;
91 break;
92 }
93 // Done Conversion

96void MapConvertor::convertLaneDataAttribute(const j2735_v2x_msgs::msg::LaneDataAttribute& in_msg,
97 carma_v2x_msgs::msg::LaneDataAttribute& out_msg)
98{
99 out_msg.choice = in_msg.choice;
100 // Convert Angles
101 // Do comparison to avoid duplicate math
102 switch (in_msg.choice)
103 {
104 case j2735_v2x_msgs::msg::LaneDataAttribute::PATH_END_POINT_ANGLE:
105 out_msg.path_end_point_angle = in_msg.path_end_point_angle;
106 break;
107 case j2735_v2x_msgs::msg::LaneDataAttribute::LANE_CROWN_POINT_CENTER:
108 out_msg.lane_crown_point_center = in_msg.lane_crown_point_center * units::THREE_TENTHS_DEG;
109 break;
110 case j2735_v2x_msgs::msg::LaneDataAttribute::LANE_CROWN_POINT_LEFT:
111 out_msg.lane_crown_point_left = in_msg.lane_crown_point_left * units::THREE_TENTHS_DEG;
112 break;
113 case j2735_v2x_msgs::msg::LaneDataAttribute::LANE_CROWN_POINT_RIGHT:
114 out_msg.lane_crown_point_right = in_msg.lane_crown_point_right * units::THREE_TENTHS_DEG;
115 break;
116 case j2735_v2x_msgs::msg::LaneDataAttribute::LANE_ANGLE:
117 out_msg.lane_angle = in_msg.lane_angle * units::ONE_AND_A_HALF_DEG;
118 break;
119 case j2735_v2x_msgs::msg::LaneDataAttribute::SPEED_LIMITS:
120 // Convert SpeedLimitsList
121 for (j2735_v2x_msgs::msg::RegulatorySpeedLimit limit : in_msg.speed_limits.speed_limits)
122 {
123 carma_v2x_msgs::msg::RegulatorySpeedLimit cav_limit;
124 convertRegulatorySpeedLimit(limit, cav_limit);
125 out_msg.speed_limits.push_back(cav_limit);
126 }
127 break;
128 }
129 // Done Conversion
