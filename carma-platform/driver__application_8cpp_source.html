46int DriverApplication::run()
47{
48 ros::init(argc_,argv_,name_);
49 //Initialize nodehandles
50 nh_.reset(new ros::NodeHandle());
51 pnh_.reset(new ros::NodeHandle("~"));
52
53 //Setup initial status
54 status_.name = ros::this_node::getName();
55 status_.status = cav_msgs::DriverStatus::OFF;
56
57 //Initialize DriverApplication services
58 ros::ServiceServer bind_service = pnh_->advertiseService("bind", &DriverApplication::bind_service_cb,this);
59 ros::ServiceServer api_service = pnh_->advertiseService("get_driver_api",
60 &DriverApplication::get_driver_api_cb, this);
61 ros::ServiceServer driver_status_svs = pnh_->advertiseService("get_status", &DriverApplication::get_status_cb,this);
62
63 driver_status_pub_ = nh_->advertise<cav_msgs::DriverStatus>("driver_discovery",1);
64 system_alert_sub_ = nh_->subscribe("/system_alert",10,&DriverApplication::system_alert_cb, this);
65 ros::Timer timer = pnh_->createTimer(ros::Duration(1), &DriverApplication::status_publish_timer,this);
66 ROS_INFO("Driver services initialized");
67
68 initialize();
69
70 ros::Rate r(spin_rate);
71 while(ros::ok() && !shutdown_)
72 {
73 pre_spin();
74 ros::spinOnce();
75 post_spin();
76 r.sleep();
77 }
78
79 ROS_INFO_STREAM("Driver Shutting Down");
80 shutdown();
81 ros::shutdown();
82}

97void DriverApplication::setStatus(cav_msgs::DriverStatus status)
98{
99 //If the driver status has been changed we need to release bonds
100 //notifying connected nodes there has been a change in the status of this node
101 if(!DriverStatusEquals(status,status_))
102 {
103 std::unique_lock<std::mutex> lock(bond_mutex_);
104 ROS_INFO_STREAM("Status updated releasing bonds, Status: " << (int) status.status);
105 bond_map_.clear();
106
107 status_ = status;
108 status_.name = ros::this_node::getName();
109 }
110}

117bool DriverApplication::bind_service_cb(cav_srvs::Bind::Request &req, cav_srvs::Bind::Response &res)
118{
119 std::unique_lock<std::mutex> lock(bond_mutex_);
120 if(bond_map_.find(req.id) != bond_map_.end())
121 {
122 ROS_WARN_STREAM(ros::this_node::getName() << " already bonded to id: " << req.id);
123 return true;
124 }
125
126 ROS_DEBUG_STREAM(ros::this_node::getName() << " binding to id: " << req.id);
127
128 std::shared_ptr<bond::Bond> bond(new bond::Bond(ros::this_node::getName() + "/bond", req.id));
129 bond->start();
130
131 bond_map_[req.id] = bond;
132
133
134 return true;
135}
