502 boost::optional<j2735_v2x_msgs::msg::TrafficControlMessage> Node::decode_geofence_control(std::vector<uint8_t>& binary_array)
503 {
504 j2735_v2x_msgs::msg::TrafficControlMessage output;
505 // decode results
506 asn_dec_rval_t rval;
507 MessageFrame_t* message = 0;
508 // copy from vector to array
509 auto len = binary_array.size();
510 uint8_t buf[len] = {0};
511 for(auto i = 0; i < len; i++) {
512 buf[i] = binary_array[i];
513 }
514
515 // use asn1c lib to decode
516 rval = uper_decode(0, &asn_DEF_MessageFrame, (void **) &message, buf, len, 0, 0);
517
518 // if decode succeed
519 if(rval.code == RC_OK) {
520 if (message->value.choice.TestMessage05.body.present == TrafficControlMessage_PR_reserved)
521 {
522 output.choice = j2735_v2x_msgs::msg::TrafficControlMessage::RESERVED;
523 }
524 else if (message->value.choice.TestMessage05.body.present == TrafficControlMessage_PR_tcmV01)
525 {
526 output.choice = j2735_v2x_msgs::msg::TrafficControlMessage::TCMV01;
527 output.tcm_v01 = decode_geofence_control_v01(message->value.choice.TestMessage05.body.choice.tcmV01);
528 }
529 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
530 return output;
531 }
532 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
533 return boost::optional<j2735_v2x_msgs::msg::TrafficControlMessage>{};
534 }

536 j2735_v2x_msgs::msg::TrafficControlMessageV01 Node::decode_geofence_control_v01(const TrafficControlMessageV01_t& message)
537 {
538 j2735_v2x_msgs::msg::TrafficControlMessageV01 output;
539
540 // decode reqid
541 output.reqid = decode_id64b(message.reqid);
542
543 // decode reqseq
544 output.reqseq = message.reqseq;
545
546 // decode msgtot
547 output.msgtot = message.msgtot;
548
549 // decode msgnum
550 output.msgnum = message.msgnum;
551
552 // decode id
553 output.id = decode_id128b(message.id);
554
555 // decode updated
556 // recover a long value from 8-bit array
557 uint64_t tmp_update=0;
558 auto update_bits_size = message.updated.size;
559 for (auto i=0; i< update_bits_size; i++){
560 tmp_update |= message.updated.buf[i];
561 if (i != update_bits_size - 1) tmp_update = tmp_update << 8;
562 }
563 output.updated = tmp_update;
564
565 // decode package optional
566 output.package_exists = false;
567 if (message.package)
568 {
569 output.package_exists = true;
570
571 output.package = decode_geofence_control_package(*message.package);
572 }
573
574 // decode params optional
575 output.params_exists = false;
576 if (message.params)
577 {
578 output.params_exists = true;
579 output.params = decode_geofence_control_params(*message.params);
580 }
581
582 // decode geometry optional
583 output.geometry_exists = false;
584 if (message.geometry)
585 {
586 output.geometry_exists = true;
587 output.geometry = decode_geofence_control_geometry(*message.geometry);
588 }
589
590 return output;
591 }
