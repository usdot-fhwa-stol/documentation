136QString OAIHttpRequestWorker::http_attribute_encode(QString attribute_name, QString input) {
137 // result structure follows RFC 5987
138 bool need_utf_encoding = false;
139 QString result = "";
140 QByteArray input_c = input.toLocal8Bit();
141 char c;
142 for (int i = 0; i < input_c.length(); i++) {
143 c = input_c.at(i);
144 if (c == '\\' || c == '/' || c == '\0' || c < ' ' || c > '~') {
145 // ignore and request utf-8 version
146 need_utf_encoding = true;
147 } else if (c == '"') {
148 result += "\\\"";
149 } else {
150 result += c;
151 }
152 }
153
154 if (result.length() == 0) {
155 need_utf_encoding = true;
156 }
157
158 if (!need_utf_encoding) {
159 // return simple version
160 return QString("%1=\"%2\"").arg(attribute_name, result);
161 }
162
163 QString result_utf8 = "";
164 for (int i = 0; i < input_c.length(); i++) {
165 c = input_c.at(i);
166 if ((c >= '0' && c <= '9') || (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')) {
167 result_utf8 += c;
168 } else {
169 result_utf8 += "%" + QString::number(static_cast<unsigned char>(input_c.at(i)), 16).toUpper();
170 }
171 }
172
173 // return enhanced version with UTF-8 support
174 return QString("%1=\"%2\"; %1*=utf-8''%3").arg(attribute_name, result, result_utf8);
175}

188
189 if (input->files.length() > 0) {
190 input->var_layout = MULTIPART;

194 }
195
196 // prepare request content
197
198 QString boundary = "";
199
200 if (input->var_layout == ADDRESS || input->var_layout == URL_ENCODED) {
201 // variable layout is ADDRESS or URL_ENCODED
202
203 if (input->vars.count() > 0) {
204 bool first = true;
205 isFormData = true;
206 foreach (QString key, input->vars.keys()) {
207 if (!first) {
208 request_content.append("&");
209 }
210 first = false;
211
212 request_content.append(QUrl::toPercentEncoding(key));
213 request_content.append("=");
214 request_content.append(QUrl::toPercentEncoding(input->vars.value(key)));
215 }
216
217 if (input->var_layout == ADDRESS) {
218 input->url_str += "?" + request_content;
219 request_content = "";
220 }
221 }
222 } else {
223 // variable layout is MULTIPART
224
225 boundary = QString("__-----------------------%1%2")
226 #if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
227 .arg(QDateTime::currentDateTime().toSecsSinceEpoch())
228 .arg(randomGenerator.generate());
229 #else
230 .arg(QDateTime::currentDateTime().toTime_t())
231 .arg(qrand());
232 #endif
233 QString boundary_delimiter = "--";
234 QString new_line = "\r\n";
235
236 // add variables
237 foreach (QString key, input->vars.keys()) {
238 // add boundary
239 request_content.append(boundary_delimiter.toUtf8());
240 request_content.append(boundary.toUtf8());
241 request_content.append(new_line.toUtf8());
242
243 // add header
244 request_content.append("Content-Disposition: form-data; ");
245 request_content.append(http_attribute_encode("name", key).toUtf8());
246 request_content.append(new_line.toUtf8());
247 request_content.append("Content-Type: text/plain");
248 request_content.append(new_line.toUtf8());
249
250 // add header to body splitter
251 request_content.append(new_line.toUtf8());
252
253 // add variable content
254 request_content.append(input->vars.value(key).toUtf8());
255 request_content.append(new_line.toUtf8());
256 }
257
258 // add files
259 for (QList<OAIHttpFileElement>::iterator file_info = input->files.begin(); file_info != input->files.end(); file_info++) {
260 QFileInfo fi(file_info->local_filename);
261
262 // ensure necessary variables are available
263 if (file_info->local_filename == nullptr
264 || file_info->local_filename.isEmpty()
265 || file_info->variable_name == nullptr
266 || file_info->variable_name.isEmpty()
267 || !fi.exists()
268 || !fi.isFile()
269 || !fi.isReadable()) {
270 // silent abort for the current file
271 continue;
272 }
273
274 QFile file(file_info->local_filename);
275 if (!file.open(QIODevice::ReadOnly)) {
276 // silent abort for the current file
277 continue;
278 }
279
280 // ensure filename for the request
281 if (file_info->request_filename == nullptr || file_info->request_filename.isEmpty()) {
282 file_info->request_filename = fi.fileName();
283 if (file_info->request_filename.isEmpty()) {
284 file_info->request_filename = "file";
285 }
286 }
287
288 // add boundary
289 request_content.append(boundary_delimiter.toUtf8());
290 request_content.append(boundary.toUtf8());
291 request_content.append(new_line.toUtf8());
292
293 // add header
294 request_content.append(
295 QString("Content-Disposition: form-data; %1; %2").arg(http_attribute_encode("name", file_info->variable_name), http_attribute_encode("filename", file_info->request_filename)).toUtf8());
296 request_content.append(new_line.toUtf8());
297
298 if (file_info->mime_type != nullptr && !file_info->mime_type.isEmpty()) {
299 request_content.append("Content-Type: ");
300 request_content.append(file_info->mime_type.toUtf8());
301 request_content.append(new_line.toUtf8());
302 }
303
304 request_content.append("Content-Transfer-Encoding: binary");
305 request_content.append(new_line.toUtf8());
306
307 // add header to body splitter
308 request_content.append(new_line.toUtf8());
309
310 // add file content
311 request_content.append(file.readAll());
312 request_content.append(new_line.toUtf8());
313
314 file.close();
315 }
316
317 // add end of body
318 request_content.append(boundary_delimiter.toUtf8());
319 request_content.append(boundary.toUtf8());
320 request_content.append(boundary_delimiter.toUtf8());
321 }
322
323 if (input->request_body.size() > 0) {
324 qDebug() << "got a request body";
325 request_content.clear();
326 if(!isFormData && (input->var_layout != MULTIPART) && isRequestCompressionEnabled){
327 request_content.append(compress(input->request_body, 7, OAICompressionType::Gzip));
328 } else {
329 request_content.append(input->request_body);
330 }
331 }
332 // prepare connection
333
334 QNetworkRequest request = QNetworkRequest(QUrl(input->url_str));
335 if (OAIHttpRequestWorker::sslDefaultConfiguration != nullptr) {
336 request.setSslConfiguration(*OAIHttpRequestWorker::sslDefaultConfiguration);
337 }
338 request.setRawHeader("User-Agent", "OpenAPI-Generator/1.0.0/cpp-qt");
339 foreach (QString key, input->headers.keys()) { request.setRawHeader(key.toStdString().c_str(), input->headers.value(key).toStdString().c_str()); }
340
341 if (request_content.size() > 0 && !isFormData && (input->var_layout != MULTIPART)) {
342 if (!input->headers.contains("Content-Type")) {
343 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
344 } else {
345 request.setHeader(QNetworkRequest::ContentTypeHeader, input->headers.value("Content-Type"));
346 }
347 if(isRequestCompressionEnabled){
348 request.setRawHeader("Content-Encoding", "gzip");
349 }
350 } else if (input->var_layout == URL_ENCODED) {
351 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded");
352 } else if (input->var_layout == MULTIPART) {
353 request.setHeader(QNetworkRequest::ContentTypeHeader, "multipart/form-data; boundary=" + boundary);
354 }
355
356 if(isResponseCompressionEnabled){
357 request.setRawHeader("Accept-Encoding", "gzip");
358 } else {
359 request.setRawHeader("Accept-Encoding", "identity");
360 }
361
362 if (input->http_method == "GET") {
363 reply = manager->get(request);
364 } else if (input->http_method == "POST") {
365 reply = manager->post(request, request_content);
366 } else if (input->http_method == "PUT") {
367 reply = manager->put(request, request_content);
368 } else if (input->http_method == "HEAD") {
369 reply = manager->head(request);
370 } else if (input->http_method == "DELETE") {
371 reply = manager->deleteResource(request);
372 } else {
373#if (QT_VERSION >= 0x050800)
374 reply = manager->sendCustomRequest(request, input->http_method.toLatin1(), request_content);
375#else
376 QBuffer *buffer = new QBuffer;
377 buffer->setData(request_content);
378 buffer->open(QIODevice::ReadOnly);
379
380 reply = manager->sendCustomRequest(request, input->http_method.toLatin1(), buffer);
381 buffer->setParent(reply);
382#endif
383 }
384 if (reply != nullptr) {
385 reply->setParent(this);
386 connect(reply, &QNetworkReply::finished, [this, reply] {
387 on_reply_finished(reply);
388 });
389 }
390 if (timeOutTimer.interval() > 0) {
391 QObject::connect(&timeOutTimer, &QTimer::timeout, [this, reply] {
392 on_reply_timeout(reply);
393 });
394 timeOutTimer.start();
395 }

398void OAIHttpRequestWorker::on_reply_finished(QNetworkReply *reply) {
399 bool codeSts = false;
400 if(timeOutTimer.isActive()) {
401 QObject::disconnect(&timeOutTimer, &QTimer::timeout, nullptr, nullptr);
402 timeOutTimer.stop();
403 }
404 error_type = reply->error();
405 error_str = reply->errorString();
406 if (reply->rawHeaderPairs().count() > 0) {
407 for (const auto &item : reply->rawHeaderPairs()) {
408 headers.insert(item.first, item.second);
409 }
410 }
411 auto rescode = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt(&codeSts);
412 if(codeSts){
413 httpResponseCode = rescode;
414 } else{
415 httpResponseCode = -1;
416 }
417 process_response(reply);
418 reply->deleteLater();
419 emit on_execution_finished(this);
420}

432void OAIHttpRequestWorker::process_response(QNetworkReply *reply) {
433 QString contentDispositionHdr;
434 QString contentTypeHdr;
435 QString contentEncodingHdr;
436
437 for(auto hdr: getResponseHeaders().keys()){
438 if(hdr.compare(QString("Content-Disposition"), Qt::CaseInsensitive) == 0){
439 contentDispositionHdr = getResponseHeaders().value(hdr);
440 }
441 if(hdr.compare(QString("Content-Type"), Qt::CaseInsensitive) == 0){
442 contentTypeHdr = getResponseHeaders().value(hdr);
443 }
444 if(hdr.compare(QString("Content-Encoding"), Qt::CaseInsensitive) == 0){
445 contentEncodingHdr = getResponseHeaders().value(hdr);
446 }
447 }
448
449 if (!contentDispositionHdr.isEmpty()) {
450 auto contentDisposition = contentDispositionHdr.split(QString(";"), SKIP_EMPTY_PARTS);
451 auto contentType =
452 !contentTypeHdr.isEmpty() ? contentTypeHdr.split(QString(";"), SKIP_EMPTY_PARTS).first() : QString();
453 if ((contentDisposition.count() > 0) && (contentDisposition.first() == QString("attachment"))) {
454 QString filename = QUuid::createUuid().toString();
455 for (const auto &file : contentDisposition) {
456 if (file.contains(QString("filename"))) {
457 filename = file.split(QString("="), SKIP_EMPTY_PARTS).at(1);
458 break;
459 }
460 }
461 OAIHttpFileElement felement;
462 felement.saveToFile(QString(), workingDirectory + QDir::separator() + filename, filename, contentType, reply->readAll());
463 files.insert(filename, felement);
464 }
465
466 } else if (!contentTypeHdr.isEmpty()) {
467 auto contentType = contentTypeHdr.split(QString(";"), SKIP_EMPTY_PARTS);
468 if ((contentType.count() > 0) && (contentType.first() == QString("multipart/form-data"))) {
469 // TODO : Handle Multipart responses
470 } else {
471 if(!contentEncodingHdr.isEmpty()){
472 auto encoding = contentEncodingHdr.split(QString(";"), SKIP_EMPTY_PARTS);
473 if(encoding.count() > 0){
474 auto compressionTypes = encoding.first().split(',', SKIP_EMPTY_PARTS);
475 if(compressionTypes.contains("gzip", Qt::CaseInsensitive) || compressionTypes.contains("deflate", Qt::CaseInsensitive)){
476 response = decompress(reply->readAll());
477 } else if(compressionTypes.contains("identity", Qt::CaseInsensitive)){
478 response = reply->readAll();
479 }
480 }
481 }
482 else {
483 response = reply->readAll();
484 }
485 }
486 }
