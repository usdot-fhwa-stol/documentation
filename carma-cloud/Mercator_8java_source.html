178 public static double eMercY(double lat) {
179 if (lat > 89.5) {
180 lat = 89.5;
181 }
182 if (lat < -89.5) {
183 lat = -89.5;
184 }
185 double phi = Math.toRadians(lat);
186 double con = ECC * Math.sin(phi);
187 con = Math.pow(((1.0-con)/(1.0+con)), ECC_OVER_TWO);
188 double ts = Math.tan(0.5 * ((PI_OVER_TWO) - phi))/con;
189 double y = 0 - R_MAJOR * Math.log(ts);
190 return y;
191 }

200 public static double eLat(double dY)
201 {
202 double ts = Math.exp(-dY / R_MAJOR);
203 double phi = PI_OVER_TWO - 2 * Math.atan(ts);
204 double dphi = 1.0;
205 for (int i = 0; Math.abs(dphi) > 0.000000001 && i < 15; i++)
206 {
207 double con = ECC * Math.sin(phi);
208 dphi = PI_OVER_TWO - 2 * Math.atan(ts * Math.pow((1.0 - con) / (1.0 + con), ECC_OVER_TWO)) - phi;
209 phi += dphi;
210 }
211
212 return Math.toDegrees(phi);
213 }
