51double CostPluginWorker::compute_final_score(cav_msgs::ManeuverPlan plan)
52{
53 cost_plugin_system::CostofComfort coc(max_decelaration_);
54 cost_plugin_system::CostofEfficiency coe(speed_limit_, speed_buffer_);
55 cost_plugin_system::CostofFeasibility cofe(max_accelaration_, max_decelaration_);
56 cost_plugin_system::CostofFuel cof;
57 cost_plugin_system::CostofLegality col;
58 cost_plugin_system::CostofSafety cos(speed_limit_);
59
60 double total_cost = 0.0;
61 if (col.compute_cost(plan) == 0)
62 {
63 double cost_of_comfort = coc.compute_cost(plan);
64 double cost_of_efficiency = coe.compute_cost(plan);
65 double cost_of_feasibility = cofe.compute_cost(plan);
66 double cost_of_fuel = cof.compute_cost(plan);
67 double cost_of_safety = cos.compute_cost(plan);
68
69 total_cost = weight_of_comfort_ * cost_of_comfort + weight_of_efficiency_ * cost_of_efficiency +
70 weight_of_feasibility_ * cost_of_feasibility + weight_of_fuel_ * cost_of_fuel +
71 weight_of_safety_ * cost_of_safety;
72 }
73 else
74 {
75 total_cost = -999.0;
76 }
77 return total_cost;
78}
