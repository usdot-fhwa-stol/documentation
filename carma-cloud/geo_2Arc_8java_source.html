32 public void addPoints(Road oRoad, double dMaxStep, boolean bIncludeEnd, Proj oProj, double[] dProjPt)
33 {
34 try
35 {
36 int nLimit;
37 double dEndDistOnRoad = m_dS + m_dL;
38 ArrayDeque<LaneSection> oDeque = new ArrayDeque();
39 oRoad.getLaneSections(oDeque, m_dS, dEndDistOnRoad);
40 double dR = 1.0 / m_dC;
41 double dH = m_dX + dR * Math.cos(m_dHdg + Mercator.PI_OVER_TWO);
42 double dK = m_dY + dR * Math.sin(m_dHdg + Mercator.PI_OVER_TWO);
43 double dInitTheta = m_dHdg + 3.0 * Mercator.PI_OVER_TWO;
44 double dTotalAngle = 0;
45 while (!oDeque.isEmpty())
46 {
47 LaneSection oSection = oDeque.removeFirst();
48 double dStart = Math.max(m_dS, oSection.m_dS);
49 double dEnd = oDeque.isEmpty() ? dEndDistOnRoad : oDeque.getFirst().m_dS;
50 double dLength = dEnd - dStart;
51 if (dMaxStep > dLength)
52 nLimit = 1;
53 else
54 nLimit = (int)(dLength / dMaxStep) + 1;
55
56 double dTheta = dLength * m_dC;
57
58 oRoad.m_dLaneZero = Arrays.ensureCapacity(oRoad.m_dLaneZero, nLimit * 2);
59 oRoad.m_dTrack = Arrays.ensureCapacity(oRoad.m_dTrack, nLimit * 2);
60 oRoad.m_dNoProj = Arrays.ensureCapacity(oRoad.m_dNoProj, nLimit * 2);
61 double dAngleStep = dTheta / nLimit;
62 double dLengthStep = dLength / nLimit;
63 if (bIncludeEnd) // add last point for last section and last geometry
64 ++nLimit;
65 for (int i = 0; i < nLimit; i++)
66 {
67 double dAngle = dInitTheta + dTotalAngle;
68 dTotalAngle += dAngleStep;
69 double dX = dH + dR * Math.cos(dAngle);
70 double dY = dK + dR * Math.sin(dAngle);
71 oRoad.m_dTrack = addProjectedPoint(dX, dY, oRoad.m_dTrack, oProj, dProjPt);
72 oRoad.m_dTrack = Arrays.add(oRoad.m_dTrack, 0);
73
74 double dLengthAlongRoad = dLengthStep * i + dStart;
75 LaneOffset oLaneOffset = LaneOffset.getLaneOffset(dLengthAlongRoad, oRoad.m_oLaneOffsets);
76 double dHdg = m_dHdg + dTotalAngle;
77 double dT = MathUtil.cubic(dLengthAlongRoad - oLaneOffset.m_dS, oLaneOffset.m_dA, oLaneOffset.m_dB, oLaneOffset.m_dC, oLaneOffset.m_dD);
78 double dXLaneZero = dX - Math.sin(dHdg) * dT;
79 double dYLaneZero = dY + Math.cos(dHdg) * dT;
80 oRoad.m_dLaneZero = addProjectedPoint(dXLaneZero, dYLaneZero, oRoad.m_dLaneZero, oProj, dProjPt);
81 oRoad.m_dLaneZero = Arrays.add(oRoad.m_dLaneZero, 0);
82
83 addLanePoints(oRoad, dHdg, dXLaneZero, dYLaneZero, dLengthAlongRoad, oSection, oProj, dProjPt);
84
85 }
86 dTotalAngle -= dAngleStep;
87 }
88 }
89 catch (Exception oEx)
90 {
91 oEx.printStackTrace();
92 }
93 }
