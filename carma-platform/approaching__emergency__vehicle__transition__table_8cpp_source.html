27void ApproachingEmergencyVehicleTransitionTable::event(ApproachingEmergencyVehicleEvent event)
28{
29 switch (state_)
30 {
31 case ApproachingEmergencyVehicleState::NO_APPROACHING_ERV:
32 eventWhenNO_APPROACHING_ERV(event);
33 break;
34
35 case ApproachingEmergencyVehicleState::MOVING_OVER_FOR_APPROACHING_ERV:
36 eventWhenMOVING_OVER_FOR_APPROACHING_ERV(event);
37 break;
38
39 case ApproachingEmergencyVehicleState::WAITING_FOR_APPROACHING_ERV:
40 eventWhenWAITING_FOR_APPROACHING_ERV(event);
41 break;
42
43 case ApproachingEmergencyVehicleState::SLOWING_DOWN_FOR_ERV:
44 eventWhenSLOWING_DOWN_FOR_ERV(event);
45 break;
46
47 default:
48 throw std::invalid_argument("Transition table in unsupported state");
49 }
50}

57void ApproachingEmergencyVehicleTransitionTable::eventWhenNO_APPROACHING_ERV(ApproachingEmergencyVehicleEvent event)
58{
59 switch (event)
60 {
61 case ApproachingEmergencyVehicleEvent::NO_APPROACHING_ERV:
62 setAndLogState(ApproachingEmergencyVehicleState::NO_APPROACHING_ERV, event);
63 break;
64
65 case ApproachingEmergencyVehicleEvent::APPROACHING_ERV_IN_PATH:
66 setAndLogState(ApproachingEmergencyVehicleState::MOVING_OVER_FOR_APPROACHING_ERV, event);
67 break;
68
69 case ApproachingEmergencyVehicleEvent::APPROACHING_ERV_NOT_IN_PATH:
70 setAndLogState(ApproachingEmergencyVehicleState::WAITING_FOR_APPROACHING_ERV, event);
71 break;
72
73 case ApproachingEmergencyVehicleEvent::ERV_ABOUT_TO_PASS_IN_PATH:
74 setAndLogState(ApproachingEmergencyVehicleState::WAITING_FOR_APPROACHING_ERV, event);
75 break;
76
77 case ApproachingEmergencyVehicleEvent::ERV_ABOUT_TO_PASS_NOT_IN_PATH:
78 setAndLogState(ApproachingEmergencyVehicleState::WAITING_FOR_APPROACHING_ERV, event);
79 break;
80
81 case ApproachingEmergencyVehicleEvent::ERV_PASSING_IN_PATH:
82 setAndLogState(ApproachingEmergencyVehicleState::SLOWING_DOWN_FOR_ERV, event);
83 break;
84
85 case ApproachingEmergencyVehicleEvent::ERV_PASSING_NOT_IN_PATH:
86 setAndLogState(ApproachingEmergencyVehicleState::SLOWING_DOWN_FOR_ERV, event);
87 break;
88
89 default:
90 logDebugEvent(event);
91 break;
92 }

203void ApproachingEmergencyVehicleTransitionTable::setAndLogState(ApproachingEmergencyVehicleState new_state, ApproachingEmergencyVehicleEvent source_event)
204{
205 if (new_state == state_)
206 {
207 return; // State was unchanged no need to log or trigger callbacks
208 }
209
210 RCLCPP_INFO_STREAM(rclcpp::get_logger("approaching_emergency_vehicle_plugin"), "ApproachingEmergencyVehicleTransitionTable changed ApproachingEmergencyVehicle Strategic Plugin state from "
211 << state_ << " to " << new_state << " because of event " << source_event);
212
213 ApproachingEmergencyVehicleState prev_state = state_;
214 state_ = new_state; // Set new state
215
216 if (transition_callback_) // Trigger callback if available
217 {
218 transition_callback_(prev_state, state_, source_event);
219 }
220}
