74 {
75 if (!m_bRunning.compareAndSet(false, true))
76 {
77 LOGGER.debug("Collecting rsm did not run. Already running");

79 }
80
81 if (!getToken())

87 int nAdded = 0;
88 int nRemoved = 0;
89 if (xmlList(true))
90 {
91 int nIndex = m_oCurrentRsms.size();
92 while (nIndex-- > 0)
93 {
94 RsmRecord oRec = m_oCurrentRsms.get(nIndex);
95 if (oRec.m_bNew)
96 {
97 xmlFile(oRec);
98 ++nAdded;
99 }
100 else if (!oRec.m_bCurrent)
101 {
102 m_oCurrentRsms.remove(nIndex);
103 oRec.cleanup();
104 ++nRemoved;
105 }
106 }
107 }
108 m_bRunning.set(false);
109
110 StringBuilder sMsg = new StringBuilder("RSM ");
111 sMsg.append(nAdded);
112 sMsg.append(" added ");
113 sMsg.append(nRemoved);
114 sMsg.append(" removed ");
115 sMsg.append(m_oCurrentRsms.size());
116 sMsg.append(" total");
117 LOGGER.debug(sMsg);
118 }
