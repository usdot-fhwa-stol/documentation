23// THERE SHOULD BE FOUR OPERATIONAL MODES FOR THIS PLUGIN
24// UNAVAILABLE { On In Intersection Range Event -> APPROACH }
25// APPROACHING { On Stopped Event -> WAITING, On Crossed Stop Bar -> DEPARTURE}
26// WAITING { On Green Light Event -> DEPARTURE}
27// DEPARTING { On exit intersection -> UNAVAILABLE}

34void LCIStrategicStateTransitionTable::signal(TransitEvent signal)
35{
36 switch (state_)
37 {
38 case TransitState::UNAVAILABLE:
39 signalWhenUNAVAILABLE(signal);
40 break;
41
42 case TransitState::APPROACHING:
43 signalWhenAPPROACHING(signal);
44 break;
45
46 case TransitState::WAITING:
47 signalWhenWAITING(signal);
48 break;
49
50 case TransitState::DEPARTING:
51 signalWhenDEPARTING(signal);
52 break;
53
54 default:
55 throw std::invalid_argument("Transition table in unsupported state");
56 }
57}

64void LCIStrategicStateTransitionTable::signalWhenUNAVAILABLE(TransitEvent signal)
65{
66 if (signal == TransitEvent::IN_STOPPING_RANGE)
67 {
68 setAndLogState(TransitState::APPROACHING, signal);
69 }
70 else
71 {
72 logDebugSignal(signal);
73 }
74}

76void LCIStrategicStateTransitionTable::signalWhenAPPROACHING(TransitEvent signal)
77{
78 switch (signal)
79 {
80 case TransitEvent::STOPPED:
81 setAndLogState(TransitState::WAITING, signal);
82 break;
83
84 case TransitEvent::CROSSED_STOP_BAR:
85 setAndLogState(TransitState::DEPARTING, signal);
86 break;
87
88 default:
89 logDebugSignal(signal);
90 break;
91 }
92}

94void LCIStrategicStateTransitionTable::signalWhenWAITING(TransitEvent signal)
95{
96 if (signal == TransitEvent::RED_TO_GREEN_LIGHT)
97 {
98 setAndLogState(TransitState::DEPARTING, signal);
99 }
100 else
101 {
102 logDebugSignal(signal);
103 }
104}
105
106void LCIStrategicStateTransitionTable::signalWhenDEPARTING(TransitEvent signal)
107{
108 if (signal == TransitEvent::INTERSECTION_EXIT)
109 {
110 setAndLogState(TransitState::UNAVAILABLE, signal);
111 }
112 else
113 {
114 logDebugSignal(signal);
115 }
116}

124void LCIStrategicStateTransitionTable::setAndLogState(TransitState new_state, TransitEvent source_signal)
125{
126 if (new_state == state_)
127 {
128 return; // State was unchanged no need to log or trigger callbacks
129 }
130
131 RCLCPP_INFO_STREAM(rclcpp::get_logger("lci_strategic_plugin"), "LCIStrategicStateTransitionTable changed LCIStrategic Strategic Plugin state from "
132 << state_ << " to " << new_state << " because of signal " << source_signal);
133
134 TransitState prev_state = state_;
135 state_ = new_state; // Set new state
136
137 if (transition_callback_) // Trigger callback if available
138 {
139 transition_callback_(prev_state, state_, source_signal);
140 }
141}
