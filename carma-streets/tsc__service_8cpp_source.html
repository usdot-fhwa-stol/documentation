190 void tsc_service::produce_spat_json() const {
191 try {
192 int count = 0;
193 uint64_t spat_processing_time = 0;
194 SPDLOG_DEBUG("Starting SPaT producer");
195 while(spat_worker_ptr && tsc_state_ptr && spat_producer) {
196 try {
197 spat_worker_ptr->receive_spat(spat_ptr);
198 SPDLOG_DEBUG("Current SPaT : {0} ", spat_ptr->toJson());
199 switch (spat_proj_mode)
200 {
201 case spat_projection_mode::fixed_projection: {
202 tsc_state_ptr->add_future_movement_events(spat_ptr);
203 break;
204 }
205 case spat_projection_mode::dpp_projection : {
206 std::scoped_lock<std::mutex> lck{dpp_mtx};
207 // throws desired phase plan exception when no previous green information present
208 monitor_dpp_ptr->update_spat_future_movement_events(spat_ptr, tsc_state_ptr);
209 break;
210 }
211 default: {
212 // Poll and log vehicle and pedestrian call information every 10 spat messages or at 10 Hz
213 if ( count%10 == 0) {
214 tsc_state_ptr->poll_vehicle_pedestrian_calls();
215 if(auto logger = spdlog::get(VEH_PED_CALL_LOGGER_NAME); logger != nullptr ){
216 logger->info("{0}, {1}, {2}", streets_clock_singleton::time_in_ms(),
217 tsc_state_ptr->vector_to_string(tsc_state_ptr->get_vehicle_calls()),
218 tsc_state_ptr->vector_to_string(tsc_state_ptr->get_pedestrian_calls())
219 );
220 }
221 }
222 }
223 }
224 if (spat_ptr && spat_producer) {
225 spat_producer->send(spat_ptr->toJson());
226 // No utility in calculating spat latency in simulation mode
227 if ( !this->is_simulation_mode() ) {
228 log_spat_latency(count, spat_processing_time, spat_ptr->get_intersection().get_epoch_timestamp()) ;
229 }
230 }
231 }
232 catch( const signal_phase_and_timing::signal_phase_and_timing_exception &e ) {
233 SPDLOG_ERROR("Encountered exception, spat not published : \n {0}", e.what());
234 }
235 catch( const traffic_signal_controller_service::monitor_desired_phase_plan_exception &e) {
236 SPDLOG_ERROR("Could not update movement events, spat not published. Encountered exception : \n {0}", e.what());
237 }
238 catch(const traffic_signal_controller_service::monitor_states_exception &e){
239 SPDLOG_ERROR("Could not update movement events, spat not published. Encountered exception : \n {0}", e.what());
240 }
241 }
242 SPDLOG_WARN("Stopping produce_spat_json! Are pointers null: spat_worker {0}, spat_producer {1}, tsc_state {2}",
243 spat_worker_ptr == nullptr, spat_ptr == nullptr, tsc_state_ptr == nullptr);

332 void tsc_service::set_tsc_hold_and_omit_forceoff_call()
333 {
334 if(control_tsc_state_ptr_ && !tsc_set_command_queue_.empty())
335 {
336 //Clear all phase controls from the traffic signal controller
337 SPDLOG_DEBUG("Clear all phase controls from TSC!");
338 control_tsc_state_ptr_->run_clear_all_snmp_commands();
339 }
340 while(!tsc_set_command_queue_.empty())
341 {
342 auto current_command = tsc_set_command_queue_.front();
343 SPDLOG_DEBUG("Checking if front command {0} is expired", current_command.get_cmd_info());
344 //Check if event is expired
345 long duration = current_command.start_time_ - streets_clock_singleton::time_in_ms();
346 if(duration < 0){
347 throw control_tsc_state_exception("SNMP set command is expired! Start time was "
348 + std::to_string(current_command.start_time_) + " and current time is "
349 + std::to_string(streets_clock_singleton::time_in_ms() ) + ".");
350 }
351 SPDLOG_DEBUG("Sleeping for {0}ms.", duration);
352 streets_clock_singleton::sleep_for(duration);
353
354 //After sleep duration, check the snmp queue again. If empty, stop the current command execution
355 if(tsc_set_command_queue_.empty())
356 {
357 SPDLOG_DEBUG("SNMP command queue is empty, stop the current command execution: {0}", current_command.get_cmd_info());
358 break;
359 }
360 //Check if queue commands has been updated
361 else if (tsc_set_command_queue_.front().get_cmd_info() != current_command.get_cmd_info())
362 {
363 SPDLOG_DEBUG("SNMP command queue is updated with new schedule, stop the current command execution: {0}", current_command.get_cmd_info());
364 break;
365 }
366
367 if(!control_tsc_state_ptr_)
368 {
369 throw control_tsc_state_exception("Control TSC state is not initialized.");
370 }
371
372 //Checking all the commands from the queue. If there are any commands executed at the same time as the current_command, and execute these commands now.
373 while (!tsc_set_command_queue_.empty())
374 {
375 auto command_to_execute = tsc_set_command_queue_.front();
376 if(command_to_execute.start_time_ == current_command.start_time_)
377 {
378 if(control_tsc_state_ptr_ && !control_tsc_state_ptr_->run_snmp_cmd_set_request(command_to_execute))
379 {
380 throw control_tsc_state_exception("Could not set state for movement group in desired phase plan/phase control schedule.");
381 }
382 // Log command info sent
383 SPDLOG_INFO(command_to_execute.get_cmd_info());
384 if ( enable_snmp_cmd_logging_ ){
385 if(auto logger = spdlog::get("snmp_cmd_logger"); logger != nullptr ){
386 logger->info( command_to_execute.get_cmd_info());
387 }
388 }
389 //SNMP command Queue has no update nor has been cleared, pop the existing front command.
390 std::scoped_lock<std::mutex> snmp_cmd_lck(snmp_cmd_queue_mtx);
391 tsc_set_command_queue_.pop();
392 }
393 else
394 {
395 break;
396 }
397 }
398 }

424 void tsc_service::log_spat_latency(int &count, uint64_t &spat_processing_time, uint64_t spat_time_stamp) const {
425 // Calculate and average spat processing time over 20 messages sent
426 if (count <= 20 ) {
427 uint64_t timestamp = streets_clock_singleton::time_in_ms();
428 spat_processing_time += timestamp - spat_time_stamp;
429 count++;
430 }
431 // Log result at 20 and clear counts
432 if (count >= 20 ) {
433 double total_processing_time = ((double)(spat_processing_time))/20.0;
434 SPDLOG_INFO("SPat average processing time over 20 messages is {0} ms and total processing time for 20 messages is {1} ms!", total_processing_time, spat_processing_time);
435 spat_processing_time = 0;
436 count = 0;
437 }

442 streets_service::streets_service::start();
443 SPDLOG_INFO("Starting TSC Service threads");
444 std::thread tsc_config_thread(&tsc_service::produce_tsc_config_json, this);
445
446 std::thread spat_t(&tsc_service::produce_spat_json, this);
447
448 std::thread desired_phase_plan_t(&tsc_service::consume_desired_phase_plan, this);
449
450 std::thread consume_phase_control_schedule_t(&tsc_service::consume_phase_control_schedule, this);
451
452 std::thread control_phases_t(&tsc_service::control_tsc_phases, this);
453
454 // Run threads as joint so that they dont overlap execution
455 tsc_config_thread.join();
456 spat_t.join();
457
458 // An indicator to control whether launching a thread to consume carma-streets internal desired phase plan generated by singal optimization (SO) service
459 // or a thread to consume MMITSS external phase control schedule generated by MRP (https://github.com/mmitss/mmitss-az)
460 if (use_mmitss_mrp)
461 {
462 SPDLOG_DEBUG("Thread joined to consume phase control schedule generated by MRP.");
463 consume_phase_control_schedule_t.join();
464 }else{
465 SPDLOG_DEBUG("Thread joined to consume desired phase plan generated by carma-streets SO service.");
466 desired_phase_plan_t.join();
467 }
468
469 control_phases_t.join();
470 }
