22int CarmaRecordNode::run() const
23{
24
25 // boolean values that will load in from the rosparam server
26 // they control if a set of topics will be excluded
27 bool exclude_default;
28 bool exclude_lidar;
29 bool exclude_camera;
30 bool exclude_can;
31
32 cnh_.getParam("/exclude_default", exclude_default);
33 cnh_.getParam("/exclude_lidar", exclude_lidar);
34 cnh_.getParam("/exclude_camera", exclude_camera);
35 cnh_.getParam("/exclude_can", exclude_can);
36
37 // exclude_regex is the final list of topics to be excluded
38 std::string exclude_regex = ""; // Add default exclusion to this string
39
40 // if a set of topics needs to be excluded, it will be iterated through and added to the exclude regex
41 if(exclude_default){
42 std::vector<std::string> excluded_default_topics;
43 cnh_.getParam("/excluded_default_topics", excluded_default_topics);
44
45 for (const auto& exclusion_string : excluded_default_topics){
46 exclude_regex += exclusion_string + "|";
47 }
48 }
49
50 if(exclude_lidar){
51 std::vector<std::string> excluded_lidar_topics;
52 cnh_.getParam("/excluded_lidar_topics", excluded_lidar_topics);
53
54 for (const auto& exclusion_string : excluded_lidar_topics){
55 exclude_regex += exclusion_string + "|";
56 }
57 }
58
59 if(exclude_camera){
60 std::vector<std::string> excluded_camera_topics;
61 cnh_.getParam("/excluded_camera_topics", excluded_camera_topics);
62
63 for (const auto& exclusion_string : excluded_camera_topics){
64 exclude_regex += exclusion_string + "|";
65 }
66 }
67
68 if(exclude_can){
69 std::vector<std::string> excluded_can_topics;
70 cnh_.getParam("/excluded_can_topics", excluded_can_topics);
71
72 for (const auto& exclusion_string : excluded_can_topics){
73 exclude_regex += exclusion_string + "|";
74 }
75 }
76
77 // remove the final "|" from the regex
78 exclude_regex.pop_back();
79
80 // set the exclude_regex as a param in the param server
81 cnh_.setParam("exclude_regex", exclude_regex);
82
83 // if no topics are being excluded, set the no_exclusions parameter so the record script will still run
84 if(!(exclude_default || exclude_lidar || exclude_camera || exclude_can)){
85 cnh_.setParam("no_exclusions", true);
86 } else {
87 cnh_.setParam("no_exclusions", false);
88 }
89
90 // End node because there is no need to spin
91 return 0;
92}
