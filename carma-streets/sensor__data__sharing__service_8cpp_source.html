36 bool sds_service::initialize() {
37 if (!streets_service::initialize() ) {
38 SPDLOG_ERROR("Failed to initialize streets service base!");
39 return false;
40 }
41 SPDLOG_DEBUG("Intializing Sensor Data Sharing Service");
42 const std::string lanlet2_map = streets_service::get_system_config("LANELET2_MAP", "/home/carma-streets/MAP/Intersection.osm");
43 if (!read_lanelet_map(lanlet2_map)){
44 SPDLOG_ERROR("Failed to read lanlet2 map {0} !", lanlet2_map);
45 return false;
46 }
47 // Read sensor configuration file and get WSG84 location/origin reference frame.
48 const std::string sensor_config_file = streets_service::get_system_config("SENSOR_JSON_FILE_PATH", "/home/carma-streets/sensor_configurations/sensors.json");
49 const std::string sensor_id = ss::streets_configuration::get_string_config("sensor_id");
50 const lanelet::BasicPoint3d pose = parse_sensor_location(sensor_config_file, sensor_id);
51 this->sdsm_reference_point = this->map_projector->reverse(pose);
52
53 // Initialize SDSM Kafka producer
54 const std::string sdsm_topic = ss::streets_configuration::get_string_config("sdsm_producer_topic");
55 const std::string detection_topic = ss::streets_configuration::get_string_config("detection_consumer_topic");
56 const std::string sdsm_geo_reference = ss::streets_configuration::get_string_config("sdsm_geo_reference");
57 // Get Infrastructure ID for SDSM messages
58 this->_infrastructure_id = streets_service::get_system_config("INFRASTRUCTURE_ID", "");
59 return initialize_kafka_producer(sdsm_topic, sdsm_producer) && initialize_kafka_consumer(detection_topic, detection_consumer);
60 }

62 bool sds_service::read_lanelet_map(const std::string &filepath) {
63
64 try
65 {
66 int projector_type = 1;
67 std::string target_frame;
68 lanelet::ErrorMessages errors;
69 // Parse geo reference info from the lanelet map (.osm)
70 lanelet::io_handlers::AutowareOsmParser::parseMapParams(filepath, &projector_type, &target_frame);
71 this->map_projector = std::make_unique<lanelet::projection::LocalFrameProjector>(target_frame.c_str());
72 this->map_ptr = lanelet::load(filepath, *map_projector.get(), &errors);
73 //
74
75 if (!this->map_ptr->empty())
76 {
77 return true;
78 }
79 }
80 catch (const lanelet::ParseError &ex)
81 {
82 SPDLOG_ERROR("Cannot read osm file {0}. Error message: {1} .", filepath, ex.what());
83 }

87 void sds_service::consume_detections(){
88 if ( !detection_consumer ) {
89 throw std::runtime_error("Detection consumer is null!");
90 }
91 SPDLOG_DEBUG("Attempting to consume detections ...");
92 detection_consumer->subscribe();
93 while ( detection_consumer->is_running() ) {
94 try{
95 const std::string payload = detection_consumer->consume(1000);
96 if (payload.length() != 0)
97 {
98 auto detected_object = streets_utils::messages::detected_objects_msg::from_json(payload);
99 // Write Lock
100 std::unique_lock lock(detected_objects_lock);
101 detected_objects[detected_object._object_id] = detected_object;
102 SPDLOG_DEBUG("Detected Object List Size {0} after consumed: {1}", detected_objects.size(), payload);
103
104 }
105 }
106 catch (const streets_utils::json_utils::json_parse_exception &e) {
107 SPDLOG_ERROR("Exception occured consuming detection message : {0}", e.what());
108 }
109 }
110 SPDLOG_ERROR("Something went wrong, no longer consuming detections." );
111
112 }

114 void sds_service::produce_sdsms() {
115 if ( !sdsm_producer ) {
116 throw std::runtime_error("SDSM consumer is null!");
117 }
118 SPDLOG_INFO("Starting SDSM Producer!");
119 while ( sdsm_producer->is_running() ) {
120 try{
121 if ( !detected_objects.empty() ) {
122 streets_utils::messages::sdsm::sensor_data_sharing_msg msg = create_sdsm();
123 const std::string json_msg = streets_utils::messages::sdsm::to_json(msg);
124 SPDLOG_DEBUG("Sending SDSM : {0}", json_msg);
125 sdsm_producer->send(json_msg);
126 // Message Count max is 127, reset after max value
127 if ( this->_message_count <= 127) {
128 this->_message_count++;
129 }else {
130 this->_message_count = 0;
131 }
132 // Clear detected object
133 detected_objects.clear();
134 }
135 }
136 catch( const streets_utils::json_utils::json_parse_exception &e) {
137 SPDLOG_ERROR("Exception occurred producing SDSM : {0}", e.what());
138 }
139 ss::streets_clock_singleton::sleep_for(100); // Sleep for 100 ms between publish
140 }
141
142 }

144 streets_utils::messages::sdsm::sensor_data_sharing_msg sds_service::create_sdsm() {
145 streets_utils::messages::sdsm::sensor_data_sharing_msg msg;
146 // Read lock
147 uint64_t timestamp = ss::streets_clock_singleton::time_in_ms();
148 msg._time_stamp = to_sdsm_timestamp(timestamp);
149 // Populate with rolling counter
150 msg._msg_count = this->_message_count;
151 // Populate with infrastructure id
152 msg._source_id = this->_infrastructure_id;
153 msg._equipment_type = sdsm::equipment_type::RSU;
154 std::shared_lock lock(detected_objects_lock);
155 for (const auto &[object_id, object] : detected_objects){
156 auto detected_object_data = to_detected_object_data(object);
157 // TODO: Update time offset. Currently CARMA-Streets detected object message does not support timestamp
158 // This is a bug and needs to be addressed.
159 msg._objects.push_back(detected_object_data);
160 }
161 return msg;
162 }
