96 public static void renderTiledData(ArrayList<TrafCtrl> oCtrls, ArrayList<int[]> nTiles) throws IOException
97 {
98 TdLayer oLayer = new TdLayer("pavement", new String[]{"type"}, new String[]{"d", "s"}, TdLayer.POLYGON);
99 int nDriving = XodrUtil.getLaneType("driving");
100 int[] nTags = new int[2];
101 for (int[] nTile : nTiles)
102 {
103 int nX = nTile[0];
104 int nY = nTile[1];
105 Files.createDirectories(Paths.get(String.format(g_sTdFileFormat, nX, 0, 0, 0)).getParent(), FileUtil.DIRPERS);
106 writeIndexFile(oCtrls, nX, nY);
107 oLayer.clear();
108
109 ArrayList<ArrayList<double[]>> oClippedLanes = new ArrayList();
110 double[] dClipBounds = TileUtil.getClippingBounds(g_nDefaultZoom, nX, nY);
111 double[][] dClips = new double[][]{Arrays.newDoubleArray(512), Arrays.newDoubleArray(512), Arrays.newDoubleArray(512)};
112 oClippedLanes.add(new ArrayList());
113 oClippedLanes.add(new ArrayList());
114 oClippedLanes.add(new ArrayList());
115 for (TrafCtrl oCtrl : oCtrls)
116 {
117 if (Collections.binarySearch(oCtrl.m_oFullGeo.m_oTiles, nTile, Mercator.TILECOMP) < 0)
118 continue;
119
120 for (ArrayList oClipped : oClippedLanes)
121 oClipped.clear();
122 TileUtil.clipCtrlGeoForTile(oCtrl.m_oFullGeo, dClips, dClipBounds, oClippedLanes);
123 int nParts = oClippedLanes.get(0).size();
124 for (int nPart = 0; nPart < nParts; nPart++)
125 {
126 double[] dNT = oClippedLanes.get(1).get(nPart);
127 double[] dPT = oClippedLanes.get(2).get(nPart);
128
129 double[] dClippedPavement = Geo.createPolygon(dPT, dNT);
130 if (!Geo.isClockwise(dClippedPavement, 1))
131 {
132 dClippedPavement = Geo.reverseOrder(dClippedPavement);
133 }
134
135 nTags[1] = MathUtil.bytesToInt(oCtrl.m_yControlValue) == nDriving ? 0 : 1;
136 oLayer.add(new TdFeature(dClippedPavement, nTags, oCtrl));
137 }
138 }
139
140 if (!oLayer.isEmpty())
141 {
142 try (DataOutputStream oOut = new DataOutputStream(new BufferedOutputStream(FileUtil.newOutputStream(Paths.get(String.format(g_sTdFileFormat, nX, g_nDefaultZoom, nX, nY)), FileUtil.APPENDTO, FileUtil.FILEPERS))))
143 {
144 oLayer.write(oOut);
145 }
146 }
147 }
