18 streets_utils::messages::sdsm::time_stamp to_sdsm_timestamp(const uint64_t _epoch_time_ms) {
19 streets_utils::messages::sdsm::time_stamp sdsm_timestamp;
20
21 // From millisecond time stamp
22 boost::posix_time::ptime posix_time = boost::posix_time::from_time_t(_epoch_time_ms/SECONDS_TO_MILLISECONDS) +
23 boost::posix_time::millisec( _epoch_time_ms % SECONDS_TO_MILLISECONDS);
24 sdsm_timestamp.year = posix_time.date().year();
25 sdsm_timestamp.month = posix_time.date().month();
26 sdsm_timestamp.day = posix_time.date().day();
27
28 sdsm_timestamp.hour = (unsigned int) posix_time.time_of_day().hours();
29 sdsm_timestamp.minute = (unsigned int) posix_time.time_of_day().minutes();
30 // Milliseconds of the current minute. The SDSM field is named seconds but is in the unit of milliseconds (see DDateTime from J2735).
31 // Fractional_seconds returns microseconds from the current second since default time resolution is microseconds
32 sdsm_timestamp.second = (unsigned int) (posix_time.time_of_day().seconds()*SECONDS_TO_MILLISECONDS + posix_time.time_of_day().fractional_seconds()/MILLISECONDS_TO_MICROSECONDS);
33 return sdsm_timestamp;
34 }

36 streets_utils::messages::sdsm::detected_object_data to_detected_object_data(const streets_utils::messages::detected_objects_msg::detected_objects_msg &msg) {
37 streets_utils::messages::sdsm::detected_object_data detected_object;
38 detected_object._detected_object_common_data._object_type = to_object_type(msg._type);
39 if (detected_object._detected_object_common_data._object_type == streets_utils::messages::sdsm::object_type::VEHICLE ) {
40 streets_utils::messages::sdsm::detected_vehicle_data optional_data;
41 // Size in cm
42 streets_utils::messages::sdsm::vehicle_size veh_size;
43 veh_size._length= static_cast<unsigned int>(msg._size._length*METERS_TO_CM);
44 veh_size._width= static_cast<unsigned int>(msg._size._width*METERS_TO_CM);
45 optional_data._size = veh_size;
46 // Height in 5 cm
47 optional_data._vehicle_height = static_cast<unsigned int>(msg._size._height * METERS_TO_5_CM);
48
49
50 detected_object._detected_object_optional_data = optional_data;
51 }
52 else if ( detected_object._detected_object_common_data._object_type == streets_utils::messages::sdsm::object_type::VRU ) {
53 streets_utils::messages::sdsm::detected_vru_data optional_data;
54 // Populate Optional VRU data
55 detected_object._detected_object_optional_data = optional_data;
56 }
57 else if (detected_object._detected_object_common_data._object_type == streets_utils::messages::sdsm::object_type::UNKNOWN ){
58 streets_utils::messages::sdsm::detected_obstacle_data optional_data;
59 // size dimensions in units of 0.1 m
60 streets_utils::messages::sdsm::obstacle_size obs_size;
61 obs_size._length = static_cast<unsigned int>(msg._size._length*METERS_TO_10_CM);
62 obs_size._width = static_cast<unsigned int>(msg._size._width*METERS_TO_10_CM);
63 obs_size._height = static_cast<unsigned int>(msg._size._height*METERS_TO_10_CM);
64 optional_data._size = obs_size;
65
66 detected_object._detected_object_optional_data = optional_data;
67
68 }
69 detected_object._detected_object_common_data._classification_confidence = static_cast<unsigned int>(msg._confidence*100);
70 // TODO: Change Detected Object ID to int
71 detected_object._detected_object_common_data._object_id = std::stoi(msg._object_id);
72 // Units are 0.1 m
73 detected_object._detected_object_common_data._position_offset._offset_x = static_cast<unsigned int>(msg._position._x*METERS_TO_10_CM);
74 detected_object._detected_object_common_data._position_offset._offset_y = static_cast<unsigned int>(msg._position._y*METERS_TO_10_CM);
75 detected_object._detected_object_common_data._position_offset._offset_z = static_cast<unsigned int>(msg._position._z*METERS_TO_10_CM);
76 // Units are 0.02 m/s
77 detected_object._detected_object_common_data._speed = static_cast<unsigned int>(std::hypot( msg._velocity._x* METERS_PER_SECOND_TO_2_CM_PER_SECOND, msg._velocity._y* METERS_PER_SECOND_TO_2_CM_PER_SECOND));
78 detected_object._detected_object_common_data._speed_z = static_cast<unsigned int>(msg._velocity._z* METERS_PER_SECOND_TO_2_CM_PER_SECOND);
79 return detected_object;
80 }
