68 }

71 SPDLOG_WARN("The list of possible desired phase plan is empty.");
72 }

102 float streets_desired_phase_plan_arbitrator::calculate_delay_measure(
103 const std::shared_ptr<streets_vehicle_scheduler::signalized_intersection_schedule> schedule_ptr,
104 const streets_desired_phase_plan::streets_desired_phase_plan &candidate_dpp) const
105 {
106 if (schedule_ptr->vehicle_schedules.empty())
107 {
108 throw(streets_desired_phase_plan_arbitrator_exception("Vehicle schedule cannot be empty. Vehicles maybe outside of the radius."));
109 }
110 u_int64_t TBD_delay = 0;
111 u_int64_t candidate_vehicles_delay = 0;
112 for (const auto &veh_schedule : schedule_ptr->vehicle_schedules)
113 {
114 // If find the vehicles’ schedule ET > candidate desired phase plan end time, these vehicles are considered TBD.
115 if (veh_schedule.et > candidate_dpp.desired_phase_plan.back().end_time)
116 {
117 TBD_delay += veh_schedule.get_delay();
118 SPDLOG_DEBUG("TBD veh_schedule v_ID= {0}, get_delay = {1}", veh_schedule.v_id, veh_schedule.get_delay());
119 }
120 // If find the vehicles’ schedule ET <= candidate desired phase plan end time and the vehicles’ schedule ET => candidate desired phase plan start time.
121 else if (veh_schedule.et >= candidate_dpp.desired_phase_plan.back().start_time)
122 {
123 candidate_vehicles_delay += veh_schedule.get_delay();
124 SPDLOG_DEBUG("candidate veh_schedule v_ID= {0}, get_delay = {1}", veh_schedule.v_id, veh_schedule.get_delay());
125 }
126 }
127
128 // The minimum value of the calculated delay can be 1 millisecond.
129 float delay_measure = (float)std::max(candidate_vehicles_delay, u_int64_t(1)) / (float)std::max(TBD_delay, u_int64_t(1));
130
131 SPDLOG_DEBUG("calculated delay_measure (= candidate/TBD) = {0}", delay_measure);
132
133 if (enable_so_logging) {
134 auto logger = spdlog::get("so_csv_logger");
135 SPDLOG_DEBUG("getting so_csv_logger!");
136 if ( logger != nullptr ) {
137 SPDLOG_DEBUG("logging info: {0} ", dpp_delay_toCSV(candidate_dpp, candidate_vehicles_delay, TBD_delay, delay_measure));
138 logger->info( dpp_delay_toCSV(candidate_dpp, candidate_vehicles_delay, TBD_delay, delay_measure) );
139 }
140 }
141 else {
142 SPDLOG_WARN("so_csv_logger in arbitrator pointer is not enabled!");
143 }
144
145 return delay_measure;
146 }

148 streets_desired_phase_plan::streets_desired_phase_plan streets_desired_phase_plan_arbitrator::identify_ddp_by_delay_measures(
149 const std::vector<streets_desired_phase_plan::streets_desired_phase_plan> &dpp_list,
150 const std::unordered_map<int, float> &ddp_index_delay_measure_mappings) const
151 {
152 streets_desired_phase_plan::streets_desired_phase_plan dpp_result;
153 if (!ddp_index_delay_measure_mappings.empty())
154 {
155 int dpp_index = ddp_index_delay_measure_mappings.begin()->first;
156 float max_delay_measure = ddp_index_delay_measure_mappings.begin()->second;
157
158 // Find the highest delay_measure value, and return its desired phase plan index
159 for (const auto &[cur_dpp_index, delay_measure] : ddp_index_delay_measure_mappings)
160 {
161 SPDLOG_DEBUG("Curent Desired phase plan = {0} , delay measure = {1}", dpp_list.at(dpp_index).toJson(), delay_measure);
162 if (max_delay_measure < delay_measure)
163 {
164 max_delay_measure = delay_measure;
165 dpp_index = cur_dpp_index;
166 }
167 }
168 dpp_result = dpp_list.at(dpp_index);
169 SPDLOG_DEBUG("Curent Desired phase plan = {0} , MAX delay measure = {1}", dpp_list.at(dpp_index).toJson(), max_delay_measure);
170 }
171
172 return dpp_result;
173 }
