102 getLanes(oLanes);
103 Collections.sort(oLanes, LEFTLANES);
104
105 double[] dEdgeR = oLanes.get(0).m_dOuter;
106 double[] dEdgeL = oLanes.get(oLanes.size() - 1).m_dOuter;
107 double[] dPavementCenter = Arrays.newDoubleArray(Arrays.size(dEdgeL));
108 dPavementCenter = Arrays.add(dPavementCenter, new double[] {Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE});
109
110 int nLimit = Arrays.size(dEdgeR);
111 for (int nIndex = 5; nIndex < nLimit; nIndex += 2)
112 {
113 double dXr = dEdgeR[nIndex];
114 double dYr = dEdgeR[nIndex + 1];
115 double dXl = dEdgeL[nIndex];
116 double dYl = dEdgeL[nIndex + 1];
117
118 double dPathX = (dXr + dXl) / 2;
119 double dPathY = (dYr + dYl) / 2;
120 double dW = Geo.distance(dXr, dYr, dXl, dYl);
121 dPavementCenter = Arrays.addAndUpdate(dPavementCenter, dPathX, dPathY);
122 dPavementCenter = Arrays.add(dPavementCenter, dW);
123 }
124
125 m_dPavement = dPavementCenter; // save local reference
126 }
