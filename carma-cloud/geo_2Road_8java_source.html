55 public void adjustSections(double dMaxStep)
56 {
57 int nIndex = size();

59 return;
60
61 double dPrevOffset = m_dLength;
62 while (nIndex-- > 0)
63 {

66 remove(nIndex);

68 dPrevOffset = get(nIndex).m_dS;
69 }
70
71 if (get(0).m_dS != 0.0)

86 public void setLaneIds(double dMaxStep)
87 {
88 double dTol = dMaxStep * 2;
89 ArrayList<Lane> oPrevLanes = new ArrayList();
90 get(0).getLanes(oPrevLanes);
91 int nNextId = Integer.MIN_VALUE;
92 for (Lane oLane : oPrevLanes)
93 {
94 oLane.m_nLaneIdByRoad = oLane.m_nLaneIndex;
95 int nComp = Math.abs(oLane.m_nLaneIndex);
96 if (nComp > nNextId)
97 nNextId = nComp;
98 }
99
100 for (int nIndex = 1; nIndex < size(); nIndex++)
101 {
102 ArrayList<Lane> oCurrLanes = new ArrayList();
103 get(nIndex).getLanes(oCurrLanes);
104 for (Lane oCurr: oCurrLanes)
105 {
106 if (oCurr.m_nLaneIndex == 0)
107 continue;
108 double[] dCurr = oCurr.m_dCenter;
109 oCurr.m_nLaneIdByRoad = Integer.MIN_VALUE;
110 for (Lane oPrev: oPrevLanes)
111 {
112 if (oPrev.m_nLaneIndex == 0)
113 continue;
114 double[] dPrev = oPrev.m_dCenter;
115 int nSize = Arrays.size(dPrev);
116
117 double dDist = Geo.distance(dCurr[5], dCurr[6], dPrev[nSize - 3], dPrev[nSize - 2]);
118// if (dDist > dTol)
119// System.out.println(dDist);
120 if (dDist <= dTol && oCurr.m_nLaneType == oPrev.m_nLaneType)
121 {
122 oCurr.m_nLaneIdByRoad = oPrev.m_nLaneIdByRoad;
123 break;
124 }
125 }
126
127 if (oCurr.m_nLaneIdByRoad == Integer.MIN_VALUE)
128 {
129 ++nNextId;
130 if (oCurr.m_nLaneIndex < 0)
131 oCurr.m_nLaneIdByRoad = -nNextId;
132 else
133 oCurr.m_nLaneIdByRoad = nNextId;
134 }
135 }
136 oPrevLanes.clear();
137 oPrevLanes.addAll(oCurrLanes);
138 }
