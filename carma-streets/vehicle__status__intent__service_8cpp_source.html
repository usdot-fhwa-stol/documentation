265 vsi.setVehicle_length(bsm.getCore_data().size.length);
266 vsi.setCur_speed(bsm.getCore_data().speed * 0.02); //convert to mps
267 vsi.setCur_accel(bsm.getCore_data().accelSet.Long * 0.01); //convert to mps2
268 std::string turn_direction = mo.get_value_from_strategy_params("turn_direction");

270 double cur_lat = bsm.getCore_data().latitude / 10000000;
271 double cur_lon = bsm.getCore_data().longitude / 10000000;
272 double cur_elev = bsm.getCore_data().elev;
273 SPDLOG_DEBUG("cur_lat = {0}", cur_lat);
274 SPDLOG_DEBUG("cur_lon = {0}", cur_lon);
275 SPDLOG_DEBUG("cur_elev = {0}", cur_elev);
276
277 SPDLOG_DEBUG("MobilityPath trajectory offset size: {0}", mp.getTrajectory().offsets.size());
278 message_services::models::trajectory trajectory = mp.getTrajectory();
279 lanelet::BasicPoint3d cur_basic_point3d = _msg_lanelet2_translate_ptr->gps_2_map_point(cur_lat, cur_lon, cur_elev);
280 lanelet::Lanelet cur_lanelet = _msg_lanelet2_translate_ptr->get_cur_lanelet_by_point_and_direction(cur_basic_point3d, turn_direction, trajectory);
281 vsi.setCur_lanelet_id(cur_lanelet.id());
282 vsi.setCur_distance(_msg_lanelet2_translate_ptr->distance2_cur_lanelet_end(cur_basic_point3d, cur_lanelet, turn_direction, trajectory));
283
284 if(!disable_est_path)
285 {
286 // Update vehicle status intent with MobilityPath
287 models::est_path_t est_path;
288 std::vector<models::est_path_t> est_path_v;
289 int32_t ecef_x = mp.getTrajectory().location.ecef_x;
290 int32_t ecef_y = mp.getTrajectory().location.ecef_y;
291 int32_t ecef_z = mp.getTrajectory().location.ecef_z;
292 long timestamp = mp.getHeader().timestamp;
293
294 SPDLOG_DEBUG("MobilityPath location ecef_x: {0}", ecef_x);
295 SPDLOG_DEBUG("MobilityPath location ecef_y: {0}", ecef_y);
296 SPDLOG_DEBUG("MobilityPath location ecef_z: {0}", ecef_z);

302 }

304 {

307 est_path.lanelet_id = mp_point_lanelet.id();
308 }
309 est_path.timestamp = timestamp;
310 est_path_v.push_back(est_path);
311
312 int32_t count = 1;
313 size_t next_index = 0;
314 double accumulated_distance_to_previous_point = 0;
315 lanelet::BasicPoint3d mp_cur_point = mp_start_point;
316 for (size_t offset_index = 0; offset_index < trajectory.offsets.size(); offset_index++)
317 {
318 ecef_x += trajectory.offsets.at(offset_index).offset_x;
319 ecef_y += trajectory.offsets.at(offset_index).offset_y;
320 ecef_z += trajectory.offsets.at(offset_index).offset_z;
321
322 SPDLOG_DEBUG("MobilityPath location offset_x: {0}", trajectory.offsets.at(offset_index).offset_x);
323 SPDLOG_DEBUG("MobilityPath location offset_y: {0}", trajectory.offsets.at(offset_index).offset_y);
324 SPDLOG_DEBUG("MobilityPath location offset_z: {0}", trajectory.offsets.at(offset_index).offset_z);
325 est_path.timestamp += 100; // The duration between two points is 0.1 sec
326
327 //Calculate the distance between two points (interval 0.1 secs) from the MobilityPath message starting from the current vehicle location
328 if(is_est_path_p2p_distance_only)
329 {
330 lanelet::BasicPoint3d mp_previous_point = mp_cur_point;
331 mp_cur_point = _msg_lanelet2_translate_ptr->ecef_2_map_point(ecef_x, ecef_y, ecef_z);

333 }
334
335 if (next_index != offset_index)
336 {
337 continue;
338 }
339 next_index += this->MOBILITY_PATH_TRAJECTORY_OFFSET_DURATION;

341 // Skip the first point
342 if (offset_index == 0)

348 if(is_est_path_p2p_distance_only)

365 count++;
366
367 // Allow to configure the number of mobilityPath offsets sent as part of VSI (vehicle status and intent)
368 if (this->vsi_est_path_point_count != 0 && count > this->vsi_est_path_point_count)
369 {
370 break;
371 }
372 }
373
374 vsi.setEst_path_v(est_path_v);
375 }
376
377 std::map<int64_t, models::intersection_lanelet_type> lanelet_id_type_m = _msg_lanelet2_translate_ptr->get_lanelet_types_ids(cur_lanelet, turn_direction);
378 for (auto itr = lanelet_id_type_m.begin(); itr != lanelet_id_type_m.end(); itr++)
379 {
380 if (itr->second == models::intersection_lanelet_type::link)
381 {
382 vsi.setLink_lanelet_id(itr->first);

384 if (itr->second == models::intersection_lanelet_type::departure)

386 vsi.setDest_lanelet_id(itr->first);
387 }
388 if (itr->second == models::intersection_lanelet_type::entry)

392 }
393 if (lanelet_id_type_m.empty() ) {

395 }
396 return vsi;

398 catch (const lanelet::LaneletError &e)
