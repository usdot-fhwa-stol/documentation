131 {
132 const char *return_msg_str = "";
133 switch (message->err())
134 {
135 case RdKafka::ERR__TIMED_OUT:
136 break;
137 case RdKafka::ERR_NO_ERROR:

140 _last_offset = message->offset();
141 return_msg_str = static_cast<const char *>(message->payload());
142 break;
143 case RdKafka::ERR__PARTITION_EOF:
144 SPDLOG_TRACE("{0} Reached the end of the queue, offset : {1}", _consumer->name(), _last_offset);
145 break;
146 case RdKafka::ERR__UNKNOWN_TOPIC:
147 case RdKafka::ERR__UNKNOWN_PARTITION:
148 SPDLOG_CRITICAL("{0} Consume failed: {1} ", _consumer->name(), message->errstr());
149 stop();
150 break;
151
152 default:
153 /* Errors */
154 SPDLOG_CRITICAL("{0} Consume failed: {1} ", _consumer->name(), message->errstr());
155 stop();
156 break;
