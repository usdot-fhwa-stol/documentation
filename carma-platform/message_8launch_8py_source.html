35def generate_launch_description():
36 """
37 Launch V2X subsystem nodes.
38 """
39
40 env_log_levels = EnvironmentVariable('CARMA_ROS_LOGGING_CONFIG', default_value='{ "default_level" : "WARN" }')
41
42 subsystem_controller_default_param_file = os.path.join(
43 get_package_share_directory('subsystem_controllers'), 'config/v2x_controller_config.yaml')
44
45 mobilitypath_publisher_param_file = os.path.join(
46 get_package_share_directory('mobilitypath_publisher'), 'config/parameters.yaml')
47
48 bsm_generator_param_file = os.path.join(
49 get_package_share_directory('bsm_generator'), 'config/parameters.yaml')
50
51 vehicle_characteristics_param_file = LaunchConfiguration('vehicle_characteristics_param_file')
52 declare_vehicle_characteristics_param_file_arg = DeclareLaunchArgument(
53 name = 'vehicle_characteristics_param_file',
54 default_value = "/opt/carma/vehicle/calibration/identifiers/UniqueVehicleParams.yaml",
55 description = "Path to file containing unique vehicle calibrations"
56 )
57
58
59 vehicle_config_param_file = LaunchConfiguration('vehicle_config_param_file')
60 declare_vehicle_config_param_file_arg = DeclareLaunchArgument(
61 name = 'vehicle_config_param_file',
62 default_value = "/opt/carma/vehicle/config/VehicleConfigParams.yaml",
63 description = "Path to file contain vehicle configuration parameters"
64 )
65
66 subsystem_controller_param_file = LaunchConfiguration('subsystem_controller_param_file')
67 declare_subsystem_controller_param_file_arg = DeclareLaunchArgument(
68 name = 'subsystem_controller_param_file',
69 default_value = subsystem_controller_default_param_file,
70 description = "Path to file containing override parameters for the subsystem controller"
71 )
72
73
74 # Nodes
75 carma_v2x_container = ComposableNodeContainer(
76 package='carma_ros2_utils',
77 name='carma_v2x_container',
78 executable='carma_component_container_mt',
79 namespace=GetCurrentNamespace(),
80 composable_node_descriptions=[
81
82 ComposableNode(
83 package='mobilitypath_publisher',
84 plugin='mobilitypath_publisher::MobilityPathPublication',

88 {'--log-level' : GetLogLevel('mobilitypath_publisher', env_log_levels) }

90 remappings=[

92 ("georeference", [ EnvironmentVariable('CARMA_LOCZ_NS', default_value=''), "/map_param_loader/georeference" ] ),
93 ("mobility_path_msg", [ EnvironmentVariable('CARMA_MSG_NS', default_value=''), "/outgoing_mobility_path" ] )
94 ],
95 parameters=[
96 mobilitypath_publisher_param_file,
97 vehicle_characteristics_param_file
98 ]
99 ),
100 ComposableNode(
101 package='bsm_generator',
102 plugin='bsm_generator::BSMGenerator',
103 name='bsm_generator_node',
104 extra_arguments=[
105 {'use_intra_process_comms': True},
106 {'--log-level' : GetLogLevel('bsm_generator', env_log_levels) }

111 ("imu_raw", [ EnvironmentVariable('CARMA_INTR_NS', default_value=''), "/imu_raw" ] ),
112 ("transmission_state", [ EnvironmentVariable('CARMA_INTR_NS', default_value=''), "/can/transmission_state" ] ),
113 ("brake_position", [ EnvironmentVariable('CARMA_INTR_NS', default_value=''), "/can/brake_position" ] ),
114 ("steering_wheel_angle", [ EnvironmentVariable('CARMA_INTR_NS', default_value=''), "/can/steering_wheel_angle" ] ),
115 ("gnss_fix_fused", [ EnvironmentVariable('CARMA_INTR_NS', default_value=''), "/gnss_fix_fused" ] ),

153 # subsystem_controller which orchestrates the lifecycle of this subsystem's components
154 subsystem_controller = Node(
155 package='subsystem_controllers',
156 name='v2x_controller',
157 executable='v2x_controller',
158 parameters=[ subsystem_controller_default_param_file, subsystem_controller_param_file ], # Default file is loaded first followed by config file
159 on_exit= Shutdown(), # Mark the subsystem controller as required
160 arguments=['--ros-args', '--log-level', GetLogLevel('subsystem_controllers', env_log_levels)]

163 return LaunchDescription([
164 declare_vehicle_config_param_file_arg,

168 subsystem_controller
169 ])
