49 void sds_service::consume_detections(){
50 if ( !detection_consumer ) {
51 throw std::runtime_error("Detection consumer is null!");
52 }
53 SPDLOG_DEBUG("Attempting to consume detections ...");
54 detection_consumer->subscribe();
55 while ( detection_consumer->is_running() ) {
56 try{
57 const std::string payload = detection_consumer->consume(1000);
58 if (payload.length() != 0)
59 {
60 auto detected_object = streets_utils::messages::detected_objects_msg::from_json(payload);
61 std::unique_lock lock(detected_objects_lock);
62 detected_objects[detected_object._object_id] = detected_object;
63 SPDLOG_DEBUG("Detected Object List Size {0} after consumed: {1}", detected_objects.size(), payload);
64
65 }
66 }
67 catch (const streets_utils::json_utils::json_parse_exception &e) {
68 SPDLOG_ERROR("Exception occured consuming detection message : {0}", e.what());
69 }
70 }
71 SPDLOG_ERROR("Something went wrong, no longer consuming detections." );
72
73 }

75 void sds_service::produce_sdsms() {
76 if ( !sdsm_producer ) {
77 throw std::runtime_error("SDSM consumer is null!");
78 }
79 SPDLOG_INFO("Starting SDSM Producer!");
80 while ( sdsm_producer->is_running() ) {
81 try{
82 if ( !detected_objects.empty() ) {
83 std::unique_lock lock(detected_objects_lock);
84 streets_utils::messages::sdsm::sensor_data_sharing_msg msg;
85 // TODO: Populate SDSM with detected objects
86 const std::string json_msg = streets_utils::messages::sdsm::to_json(msg);
87 SPDLOG_DEBUG("Sending SDSM : {0}", json_msg);
88 sdsm_producer->send(json_msg);
89 // Clear detected object
90 detected_objects.clear();
91 }
92 }
93 catch( const streets_utils::json_utils::json_parse_exception &e) {
94 SPDLOG_ERROR("Exception occurred producing SDSM : {0}", e.what());
95 }
96 ss::streets_clock_singleton::sleep_for(1000); // Sleep for 10 second between publish
97 }
98
99 }
