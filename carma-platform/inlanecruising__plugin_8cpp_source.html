49void InLaneCruisingPlugin::plan_trajectory_callback(
50 carma_planning_msgs::srv::PlanTrajectory::Request::SharedPtr req,
51 carma_planning_msgs::srv::PlanTrajectory::Response::SharedPtr resp)
52{
53 std::chrono::system_clock::time_point start_time = std::chrono::system_clock::now(); // Start timing the execution time for planning so it can be logged
54
55 lanelet::BasicPoint2d veh_pos(req->vehicle_state.x_pos_global, req->vehicle_state.y_pos_global);
56 double current_downtrack = wm_->routeTrackPos(veh_pos).downtrack;
57
58 // Only plan the trajectory for the initial LANE_FOLLOWING maneuver and any immediately sequential maneuvers of the same type
59 std::vector<carma_planning_msgs::msg::Maneuver> maneuver_plan;
60 for(size_t i = req->maneuver_index_to_plan; i < req->maneuver_plan.maneuvers.size(); i++)
61 {
62 if(req->maneuver_plan.maneuvers[i].type == carma_planning_msgs::msg::Maneuver::LANE_FOLLOWING)
63 {
64 maneuver_plan.push_back(req->maneuver_plan.maneuvers[i]);
65 resp->related_maneuvers.push_back((uint8_t)i);
66 }
67 else
68 {
69 break;
70 }
71 }
72
73 basic_autonomy:: waypoint_generation::DetailedTrajConfig wpg_detail_config;
74 basic_autonomy:: waypoint_generation::GeneralTrajConfig wpg_general_config;
75
76 wpg_general_config = basic_autonomy:: waypoint_generation::compose_general_trajectory_config("inlanecruising",
77 config_.default_downsample_ratio,
78 config_.turn_downsample_ratio);
79
80 wpg_detail_config = basic_autonomy:: waypoint_generation::compose_detailed_trajectory_config(config_.trajectory_time_length,
81 config_.curve_resample_step_size, config_.minimum_speed,
82 config_.max_accel * config_.max_accel_multiplier,
83 config_.lateral_accel_limit * config_.lat_accel_multiplier,
84 config_.speed_moving_average_window_size,
85 config_.curvature_moving_average_window_size, config_.back_distance,
86 config_.buffer_ending_downtrack);
87
88 auto points_and_target_speeds = basic_autonomy::waypoint_generation::create_geometry_profile(maneuver_plan, std::max((double)0, current_downtrack - config_.back_distance),
89 wm_, ending_state_before_buffer_, req->vehicle_state, wpg_general_config, wpg_detail_config);
90
91 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "points_and_target_speeds: " << points_and_target_speeds.size());
92
93 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "PlanTrajectory");
94
95 carma_planning_msgs::msg::TrajectoryPlan original_trajectory;
96 original_trajectory.header.frame_id = "map";
97 original_trajectory.header.stamp = nh_->now();
98 original_trajectory.trajectory_id = boost::uuids::to_string(boost::uuids::random_generator()());
99
100 original_trajectory.trajectory_points = basic_autonomy:: waypoint_generation::compose_lanefollow_trajectory_from_path(points_and_target_speeds,
101 req->vehicle_state, req->header.stamp, wm_, ending_state_before_buffer_, debug_msg_,
102 wpg_detail_config); // Compute the trajectory
103 original_trajectory.initial_longitudinal_velocity = std::max(req->vehicle_state.longitudinal_vel, config_.minimum_speed);
104
105 // Set the planning plugin field name
106 for (auto& p : original_trajectory.trajectory_points) {
107 p.planner_plugin_name = plugin_name_;
108 }
109
110 if (config_.enable_object_avoidance)
111 {
112 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Activate Object Avoidance");
113
114 if (yield_client_ && yield_client_->service_is_ready())
115 {
116 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Yield Client is valid");
117
118 auto yield_srv = std::make_shared<carma_planning_msgs::srv::PlanTrajectory::Request>();
119 yield_srv->initial_trajectory_plan = original_trajectory;
120 yield_srv->vehicle_state = req->vehicle_state;
121
122 auto yield_resp = yield_client_->async_send_request(yield_srv);
123
124 auto future_status = yield_resp.wait_for(std::chrono::milliseconds(100));
125
126 if (future_status == std::future_status::ready)
127 {
128 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Received Traj from Yield");
129 carma_planning_msgs::msg::TrajectoryPlan yield_plan = yield_resp.get()->trajectory_plan;
130 if (validate_yield_plan(yield_plan))
131 {
132 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Yield trajectory validated");
133 resp->trajectory_plan = yield_plan;
134 }
135 else
136 {
137 throw std::invalid_argument("Invalid Yield Trajectory");
138 }
139 }
140 else
141 {
142 throw std::invalid_argument("Unable to Call Yield Plugin");
143 }
144 }
145 else
146 {
147 throw std::invalid_argument("Yield Client is unavailable");
148 }
149
150 }
151 else
152 {
153 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Ignored Object Avoidance");
154 resp->trajectory_plan = original_trajectory;
155 }
156
157 if (config_.publish_debug) { // Publish the debug message if in debug logging mode
158 debug_msg_.trajectory_plan = resp->trajectory_plan;
159 debug_publisher_(debug_msg_);
160 }
161
162 resp->maneuver_status.push_back(carma_planning_msgs::srv::PlanTrajectory::Response::MANEUVER_IN_PROGRESS);
163
164 std::chrono::system_clock::time_point end_time = std::chrono::system_clock::now(); // Planning complete
165
166 auto duration = end_time - start_time;
167 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "ExecutionTime: " << std::chrono::duration<double>(duration).count());
168}

175bool InLaneCruisingPlugin::validate_yield_plan(const carma_planning_msgs::msg::TrajectoryPlan& yield_plan) const
176{
177 if (yield_plan.trajectory_points.size()>= 2)
178 {
179 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Yield Trajectory Time" << rclcpp::Time(yield_plan.trajectory_points[0].target_time).seconds());
180 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Now:" << nh_->now().seconds());
181 if (rclcpp::Time(yield_plan.trajectory_points[0].target_time) + rclcpp::Duration(5.0, 0) > nh_->now())
182 {
183 return true;
184 }
185 else
186 {
187 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Old Yield Trajectory");
188 }
189 }
190 else
191 {
192 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "Invalid Yield Trajectory");
193 }
194 return false;
195}
