29 void LightControlledIntersectionTacticalPlugin::planTrajectoryCB(
30 carma_planning_msgs::srv::PlanTrajectory::Request::SharedPtr req,
31 carma_planning_msgs::srv::PlanTrajectory::Response::SharedPtr resp)
32 {
33 std::chrono::system_clock::time_point start_time = std::chrono::system_clock::now(); // Start timing the execution time for planning so it can be logged
34
35 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Starting light controlled intersection trajectory planning");
36
37 if(req->maneuver_index_to_plan >= req->maneuver_plan.maneuvers.size())
38 {
39 throw std::invalid_argument(
40 "Light Control Intersection Plugin asked to plan invalid maneuver index: " + std::to_string(req->maneuver_index_to_plan) +
41 " for plan of size: " + std::to_string(req->maneuver_plan.maneuvers.size()));
42 }
43
44 std::vector<carma_planning_msgs::msg::Maneuver> maneuver_plan;
45 // Expecting only one maneuver for an intersection, which corresponds to the maneuver_index_to_plan value provided in the request
46 if(req->maneuver_plan.maneuvers[req->maneuver_index_to_plan].type == carma_planning_msgs::msg::Maneuver::LANE_FOLLOWING
47 && GET_MANEUVER_PROPERTY(req->maneuver_plan.maneuvers[req->maneuver_index_to_plan], parameters.string_valued_meta_data.front()) == light_controlled_intersection_strategy_)
48 {
49 maneuver_plan.push_back(req->maneuver_plan.maneuvers[req->maneuver_index_to_plan]);
50 resp->related_maneuvers.push_back(req->maneuver_index_to_plan);
51 }
52 else
53 {
54 throw std::invalid_argument("Light Control Intersection Plugin asked to plan unsupported maneuver");
55 }
56
57 lanelet::BasicPoint2d veh_pos(req->vehicle_state.x_pos_global, req->vehicle_state.y_pos_global);
58 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Planning state x:" << req->vehicle_state.x_pos_global << " , y: " << req->vehicle_state.y_pos_global);
59
60 current_downtrack_ = wm_->routeTrackPos(veh_pos).downtrack;
61
62 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Current_downtrack: "<< current_downtrack_);
63
64 auto current_lanelets = wm_->getLaneletsFromPoint({req->vehicle_state.x_pos_global, req->vehicle_state.y_pos_global});
65 lanelet::ConstLanelet current_lanelet;
66
67 if (current_lanelets.empty())
68 {
69 RCLCPP_ERROR_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Given vehicle position is not on the road! Returning...");
70 return;
71 }
72
73 // get the lanelet that is on the route in case overlapping ones found
74 for (auto llt : current_lanelets)
75 {
76 auto route = wm_->getRoute()->shortestPath();
77 if (std::find(route.begin(), route.end(), llt) != route.end())
78 {
79 current_lanelet = llt;
80 break;
81 }
82 }
83
84 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Current_lanelet: " << current_lanelet.id());
85
86 speed_limit_ = findSpeedLimit(current_lanelet, wm_);
87
88 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "speed_limit_: " << speed_limit_);
89
90 DetailedTrajConfig wpg_detail_config;
91 GeneralTrajConfig wpg_general_config;
92
93 wpg_general_config = basic_autonomy:: waypoint_generation::compose_general_trajectory_config("intersection_transit",
94 config_.default_downsample_ratio,
95 config_.turn_downsample_ratio);
96
97 wpg_detail_config = basic_autonomy:: waypoint_generation::compose_detailed_trajectory_config(config_.trajectory_time_length,
98 config_.curve_resample_step_size, config_.minimum_speed,
99 config_.vehicle_accel_limit,
100 config_.lateral_accel_limit,
101 config_.speed_moving_average_window_size,
102 config_.curvature_moving_average_window_size, config_.back_distance,
103 config_.buffer_ending_downtrack);
104
105 // CHECK IF LAST TRAJECTORY WILL BE USED
106
107 bool is_new_case_successful = GET_MANEUVER_PROPERTY(maneuver_plan.front(), parameters.int_valued_meta_data[1]);
108 TSCase new_case = static_cast<TSCase>GET_MANEUVER_PROPERTY(maneuver_plan.front(), parameters.int_valued_meta_data[0]);
109
110 if (is_last_case_successful_ != boost::none && last_case_ != boost::none)
111 {
112 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "all variables are set!");
113 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "is_last_case_successful_.get(): " << (int)is_last_case_successful_.get());
114 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "evaluation distance: " << last_successful_ending_downtrack_ - current_downtrack_);
115
116 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "evaluation time: " << std::to_string(last_successful_scheduled_entry_time_ - rclcpp::Time(req->header.stamp).seconds()));
117 }
118 else
119 {
120 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Not all variables are set...");
121 }
122
123
124
125 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "traj points size: " << last_trajectory_.trajectory_points.size() << ", last_final_speeds_ size: " <<
126 last_final_speeds_.size() );
127
128 size_t idx_to_start_new_traj = 0;
129 for (size_t i = 0; i < last_trajectory_.trajectory_points.size(); i++)
130 {
131 auto dist = sqrt(pow(veh_pos.x() - last_trajectory_.trajectory_points.at(i).x, 2) + std::pow(veh_pos.y() - last_trajectory_.trajectory_points.at(i).y, 2));
132
133 if (dist < 2.0) // reduce until if 2 meters away
134 {
135 idx_to_start_new_traj = i;
136 break;
137 }
138 }
139
140 last_final_speeds_ = std::vector<double>(last_final_speeds_.begin() + idx_to_start_new_traj, last_final_speeds_.end());
141 last_trajectory_.trajectory_points = std::vector<carma_planning_msgs::msg::TrajectoryPlanPoint>(last_trajectory_.trajectory_points.begin() + idx_to_start_new_traj, last_trajectory_.trajectory_points.end());
142
143 if (is_last_case_successful_ != boost::none && last_case_ != boost::none
144 && last_case_.get() == new_case
145 && is_new_case_successful == true
146 && last_trajectory_.trajectory_points.size() >= 2
147 && rclcpp::Time(last_trajectory_.trajectory_points.back().target_time) > rclcpp::Time(req->header.stamp) + rclcpp::Duration(1 * 1e9)) // Duration is in nanoseconds
148 {
149 resp->trajectory_plan = last_trajectory_;
150 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Last TRAJ's target time: " << std::to_string(rclcpp::Time(last_trajectory_.trajectory_points.back().target_time).seconds()) << ", and stamp:" << std::to_string(rclcpp::Time(req->header.stamp).seconds()));
151 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "USING LAST TRAJ: " << (int)last_case_.get());
152 }
153 else if (is_last_case_successful_ != boost::none && last_case_ != boost::none
154 && is_last_case_successful_.get() == true
155 && is_new_case_successful == false
156 && last_successful_ending_downtrack_ - current_downtrack_ < config_.algorithm_evaluation_distance
157 && last_successful_scheduled_entry_time_ - rclcpp::Time(req->header.stamp).seconds() < config_.algorithm_evaluation_period
158 && last_trajectory_.trajectory_points.size() >= 2
159 && rclcpp::Time(last_trajectory_.trajectory_points.back().target_time).seconds() > rclcpp::Time(req->header.stamp).seconds())
160 {
161 resp->trajectory_plan = last_trajectory_;
162 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Last Traj's target time: " << std::to_string(rclcpp::Time(last_trajectory_.trajectory_points.back().target_time).seconds()) << ", and stamp:" << std::to_string(rclcpp::Time(req->header.stamp).seconds()) << ", and scheduled: " << std::to_string(last_successful_scheduled_entry_time_));
163 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "EDGE CASE: USING LAST TRAJ: " << (int)last_case_.get());
164 }
165
166 if (!resp->trajectory_plan.trajectory_points.empty()) // if has valid trajectory saved from before return
167 {
168 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Debug: new case:" << (int) new_case << ", is_new_case_successful: " << is_new_case_successful);
169
170 resp->trajectory_plan.initial_longitudinal_velocity = last_final_speeds_.front();
171
172 resp->maneuver_status.push_back(carma_planning_msgs::srv::PlanTrajectory::Response::MANEUVER_IN_PROGRESS);
173
174 std::chrono::system_clock::time_point end_time = std::chrono::system_clock::now(); // Planning complete
175
176 auto duration = end_time - start_time;
177 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "ExecutionTime Using New: " << std::chrono::duration<double>(duration).count());
178
179 return;
180 }
181
182 // IF NOT USING LAST TRAJECTORY PLAN NEW TRAJECTORY
183
184 // Create curve-fitting compatible trajectories (with extra back and front attached points) with raw speed limits from maneuver
185 auto points_and_target_speeds = createGeometryProfile(maneuver_plan, std::max((double)0, current_downtrack_ - config_.back_distance),
186 wm_, ending_state_before_buffer_, req->vehicle_state, wpg_general_config, wpg_detail_config);
187
188 // Change raw speed limit values to target speeds specified by the algorithm
189 applyOptimizedTargetSpeedProfile(maneuver_plan.front(), req->vehicle_state.longitudinal_vel, points_and_target_speeds);
190
191 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "points_and_target_speeds: " << points_and_target_speeds.size());
192
193 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "PlanTrajectory");
194
195 //Trajectory Plan
196 carma_planning_msgs::msg::TrajectoryPlan trajectory;
197 trajectory.header.frame_id = "map";
198 trajectory.header.stamp = req->header.stamp;
199 trajectory.trajectory_id = boost::uuids::to_string(boost::uuids::random_generator()());
200
201 // Compose smooth trajectory/speed by resampling
202 trajectory.trajectory_points = basic_autonomy::waypoint_generation::compose_lanefollow_trajectory_from_path(points_and_target_speeds,
203 req->vehicle_state, req->header.stamp, wm_, ending_state_before_buffer_, debug_msg_,
204 wpg_detail_config); // Compute the trajectory
205
206 // Set the planning plugin field name
207 for (auto& p : trajectory.trajectory_points) {
208 p.planner_plugin_name = plugin_name_;
209 }
210
211 if (trajectory.trajectory_points.size () < 2)
212 {
213 if (last_trajectory_.trajectory_points.size() >= 2
214 && rclcpp::Time(last_trajectory_.trajectory_points.back().target_time) > rclcpp::Time(req->header.stamp))
215 {
216 resp->trajectory_plan = last_trajectory_;
217 RCLCPP_WARN_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Failed to generate new trajectory, so using last valid trajectory!");
218 }
219 else
220 {
221 resp->trajectory_plan = trajectory;
222 RCLCPP_WARN_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Failed to generate new trajectory or use old valid trajectory, so returning empty/invalid trajectory!");
223 }
224 }
225 else
226 {
227 last_trajectory_ = trajectory;
228 resp->trajectory_plan = trajectory;
229 last_case_ = new_case;
230 last_final_speeds_ = debug_msg_.velocity_profile;
231 is_last_case_successful_ = is_new_case_successful;
232 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "USING NEW: Target time: " << std::to_string(rclcpp::Time(last_trajectory_.trajectory_points.back().target_time).seconds()) << ", and stamp:" << std::to_string(rclcpp::Time(req->header.stamp).seconds()));
233 if (is_new_case_successful)
234 {
235 last_successful_ending_downtrack_ = GET_MANEUVER_PROPERTY(maneuver_plan.front(), end_dist); // if algorithm was successful, this is traffic_light_downtrack
236 last_successful_scheduled_entry_time_ = rclcpp::Time(GET_MANEUVER_PROPERTY(maneuver_plan.front(), end_time)).seconds(); // if algorithm was successful, this is also scheduled entry time (ET in TSMO UC2 Algo)
237 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "last_successful_ending_downtrack_:" << last_successful_ending_downtrack_ << ", last_successful_scheduled_entry_time_: " << std::to_string(last_successful_scheduled_entry_time_));
238 }
239 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "USING NEW CASE!!! : " << (int)last_case_.get());
240
241 }
242 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Debug: new case:" << (int) new_case << ", is_new_case_successful: " << is_new_case_successful);
243
244 resp->trajectory_plan.initial_longitudinal_velocity = last_final_speeds_.front();
245
246 // Yield for potential obstacles in the road
247 // Aside from the flag, yield_plugin should not be called on invalid trajectories
248 if (config_.enable_object_avoidance && resp->trajectory_plan.trajectory_points.size() >= 2)
249 {
250 basic_autonomy::waypoint_generation::modify_trajectory_to_yield_to_obstacles(nh_, req, resp, yield_client_, config_.tactical_plugin_service_call_timeout);
251 }
252 else
253 {
254 RCLCPP_DEBUG(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Ignored Object Avoidance");
255 }
256
257 resp->maneuver_status.push_back(carma_planning_msgs::srv::PlanTrajectory::Response::MANEUVER_IN_PROGRESS);
258
259 std::chrono::system_clock::time_point end_time = std::chrono::system_clock::now(); // Planning complete
260
261 auto duration = end_time - start_time;
262 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "ExecutionTime Using Last: " << std::chrono::duration<double>(duration).count());
263
264 return;
265 }

267 void LightControlledIntersectionTacticalPlugin::applyTrajectorySmoothingAlgorithm(const carma_wm::WorldModelConstPtr& wm, std::vector<PointSpeedPair>& points_and_target_speeds, double start_dist, double remaining_dist,
268 double starting_speed, double departure_speed, TrajectoryParams tsp)
269 {
270 if (points_and_target_speeds.empty())
271 {
272 throw std::invalid_argument("Point and target speed list is empty! Unable to apply case one speed profile...");
273 }
274
275 // Checking route geometry start against start_dist and adjust profile
276 double planning_downtrack_start = wm->routeTrackPos(points_and_target_speeds[0].point).downtrack; // this can include buffered points earlier than maneuver start_dist
277
278 //Check calculated total dist against maneuver limits
279 double total_distance_needed = remaining_dist;
280 double dist1 = tsp.x1_ - start_dist;
281 double dist2 = tsp.x2_ - start_dist;
282 double dist3 = tsp.x3_ - start_dist;
283
284 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "total_distance_needed: " << total_distance_needed << "\n" <<
285 "dist1: " << dist1 << "\n" <<
286 "dist2: " << dist2 << "\n" <<
287 "dist3: " << dist3);
288 double algo_min_speed = std::min({tsp.v1_,tsp.v2_,tsp.v3_});
289 double algo_max_speed = std::max({tsp.v1_,tsp.v2_,tsp.v3_});
290
291 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "found algo_minimum_speed: " << algo_min_speed << "\n" <<
292 "algo_max_speed: " << algo_max_speed);
293
294 double total_dist_planned = 0; //Starting dist for maneuver treated as 0.0
295
296 if (planning_downtrack_start < start_dist)
297 {
298 //Account for the buffer distance that is technically not part of this maneuver
299
300 total_dist_planned = planning_downtrack_start - start_dist;
301 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "buffered section is present. Adjusted total_dist_planned to: " << total_dist_planned);
302 }
303
304 double prev_speed = starting_speed;
305 auto prev_point = points_and_target_speeds.front();
306
307 for(auto& p : points_and_target_speeds)
308 {
309 double delta_d = lanelet::geometry::distance2d(prev_point.point, p.point);
310 total_dist_planned += delta_d;
311
312 //Apply the speed from algorithm at dist covered
313 //Kinematic: v_f = sqrt(v_o^2 + 2*a*d)
314 double speed_i;
315 if (total_dist_planned <= epsilon_)
316 {
317 //Keep target speed same for buffer distance portion
318 speed_i = starting_speed;
319 }
320 else if(total_dist_planned <= dist1 + epsilon_){
321 //First segment
322 speed_i = sqrt(pow(starting_speed, 2) + 2 * tsp.a1_ * total_dist_planned);
323 }
324 else if(total_dist_planned > dist1 && total_dist_planned <= dist2 + epsilon_){
325 //Second segment
326 speed_i = sqrt(std::max(pow(tsp.v1_, 2) + 2 * tsp.a2_ * (total_dist_planned - dist1), 0.0)); //std::max to ensure negative value is not sqrt
327 }
328 else if (total_dist_planned > dist2 && total_dist_planned <= dist3 + epsilon_)
329 {
330 //Third segment
331 speed_i = sqrt(std::max(pow(tsp.v2_, 2) + 2 * tsp.a3_ * (total_dist_planned - dist2), 0.0)); //std::max to ensure negative value is not sqrt
332 }
333 else
334 {
335 //buffer points that will be cut
336 speed_i = prev_speed;
337 }
338
339 if (isnan(speed_i))
340 {
341 speed_i = std::max(config_.minimum_speed, algo_min_speed);
342 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Detected nan number from equations. Set to " << speed_i);
343 }
344
345 p.speed = std::max({speed_i, config_.minimum_speed, algo_min_speed});
346 p.speed = std::min({p.speed, speed_limit_, algo_max_speed});
347 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Applied speed: " << p.speed << ", at dist: " << total_dist_planned);
348
349 prev_point = p;
350 prev_speed = p.speed;
351 }

354 void LightControlledIntersectionTacticalPlugin::applyOptimizedTargetSpeedProfile(const carma_planning_msgs::msg::Maneuver& maneuver, const double starting_speed, std::vector<PointSpeedPair>& points_and_target_speeds)
355 {
356 if(GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data).size() < 9 ||
357 GET_MANEUVER_PROPERTY(maneuver, parameters.int_valued_meta_data).size() < 2 ){
358 throw std::invalid_argument("There must be 9 float_valued_meta_data and 2 int_valued_meta_data to apply algorithm's parameters.");
359 }
360
361 TrajectoryParams tsp;
362
363 tsp.a1_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[0]);
364 tsp.v1_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[1]);
365 tsp.x1_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[2]);
366
367 tsp.a2_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[3]);
368 tsp.v2_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[4]);
369 tsp.x2_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[5]);
370
371 tsp.a3_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[6]);
372 tsp.v3_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[7]);
373 tsp.x3_ = GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data[8]);
374
375 double starting_downtrack = GET_MANEUVER_PROPERTY(maneuver, start_dist);
376 double ending_downtrack = GET_MANEUVER_PROPERTY(maneuver, end_dist);
377 double departure_speed = GET_MANEUVER_PROPERTY(maneuver, end_speed);
378 double scheduled_entry_time = rclcpp::Time(GET_MANEUVER_PROPERTY(maneuver, end_time)).seconds();
379 double entry_dist = ending_downtrack - starting_downtrack;
380
381 // change speed profile depending on algorithm case starting from maneuver start_dist
382 applyTrajectorySmoothingAlgorithm(wm_, points_and_target_speeds, starting_downtrack, entry_dist, starting_speed,
383 departure_speed, tsp);
384 }

399 std::vector<PointSpeedPair> LightControlledIntersectionTacticalPlugin::createGeometryProfile(const std::vector<carma_planning_msgs::msg::Maneuver> &maneuvers, double max_starting_downtrack,const carma_wm::WorldModelConstPtr &wm,
400 carma_planning_msgs::msg::VehicleState &ending_state_before_buffer,const carma_planning_msgs::msg::VehicleState& state,
401 const GeneralTrajConfig &general_config, const DetailedTrajConfig &detailed_config)
402 {
403 std::vector<PointSpeedPair> points_and_target_speeds;
404
405 bool first = true;
406 std::unordered_set<lanelet::Id> visited_lanelets;
407 std::vector<carma_planning_msgs::msg::Maneuver> processed_maneuvers;
408 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "VehDowntrack: "<<max_starting_downtrack);
409
410 // Only one maneuver is expected in the received maneuver plan
411 if(maneuvers.size() == 1)
412 {
413 auto maneuver = maneuvers.front();
414
415 double starting_downtrack = GET_MANEUVER_PROPERTY(maneuver, start_dist);
416
417 starting_downtrack = std::min(starting_downtrack, max_starting_downtrack);
418
419 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Used downtrack: " << starting_downtrack);
420
421 // check if required parameter from strategic planner is present
422 if(GET_MANEUVER_PROPERTY(maneuver, parameters.float_valued_meta_data).empty())
423 {
424 throw std::invalid_argument("No time_to_schedule_entry is provided in float_valued_meta_data");
425 }
426
427 RCLCPP_DEBUG_STREAM(rclcpp::get_logger("light_controlled_intersection_tactical_plugin"), "Creating Lane Follow Geometry");
