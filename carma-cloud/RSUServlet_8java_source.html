210 private int requestRSU(HttpServletRequest oReq, JSONObject oResponse) throws Exception {
211 long start_ts = Instant.now().toEpochMilli();
212 ArrayList<RSUBoundingbox> registeredRSUs = (ArrayList<RSUBoundingbox>) getServletContext()
213 .getAttribute(RSULIST);
214 if (registeredRSUs == null || registeredRSUs.size() == 0) {
215 LOGGER.error("No RSU is registered!");
216 return HttpServletResponse.SC_ACCEPTED;
217 }
218 StringBuilder sReq = new StringBuilder();
219 try (BufferedInputStream oIn = new BufferedInputStream(oReq.getInputStream())) {
220 int nByte = 0;
221 while ((nByte = oIn.read()) >= 0) {
222 sReq.append((char) nByte);
223 }
224 }
225
226 // Tracking list for BSM request
227 BSMRequestParser parser = new BSMRequestParser();
228 BSMRequest newBsmReq = parser
229 .parseRequest(new ByteArrayInputStream(sReq.toString().getBytes(StandardCharsets.UTF_8)));
230 newBsmReq.setLast_update_at(Instant.now());
231 LOGGER.info("Received: " + newBsmReq);
232 if (newBsmReq.getRoute().size() == 1) {
233 LOGGER.info(
234 "Ignore BSM Request: Only 1 point from BSM request route (Note: BSM Request route includes vehicle current location) ");
235 // If there is only one point in the BSM request, it means the vehicle arrived
236 // at the destination
237 return HttpServletResponse.SC_OK;
238 }
239 ArrayList<BSMRequest> updatedBsmReqsList = new ArrayList<BSMRequest>();
240 Boolean isIgnore = false;
241 try {
242 // Already processed existing BSM list
243 ArrayList<BSMRequest> existingBsmReqList = (ArrayList<BSMRequest>) getServletContext()
244 .getAttribute(BSMREQLIST);
245 if (existingBsmReqList != null) {
246 // Check if the new BSM already processed. If processed, ignore the BSM request
247 // and update existing matching BSM request update time to now
248 for (BSMRequest bsmReq : existingBsmReqList) {
249 if (bsmReq.getId().equals(newBsmReq.getId())) {
250 LOGGER.debug("IGNORE already Processed BSM: " + newBsmReq.getId());
251 bsmReq.setLast_update_at(Instant.now());
252 isIgnore = true;
253 }
254 updatedBsmReqsList.add(bsmReq);
255 }
256 }
257 } catch (NullPointerException oEx) {
258 LOGGER.error("No existing BSM");
259 }
260
261 if (isIgnore) {
262 // Update the tracking list
263 getServletContext().setAttribute(BSMREQLIST, updatedBsmReqsList);
264 return HttpServletResponse.SC_CONFLICT;
265 }
266
267 // Processing new BSM request
268 ExecutorService singleExector = Executors.newSingleThreadExecutor();
269 singleExector.submit(new RSUIdentificationTask(newBsmReq, registeredRSUs));
270 singleExector.shutdown();
271
272 // Add newly processed BSM request to tracking list
273 updatedBsmReqsList.add(newBsmReq);
274 getServletContext().setAttribute(BSMREQLIST, updatedBsmReqsList);
275 long end_ts = Instant.now().toEpochMilli();
276 LOGGER.warn("FER-13-1: TOTAL BSM REQUEST time (ms): " + (end_ts - start_ts) + "; BSM:" + newBsmReq.getId());
277 return HttpServletResponse.SC_OK;
278 }
