119 public Subsegment(CsvReader oIn, int nCols)
120 throws IOException
121 {
122 double[] dC = Arrays.newDoubleArray();
123 double[] dPT = Arrays.newDoubleArray();
124 double[] dNT = Arrays.newDoubleArray();
125 double[] dBb = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
126 for (int nIndex = 0; nIndex < nCols;)
127 {
128 double dX = Mercator.lonToMeters(oIn.parseDouble(nIndex++));
129 double dY = Mercator.latToMeters(oIn.parseDouble(nIndex++));
130 double dW = oIn.parseDouble(nIndex++);
131
132 dC = Arrays.add(dC, new double[]{dX, dY, dW});
133 }
134 m_dCenterLine = Arrays.newDoubleArray(Arrays.size(dC) * 2);
135 m_dCenterLine = Arrays.add(m_dCenterLine, new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE});
136 double[] dSeg = new double[6];
137 double dHeading = 0;
138 Iterator<double[]> oIt = Arrays.iterator(dC, dSeg, 1, 3);
139
140 while (oIt.hasNext())
141 {
142 oIt.next();
143 dHeading = Geo.heading(dSeg[0], dSeg[1], dSeg[3], dSeg[4]);
144 double dW = dSeg[2] / 2;
145 double dXPrime = dSeg[0] + Math.sin(dHeading) * dW; // cos(x - pi/2) = sin(x)
146 double dYPrime = dSeg[1] - Math.cos(dHeading) * dW; // sin(x - pi/2) = -cos(x)
147 dNT = Arrays.add(dNT, dXPrime, dYPrime);
148 updateBoundingBox(dXPrime, dYPrime, dBb);
149
150 dXPrime = dSeg[0] - Math.sin(dHeading) * dW; // cos(x + pi/2 = -sin(x)
151 dYPrime = dSeg[1] + Math.cos(dHeading) * dW; // sin(x + pi/2) = cos(x)
152 dPT = Arrays.add(dPT, dXPrime, dYPrime);
153 updateBoundingBox(dXPrime, dYPrime, dBb);
154
155 m_dLength += Geo.distance(dSeg[0], dSeg[1], dSeg[3], dSeg[4]);
156 m_dCenterLine = Arrays.add(m_dCenterLine, new double[]{dSeg[0], dSeg[1], dSeg[2]});
157 m_dCenterLine = Arrays.add(m_dCenterLine, new double[]{(dSeg[0] + dSeg[3]) / 2, (dSeg[1] + dSeg[4]) / 2, dSeg[2]});
158 }
159
160 m_dCenterLine = Arrays.add(m_dCenterLine, new double[]{dSeg[3], dSeg[4], dSeg[5]});
161
162 double dW = dSeg[5] / 2;
163 double dXPrime = dSeg[3] + Math.sin(dHeading) * dW; // cos(x - pi/2) = sin(x)
164 double dYPrime = dSeg[4] - Math.cos(dHeading) * dW; // sin(x - pi/2) = -cos(x)
165 dNT = Arrays.add(dNT, dXPrime, dYPrime);
166 updateBoundingBox(dXPrime, dYPrime, dBb);
167
168 dXPrime = dSeg[3] - Math.sin(dHeading) * dW; // cos(x + pi/2 = -sin(x)
169 dYPrime = dSeg[4] + Math.cos(dHeading) * dW; // sin(x + pi/2) = cos(x)
170 dPT = Arrays.add(dPT, dXPrime, dYPrime);
171 updateBoundingBox(dXPrime, dYPrime, dBb);
172
173 m_dGeo = Geo.createPolygon(dPT, dNT);
174 m_dGeo = Arrays.add(m_dGeo, m_dGeo[1], m_dGeo[2]);
175 m_dBb = dBb;
