23def generate_launch_description():
24 """
25 Launch file for launching the nodes in the CARMA Messenger user interface stack
26 """
27
28 params = os.path.join(
29 get_package_share_directory('carma-messenger'),
30 'ui',
31 'config',
32 'CommandAPIParams.yaml'
33 )
34
35 ui_group = GroupAction(
36 actions=[
37 SetRemap("system_alert", "/system_alert"),
38 SetRemap("bsm", "incoming_bsm"),
39 IncludeLaunchDescription(
40 XMLLaunchDescriptionSource(os.path.join(get_package_share_directory("rosbridge_server"), 'launch', 'rosbridge_websocket_launch.xml')),
41 launch_arguments = {
42 'params_glob' : [params],
43 }.items()
44 ),
45 ]
46 )
47
48 return LaunchDescription([
49 ui_group
