36void GNSSToMapConvertor::gnssFixCb(gps_msgs::msg::GPSFix::UniquePtr fix_msg)
37{
38 if (!baselink_in_sensor_) // Extract the assumed static transform between baselink and the sensor if it was not
39 // already optained
40 {
41 auto tf_msg = tf_lookup_(fix_msg->header.frame_id, base_link_frame_id_);
42 if (!tf_msg) // Failed to get transform
43 {
44 RCLCPP_WARN_STREAM(logger_->get_logger(),"Ignoring fix message: Could not locate static transform between "
45 << fix_msg->header.frame_id << " and " << base_link_frame_id_);
46 return;
47 }
48
49 tf2::Transform tf;
50 tf2::convert(tf_msg.get().transform, tf);
51
52 baselink_in_sensor_ = tf;
53 }
54
55 if (!sensor_in_ned_heading_rotation_) // Extract the assumed static transform between heading frame and the position
56 // sensor frame if it was not already optained
57 {
58 auto tf_msg = tf_lookup_(fix_msg->header.frame_id, heading_frame_id_);
59 if (!tf_msg) // Failed to get transform
60 {
61 RCLCPP_WARN_STREAM(logger_->get_logger(), "Ignoring fix message: Could not locate static transform between " << heading_frame_id_ << " and "
62 << fix_msg->header.frame_id);
63 return;
64 }
65
66 tf2::Transform tf;
67 tf2::convert(tf_msg.get().transform, tf);
68
69 if (tf.getOrigin().x() != 0.0 || tf.getOrigin().y() != 0.0 || tf.getOrigin().z() != 0.0)
70 {
71 RCLCPP_WARN_STREAM(logger_->get_logger(), "Heading frame does not have rotation only transform with sensor frame. The translation will not "
72 "be handled by the GNSS convertor");
73 }
74
75 sensor_in_ned_heading_rotation_ = tf.getRotation();
76 }
77
78 if (!map_projector_ || !ned_in_map_rotation_) // Check if map projection is available
79 {
80 RCLCPP_WARN_STREAM(logger_->get_logger(), "Ignoring fix message as no map projection has been recieved.");
81 return;
82 }
83
84 geometry_msgs::msg::PoseWithCovarianceStamped pose_msg =
85 poseFromGnss(baselink_in_sensor_.get(), sensor_in_ned_heading_rotation_.get(), *map_projector_, ned_in_map_rotation_.get(), *fix_msg); // Convert to pose
86
87 geometry_msgs::msg::PoseStamped msg; // TODO until covariance is added drop it from the output
88 msg.header = pose_msg.header;
89 msg.header.frame_id = map_frame_id_;
90 msg.pose = pose_msg.pose.pose;
91
92 pose_pub_(msg);
93}

95void GNSSToMapConvertor::geoReferenceCallback(std_msgs::msg::String::UniquePtr geo_ref)
96{
97
98 map_projector_ = std::make_shared<lanelet::projection::LocalFrameProjector>(
99 geo_ref->data.c_str()); // Build projector from proj string
100
101 RCLCPP_INFO_STREAM(logger_->get_logger(), "Recieved map georeference: " << geo_ref->data);
102
103 std::string axis = wgs84_utils::proj_tools::getAxisFromProjString(geo_ref->data); // Extract axis for orientation calc
104
105 RCLCPP_INFO_STREAM(logger_->get_logger(), "Extracted Axis: " << axis);
106
107 ned_in_map_rotation_ = wgs84_utils::proj_tools::getRotationOfNEDFromProjAxis(axis); // Extract map rotation from axis
108
109 RCLCPP_INFO_STREAM(logger_->get_logger(), "Extracted NED in Map Rotation (x,y,z,w) : ( "
110 << ned_in_map_rotation_.get().x() << ", " << ned_in_map_rotation_.get().y() << ", "
111 << ned_in_map_rotation_.get().z() << ", " << ned_in_map_rotation_.get().w());
112}

125geometry_msgs::msg::PoseWithCovarianceStamped GNSSToMapConvertor::poseFromGnss(
126 const tf2::Transform& baselink_in_sensor,
127 const tf2::Quaternion& sensor_in_ned_heading_rotation,
128 const lanelet::projection::LocalFrameProjector& projector,
129 const tf2::Quaternion& ned_in_map_rotation,
130 gps_msgs::msg::GPSFix fix_msg)
131{
133 const double lat = fix_msg.latitude;
134 const double lon = fix_msg.longitude;
135 const double alt = fix_msg.altitude;
136
137 lanelet::BasicPoint3d map_point = projector.forward({ lat, lon, alt });
138 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "map_point: " << map_point.x() << ", " << map_point.y() << ", " << map_point.z());
139
140 if (fabs(map_point.x()) > 10000.0 || fabs(map_point.y()) > 10000.0)
141 { // Above 10km from map origin earth curvature will start to have a negative impact on system performance
142
143 RCLCPP_WARN_STREAM(logger_->get_logger(), "Distance from map origin is larger than supported by system assumptions. Strongly advise "
144 "alternative map origin be used. ");
145 }
146
148 // This logic assumes that the orientation difference between an NED frame located at the map origin and an NED frame
149 // located at the GNSS point are sufficiently small that they can be ignored. Therefore it is assumed the heading
150 // report of the GNSS system reguardless of its poition in the map without change in its orientation will give the
151 // same result (as far as we are concered).
152
153 tf2::Quaternion R_m_n(ned_in_map_rotation); // Rotation of NED frame in map frame
154 tf2::Quaternion R_n_h; // Rotation of sensor heading report in NED frame
155 R_n_h.setRPY(0, 0, fix_msg.track * wgs84_utils::DEG2RAD);
156
157 tf2::Quaternion R_h_s = sensor_in_ned_heading_rotation; // Rotation of heading report in sensor frame
158
159 tf2::Quaternion R_m_s =
160 R_m_n * R_n_h * R_h_s; // Rotation of sensor in map frame under assumption that distance from map origin is
161 // sufficiently small so as to ignore local changes in NED orientation
162
163 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "R_m_n (x,y,z,w) : ( "
164 << R_m_n.x() << ", " << R_m_n.y() << ", "
165 << R_m_n.z() << ", " << R_m_n.w());
166
167 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "R_n_h (x,y,z,w) : ( "
168 << R_n_h.x() << ", " << R_n_h.y() << ", "
169 << R_n_h.z() << ", " << R_n_h.w());
170
171 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "R_h_s (x,y,z,w) : ( "
172 << R_h_s.x() << ", " << R_h_s.y() << ", "
173 << R_h_s.z() << ", " << R_h_s.w());
174
175 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "R_m_s (x,y,z,w) : ( "
176 << R_m_s.x() << ", " << R_m_s.y() << ", "
177 << R_m_s.z() << ", " << R_m_s.w());
178
179 tf2::Transform T_m_s(R_m_s,
180 tf2::Vector3(map_point.x(), map_point.y(), map_point.z())); // Reported position and orientation
181 // of sensor frame in map frame
182 tf2::Transform T_s_b(baselink_in_sensor); // Transform between sensor and baselink frame
183 tf2::Transform T_m_b = T_m_s * T_s_b; // Transform between map and baselink frame
184
185 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "T_m_s (x,y,z,w) : ( "
186 << T_m_s.getRotation().x() << ", " << T_m_s.getRotation().y() << ", "
187 << T_m_s.getRotation().z() << ", " << T_m_s.getRotation().w());
188 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "T_s_b (x,y,z,w) : ( "
189 << T_s_b.getRotation().x() << ", " << T_s_b.getRotation().y() << ", "
190 << T_s_b.getRotation().z() << ", " << T_s_b.getRotation().w());
191 RCLCPP_DEBUG_STREAM(logger_->get_logger(), "T_m_b (x,y,z,w) : ( "
192 << T_m_b.getRotation().x() << ", " << T_m_b.getRotation().y() << ", "
193 << T_m_b.getRotation().z() << ", " << T_m_b.getRotation().w());
194
195 // TODO handle covariance
196
197 // Populate message
198 geometry_msgs::msg::PoseWithCovarianceStamped pose;
199 pose.header = fix_msg.header;
200
201 pose.pose.pose.position.x = T_m_b.getOrigin().getX();
202 pose.pose.pose.position.y = T_m_b.getOrigin().getY();
203 pose.pose.pose.position.z = T_m_b.getOrigin().getZ();
204
205 pose.pose.pose.orientation.x = T_m_b.getRotation().getX();
206 pose.pose.pose.orientation.y = T_m_b.getRotation().getY();
207 pose.pose.pose.orientation.z = T_m_b.getRotation().getZ();
208 pose.pose.pose.orientation.w = T_m_b.getRotation().getW();
209
210 return pose;
211}
