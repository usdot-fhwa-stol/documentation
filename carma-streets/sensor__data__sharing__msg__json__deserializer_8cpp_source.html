19 sensor_data_sharing_msg from_json( const std::string &json ){
20 rapidjson::Document document = streets_utils::json_utils::parse_json(json);
21 sensor_data_sharing_msg msg;
22 msg._msg_count = parse_uint_member("msg_cnt", document, true).value();
23 msg._source_id = parse_string_member("source_id", document, true).value();
24 msg._equipment_type = equipment_type_from_int(parse_uint_member("equipment_type", document, true).value());
25 msg._time_stamp = parse_time_stamp(parse_object_member("sdsm_time_stamp", document, true).value());
26 msg._ref_positon = parse_position_3d(parse_object_member("ref_pos", document, true).value());
27 msg._ref_position_confidence = parse_positional_accuracy(parse_object_member("ref_pos_xy_conf", document, true).value());
28 msg._ref_position_elevation_confidence = parse_elevation_confidence( document);
29 msg._objects = parse_detected_object_list(document);
30 return msg;
31 }

73 std::vector<detected_object_data> parse_detected_object_list(const rapidjson::Value &val){
74 std::vector<detected_object_data> detected_object_list;
75 auto json_detected_object_list = parse_array_member("objects",val, true).value();
76 for (const auto &object: json_detected_object_list){
77 detected_object_data data;
78 auto detected_object = parse_object_member("detected_object_data",object, true);
79 data._detected_object_common_data = parse_detected_object_data_common(parse_object_member("detected_object_common_data", detected_object.value(), true).value());
80 if ( auto data_optional = parse_object_member("detected_object_optional_data", detected_object.value(), false); data_optional.has_value()) {
81 data._detected_object_optional_data = parse_detected_object_data_optional( data_optional.value() );
82 }
83 detected_object_list.push_back(data);
84 }
85 return detected_object_list;
86 }

88 detected_object_data_common parse_detected_object_data_common(const rapidjson::Value &val){
89 detected_object_data_common _detected_object_common_data;

91 _detected_object_common_data._classification_confidence = parse_uint_member("obj_type_cfd",val,true).value();
92 _detected_object_common_data._object_id = parse_uint_member("object_id", val, true).value();
93 _detected_object_common_data._time_measurement_offset = parse_int_member("measurement_time", val, true).value();

95 _detected_object_common_data._position_offset = parse_position_offset(parse_object_member("pos" ,val, true).value());
96 _detected_object_common_data._pos_confidence = parse_position_confidence_set(parse_object_member("pos_confidence", val, true).value());
97 _detected_object_common_data._speed = parse_uint_member("speed", val, true).value();
98 _detected_object_common_data._speed_confidence = speed_confidence_from_int(parse_uint_member("speed_confidence", val, true).value());
99 // Optional
100 _detected_object_common_data._speed_z = parse_uint_member("speed_z", val, false);
101 // Optional enumeration
102 if ( val.HasMember("speed_z_confidence")) {
103 _detected_object_common_data._speed_z_confidence = speed_confidence_from_int( parse_uint_member("speed_z_confidence", val, true).value());
104 }
105 _detected_object_common_data._heading = parse_uint_member("heading", val, true).value();
106 _detected_object_common_data._heading_confidence = heading_confidence_from_int(parse_uint_member("heading_conf", val, true).value());
107
108 _detected_object_common_data._acceleration_4_way = parse_acceleration_4_way(parse_object_member("accel_4_way", val, true ).value());
109 if ( val.HasMember("acc_cfd_x")) {
110 _detected_object_common_data._lateral_acceleration_confidence = acceleration_confidence_from_int(parse_uint_member("acc_cfd_x", val, true).value());
111 }
112 if ( val.HasMember("acc_cfd_y")) {
113 _detected_object_common_data._longitudinal_acceleration_confidence = acceleration_confidence_from_int(parse_uint_member("acc_cfd_y", val, true).value());
114 }
115 if ( val.HasMember("acc_cfd_z")) {
116 _detected_object_common_data._vertical_accelaration_confidence = acceleration_confidence_from_int(parse_uint_member("acc_cfd_z", val, true).value());
117 }
118 if ( val.HasMember("acc_cfd_yaw")) {
119 _detected_object_common_data._yaw_rate_confidence = angular_velocity_confidence_from_int(parse_uint_member("acc_cfd_yaw", val, true).value());
120 }
121 return _detected_object_common_data;

263 std::optional<std::variant<human_propelled_type, motorized_propelled_type, animal_propelled_type>> parse_propelled_information( const rapidjson::Value &val){
264 std::optional<std::variant<human_propelled_type, motorized_propelled_type, animal_propelled_type>> data;
265 if (val.HasMember("human")) {

267 }

269 data = animal_propelled_type_from_int(parse_uint_member("animal", val, true).value());

272 data = motorized_propelled_type_from_int(parse_uint_member("motor", val, true).value());
273 }
274 else {
275 throw json_parse_exception("Propelled information requires one of the following to be define: human propelled type, animal propelled type, motorized propelled type.");
276 }
