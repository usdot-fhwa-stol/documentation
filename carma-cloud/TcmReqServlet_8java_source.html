107 {

109 {
110 long lNow = System.currentTimeMillis();
111 String sReq = Thread.currentThread().getName();
112 TcmReqParser oReqParser;
113 if (sReq.indexOf("<tcrV01>") >= 0)
114 oReqParser = new TcmReqParser();
115 else
116 oReqParser = new TcmReqParser2();
117
118 TcmReq oTcmReq = oReqParser.parseRequest(new ByteArrayInputStream(sReq.getBytes(StandardCharsets.UTF_8)));
119
120 ArrayList<TrafCtrl> oResCtrls = new ArrayList();
121 ArrayList<TileIds> oTiles = new ArrayList(50);
122 ArrayList<byte[]> oProcessed = new ArrayList(1000);
123 TrafCtrl oTrafCtrlSearch = new TrafCtrl();
124 oTrafCtrlSearch.m_yId = new byte[16];
125 TileIds oTileIdsSearch = new TileIds();
126 int nMsgTot = 0;
127 for (TcBounds oBounds : oTcmReq.m_oBounds)
128 {
129 double[] dTcMercBounds = new double[]{Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE};
130 double[] dCorners = new double[oBounds.m_nCorners.length];
131 for (int nIndex = 0; nIndex < oBounds.m_nCorners.length;)
132 {
133 double dX = Mercator.lonToMeters(Geo.fromIntDeg(oBounds.m_nCorners[nIndex]));
134 dCorners[nIndex++] = dX;
135 double dY = Mercator.latToMeters(Geo.fromIntDeg(oBounds.m_nCorners[nIndex]));
136 dCorners[nIndex++] = dY;
137 if (dX < dTcMercBounds[0])
138 dTcMercBounds[0] = dX;
139 if (dY < dTcMercBounds[1])
140 dTcMercBounds[1] = dY;
141 if (dX > dTcMercBounds[2])
142 dTcMercBounds[2] = dX;
143 if (dY > dTcMercBounds[3])
144 dTcMercBounds[3] = dY;
145 }
146 Geo.untwist(dCorners);
147 Mercator oM = Mercator.getInstance();
148 int[] nTiles = new int[2];
149 int[] nTileIndices = new int[4];
150 oM.metersToTile(dTcMercBounds[0], dTcMercBounds[3], CtrlTiles.g_nZoom, nTiles); // determine the tiles the bounds intersect
151 nTileIndices[0] = nTiles[0];
152 nTileIndices[1] = nTiles[1];
153 oM.metersToTile(dTcMercBounds[2], dTcMercBounds[1], CtrlTiles.g_nZoom, nTiles);
154 nTileIndices[2] = nTiles[0];
155 nTileIndices[3] = nTiles[1];
156 for (int nTileX = nTileIndices[0]; nTileX <= nTileIndices[2]; nTileX++) // for each tile
157 {
158 for (int nTileY = nTileIndices[1]; nTileY <= nTileIndices[3]; nTileY++)
159 {
160 Path oIndexFile = Paths.get(String.format(CtrlTiles.g_sTdFileFormat, nTileX, CtrlTiles.g_nZoom, nTileX, nTileY) + ".ndx");
161 if (!Files.exists(oIndexFile)) // check if the index file exists
162 continue;
163
164 oTileIdsSearch.setIndices(nTileX, nTileY);
165 int nIndex = Collections.binarySearch(oTiles, oTileIdsSearch);
166 if (nIndex < 0) // if the tile index file has not been loaded
167 {
168 TileIds oTile = new TileIds(nTileX, nTileY, 1000);
169 nIndex = ~nIndex;
170 oTiles.add(nIndex, oTile);
171 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oIndexFile)))) // read the index file to find which controls are in the tile
172 {
173 while (oIn.available() > 0)
174 {
175 CtrlIndex oIndex = new CtrlIndex(oIn);
176 if (oBounds.m_lOldest <= oIndex.m_lStart && oIndex.m_lEnd > lNow && java.util.Arrays.binarySearch(IGNORE_CTRLS, oIndex.m_nType) < 0 && Geo.boundingBoxesIntersect(dTcMercBounds, oIndex.m_dBB)) // skip out controls and control types to ignore
177 oTile.add(oIndex.m_yId);
178 }
179 }
180 }
181
182 TileIds oTile = oTiles.get(nIndex);
183 oResCtrls.ensureCapacity(oResCtrls.size() + oTile.size());
184 for (byte[] yId : oTile)
185 {
186 int nProcessIndex = Collections.binarySearch(oProcessed, yId, TrafCtrl.ID_COMP);
187 if (nProcessIndex < 0)
188 oProcessed.add(~nProcessIndex, yId);
189 else
190 continue;
191
192 String sFile = CtrlTiles.g_sCtrlDir + TrafCtrl.getId(yId) + ".bin";
193 Path oCtrlFile = Paths.get(sFile);
194
195 if (!Files.exists(oCtrlFile))
196 continue;
197 TrafCtrl oCtrl;
198 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oCtrlFile))))
199 {
200 oCtrl = new TrafCtrl(oIn, true, false);
201 }
202
203 if (Geo.ctrlIntersectBounds(dCorners, oCtrl, m_nPolyStep))
204 {
205 oResCtrls.add(oCtrl);
206 oCtrl.preparePoints(m_nExplodeDistForXml);
207 nMsgTot += (oCtrl.size() / 256 + 1);
208 }
209 }
210 }
211 }
212 }
213
214 int nMsgCount = 1;
215 if (oReqParser.m_bList)
216 nMsgTot = oResCtrls.size();
217
218 StringBuilder sBuf = new StringBuilder(nMsgTot * 2048);
219
220 for (int nCtrlIndex = 0; nCtrlIndex < oResCtrls.size(); nCtrlIndex++)
221 {
222 TrafCtrl oCtrl = oResCtrls.get(nCtrlIndex);
223 int nParts = oCtrl.size() / 256 + 1;
224 if (oReqParser.m_bList)
225 nParts = 1;
226 for (int nIndex = 0; nIndex < nParts; nIndex++)
227 {
228 if (nIndex == 0)
229 oCtrl.getXml(sBuf, oTcmReq.m_sReqId, oTcmReq.m_nReqSeq, nMsgCount, nMsgTot, oTcmReq.m_sVersion, true, 0, m_bRemoveWidth, oReqParser.m_bList);
230 else
231 oCtrl.getXml(sBuf, oTcmReq.m_sReqId, oTcmReq.m_nReqSeq, nMsgCount, nMsgTot, oTcmReq.m_sVersion, false, nIndex * 256 - 1, m_bRemoveWidth, oReqParser.m_bList);
232 ++nMsgCount;
233
234 if (!oReqParser.m_bList)
235 {
236 HttpURLConnection oHttpClient = (HttpURLConnection)new URL(String.format("http://tcmreplyhost:%d/tcmreply", oReqParser.m_nPort)).openConnection();
237 oHttpClient.setFixedLengthStreamingMode(sBuf.length());
238 oHttpClient.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
239 oHttpClient.setDoOutput(true);
240 oHttpClient.setRequestMethod("POST");
241 oHttpClient.setConnectTimeout(1000);
242
243 try
244 {
245 oHttpClient.connect(); // send post request
246
247 try (BufferedWriter oOut = new BufferedWriter(new OutputStreamWriter(oHttpClient.getOutputStream())))
248 {
249 oOut.append(sBuf);
250 }
251
252 oHttpClient.disconnect();
253 }
254 catch (Exception oEx)
255 {
256 oEx.printStackTrace();
257 }
258 LOGGER.debug(sBuf);
259 }
260 }
261 }
262

264 {
265 HttpURLConnection oHttpClient = (HttpURLConnection)new URL(String.format("http://tcmreplyhost:%d/tcmreply", oReqParser.m_nPort)).openConnection();
266 oHttpClient.setRequestProperty("Content-Encoding", "gzip");
267 oHttpClient.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
268 oHttpClient.setDoOutput(true);
269 oHttpClient.setRequestMethod("POST");

274 {
275 oOut.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?><TrafficControlMessageList>").append(sBuf).append("</TrafficControlMessageList>");

290 oEx.printStackTrace();

293 sBuf.insert(0, "<?xml version=\"1.0\" encoding=\"UTF-8\"?><TrafficControlMessageList>");
294 sBuf.append("</TrafficControlMessageList>");
295 LOGGER.debug(sBuf);
296 }
297 }

300 LOGGER.error(oEx, oEx);
