25WMListener::WMListener(
26 rclcpp::node_interfaces::NodeBaseInterface::SharedPtr node_base,
27 rclcpp::node_interfaces::NodeLoggingInterface::SharedPtr node_logging,
28 rclcpp::node_interfaces::NodeTopicsInterface::SharedPtr node_topics,
29 rclcpp::node_interfaces::NodeParametersInterface::SharedPtr node_params_,
30 bool multi_thread)
31 :node_base_(node_base),node_logging_(node_logging),node_topics_(node_topics),worker_(std::unique_ptr<WMListenerWorker>(new WMListenerWorker)), multi_threaded_(multi_thread)
32{
33
34 RCLCPP_DEBUG_STREAM(node_logging_->get_logger(), "WMListener: Creating world model listener");
35
36 //Declare parameter if it doesn't exist
37 rclcpp::Parameter config_speed_limit_param("config_speed_limit");
38 if(!node_params_->get_parameter("config_speed_limit", config_speed_limit_param)){
39 rclcpp::ParameterValue config_speed_limit_param_value;
40 config_speed_limit_param_value = node_params_->declare_parameter("config_speed_limit", rclcpp::ParameterValue (0.0));
41 }
42
43 rclcpp::Parameter participant_param("vehicle_participant_type");
44 if(!node_params_->get_parameter("vehicle_participant_type", participant_param)){
45 rclcpp::ParameterValue participant_param_value;
46 participant_param_value = node_params_->declare_parameter("vehicle_participant_type", rclcpp::ParameterValue(""));
47 }
48
49 // Get params
50 config_speed_limit_param = node_params_->get_parameter("config_speed_limit");
51 participant_param = node_params_->get_parameter("vehicle_participant_type");
52
53 RCLCPP_INFO_STREAM(node_logging->get_logger(), "Loaded config speed limit: " << config_speed_limit_param.as_double());
54 RCLCPP_INFO_STREAM(node_logging->get_logger(), "Loaded vehicle participant type: " << participant_param.as_string());
55
56 setConfigSpeedLimit(config_speed_limit_param.as_double());
57 worker_->setVehicleParticipationType(participant_param.as_string());
58
59 rclcpp::SubscriptionOptions map_update_options;
60 rclcpp::SubscriptionOptions map_options;
61 rclcpp::SubscriptionOptions route_options;
62 rclcpp::SubscriptionOptions roadway_objects_options;
63 rclcpp::SubscriptionOptions traffic_spat_options;
64
65 if(multi_threaded_)
66 {
67 RCLCPP_DEBUG_STREAM(node_logging_->get_logger(), "WMListener: Using a multi-threaded subscription");
68
69 auto map_update_cb_group = node_base_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
70 map_update_options.callback_group = map_update_cb_group;
71
72 auto map_cb_group = node_base_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
73 map_options.callback_group = map_cb_group;
74
75 auto route_cb_group = node_base_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
76 route_options.callback_group = route_cb_group;
77
78 auto roadway_objects_group = node_base_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
79 roadway_objects_options.callback_group = roadway_objects_group;
80
81 auto traffic_spat_group = node_base_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
82 traffic_spat_options.callback_group = traffic_spat_group;
83
84 }
85
86 // Setup subscribers
87 route_sub_ = rclcpp::create_subscription<carma_planning_msgs::msg::Route>(node_topics_, "route", 1,
88 std::bind(&WMListenerWorker::routeCallback, worker_.get(), std::placeholders::_1), route_options);
89
90 roadway_objects_sub_ = rclcpp::create_subscription<carma_perception_msgs::msg::RoadwayObstacleList>(node_topics_, "roadway_objects", 1,
91 std::bind(&WMListenerWorker::roadwayObjectListCallback, worker_.get(), std::placeholders::_1), roadway_objects_options);
92
93 traffic_spat_sub_ = rclcpp::create_subscription<carma_v2x_msgs::msg::SPAT>(node_topics_, "incoming_spat", 1,
94 std::bind(&WMListenerWorker::incomingSpatCallback, worker_.get(), std::placeholders::_1), traffic_spat_options);
95
96 // NOTE: Currently, intra-process comms must be disabled for subscribers that are transient_local: https://github.com/ros2/rclcpp/issues/1753
97 map_update_options.use_intra_process_comm = rclcpp::IntraProcessSetting::Disable; // Disable intra-process comms for the map update subscriber
98 auto map_update_sub_qos = rclcpp::QoS(rclcpp::KeepLast(10)); // Set the queue size for the map update subscriber
99 map_update_sub_qos.transient_local(); // If it is possible that this node is a late-joiner to its topic, it must be set to transient_local to receive earlier messages that were missed.
100 // NOTE: The publisher's QoS must be set to transisent_local() as well for earlier messages to be resent to this later-joiner.
101
102 // Create map update subscriber that will receive earlier messages that were missed ONLY if the publisher is transient_local too
103 map_update_sub_ = rclcpp::create_subscription<autoware_lanelet2_msgs::msg::MapBin>(node_topics_, "map_update", map_update_sub_qos,
104 std::bind(&WMListener::mapUpdateCallback, this, std::placeholders::_1), map_update_options);
105
106 map_options.use_intra_process_comm = rclcpp::IntraProcessSetting::Disable; // Disable intra-process comms for the semantic map subscriber
107 auto map_sub_qos = rclcpp::QoS(rclcpp::KeepLast(2)); // Set the queue size for the semantic map subscriber
108 map_sub_qos.transient_local(); // If it is possible that this node is a late-joiner to its topic, it must be set to transient_local to receive earlier messages that were missed.
109 // NOTE: The publisher's QoS must be set to transisent_local() as well for earlier messages to be resent to this later-joiner.
110
111 // Create semantic mab subscriber that will receive earlier messages that were missed ONLY if the publisher is transient_local too
112 map_sub_ = rclcpp::create_subscription<autoware_lanelet2_msgs::msg::MapBin>(node_topics_, "semantic_map", map_sub_qos,
113 std::bind(&WMListenerWorker::mapCallback, worker_.get(), std::placeholders::_1), map_options);
114}
