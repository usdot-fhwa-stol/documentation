25 boost::optional<j2735_v2x_msgs::msg::BSM> BSM_Message::decode_bsm_message(std::vector<uint8_t>& binary_array){
26
27 j2735_v2x_msgs::msg::BSM output;
28 //decode results - stored in binary_array
29 asn_dec_rval_t rval;
30 MessageFrame_t* message = nullptr;
31
32 //copy from vector to array
33 size_t len=binary_array.size();
34
35 uint8_t buf[len];
36 std::copy(binary_array.begin(),binary_array.end(),buf);
37 //use asn1c lib to decode
38
39 rval=uper_decode(0, &asn_DEF_MessageFrame,(void **) &message, buf, len, 0, 0);
40
41 //if decode success
42 if(rval.code==RC_OK)
43 {
44 BSMcoreData_t core_data_msg = message->value.choice.BasicSafetyMessage.coreData;
45 output.core_data.msg_count = core_data_msg.msgCnt;
46 auto id_len = core_data_msg.id.size;
47 for(auto i = 0; i < id_len; i++)
48 {
49 output.core_data.id.push_back(core_data_msg.id.buf[i]);
50 }
51 output.core_data.sec_mark = core_data_msg.secMark;
52 output.core_data.latitude = core_data_msg.lat;
53 output.core_data.longitude = core_data_msg.Long;
54 output.core_data.elev = core_data_msg.elev;
55 output.core_data.accuracy.orientation = core_data_msg.accuracy.orientation;
56 output.core_data.accuracy.semi_major = core_data_msg.accuracy.semiMajor;
57 output.core_data.accuracy.semi_minor = core_data_msg.accuracy.semiMinor;
58 output.core_data.transmission.transmission_state = core_data_msg.transmission;
59 output.core_data.speed = core_data_msg.speed;
60 output.core_data.heading = core_data_msg.heading;
61 output.core_data.angle = core_data_msg.angle;
62 output.core_data.accel_set.lateral = core_data_msg.accelSet.lat;
63 output.core_data.accel_set.longitudinal =core_data_msg.accelSet.Long;
64 output.core_data.accel_set.vert = core_data_msg.accelSet.vert;
65 output.core_data.accel_set.yaw_rate = core_data_msg.accelSet.yaw;
66 // brake_applied_status decoding
67 // e.g. make 0b0100000 to 0b0000100
68 uint8_t binary = core_data_msg.brakes.wheelBrakes.buf[0] >> 3;
69 unsigned int brake_applied_status_type = 4;
70 // e.g. shift the binary right until it equals to 1 (0b00000001) to determine the location of the non-zero bit
71
72 for (int i = 0; i < 4; i ++)
73 {
74 if ((int)binary == 1)
75 {
76 output.core_data.brakes.wheel_brakes.brake_applied_status = brake_applied_status_type;
77 break;
78 }
79 else
80 {
81 brake_applied_status_type -= 1;
82 binary = binary >> 1;
83 }
84 }
85 output.core_data.brakes.traction.traction_control_status = core_data_msg.brakes.traction;
86 output.core_data.brakes.abs.anti_lock_brake_status = core_data_msg.brakes.abs;
87 output.core_data.brakes.scs.stability_control_status = core_data_msg.brakes.scs;
88 output.core_data.brakes.brake_boost.brake_boost_applied = core_data_msg.brakes.brakeBoost;
89 output.core_data.brakes.aux_brakes.auxiliary_brake_status = core_data_msg.brakes.auxBrakes;
90 output.core_data.size.vehicle_length = core_data_msg.size.length;
91 output.core_data.size.vehicle_width = core_data_msg.size.width;
92
93 return boost::optional<j2735_v2x_msgs::msg::BSM>(output);
94 }
95 RCLCPP_WARN_STREAM( node_logging_->get_logger(), "BasicSafetyMessage decoding failed");
96 return boost::optional<j2735_v2x_msgs::msg::BSM>{};
97
98 }

100 boost::optional<std::vector<uint8_t>> BSM_Message::encode_bsm_message(const j2735_v2x_msgs::msg::BSM& plain_msg)
101 {
102 //Uncomment below (and one line at the end of the function) to print the message in human readable form
103 //FILE *fp;
104 //fp = fopen("/absolute/directory/log_C.txt", "w");
105 //fprintf(fp, "encodeBSM function is called\n");
106
107 //encode result placeholder
108 uint8_t buffer[544];
109 size_t buffer_size=sizeof(buffer);
110 asn_enc_rval_t ec;
111 MessageFrame_t message;
112
113 //set message type to BasicSafetyMessage
114 message.messageId = 20;
115 message.value.present = MessageFrame__value_PR_BasicSafetyMessage;
116
117 BasicSafetyMessage bsm_msg;
118 bsm_msg.partII = nullptr;
119 bsm_msg.regional = nullptr;
120
121 // Encode coreData
122 BSMcoreData_t core_data;
123 core_data.msgCnt = plain_msg.core_data.msg_count;
124 //Set the fields
125 uint8_t id_content[4] = {0};
126 for(auto i = 0; i < 4; i++)
127 {
128 id_content[i] = (char) plain_msg.core_data.id[i];
129 }
130 TemporaryID_t temp_id;
131 temp_id.buf = id_content;
132 temp_id.size = 4;
133 core_data.id = temp_id;
134 core_data.secMark = plain_msg.core_data.sec_mark;
135
136 core_data.lat = plain_msg.core_data.latitude;
137 core_data.Long = plain_msg.core_data.longitude;
138 core_data.elev = plain_msg.core_data.elev;
139 PositionalAccuracy_t pos_acc;
140 pos_acc.orientation = plain_msg.core_data.accuracy.orientation;
141 pos_acc.semiMajor = plain_msg.core_data.accuracy.semi_major;
142 pos_acc.semiMinor = plain_msg.core_data.accuracy.semi_minor;
143 core_data.accuracy = pos_acc;
144 core_data.transmission = plain_msg.core_data.transmission.transmission_state;
145 core_data.speed = plain_msg.core_data.speed;
146 core_data.heading = plain_msg.core_data.heading;
147 core_data.angle = plain_msg.core_data.angle;
148 AccelerationSet4Way_t accel;
149 accel.lat = plain_msg.core_data.accel_set.lateral;
150 accel.Long = plain_msg.core_data.accel_set.longitudinal;
151 accel.vert= plain_msg.core_data.accel_set.vert;
152 accel.yaw = plain_msg.core_data.accel_set.yaw_rate;
153 core_data.accelSet = accel;
154 VehicleSize_t vehicle_size;
155 vehicle_size.length = plain_msg.core_data.size.vehicle_length;
156 vehicle_size.width = plain_msg.core_data.size.vehicle_width;
157 core_data.size = vehicle_size;
158 BrakeSystemStatus_t brakes;
159
160 brakes.traction = plain_msg.core_data.brakes.traction.traction_control_status;
161 brakes.abs = plain_msg.core_data.brakes.abs.anti_lock_brake_status;
162 brakes.scs = plain_msg.core_data.brakes.scs.stability_control_status;
163 brakes.brakeBoost = plain_msg.core_data.brakes.brake_boost.brake_boost_applied;
164 brakes.auxBrakes = plain_msg.core_data.brakes.aux_brakes.auxiliary_brake_status;
165
166 BrakeAppliedStatus_t brake_applied_status;
167
168 uint8_t wheel_brake[1] = {8}; // dummy 8 value
169
170 // there are 3 unused bits in the end: 0b000
171 // which makes every possible encoded value to be multiples of 8: 0b00001000 (8), 0b00010000 (16), 0b00011000 (24) etc
172 // so num in brackets indicate the position in the bit string:
173 // unavailable: 0b10000000, leftFront: 0b01000000 etc
174 wheel_brake[0] = (char) (8 << (4 - plain_msg.core_data.brakes.wheel_brakes.brake_applied_status));
175 brake_applied_status.buf = wheel_brake;
176 brake_applied_status.size = 1;
177 brake_applied_status.bits_unused = 3;
178 brakes.wheelBrakes = brake_applied_status;
179
180 core_data.brakes = brakes;
181 bsm_msg.coreData = core_data;
182 message.value.choice.BasicSafetyMessage = bsm_msg;
183 //encode message
184 ec=uper_encode_to_buffer(&asn_DEF_MessageFrame, 0 , &message , buffer , buffer_size);
185 // Uncomment below to enable logging in human readable form
186 //asn_fprint(fp, &asn_DEF_MessageFrame, message);
187
188 //log a warning if that fails
189 if(ec.encoded == -1) {
190 RCLCPP_WARN_STREAM( node_logging_->get_logger(), "Encoding for BasicSafetyMessage has failed");
191 std::cout << "Failed: " << ec.failed_type->name << std::endl;
192 return boost::optional<std::vector<uint8_t>>{};
193 }
194
195 //copy to byte array msg
196 size_t array_length=(ec.encoded + 7) / 8;
197 std::vector<uint8_t> b_array(array_length);
198 for(size_t i=0;i<array_length;i++)b_array[i]=buffer[i];
199
200 //Debugging/Unit Testing
201 //for(size_t i = 0; i < array_length; i++) std::cout<< int(b_array[i])<< ", ";
202 return boost::optional<std::vector<uint8_t>>(b_array);
203 }
