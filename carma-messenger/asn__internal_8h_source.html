147static int CC_NOTUSED
148ASN__STACK_OVERFLOW_CHECK(const asn_codec_ctx_t *ctx) {
149 if(ctx && ctx->max_stack_size) {
150
151 /* ctx MUST be allocated on the stack */
152 ptrdiff_t usedstack = ((const char *)ctx - (const char *)&ctx);
153 if(usedstack > 0) usedstack = -usedstack; /* grows up! */
154
155 /* double negative required to avoid int wrap-around */
156 if(usedstack < -(ptrdiff_t)ctx->max_stack_size) {
157 ASN_DEBUG("Stack limit %ld reached",
158 (long)ctx->max_stack_size);
159 return -1;
160 }
161 }
162 return 0;
163}
