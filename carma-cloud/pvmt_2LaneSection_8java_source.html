102 public void createPavement()
103 {
104 ArrayList<Lane> oLanes = new ArrayList();
105 getLanes(oLanes);
106 Collections.sort(oLanes, LEFTLANES);
107 int nDriving = XodrUtil.getLaneType("driving");
108 double[] dEdgeR = null;
109 for (int nIndex = 0; nIndex < oLanes.size(); nIndex++)
110 {
111 Lane oLane = oLanes.get(nIndex);
112 if (oLane.m_nLaneType == nDriving)
113 {
114 dEdgeR = oLane.m_nLaneIndex < 0 ? oLane.m_dOuter : oLane.m_dInner;
115 break;
116 }
117 }
118
119 double[] dEdgeL = null;
120 for (int nIndex = oLanes.size() - 1; nIndex >= 0; nIndex--)
121 {
122 Lane oLane = oLanes.get(nIndex);
123 if (oLane.m_nLaneType == nDriving)
124 {
125 dEdgeL = oLane.m_nLaneIndex < 0 ? oLane.m_dInner : oLane.m_dOuter;
126 break;
127 }
128 }
129
130 if (dEdgeL == null || dEdgeR == null)
131 return;
132 double[] dPavementCenter = Arrays.newDoubleArray(Arrays.size(dEdgeL));
133 dPavementCenter = Arrays.add(dPavementCenter, new double[] {Double.MAX_VALUE, Double.MAX_VALUE, -Double.MAX_VALUE, -Double.MAX_VALUE});
134
135 int nLimit = Arrays.size(dEdgeR);
136 m_dEdgeEnds = new double[]{dEdgeR[nLimit - 2], dEdgeR[nLimit - 1], dEdgeL[nLimit - 2], dEdgeL[nLimit - 1]};
137 if (m_oPrevSect != null && m_oPrevSect.m_dEdgeEnds != null)
138 {
139 double dXr = m_oPrevSect.m_dEdgeEnds[0];
140 double dYr = m_oPrevSect.m_dEdgeEnds[1];
141 double dXl = m_oPrevSect.m_dEdgeEnds[2];
142 double dYl = m_oPrevSect.m_dEdgeEnds[3];
143
144 double dPathX = (dXr + dXl) / 2;
145 double dPathY = (dYr + dYl) / 2;
146 double dW = Geo.distance(dXr, dYr, dXl, dYl);
147 dPavementCenter = Arrays.addAndUpdate(dPavementCenter, dPathX, dPathY);
148 dPavementCenter = Arrays.add(dPavementCenter, dW);
149 }
150 for (int nIndex = 5; nIndex < nLimit; nIndex += 2)
151 {
152 double dXr = dEdgeR[nIndex];
153 double dYr = dEdgeR[nIndex + 1];
154 double dXl = dEdgeL[nIndex];
155 double dYl = dEdgeL[nIndex + 1];
156
157 double dPathX = (dXr + dXl) / 2;
158 double dPathY = (dYr + dYl) / 2;
159 double dW = Geo.distance(dXr, dYr, dXl, dYl);
160 dPavementCenter = Arrays.addAndUpdate(dPavementCenter, dPathX, dPathY);
161 dPavementCenter = Arrays.add(dPavementCenter, dW);
162 }
163
164 m_dPavement = dPavementCenter; // save local reference
165 }
