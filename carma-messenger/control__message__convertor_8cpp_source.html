140void convert(const j2735_v2x_msgs::msg::TrafficControlGeometry& in_msg, carma_v2x_msgs::msg::TrafficControlGeometry& out_msg)
141{
142 out_msg.proj = in_msg.proj;
143 out_msg.datum = in_msg.datum;
144
145 int32_t sec = 0;
146 // Use conditional to avoid issues from casting in std::min
147 if ((double)in_msg.reftime* units::SEC_PER_MIN > (double)std::numeric_limits<int32_t>::max())
148 {
149 sec = std::numeric_limits<int32_t>::max();
150 }
151 else
152 {
153 sec = in_msg.reftime* units::SEC_PER_MIN;
154
155 }
156
157 out_msg.reftime = rclcpp::Time(sec, 0);
158 out_msg.reflon = (double)in_msg.reflon / units::TENTH_MICRO_DEG_PER_DEG;
159 out_msg.reflat = (double)in_msg.reflat / units::TENTH_MICRO_DEG_PER_DEG;
160 out_msg.refelv = (float)in_msg.refelv / units::DECI_M_PER_M - (float) 409.6; //handle offset
161 out_msg.heading = (float)in_msg.heading / units::DECI_S_PER_S;
162
163 for (auto in_node : in_msg.nodes)
164 {
165 carma_v2x_msgs::msg::PathNode out_node;
166 convert(in_node, out_node);
167 out_msg.nodes.push_back(out_node);
168 }
