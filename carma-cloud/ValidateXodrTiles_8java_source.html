112 @Override
113 protected void doGet(HttpServletRequest oRequest, HttpServletResponse oResponse)
114 throws ServletException, IOException
115 {
116 long lNow = System.currentTimeMillis();
117 String[] sUriParts = oRequest.getRequestURI().split("/");
118 int nZ = Integer.parseInt(sUriParts[sUriParts.length - 3]);
119 int nX = Integer.parseInt(sUriParts[sUriParts.length - 2]);
120 int nY = Integer.parseInt(sUriParts[sUriParts.length - 1]);
121
122 double[] dBounds = TileUtil.getTileBounds(nZ, nX, nY);
123 double dXAdjust = 0.001;
124 double dYAdjust = 0.001;
125 int[] nTiles = new int[2];
126 Mercator.getInstance().metersToTile(dBounds[0] + dXAdjust, dBounds[3] - dYAdjust, m_nZoom, nTiles); // determine the correct tiles for the default zoom level
127 int nDefaultX = nTiles[0];
128 int nDefaultY = nTiles[1];
129 Path oIndexFile = Paths.get(String.format(m_sTdFileFormat, nDefaultX, m_nZoom, nDefaultX, nDefaultY) + ".ndx");
130 Path oTdFile = Paths.get(String.format(m_sTdFileFormat, nDefaultX, m_nZoom, nDefaultX, nDefaultY));
131 if (!Files.exists(oTdFile) || !Files.exists(oIndexFile))
132 return;
133
134 ArrayList<byte[]> yIdsToRender = new ArrayList();
135 byte[] yIdBuf = new byte[16];
136 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oIndexFile))))
137 {
138 while (oIn.available() > 0)
139 {
140 CtrlIndex oIndex = new CtrlIndex(oIn);
141
142 if (oIndex.m_lStart >= lNow || oIndex.m_lEnd > lNow) // everything valid now and in the future add to tile
143 yIdsToRender.add(oIndex.m_yId);
144 }
145 }
146 Introsort.usort(yIdsToRender, TrafCtrl.ID_COMP);
147 double[] dClipBounds = TileUtil.getClippingBounds(nZ, nX, nY, dBounds);
148
149 int nExtent = 4096;
150// int nExtent = Mercator.getExtent(nZ);
151 VectorTile.Tile.Layer.Builder oLayerBuilder = VectorTile.Tile.Layer.newBuilder();
152 oLayerBuilder.setVersion(2);
153 oLayerBuilder.setExtent(nExtent);
154 VectorTile.Tile.Feature.Builder oFeatureBuilder = VectorTile.Tile.Feature.newBuilder();
155 VectorTile.Tile.Value.Builder oValueBuilder = VectorTile.Tile.Value.newBuilder();
156 int[] nCur = new int[2];
157 int[] nPointBuffer = Arrays.newIntArray(1024);
158 Area oPolyClip = TileUtil.getClippingArea(nZ, nX, nY, dClipBounds);
159
160 ArrayList<TdLayer> oLayers = new ArrayList();
161 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oTdFile))))
162 {
163 while (oIn.available() > 0)
164 {
165 TdLayer oTemp = new TdLayer(oIn, false);
166 int nFeatures = oIn.readInt();
167 for (int nIndex = 0; nIndex < nFeatures; nIndex++)
168 {
169 oIn.read(yIdBuf);
170 int nBytesToSkip = oIn.readInt();
171 int nSearch = Collections.binarySearch(yIdsToRender, yIdBuf, TrafCtrl.ID_COMP);
172 if (nSearch < 0)
173 oIn.skipBytes(nBytesToSkip);
174 else
175 oTemp.add(new TdFeature(oIn, oTemp.m_oKeys.length * 2, yIdBuf));
176 }
177 int nIndex = Collections.binarySearch(oLayers, oTemp);
178 if (nIndex < 0)
179 {
180 oLayers.add(~nIndex, oTemp);
181 }
182 else
183 {
184 oLayers.get(nIndex).addAll(oTemp);
185 }
186
187 }
188 }
189
190 VectorTile.Tile.Builder oTile = VectorTile.Tile.newBuilder();
191 ArrayList<byte[]> oIds = new ArrayList();
192 for (TdLayer oLayer : oLayers)
193 {
194 oIds.clear();
195 oLayerBuilder.clearFeatures();
196 oLayerBuilder.clearKeys();
197 oLayerBuilder.clearValues();
198 oLayerBuilder.setName(oLayer.m_sName);
199 int nIdKeyIndex = oLayer.m_oKeys.length;
200 for (String sKey : oLayer.m_oKeys)
201 {
202 oLayerBuilder.addKeys(sKey);
203 }
204 oLayerBuilder.addKeys("sponge_id");
205
206 int nIdValueOffset = oLayer.m_oValues.length;
207 for (String sValue : oLayer.m_oValues)
208 {
209 oValueBuilder.setStringValue(sValue);
210 oLayerBuilder.addValues(oValueBuilder.build());
211 }
212 for (TdFeature oFeature : oLayer)
213 {
214 int nIndex = Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP);
215 if (nIndex < 0)
216 oIds.add(~nIndex, oFeature.m_yCtrlId);
217 }
218
219 for (byte[] yId : oIds)
220 {
221 oValueBuilder.setStringValue(TrafCtrl.getId(yId));
222 oLayerBuilder.addValues(oValueBuilder.build());
223 }
224
225 switch (oLayer.m_yGeoType)
226 {
227 case TdLayer.POINT:
228 {
229 oFeatureBuilder.setType(VectorTile.Tile.GeomType.POINT);
230 for (TdFeature oFeature : oLayer)
231 {
232 double[] dGeo = oFeature.m_dGeo;
233 if (!Geo.isInside(dGeo[1], dGeo[2], dClipBounds[3], dClipBounds[2], dClipBounds[1], dClipBounds[0], 0))
234 continue;
235
236 for (int nTag : oFeature.m_nTags)
237 oFeatureBuilder.addTags(nTag);
238 oFeatureBuilder.addTags(nIdKeyIndex);
239 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
240 TileUtil.addMercPointToFeature(oFeatureBuilder, nCur, dBounds, nExtent, dGeo[1], dGeo[2]);
241 if (oFeatureBuilder.getGeometryCount() > 0)
242 oLayerBuilder.addFeatures(oFeatureBuilder.build());
243 nCur[0] = nCur[1] = 0;
244 oFeatureBuilder.clearGeometry();
245 oFeatureBuilder.clearTags();
246 }
247 break;
248 }
249 case TdLayer.LINESTRING:
250 {
251 oFeatureBuilder.setType(VectorTile.Tile.GeomType.LINESTRING);
252 ArrayList<double[]> oClipped = new ArrayList();
253 for (TdFeature oFeature : oLayer)
254 {
255 double[] dGeo = oFeature.m_dGeo;
256 oClipped.clear();
257 TileUtil.clipLineString(dGeo, dClipBounds, oClipped);
258 if (oClipped.isEmpty())
259 continue;
260
261 for (int nTag : oFeature.m_nTags)
262 oFeatureBuilder.addTags(nTag);
263 oFeatureBuilder.addTags(nIdKeyIndex);
264 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
265 for (double[] dLine : oClipped)
266 {
267 nPointBuffer = TileUtil.addMercLinestring(oFeatureBuilder, nCur, dBounds, nExtent, dLine, nPointBuffer);
268 if (oFeatureBuilder.getGeometryCount() > 0)
269 oLayerBuilder.addFeatures(oFeatureBuilder.build());
270 nCur[0] = nCur[1] = 0;
271 oFeatureBuilder.clearGeometry();
272 oFeatureBuilder.clearTags();
273 }
274 }
275 break;
276 }
277 case TdLayer.POLYGON:
278 {
279 oFeatureBuilder.setType(VectorTile.Tile.GeomType.POLYGON);
280 for (TdFeature oFeature : oLayer)
281 {
282 double[] dGeo = oFeature.m_dGeo;
283 Area oArea = Geo.getArea(dGeo, 1);
284 if (oArea == null)
285 continue;
286 oArea.intersect(oPolyClip);
287 if (oArea.isEmpty())
288 continue;
289
290 for (int nTag : oFeature.m_nTags)
291 oFeatureBuilder.addTags(nTag);
292 oFeatureBuilder.addTags(nIdKeyIndex);
293 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
294 nPointBuffer = TileUtil.addPolygon(oFeatureBuilder, nCur, dBounds, nExtent, oArea, nPointBuffer);
295 if (oFeatureBuilder.getGeometryCount() > 0)
296 oLayerBuilder.addFeatures(oFeatureBuilder.build());
297 nCur[0] = nCur[1] = 0;
298 oFeatureBuilder.clearGeometry();
299 oFeatureBuilder.clearTags();
300 }
301 break;
302 }
303 default:
304 {
305 oFeatureBuilder.setType(VectorTile.Tile.GeomType.UNKNOWN);
306 break;
307 }
308 }
309 if (oLayerBuilder.getFeaturesCount() > 0)
310 oTile.addLayers(oLayerBuilder.build());
311 }
312
313 oResponse.setContentType("application/x-protobuf");
314 if (oTile.getLayersCount() > 0)
315 oTile.build().writeTo(oResponse.getOutputStream());
316 }
