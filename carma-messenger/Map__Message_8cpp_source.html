22 boost::optional<j2735_v2x_msgs::msg::MapData> Map_Message::decode_map_message(std::vector<uint8_t> &binary_array)
23 {
24 j2735_v2x_msgs::msg::MapData output;
25
26 // decode results
27 asn_dec_rval_t rval;
28 MessageFrame_t *message = nullptr;
29
30 // copy from vector to array
31 size_t len = binary_array.size();
32
33 uint8_t buf[len];
34 std::copy(binary_array.begin(), binary_array.end(), buf);
35 // use asn1c lib to decode
36
37 rval = uper_decode(0, &asn_DEF_MessageFrame, (void **) &message, buf, len, 0, 0);
38 if (rval.code != RC_OK)
39 {
40 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);

43

45
46 // Time Stamp

52 }
53 else
54 {
55 output.time_stamp_exists = false;
56 }
57

91 else
92 {

218 if (rseg->name)

220 rs.name_exists = true;
221 DescriptiveName_t* nm = rseg->name;
222 std::string n;
223
224 for (size_t i = 0; i < nm->size; i++)
225 {
226 n.push_back(nm->buf[i]);
227 }
228 rs.name = n;
229 }
230 else
231 {
232 rs.name_exists = false;
233 }
234
235 if (rseg->speedLimits)
236 {
237 rs.speed_limits_exists = true;
238
239 for (size_t i = 0; i < rseg->speedLimits->list.count; i++)
240 {
241 j2735_v2x_msgs::msg::RegulatorySpeedLimit sl;
242
243 sl.speed = rseg->speedLimits->list.array[i]->speed;
244 sl.type.speed_limit_type = rseg->speedLimits->list.array[i]->type;
245 rs.speed_limits.speed_limits.push_back(sl);
246 }
247 } // end speed limits
248 else
249 {
250 rs.speed_limits_exists = false;
251 }
252
253 if (rseg->refPoint.elevation)
254 {
255 rs.ref_point.elevation_exists = true;
256 rs.ref_point.elevation = *rseg->refPoint.elevation;
257 }
258 else
259 {
260 rs.ref_point.elevation_exists = false;
261 }
262 rs.ref_point.latitude = rseg->refPoint.lat;
263 rs.ref_point.longitude = rseg->refPoint.Long;
264
265 // RLS
266
267 for (size_t i = 0; i < rseg->roadLaneSet.list.count; i++)
268 {
269 j2735_v2x_msgs::msg::GenericLane gl;
270 gl = decode_generic_lane(rseg->roadLaneSet.list.array[i]);
271 rs.road_lane_set.road_lane_set_list.push_back(gl);
272 }
273
274 output.road_segments.road_segment_list.push_back(rs);
275 }
276 } // end Road Segments
277
278 // Data Parameters
279 if (map_msg.dataParameters)
280 {
281 output.data_parameters_exists = true;
282
283 for (size_t i = 0; i < map_msg.dataParameters->geoidUsed->size; i++)
284 {
285 output.data_parameters.geoid_used.push_back(map_msg.dataParameters->geoidUsed->buf[i]);
286 }
287
288 for (size_t i = 0; i < map_msg.dataParameters->lastCheckedDate->size; i++)
289 {
290 output.data_parameters.last_checked_date.push_back(map_msg.dataParameters->lastCheckedDate->buf[i]);
291 }
292
293 for (size_t i = 0; i < map_msg.dataParameters->processAgency->size; i++)
294 {
295 output.data_parameters.process_agency.push_back(map_msg.dataParameters->processAgency->buf[i]);
296 }
297
298 for (size_t i = 0; i < map_msg.dataParameters->processMethod->size; i++)
299 {
300 output.data_parameters.last_checked_date.push_back(map_msg.dataParameters->processMethod->buf[i]);
301 }
302 }
303 else
304 {
305 output.data_parameters_exists = false;
306 }
307
308 // Restriction List
309 if (map_msg.restrictionList)
310 {
311 output.restriction_list_exists = true;
312
313 for (size_t i = 0; i < map_msg.restrictionList->list.count; i++)
314 {
315 RestrictionClassAssignment_t* rca = map_msg.restrictionList->list.array[i];
316
317 j2735_v2x_msgs::msg::RestrictionClassAssignment rclass;
318 rclass.id = rca->id;
319 output.restriction_list.restriction_class_list.push_back(rclass);
320 }
321 } // end Restriction List
322 else
323 {
324 output.restriction_list_exists = false;
325 }
326
327 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
328 return output;

356 // Lane Attributes

360
361 // - LaneType
362 gl.lane_attributes.lane_type.choice = (int)g_lane->laneAttributes.laneType.present - 1; // j2735 ROS msg starts from index 0
363
364 gl.lane_attributes.lane_type.crosswalk.lane_attributes_crosswalk = *g_lane->laneAttributes.laneType.choice.crosswalk.buf >>
365 g_lane->laneAttributes.laneType.choice.crosswalk.bits_unused;
366 gl.lane_attributes.lane_type.median.lane_attributes_barrier = *g_lane->laneAttributes.laneType.choice.median.buf >>
367 g_lane->laneAttributes.laneType.choice.median.bits_unused;

373 g_lane->laneAttributes.laneType.choice.sidewalk.bits_unused;
374 gl.lane_attributes.lane_type.striping.lane_attributes_striping = *g_lane->laneAttributes.laneType.choice.striping.buf >>
375 g_lane->laneAttributes.laneType.choice.striping.bits_unused;
376 gl.lane_attributes.lane_type.tracked_vehicle.lane_attributes_trackedvehicle = *g_lane->laneAttributes.laneType.choice.trackedVehicle.buf >>

390 if (g_lane->connectsTo)
391 {
392 gl.connects_to_exists = true;
393
394 for (size_t c = 0; c < g_lane->connectsTo->list.count; c++)
395 {
396 Connection_t* ct = g_lane->connectsTo->list.array[c];
397
398 j2735_v2x_msgs::msg::Connection entry;
399
400 // Connection ID
401 if (ct->connectionID)
402 {
403 entry.connection_id_exists = true;
404 entry.connection_id = *ct->connectionID;
405 }
406 else
407 {
408 entry.connection_id_exists = false;
409 }
410
411 // Connecting Lane
412 entry.connecting_lane.lane = ct->connectingLane.lane;
413 if (ct->connectingLane.maneuver)
414 {
415 entry.connecting_lane.maneuver_exists = true;
416 uint16_t binary = ct->connectingLane.maneuver->buf[0] >> 4;
417 unsigned int maneuver_type;
418 maneuver_type = 4;
419
420 // e.g. shift the binary right until it equals to 1 (0b00000001) to determine the location of the non-zero bit
421 for (int m = 0; m < ct->connectingLane.maneuver->size; m++)
422 {
423 if ((int)binary == 1)
424 {
425 entry.connecting_lane.maneuver.allowed_maneuvers = maneuver_type;
426 break;
427 }
428 else
429 {
430 maneuver_type -= 1;
431 binary = binary >> 1;
432 }
433 }
434 }
435 else
436 {
437 entry.connecting_lane.maneuver_exists = false;
438 entry.connecting_lane.maneuver.allowed_maneuvers = 0;
439 }
440
441 // Remote Intersection
442 if (ct->remoteIntersection)

451 }

483 gl.connects_to.connect_to_list.push_back(entry);

485 }
486 else
487 {
488 gl.connects_to_exists = false;
