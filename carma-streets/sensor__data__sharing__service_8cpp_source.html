36 bool sds_service::initialize() {
37 if (!streets_service::initialize() ) {
38 SPDLOG_ERROR("Failed to initialize streets service base!");
39 return false;
40 }
41 SPDLOG_DEBUG("Intializing Sensor Data Sharing Service");
42 const std::string lanlet2_map = streets_service::get_system_config("LANELET2_MAP", "/home/carma-streets/MAP/Intersection.osm");
43 if (!read_lanelet_map(lanlet2_map)){
44 SPDLOG_ERROR("Failed to read lanlet2 map {0} !", lanlet2_map);
45 return false;
46 }
47 // Read sensor configuration file and get WSG84 location/origin reference frame.
48 const std::string sensor_config_file = streets_service::get_system_config("SENSOR_JSON_FILE_PATH", "/home/carma-streets/sensor_configurations/sensors.json");
49 const std::string sensor_id = ss::streets_configuration::get_string_config("sensor_id");
50 const lanelet::BasicPoint3d pose = parse_sensor_location(sensor_config_file, sensor_id);
51 this->sdsm_reference_point = this->map_projector->reverse(pose);
52
53 // Initialize SDSM Kafka producer
54 const std::string sdsm_topic = ss::streets_configuration::get_string_config("sdsm_producer_topic");
55 const std::string detection_topic = ss::streets_configuration::get_string_config("detection_consumer_topic");
56 const std::string sdsm_geo_reference = ss::streets_configuration::get_string_config("sdsm_geo_reference");
57 // Get Infrastructure ID for SDSM messages
58 this->_infrastructure_id = streets_service::get_system_config("INFRASTRUCTURE_ID", "");
59 return initialize_kafka_producer(sdsm_topic, sdsm_producer) && initialize_kafka_consumer(detection_topic, detection_consumer);
60 }

63

87 void sds_service::consume_detections(){
88 if ( !detection_consumer ) {
89 throw std::runtime_error("Detection consumer is null!");
90 }
91 SPDLOG_DEBUG("Attempting to consume detections ...");
92 detection_consumer->subscribe();
93 while ( detection_consumer->is_running() ) {
94 try{
95 const std::string payload = detection_consumer->consume(1000);
96 if (payload.length() != 0)
97 {
98 auto detected_object = streets_utils::messages::detected_objects_msg::from_json(payload);
99 // Get delay of detected object
100 auto delay = static_cast<int64_t>(ss::streets_clock_singleton::time_in_ms()) - static_cast<int64_t>(detected_object._timestamp);
101 SPDLOG_DEBUG("Detection Delay : {0}ms!", delay);
102 // if delay is greater than 500 ms skip detection to get more recent data
103 if ( delay >= 500 ) {
104 SPDLOG_WARN("Skipping incoming detection at {0}ms is not current or has invalid timestamp of {1}ms!" , ss::streets_clock_singleton::time_in_ms(), detected_object._timestamp );
105 continue;
106 }
107 // if delay is negative, and service is in simulation mode
108 // the detection message was processed before time sync message. Wait on time sync message
109 else if (is_simulation_mode() && delay < 0 ) {
110 SPDLOG_WARN("Current sim time {0} waiting for sim time {1}ms from detection ...",ss::streets_clock_singleton::time_in_ms(), detected_object._timestamp );
111 ss::streets_clock_singleton::sleep_for(abs(delay));
112 }
113 // If delay is negative and service is not in simulation mode
114 // indicates sensor and service are not time sychronized.
115 else if( delay < 0 ) {
116 SPDLOG_WARN(
117 R"(Skipping incoming detection. Current time is {0}ms and detection time stamp is {1}ms.
118 Sensor Data Sharing Service and sensor producing detections to not appear to be time synchronized.)",
119 ss::streets_clock_singleton::time_in_ms(),
120 detected_object._timestamp );
121 continue;
122
123 }
124
125 // Write Lock
126 std::unique_lock lock(detected_objects_lock);
127 detected_objects[detected_object._object_id] = detected_object;
128 SPDLOG_DEBUG("Detected Object List Size {0} after consumed: {1}", detected_objects.size(), payload);
129
130 }
131 }
132 catch (const streets_utils::json_utils::json_parse_exception &e) {
133 SPDLOG_ERROR("Exception occured consuming detection message : {0}", e.what());
134 }
135 }
136 SPDLOG_ERROR("Something went wrong, no longer consuming detections." );
137
138 }

140 void sds_service::produce_sdsms() {
141 if ( !sdsm_producer ) {
142 throw std::runtime_error("SDSM consumer is null!");
143 }
144 SPDLOG_INFO("Starting SDSM Producer!");
145 while ( sdsm_producer->is_running() ) {
146 try{
147 if ( !detected_objects.empty() ) {
148 streets_utils::messages::sdsm::sensor_data_sharing_msg msg = create_sdsm();
149 const std::string json_msg = streets_utils::messages::sdsm::to_json(msg);
150 SPDLOG_DEBUG("Sending SDSM : {0}", json_msg);
151 sdsm_producer->send(json_msg);
152 // Message Count max is 127, reset after max value
153 if ( this->_message_count <= 127) {
154 this->_message_count++;
155 }else {
156 this->_message_count = 0;
157 }
158 // Write Lock
159 std::unique_lock lock(detected_objects_lock);
160 // Clear detected object
161 detected_objects.clear();
162 }
163 }
164 catch( const streets_utils::json_utils::json_parse_exception &e) {
165 SPDLOG_ERROR("Exception occurred producing SDSM : {0}", e.what());
166 }
167 ss::streets_clock_singleton::sleep_for(100); // Sleep for 100 ms between publish
168 }
169 SPDLOG_CRITICAL("SDSM Producers no longer running.");
170
171 }

173 streets_utils::messages::sdsm::sensor_data_sharing_msg sds_service::create_sdsm() {
174 streets_utils::messages::sdsm::sensor_data_sharing_msg msg;
175 // Read lock
176 uint64_t timestamp = ss::streets_clock_singleton::time_in_ms();
177 msg._time_stamp = to_sdsm_timestamp(timestamp);
178 // Populate with rolling counter
179 msg._msg_count = this->_message_count;
180 // Populate with infrastructure id
181 msg._source_id = this->_infrastructure_id;
182 // Populate equipement type
183 msg._equipment_type = sdsm::equipment_type::RSU;
184 // Polulate ref position
185 msg._ref_positon = to_position_3d(this->sdsm_reference_point);
186 std::shared_lock lock(detected_objects_lock);
187 for (const auto &[object_id, object] : detected_objects){
188 auto ned_object = detected_object_enu_to_ned(object);
189 auto detected_object_data = to_detected_object_data(ned_object,timestamp);
190 // TODO: Update time offset. Currently CARMA-Streets detected object message does not support timestamp
191 // This is a bug and needs to be addressed.
192 msg._objects.push_back(detected_object_data);
193 }
194 return msg;
195 }

198 void sds_service::start() {
199 SPDLOG_DEBUG("Starting Sensor Data Sharing Service");
200 streets_service::start();
201 std::thread detection_thread(&sds_service::consume_detections, this);
202 std::thread sdsm_thread(&sds_service::produce_sdsms, this);
203 detection_thread.join();
204 sdsm_thread.join();
205 }

207 streets_utils::messages::sdsm::position_3d to_position_3d(const lanelet::GPSPoint &ref_position) {
208 streets_utils::messages::sdsm::position_3d position;
209 // Convert to 1/10 of microdegrees
210 position._longitude = static_cast<int>(ref_position.lon * 1e7);
211 position._latitude = static_cast<int>(ref_position.lat *1e7);
212 // Convert 0.1 meters
213 position._elevation = static_cast<int>(ref_position.ele * 10);
214 return position;
