21int DriverWrapper::run()
22{
23 nh_.reset(new ros::NodeHandle);
24 private_nh_.reset(new ros::NodeHandle("~"));
25
26 //Setup initial status
27 status_.name = ros::this_node::getName();
28 status_.status = cav_msgs::DriverStatus::OFF;
29
30 //Initialize pubs and subs
31 driver_status_pub_ = nh_->advertise<cav_msgs::DriverStatus>("driver_discovery", 1);
32 system_alert_sub_ = nh_->subscribe("/system_alert", 10, &DriverWrapper::system_alert_cb, this);
33 ros::Timer timer = private_nh_->createTimer(ros::Duration(1), &DriverWrapper::status_publish_timer, this);
34
35 ROS_INFO_STREAM("Driver Initializing");
36
37 initialize();
38 ros::Rate r(spin_rate_);
39 while(ros::ok() && !shutdown_)
40 {
41 pre_spin();
42 ros::spinOnce();
43 post_spin();
44 r.sleep();
45 }
46
47 ROS_INFO_STREAM("Driver Shutting Down");
48 shutdown();
49 ros::shutdown();
50}
