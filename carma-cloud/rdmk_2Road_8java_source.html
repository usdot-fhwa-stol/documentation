92 public void setLaneIds(double dMaxStep)
93 {
94 double dTol = dMaxStep * 2;
95 ArrayList<Lane> oPrevLanes = new ArrayList();
96 get(0).getLanes(oPrevLanes);
97 int nNextId = Integer.MIN_VALUE;
98 for (Lane oLane : oPrevLanes)
99 {
100 oLane.m_nLaneIdByRoad = oLane.m_nLaneIndex;
101 int nComp = Math.abs(oLane.m_nLaneIndex);
102 if (nComp > nNextId)
103 nNextId = nComp;
104 }
105
106 for (int nIndex = 1; nIndex < size(); nIndex++)
107 {
108 ArrayList<Lane> oCurrLanes = new ArrayList();
109 get(nIndex).getLanes(oCurrLanes);
110 for (Lane oCurr: oCurrLanes)
111 {
112 if (oCurr.m_nLaneIndex == 0)
113 continue;
114 double[] dCurr = oCurr.m_dCenters.get(0);
115 oCurr.m_nLaneIdByRoad = Integer.MIN_VALUE;
116 for (Lane oPrev: oPrevLanes)
117 {
118 if (oPrev.m_nLaneIndex == 0)
119 continue;
120 double[] dPrev = oPrev.m_dCenters.get(oPrev.m_dCenters.size() - 1);
121 int nSize = Arrays.size(dPrev);
122
123 double dDist = Geo.distance(dCurr[5], dCurr[6], dPrev[nSize - 3], dPrev[nSize - 2]);
124// if (dDist > dTol)
125// System.out.println(dDist);
126 if (dDist <= dTol && oCurr.m_nType == oCurr.m_nType)
127 {
128 oCurr.m_nLaneIdByRoad = oPrev.m_nLaneIdByRoad;
129 break;
130 }
131 }
132
133 if (oCurr.m_nLaneIdByRoad == Integer.MIN_VALUE)
134 {
135 ++nNextId;
136 if (oCurr.m_nLaneIndex < 0)
137 oCurr.m_nLaneIdByRoad = -nNextId;
138 else
139 oCurr.m_nLaneIdByRoad = nNextId;
140 }
141 }
142 oPrevLanes.clear();
143 oPrevLanes.addAll(oCurrLanes);
144 }

154 oSection.getLanes(oLanes);

163 int nIndex = Collections.binarySearch(oRdMkInfo, nSearch, RDMKCOMP);
164 if (nIndex < 0)
165 {
166 oLane.m_nRdMkTags = Arrays.insert(oLane.m_nRdMkTags, nSearch[1], 1);
167 oLane.m_nRdMkTags = Arrays.insert(oLane.m_nRdMkTags, nSearch[2], 2);
168 oRdMkInfo.add(~nIndex, oLane.m_nRdMkTags);

170 else

172 int[] nRdMkInfo = oRdMkInfo.get(nIndex);
173 Iterator<int[]> oIt = Arrays.iterator(oLane.m_nRdMkTags, new int[8], 1, 8);
174 while (oIt.hasNext())
175 {
176 int[] nTags = oIt.next();
177 nRdMkInfo = Arrays.add(nRdMkInfo, nTags);
178 }
179 }
180 for (double[] dCenter : oLane.m_dCenters)
181 {
182 oCtrlLineArcs.add(new CtrlLineArcs(m_nId, oSection.m_nId, oLane.m_nLaneIndex, oLane.m_nLaneIdByRoad, oLane.m_nType, dCenter, dMaxStep));
