51 std::vector<std::pair<int, lanelet::BasicPoint2d>> YieldPlugin::detect_trajectories_intersection(std::vector<lanelet::BasicPoint2d> self_trajectory, std::vector<lanelet::BasicPoint2d> incoming_trajectory) const
52 {
53 std::vector<std::pair<int, lanelet::BasicPoint2d>> intersection_points;
54 boost::geometry::model::linestring<lanelet::BasicPoint2d> self_traj;
55 for (auto tpp:self_trajectory)
56 {
57 boost::geometry::append(self_traj, tpp);
58 }
59 // distance to consider trajectories colliding (chosen based on lane width and vehicle size)
60 for (size_t i=0; i<incoming_trajectory.size(); i++)
61 {
62 double res = boost::geometry::distance(incoming_trajectory[i], self_traj);
63
64 if (fabs(res) <= config_.intervehicle_collision_distance)
65 {
66 intersection_points.push_back(std::make_pair(i, incoming_trajectory[i]));
67 }
68 }
69 return intersection_points;
70 }

72 std::vector<lanelet::BasicPoint2d> YieldPlugin::convert_eceftrajectory_to_mappoints(const carma_v2x_msgs::msg::Trajectory& ecef_trajectory) const
73 {
74 carma_planning_msgs::msg::TrajectoryPlan trajectory_plan;
75 std::vector<lanelet::BasicPoint2d> map_points;
76
77 lanelet::BasicPoint2d first_point = ecef_to_map_point(ecef_trajectory.location);
78
79 map_points.push_back(first_point);
80 auto curr_point = ecef_trajectory.location;
81
82 for (size_t i = 0; i<ecef_trajectory.offsets.size(); i++)
83 {
84 lanelet::BasicPoint2d offset_point;
85 curr_point.ecef_x += ecef_trajectory.offsets[i].offset_x;
86 curr_point.ecef_y += ecef_trajectory.offsets[i].offset_y;
87 curr_point.ecef_z += ecef_trajectory.offsets[i].offset_z;
88
89 offset_point = ecef_to_map_point(curr_point);
90
91 map_points.push_back(offset_point);
92 }
93
94 return map_points;
95 }

97 lanelet::BasicPoint2d YieldPlugin::ecef_to_map_point(const carma_v2x_msgs::msg::LocationECEF& ecef_point) const
98 {
99
100 if (!map_projector_) {
101 throw std::invalid_argument("No map projector available for ecef conversion");
102 }
103
104 lanelet::BasicPoint3d map_point = map_projector_->projectECEF( { (double)ecef_point.ecef_x/100.0, (double)ecef_point.ecef_y/100.0, (double)ecef_point.ecef_z/100.0 } , 1);
105
106 return lanelet::traits::to2D(map_point);

131 void YieldPlugin::mobilityrequest_cb(const carma_v2x_msgs::msg::MobilityRequest::UniquePtr msg)
132 {
133 carma_v2x_msgs::msg::MobilityRequest incoming_request = *msg;
134 carma_planning_msgs::msg::LaneChangeStatus lc_status_msg;
135 if (incoming_request.strategy == "carma/cooperative-lane-change")
136 {
137 if (!map_projector_) {
138 RCLCPP_ERROR_STREAM(nh_->get_logger(),"Cannot process mobility request as map projection is not yet set!");
139 return;
140 }
141 if (incoming_request.plan_type.type == carma_v2x_msgs::msg::PlanType::CHANGE_LANE_LEFT || incoming_request.plan_type.type == carma_v2x_msgs::msg::PlanType::CHANGE_LANE_RIGHT)
142 {
143 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Cooperative Lane Change Request Received");
144 lc_status_msg.status = carma_planning_msgs::msg::LaneChangeStatus::REQUEST_RECEIVED;
145 lc_status_msg.description = "Received lane merge request";
146 if (incoming_request.m_header.recipient_id == config_.vehicle_id)
147 {
148 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"CLC Request correctly received");
149 }
150 // extract mobility header
151 std::string req_sender_id = incoming_request.m_header.sender_id;
152 std::string req_plan_id = incoming_request.m_header.plan_id;
153 // extract mobility request
154 carma_v2x_msgs::msg::LocationECEF ecef_location = incoming_request.location;
155 carma_v2x_msgs::msg::Trajectory incoming_trajectory = incoming_request.trajectory;
156 std::string req_strategy_params = incoming_request.strategy_params;
157 clc_urgency_ = incoming_request.urgency;
158 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"received urgency: " << clc_urgency_);
159
160 // Parse strategy parameters
161 using boost::property_tree::ptree;
162 ptree pt;
163 std::istringstream strstream(req_strategy_params);
164 boost::property_tree::json_parser::read_json(strstream, pt);
165 int req_traj_speed_full = pt.get<int>("s");
166 int req_traj_fractional = pt.get<int>("f");
167 int start_lanelet_id = pt.get<int>("sl");
168 int end_lanelet_id = pt.get<int>("el");
169 double req_traj_speed = (double)req_traj_speed_full + (double)(req_traj_fractional)/10.0;
170 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"req_traj_speed" << req_traj_speed);
171 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"start_lanelet_id" << start_lanelet_id);
172 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"end_lanelet_id" << end_lanelet_id);
173
174 std::vector<lanelet::BasicPoint2d> req_traj_plan = {};
175
176 req_traj_plan = convert_eceftrajectory_to_mappoints(incoming_trajectory);
177
178 double req_expiration_sec = (double)incoming_request.expiration;
179 double current_time_sec = nh_->now().seconds();
180
181 bool response_to_clc_req = false;
182 // ensure there is enough time for the yield
183 double req_plan_time = req_expiration_sec - current_time_sec;
184 double req_timestamp = (double)incoming_request.m_header.timestamp / 1000.0 - current_time_sec;
185 set_incoming_request_info(req_traj_plan, req_traj_speed, req_plan_time, req_timestamp);
186
187
188 if (req_expiration_sec - current_time_sec >= config_.tpmin && cooperative_request_acceptable_)
189 {
190 timesteps_since_last_req_ = 0;
191 lc_status_msg.status = carma_planning_msgs::msg::LaneChangeStatus::REQUEST_ACCEPTED;
192 lc_status_msg.description = "Accepted lane merge request";
193 response_to_clc_req = true;
194 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"CLC accepted");
195 }
196 else
197 {
198 lc_status_msg.status = carma_planning_msgs::msg::LaneChangeStatus::REQUEST_REJECTED;
199 lc_status_msg.description = "Rejected lane merge request";
200 response_to_clc_req = false;
201 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"CLC rejected");
202 }
203 carma_v2x_msgs::msg::MobilityResponse outgoing_response = compose_mobility_response(req_sender_id, req_plan_id, response_to_clc_req);
204 mobility_response_publisher_(outgoing_response);
205 lc_status_msg.status = carma_planning_msgs::msg::LaneChangeStatus::RESPONSE_SENT;
206 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"response sent");
207 }
208 }
209 lc_status_publisher_(lc_status_msg);
210
211 }

228 void YieldPlugin::plan_trajectory_callback(
229 carma_planning_msgs::srv::PlanTrajectory::Request::SharedPtr req,
230 carma_planning_msgs::srv::PlanTrajectory::Response::SharedPtr resp)
231{
232 if (req->initial_trajectory_plan.trajectory_points.size() < 2){
233 throw std::invalid_argument("Empty Trajectory received by Yield");
234 }
235 carma_planning_msgs::msg::TrajectoryPlan original_trajectory = req->initial_trajectory_plan;
236 carma_planning_msgs::msg::TrajectoryPlan yield_trajectory;
237
238 // seperating cooperative yield with regular object detection for better performance.
239 if (config_.enable_cooperative_behavior && clc_urgency_ > config_.acceptable_urgency)
240 {
241 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Only consider high urgency clc");
242 if (timesteps_since_last_req_ < config_.acceptable_passed_timesteps)
243 {
244 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Yield for CLC. We haven't received an updated negotiation this timestep");
245 yield_trajectory = update_traj_for_cooperative_behavior(original_trajectory, req->vehicle_state.longitudinal_vel);
246 timesteps_since_last_req_++;
247 }
248 else
249 {
250 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"unreliable CLC communication, switching to object avoidance");
251 yield_trajectory = update_traj_for_object(original_trajectory, req->vehicle_state.longitudinal_vel); // Compute the trajectory
252 }
253 }
254 else
255 {
256 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Yield for object avoidance");
257 yield_trajectory = update_traj_for_object(original_trajectory, req->vehicle_state.longitudinal_vel); // Compute the trajectory
258 }
259 yield_trajectory.header.frame_id = "map";
260 yield_trajectory.header.stamp = nh_->now();
261 yield_trajectory.trajectory_id = original_trajectory.trajectory_id;
262 yield_trajectory.initial_longitudinal_velocity = original_trajectory.initial_longitudinal_velocity;//copy the original trajectory's desired speed for now.
263
264 resp->trajectory_plan = yield_trajectory;
265
266 }

268 carma_planning_msgs::msg::TrajectoryPlan YieldPlugin::update_traj_for_cooperative_behavior(const carma_planning_msgs::msg::TrajectoryPlan& original_tp, double current_speed)
269 {
270 carma_planning_msgs::msg::TrajectoryPlan cooperative_trajectory;
271
272 double initial_pos = 0;
273 double goal_pos;
274 double initial_velocity = current_speed;
275 double goal_velocity = req_target_speed_;
276 double planning_time = req_target_plan_time_;
277
278 std::vector<lanelet::BasicPoint2d> host_traj_points = {};
279 for (size_t i=0; i<original_tp.trajectory_points.size(); i++)
280 {
281 lanelet::BasicPoint2d traj_point;
282 traj_point.x() = original_tp.trajectory_points[i].x;
283 traj_point.y() = original_tp.trajectory_points[i].y;
284 host_traj_points.push_back(traj_point);
285 }
286
287 std::vector<std::pair<int, lanelet::BasicPoint2d>> intersection_points = detect_trajectories_intersection(host_traj_points, req_trajectory_points_);
288 if (!intersection_points.empty())
289 {
290 lanelet::BasicPoint2d intersection_point = intersection_points[0].second;
291 double dx = original_tp.trajectory_points[0].x - intersection_point.x();
292 double dy = original_tp.trajectory_points[0].y - intersection_point.y();
293 // check if a digital_gap is available
294 double digital_gap = check_traj_for_digital_min_gap(original_tp);
295 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"digital_gap: " << digital_gap);
296 goal_pos = sqrt(dx*dx + dy*dy) - config_.x_gap;
297 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Goal position (goal_pos): " << goal_pos);
298 double collision_time = req_timestamp_ + (intersection_points[0].first * ecef_traj_timestep_) - config_.safety_collision_time_gap;
299 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"req time stamp: " << req_timestamp_);
300 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Collision time: " << collision_time);
301 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"intersection num: " << intersection_points[0].first);
302 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Planning time: " << planning_time);
303 // calculate distance traveled from beginning of trajectory to collision point
304 double dx2 = intersection_point.x() - req_trajectory_points_[0].x();
305 double dy2 = intersection_point.y() - req_trajectory_points_[0].y();
306 // calculate incoming trajectory speed from time and distance between trajectory points
307 double incoming_trajectory_speed = sqrt(dx2*dx2 + dy2*dy2)/(intersection_points[0].first * ecef_traj_timestep_);
308 // calculate goal velocity from request trajectory
309 goal_velocity = std::min(goal_velocity, incoming_trajectory_speed);
310 double min_time = (initial_velocity - goal_velocity)/config_.yield_max_deceleration;
311
312 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"goal_velocity: " << goal_velocity);
313 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"incoming_trajectory_speed: " << incoming_trajectory_speed);
314
315 if (planning_time > min_time)
316 {
317 cooperative_request_acceptable_ = true;
318 cooperative_trajectory = generate_JMT_trajectory(original_tp, initial_pos, goal_pos, initial_velocity, goal_velocity, planning_time);
319 }
320 else
321 {
322 cooperative_request_acceptable_ = false;
323 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"The incoming requested trajectory is rejected, due to insufficient gap");
324 cooperative_trajectory = original_tp;
325 }
326
327 }
328 else
329 {
330 cooperative_request_acceptable_ = true;
331 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"The incoming requested trajectory does not overlap with host vehicle's trajectory");
332 cooperative_trajectory = original_tp;
333 }
334
335 return cooperative_trajectory;
336 }

338 carma_planning_msgs::msg::TrajectoryPlan YieldPlugin::generate_JMT_trajectory(const carma_planning_msgs::msg::TrajectoryPlan& original_tp, double initial_pos, double goal_pos, double initial_velocity, double goal_velocity, double planning_time)
339 {
340 carma_planning_msgs::msg::TrajectoryPlan jmt_trajectory;
341 std::vector<carma_planning_msgs::msg::TrajectoryPlanPoint> jmt_trajectory_points;
342 jmt_trajectory_points.push_back(original_tp.trajectory_points[0]);
343
344 double initial_time = 0;
345 double initial_accel = 0.0;
346 double goal_accel = 0.0;
347
348 double original_max_speed = max_trajectory_speed(original_tp.trajectory_points);
349 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"original_max_speed" << original_max_speed);
350 std::vector<double> values = quintic_coefficient_calculator::quintic_coefficient_calculator(initial_pos,
351 goal_pos,
352 initial_velocity,
353 goal_velocity,
354 initial_accel,
355 goal_accel,
356 initial_time,
357 planning_time);
358
359 std::vector<double> original_traj_downtracks = get_relative_downtracks(original_tp);
360 std::vector<double> calculated_speeds = {};
361 calculated_speeds.push_back(initial_velocity);
362 for(size_t i = 1; i < original_tp.trajectory_points.size(); i++ )
363 {
364 double traj_target_time = i * planning_time / original_tp.trajectory_points.size();
365 double dt_dist = polynomial_calc(values, traj_target_time);
366 double dv = polynomial_calc_d(values, traj_target_time);
367
368 if (dv >= original_max_speed)
369 {
370 dv = original_max_speed;
371 }
372 calculated_speeds.push_back(dv);
373 }
374 // moving average filter
375 std::vector<double> filtered_speeds = moving_average_filter(calculated_speeds, config_.speed_moving_average_window_size);
376
377 double prev_speed = filtered_speeds[0];
378 for(size_t i = 1; i < original_tp.trajectory_points.size(); i++ )
379 {
380 carma_planning_msgs::msg::TrajectoryPlanPoint jmt_tpp;
381 double current_speed = filtered_speeds[i];
382 double traj_target_time = i * planning_time / original_tp.trajectory_points.size();
383 if (current_speed >= config_.max_stop_speed)
384 {
385 double dt = (2 * original_traj_downtracks[i]) / (current_speed + prev_speed);
386 jmt_tpp = original_tp.trajectory_points[i];
387 jmt_tpp.target_time = rclcpp::Time(jmt_trajectory_points[i-1].target_time) + rclcpp::Duration(dt*1e9);
388 jmt_trajectory_points.push_back(jmt_tpp);
389 }
390 else
391 {
392 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"target speed is zero");
393 // if speed is zero, the vehicle will stay in previous location.
394 jmt_tpp = jmt_trajectory_points[i-1];
395 jmt_tpp.target_time = rclcpp::Time(jmt_trajectory_points[0].target_time) + rclcpp::Duration(traj_target_time*1e9);
396 jmt_trajectory_points.push_back(jmt_tpp);
397 }
398 prev_speed = current_speed;
399 }
400
401 jmt_trajectory.header = original_tp.header;
402 jmt_trajectory.trajectory_id = original_tp.trajectory_id;
403 jmt_trajectory.trajectory_points = jmt_trajectory_points;
404 return jmt_trajectory;
405 }

407 carma_planning_msgs::msg::TrajectoryPlan YieldPlugin::update_traj_for_object(const carma_planning_msgs::msg::TrajectoryPlan& original_tp, double initial_velocity)
408 {
409
410 carma_planning_msgs::msg::TrajectoryPlan update_tpp_vector;
411 geometry_msgs::msg::Twist current_velocity;
412 current_velocity.linear.x = initial_velocity;
413
414 std::vector<carma_perception_msgs::msg::RoadwayObstacle> rwol = wm_->getRoadwayObjects();
415 carma_perception_msgs::msg::RoadwayObstacleList rwol2;
416 rwol2.roadway_obstacles = rwol;
417 host_vehicle_size.x = config_.vehicle_length;
418 host_vehicle_size.y = config_.vehicle_width;
419 host_vehicle_size.z = config_.vehicle_height;
420
421
422 std::vector<carma_perception_msgs::msg::RoadwayObstacle> rwol_collision;
423
424 // std::vector<carma_perception_msgs::msg::RoadwayObstacle> rwol_collision = carma_wm::collision_detection::WorldCollisionDetection(rwol2, original_tp, host_vehicle_size, current_velocity, config_.collision_horizon);
425
426 lanelet::BasicPoint2d point(original_tp.trajectory_points[0].x,original_tp.trajectory_points[0].y);
427 double vehicle_downtrack = wm_->routeTrackPos(point).downtrack;
428
429 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"vehicle_downtrack");
430 RCLCPP_DEBUG_STREAM(nh_->get_logger(),vehicle_downtrack);
431
432 for (auto i : rwol2.roadway_obstacles){
433
434 lanelet::BasicPoint2d point_o(i.object.pose.pose.position.x, i.object.pose.pose.position.y);
435 double object_down_track = wm_->routeTrackPos(point_o).downtrack;
436
437 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"object_down_track");
438 RCLCPP_DEBUG_STREAM(nh_->get_logger(),object_down_track);
439
440 double dist_to_object = object_down_track - vehicle_downtrack;
441 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"object_down_track - vehicle_downtrack");
442 RCLCPP_DEBUG_STREAM(nh_->get_logger(),dist_to_object);
443
444 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"i.object.velocity.twist.linear.x");
445 RCLCPP_DEBUG_STREAM(nh_->get_logger(),i.object.velocity.twist.linear.x);
446
447 if(current_velocity.linear.x > 0.0) {
448 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"(object_down_downtrack - vehicle_downtrack)/current_velocity.linear.x");
449 RCLCPP_DEBUG_STREAM(nh_->get_logger(),dist_to_object/current_velocity.linear.x);
450
451 // Check to see if the object is in front of us and has a time-to-collision lower than our
452 // desired horizon
453 if(dist_to_object >= 0 &&
454 dist_to_object / current_velocity.linear.x < config_.collision_horizon) {
455 rwol_collision.push_back(i);
456 }
457 }
458
459 }
460
461 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Roadway Object List (rwol) size: " << rwol.size());
462
463 // correct the input types
464 if(!rwol_collision.empty())
465 {
466 RCLCPP_WARN_STREAM(nh_->get_logger(),"Collision Detected!");
467
468 // Distance from the original trajectory point to the lead vehicle/object
469 double dist_x = rwol_collision[0].object.pose.pose.position.x - original_tp.trajectory_points[0].x;
470 double dist_y = rwol_collision[0].object.pose.pose.position.y - original_tp.trajectory_points[0].y;
471 double x_lead = sqrt(dist_x*dist_x + dist_y*dist_y);
472
473 // roadway object position
474 double gap_time = (x_lead - config_.x_gap)/initial_velocity;
475
476 double collision_time = 0; //\TODO comming from carma_wm collision detection in future (CAR 4288)
477
478 double goal_velocity = rwol_collision[0].object.velocity.twist.linear.x;
479 // determine the safety inter-vehicle gap based on speed
480 double safety_gap = std::max(goal_velocity * gap_time, config_.x_gap);
481 if (config_.enable_adjustable_gap)
482 {
483 // check if a digital_gap is available
484 double digital_gap = check_traj_for_digital_min_gap(original_tp);
485 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"digital_gap: " << digital_gap);
486 // if a digital gap is available, it is replaced as safety gap
487 safety_gap = std::max(safety_gap, digital_gap);
488 }
489 // safety gap is implemented
490 double goal_pos = x_lead - safety_gap;
491
492 if (goal_velocity <= config_.min_obstacle_speed){
493 RCLCPP_WARN_STREAM(nh_->get_logger(),"The obstacle is not moving");
494 }
495
496
497 double initial_time = 0;
498 double initial_pos = 0.0; //relative initial position (first trajectory point)
499
500 double initial_accel = 0;
501 double goal_accel = 0;
502
503 double delta_v_max = fabs(rwol_collision[0].object.velocity.twist.linear.x - max_trajectory_speed(original_tp.trajectory_points));
504 // reference time, is the maximum time available to perform object avoidance (length of a trajectory)
505 double t_ref = (rclcpp::Time(original_tp.trajectory_points[original_tp.trajectory_points.size() - 1].target_time).seconds() - rclcpp::Time(original_tp.trajectory_points[0].target_time).seconds());
506 // time required for comfortable deceleration
507 double t_ph = config_.acceleration_adjustment_factor * delta_v_max / config_.yield_max_deceleration;
508
509 // planning time for object avoidance
510 double tp = 0;
511
512 if(t_ph > config_.tpmin && t_ref > t_ph)
513 {
514 tp = t_ph;
515 }
516 else if(t_ph < config_.tpmin)
517 {
518 tp = config_.tpmin;
519 }
520 else
521 {
522 tp = t_ref;
523 }
524
525 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"Object avoidance planning time: " << tp);
526
527 update_tpp_vector = generate_JMT_trajectory(original_tp, initial_pos, goal_pos, initial_velocity, goal_velocity, tp);
528
529 return update_tpp_vector;
530 }
531 RCLCPP_DEBUG_STREAM(nh_->get_logger(),"No collision detection, so trajectory not modified.");
532 return original_tp;
533 }

536 std::vector<double> YieldPlugin::get_relative_downtracks(const carma_planning_msgs::msg::TrajectoryPlan& trajectory_plan) const
537 {
538 std::vector<double> downtracks;
539 downtracks.reserve(trajectory_plan.trajectory_points.size());
540 // relative downtrack distance of the fist point is 0.0
541 downtracks[0] = 0.0;
542 for (size_t i=1; i < trajectory_plan.trajectory_points.size(); i++){
543 double dx = trajectory_plan.trajectory_points[i].x - trajectory_plan.trajectory_points[i-1].x;
544 double dy = trajectory_plan.trajectory_points[i].y - trajectory_plan.trajectory_points[i-1].y;
545 downtracks[i] = sqrt(dx*dx + dy*dy);
546 }
547 return downtracks;

573 {
574 double max_speed = 0;

576 {
577 double dx = trajectory_points[i + 1].x - trajectory_points[i].x;
578 double dy = trajectory_points[i + 1].y - trajectory_points[i].y;
579 double d = sqrt(dx*dx + dy*dy);
580 double t = (rclcpp::Time(trajectory_points[i + 1].target_time).seconds() - rclcpp::Time(trajectory_points[i].target_time).seconds());
581 double v = d/t;
582 if(v > max_speed)
583 {
584 max_speed = v;
585 }
586 }
