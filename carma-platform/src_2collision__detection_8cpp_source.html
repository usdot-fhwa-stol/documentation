162 collision_detection::MovingObject ConvertRoadwayObstacleToMovingObject(const cav_msgs::RoadwayObstacle& rwo){
163
164 collision_detection::MovingObject mo;
165
166 mo.object_polygon = ObjectToBoostPolygon<polygon_t>(rwo.object.pose.pose, rwo.object.size);
167
168 std::tuple <__uint64_t,polygon_t> current_pose(0 , mo.object_polygon);
169 mo.fp.push_back(current_pose);
170
171 // Add future polygons for roadway obstacle
172 for (auto i : rwo.object.predictions){
173 std::tuple <__uint64_t,polygon_t> future_object(i.header.stamp.toNSec() / 1000000, ObjectToBoostPolygon<polygon_t>(i.predicted_position, rwo.object.size));
174
175 mo.fp.push_back(future_object);
176 }
177
178 mo.linear_velocity = rwo.object.velocity.twist.linear;
179
180 return mo;

183 collision_detection::MovingObject ConvertVehicleToMovingObject(const cav_msgs::TrajectoryPlan& tp, const geometry_msgs::Vector3& size, const geometry_msgs::Twist& velocity){
184
185 collision_detection::MovingObject v;
186
187 geometry_msgs::Pose pose;
188 pose.position.x = tp.trajectory_points[0].x;
189 pose.position.y = tp.trajectory_points[0].y;
190
191 Eigen::Vector2d vehicle_vector = {tp.trajectory_points[1].x - tp.trajectory_points[0].x , tp.trajectory_points[1].y - tp.trajectory_points[0].y};
192
193 Eigen::Vector2d x_axis = {1, 0};
194
195 double yaw = std::acos(vehicle_vector.dot(x_axis)/(vehicle_vector.norm() * x_axis.norm()));
196
197 tf2::Quaternion vehicle_orientation;
198 vehicle_orientation.setRPY(0, 0, yaw);
199
200 pose.orientation.x = vehicle_orientation.getX();
201 pose.orientation.y = vehicle_orientation.getY();
202 pose.orientation.z = vehicle_orientation.getZ();
203 pose.orientation.w = vehicle_orientation.getW();
204
205 v.object_polygon = ObjectToBoostPolygon<polygon_t>(pose, size);
206 v.linear_velocity = velocity.linear;
207
208 for(size_t i=1; i < tp.trajectory_points.size() - 1; i++){
209
210 vehicle_vector = {tp.trajectory_points[i + 1].x - tp.trajectory_points[i].x , tp.trajectory_points[i+1].y - tp.trajectory_points[i].y};
211 yaw = std::acos(vehicle_vector.dot(x_axis)/(vehicle_vector.norm() * x_axis.norm()));
212
213 tf2::Quaternion orientation;
214 orientation.setRPY(0, 0, yaw);
215
216 geometry_msgs::Pose trajectory_pose;
217 trajectory_pose.position.x = tp.trajectory_points[i].x;
218 trajectory_pose.position.y = tp.trajectory_points[i].y;
219
220 trajectory_pose.orientation.x = orientation.getX();
221 trajectory_pose.orientation.y = orientation.getY();
222 trajectory_pose.orientation.z = orientation.getZ();
223 trajectory_pose.orientation.w = orientation.getW();
224
225 std::tuple <__uint64_t,polygon_t> future_object(tp.trajectory_points[i].target_time.toNSec(),ObjectToBoostPolygon<polygon_t>(trajectory_pose, size));

227 v.fp.push_back(future_object);
228 }

263 collision_detection::MovingObject PredictObjectPosition(collision_detection::MovingObject const &op, __uint64_t target_time){
264
265 int union_polygon_size = 0;
266 for (auto i : op.fp){
267 if( std::get<0>(i) <= target_time) {
268 union_polygon_size = union_polygon_size + std::get<1>(i).outer().size();
269 }
270 }
271
272 std::vector<point_t> union_future_polygon_points;
273 union_future_polygon_points.reserve(union_polygon_size);
274
275 for (auto i : op.fp){
276 if( std::get<0>(i) <= target_time) {
277 union_future_polygon_points.insert( union_future_polygon_points.end(), std::get<1>(i).outer().begin(), std::get<1>(i).outer().end());
278 }
279 }
280
281 polygon_t union_polygon;
282 boost::geometry::assign_points(union_polygon, union_future_polygon_points);
283
284 polygon_t hull_polygon;
285 boost::geometry::convex_hull(union_polygon, hull_polygon);
286
287 std::vector<std::tuple <__uint64_t, collision_detection::polygon_t>> no_future;
288 collision_detection::MovingObject output_object = {hull_polygon, op.linear_velocity, no_future};
289
290 return output_object;
291 }

293 template <class P>
294 P ObjectToBoostPolygon(const geometry_msgs::Pose& pose, const geometry_msgs::Vector3& size) {
295
296 tf2::Transform object_tf;
297 tf2::fromMsg(pose, object_tf);
298
299 double half_x_bound = size.x / 2;
300 double half_y_bound = size.y / 2;
301
302 // 4 corners of the object starting with upper left and moving in clockwise direction in pose frame
303 tf2::Vector3 obj_p1(half_x_bound, half_y_bound, 0);
304 tf2::Vector3 obj_p2(half_x_bound, -half_y_bound, 0);
305 tf2::Vector3 obj_p3(-half_x_bound, -half_y_bound, 0);
306 tf2::Vector3 obj_p4(-half_x_bound, half_y_bound, 0);
307
308 tf2::Vector3 obj_p1_map = object_tf * obj_p1;
309 tf2::Vector3 obj_p2_map = object_tf * obj_p2;
310 tf2::Vector3 obj_p3_map = object_tf * obj_p3;
311 tf2::Vector3 obj_p4_map = object_tf * obj_p4;
312
313 point_t p1(obj_p1_map.getX(), obj_p1_map.getY());
314 point_t p2(obj_p2_map.getX(), obj_p2_map.getY());
315 point_t p3(obj_p3_map.getX(), obj_p3_map.getY());
316 point_t p4(obj_p4_map.getX(), obj_p4_map.getY());
317
318 P p;
319 p.outer().push_back(p1);
320 p.outer().push_back(p2);
321 p.outer().push_back(p3);
322 p.outer().push_back(p4);
323
324 return p;
325 }
