35 public CtrlLineArcs(int nRoadId, int nSectionId, int nLaneIndex, int nLaneIdByRoad, int nLaneType, double[] dCenterLine, double dTol)
36 {
37 m_nLaneId = XodrUtil.getLaneId(nRoadId, nLaneIdByRoad);
38 m_lLaneSectionId = XodrUtil.getLaneSectionId(nRoadId, nSectionId, nLaneIndex);
39 m_nLaneType = nLaneType;
40
41 double[] dCenter;
42 if (nRoadId == -100)
43 dCenter = dCenterLine;
44 else
45 dCenter = Geo.combineLineArcs(dCenterLine, dTol);
46 m_dLineArcs = Arrays.newDoubleArray(Arrays.size(dCenter));
47 for (int i = 1; i < 5; i += 2) // add bounding box
48 {
49 m_dLineArcs = Arrays.add(m_dLineArcs, dCenter[i], dCenter[i + 1]);
50 }
51
52 if (nLaneIndex > 0)
53 {
54 for (int nIndex = Arrays.size(dCenter) - 3; nIndex >= 5; nIndex -= 3)
55 {
56 m_dLineArcs = Arrays.add(m_dLineArcs, dCenter[nIndex], dCenter[nIndex + 1]);
57 m_dLineArcs = Arrays.add(m_dLineArcs, dCenter[nIndex + 2]);
58 }
59 }
60 else
61 {
62 Iterator<double[]> oIt = Arrays.iterator(dCenter, new double[3], 5, 3);
63 while (oIt.hasNext())
64 {
65 double[] dPt = oIt.next();
66 m_dLineArcs = Arrays.add(m_dLineArcs, dPt[0], dPt[1]); // add x,y as mercator meters
67 m_dLineArcs = Arrays.add(m_dLineArcs, dPt[2]); // add w
68 }
69 }

131 public void combine(double[] dOtherLAs, int nConnect)
132 {
133 switch (nConnect)
134 {
135 case CON_TEND_OSTART: // this end and other start touch
136 {
137 int nCpLen = Arrays.size(dOtherLAs) - 8; // subtract insertion point, bounding box, and first point which is shared
138 m_dLineArcs = Arrays.ensureCapacity(m_dLineArcs, nCpLen);
139 System.arraycopy(dOtherLAs, 8, m_dLineArcs, Arrays.size(m_dLineArcs), nCpLen); // start copying from position 8 to skip insertion point, bb, first point
140 m_dLineArcs[0] += nCpLen;
141 break;
142 }
143 case CON_TEND_OEND: // this end and other end touch
144 {
145 int nIndex = Arrays.size(dOtherLAs) - 6; // minus 6 to skip the last point which is shared
146 while (nIndex > 5) // add other points in reverse order
147 {
148 m_dLineArcs = Arrays.add(m_dLineArcs, dOtherLAs[nIndex], dOtherLAs[nIndex + 1]); // add x,y
149 m_dLineArcs = Arrays.add(m_dLineArcs, dOtherLAs[nIndex + 2]); // add w
150 nIndex -= 3;
151 }
152 break;
153 }
154 case CON_TSTART_OSTART: // this start and other start touch
155 {
156 double[] dPts = Arrays.newDoubleArray(Arrays.size(m_dLineArcs) + Arrays.size(dOtherLAs) - 8); // subtract 8 for extra insertion point, bounding box, and shared first point
157 dPts = Arrays.add(dPts, m_dLineArcs[1], m_dLineArcs[2]); // add this bounding box
158 dPts = Arrays.add(dPts, m_dLineArcs[3], m_dLineArcs[4]);
159 int nIndex = Arrays.size(dOtherLAs) - 3;
160 while (nIndex > 5) // add other points in reverse order
161 {
162 dPts = Arrays.add(dPts, dOtherLAs[nIndex], dOtherLAs[nIndex + 1]);
163 dPts = Arrays.add(dPts, dOtherLAs[nIndex + 2]);
164 nIndex -= 3;
165 }
166 int nCpLen = Arrays.size(m_dLineArcs) - 8;
167 System.arraycopy(m_dLineArcs, 8, dPts, Arrays.size(dPts), nCpLen);
168 m_dLineArcs = dPts;
169 m_dLineArcs[0] += nCpLen;
170 break;
171 }
172 case CON_TSTART_OEND: // this start and other end touch
173 {
174 double[] dPts = Arrays.newDoubleArray(Arrays.size(m_dLineArcs) + Arrays.size(dOtherLAs) - 8); // subtract 8 for extra insertion point, bounding box, and shared first point
175 System.arraycopy(m_dLineArcs, 1, dPts, 1, 4); // copy this bb into pts
176 System.arraycopy(dOtherLAs, 5, dPts, 5, Arrays.size(dOtherLAs) - 5);
177 dPts[0] = dOtherLAs[0];
178 int nCpLen = Arrays.size(m_dLineArcs) - 8;
179 System.arraycopy(m_dLineArcs, 8, dPts, Arrays.size(dOtherLAs), nCpLen);
180 m_dLineArcs = dPts;
181 m_dLineArcs[0] += nCpLen;
182 break;
183 }
184 default:
185 return;
186 }
187 if (dOtherLAs[1] < m_dLineArcs[1])
188 m_dLineArcs[1] = dOtherLAs[1];
189 if (dOtherLAs[2] < m_dLineArcs[2])
190 m_dLineArcs[2] = dOtherLAs[2];
191 if (dOtherLAs[3] > m_dLineArcs[3])
192 m_dLineArcs[3] = dOtherLAs[3];
193 if (dOtherLAs[4] > m_dLineArcs[4])
194 m_dLineArcs[4] = dOtherLAs[4];
195 }

204 public int connects(double[] dOtherGeo, double dSqTol)
205 {
206 double[] dThisGeo = m_dLineArcs;
207 int nThisEnd = Arrays.size(dThisGeo) - 3;
208 int nOtherEnd = Arrays.size(dOtherGeo) - 3;
209 if (MathUtil.compareTol(Geo.sqDist(dThisGeo[nThisEnd], dThisGeo[nThisEnd + 1], dOtherGeo[5], dOtherGeo[6]), 0, dSqTol) == 0) // this end and other start touch
210 return CON_TEND_OSTART;
211 if (MathUtil.compareTol(Geo.sqDist(dThisGeo[nThisEnd], dThisGeo[nThisEnd + 1], dOtherGeo[nOtherEnd], dOtherGeo[nOtherEnd + 1]), 0, dSqTol) == 0) // this end and other end touch
212 return CON_TEND_OEND;
213 if (MathUtil.compareTol(Geo.sqDist(dThisGeo[5], dThisGeo[6], dOtherGeo[5], dOtherGeo[6]), 0, dSqTol) == 0) // this start and other start touch
214 return CON_TSTART_OSTART;
215 if (MathUtil.compareTol(Geo.sqDist(dThisGeo[5], dThisGeo[6], dOtherGeo[nOtherEnd], dOtherGeo[nOtherEnd + 1]), 0, dSqTol) == 0) // this start and other end touch
216 return CON_TSTART_OEND;
217
218 return -1;

230 public static double getLen(double[] dLineArcs)
231 {
232 Iterator<double[]> oIt = Arrays.iterator(dLineArcs, new double[9], 5, 6);
233 double[] dCenter = new double[2];
234 double dTotalLen = 0.0;
235 while (oIt.hasNext())
236 {
237 double[] dSeg = oIt.next();
238 double dR = Geo.circle(dSeg[0], dSeg[1], dSeg[3], dSeg[4], dSeg[6], dSeg[7], dCenter);
239 if (!Double.isFinite(dR) || dR >= 10000) // expand line
240 {
241 dTotalLen += Geo.distance(dSeg[0], dSeg[1], dSeg[6], dSeg[7]);
242 }
243 else
244 {
245 int nRightHand = Geo.rightHand(dSeg[3], dSeg[4], dSeg[0], dSeg[1], dSeg[6], dSeg[7]);
246 double dC = -nRightHand / dR;
247 double dCmAngleStep = dC / 100;
248 double dCmAngleStepMag = Math.abs(dCmAngleStep);
249 double dH = dCenter[0];
250 double dK = dCenter[1];
251 double dHdg = Geo.heading(dH, dK, dSeg[0], dSeg[1]);
252 int nSteps = 0;
253 double dPrevX = dSeg[0];
254 double dPrevY = dSeg[1];
255 double dPrevAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dSeg[6], dSeg[7]);
256 while (dPrevAngle > dCmAngleStepMag)
257 {
258 dPrevX = dH + dR * Math.cos(dHdg + dCmAngleStep * ++nSteps);
259 dPrevY = dK + dR * Math.sin(dHdg + dCmAngleStep * nSteps);
260 dTotalLen += 0.01;
261 dPrevAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dSeg[6], dSeg[7]);
262 }
263 double dLastAngle = Geo.angle(dPrevX, dPrevY, dH, dK, dSeg[6], dSeg[7]);
264 dTotalLen += dLastAngle * dR;
265 }
266 }
267
268 return dTotalLen;
269 }
