26void LightBarManager::turnOffAll()
27{
28 // All components lose controls and turn the indicators off by giving lightbarmanar the control
29 std::vector<LightBarIndicator> all_indicators;
30
31 ROS_INFO_STREAM("LightBarManager was commanded to turn off all indicators!");
32
33 for (std::pair <LightBarIndicator, std::string> element : lbm_->getIndicatorControllers())
34 all_indicators.push_back(element.first);
35
36 // Reset all controls
37 lbm_->setIndicatorControllers();
38
39 // Give lightbar manager the control which is guaranteed to succeed
40 lbm_->requestControl(all_indicators, node_name_);
41 int response_code = 0;
42 for (auto indicator : all_indicators)
43 {
44 response_code = setIndicator(indicator, OFF, node_name_);
45 if (response_code != 0)
46 ROS_WARN_STREAM ("In Function " << __FUNCTION__ << ": LightBarManager was not able to turn off indicator ID:"
47 << indicator << ". Response code: " << response_code);
48 }
49
50 // Once forced them off, release controls in case carma is still running so that lightbar_manager does not hog the control
51 all_indicators.erase(std::remove(all_indicators.begin(), all_indicators.end(), GREEN_FLASH), all_indicators.end());
52 all_indicators.erase(std::remove(all_indicators.begin(), all_indicators.end(), GREEN_SOLID), all_indicators.end());
53 lbm_->releaseControl(all_indicators, node_name_);
54}

56bool LightBarManager::requestControlCallBack(cav_srvs::RequestIndicatorControlRequest& req, cav_srvs::RequestIndicatorControlResponse& res)
57{
58 std::vector<LightBarIndicator> ind_list, controlled_ind_list;
59 std::vector<LightBarCDAType> controlled_cda_type_list;
60
61 // Use CDAType if the field is not empty in the request
62 if (req.cda_list.size() != 0)
63 {
64 for (auto cda_type : req.cda_list)
65 {
66 // return false if invalid cda_type number
67 if (static_cast<uint8_t>(cda_type.type) >= INDICATOR_COUNT)
68 return false;
69 ind_list.push_back(lbm_->getIndicatorFromCDAType(static_cast<LightBarCDAType>(cda_type.type)));
70 }
71 }
72 else
73 {
74 for (auto indicator : req.ind_list)
75 {
76 // return false if invalid indicator number
77 if (static_cast<uint8_t>(indicator.indicator) >= INDICATOR_COUNT)
78 return false;
79 ind_list.push_back(static_cast<LightBarIndicator>(indicator.indicator));
80 }
81
82 }
83
84 controlled_ind_list = lbm_->requestControl(ind_list, req.requester_name);
85
86 for (auto indicator : controlled_ind_list)
87 {
88 try
89 {
90 // Some indicators do not have mapped CDAType
91 controlled_cda_type_list.push_back(lbm_->getCDATypeFromIndicator(indicator));
92 }
93 catch(INDICATOR_NOT_MAPPED)
94 {
95 // 4 out of 8 does not have mapping, so skip
96 continue;
97 }
98 }
99
100 // Modify the response
101 res.cda_list = lbm_->getMsg(controlled_cda_type_list);
102 res.ind_list = lbm_->getMsg(controlled_ind_list);
103 return true;
104}

142void LightBarManager::stateChangeCallBack(const cav_msgs::GuidanceStateConstPtr& msg_ptr)
143{
144 // Relay the msg to state machine
145 LightBarState prev_lightbar_state = lbm_->getCurrentState();
146 lbm_->handleStateChange(msg_ptr);
147
148 // Change green lights depending on states, no need to check if its current owner, as it will always be for green lights
149 LightBarState curr_lightbar_state = lbm_->getCurrentState();
150 if (curr_lightbar_state != prev_lightbar_state)
151 {
152 switch(curr_lightbar_state)
153 {
154 case DISENGAGED:
155 turnOffAll();
156 break;
157 case ACTIVE:
158 setIndicator(GREEN_FLASH, ON, node_name_);
159 break;
160 case ENGAGED:
161 setIndicator(GREEN_SOLID, ON, node_name_);
162 break;
163 default:
164 break;
165 }
166 }

169void LightBarManager::turnSignalCallback(const automotive_platform_msgs::TurnSignalCommandPtr& msg_ptr)
170{
171 // if not automated
172 if (msg_ptr->mode != 1)
173 {
174 return;
175 }
176
177 // check if left or right signal should be controlled, or none at all
178 std::vector<lightbar_manager::LightBarIndicator> changed_turn_signal = lbm_->handleTurnSignal(msg_ptr);
179
180 if (changed_turn_signal.empty())
181 {
182 return; //no need to do anything if it is same turn signal changed
183 }
184
185 lightbar_manager::IndicatorStatus indicator_status;
186 // check if we should turn off or on given any indicator
187 if (msg_ptr->turn_signal == automotive_platform_msgs::TurnSignalCommand::NONE)
188 {
189 indicator_status = lightbar_manager::IndicatorStatus::OFF;
190 }
191 else
192 {
193 indicator_status = lightbar_manager::IndicatorStatus::ON;
194 prev_owners_before_turn_ = lbm_->getIndicatorControllers(); // save the owner if new turn is starting
195 }
196
197 if (lbm_->requestControl(changed_turn_signal, node_name_).empty())
198 {
199 int response_code = 0;
200 response_code = setIndicator(changed_turn_signal[0], indicator_status, node_name_);
201 if (response_code != 0)
202 ROS_ERROR_STREAM ("In Function " << __FUNCTION__ << ": LightBarManager was not able to set light of indicator ID:"
203 << changed_turn_signal[0] << ". Response code: " << response_code);
204 }
205 else
206 {
207 std::string turn_string = msg_ptr->turn_signal == automotive_platform_msgs::TurnSignalCommand::LEFT ? "left" : "right";
208 ROS_WARN_STREAM("Lightbar was not able to take control of lightbar to indicate " << turn_string << "turn!");
209 return;
210 }
211
212 // release control if it is turning off and put back the previous owners
213 if (indicator_status == lightbar_manager::IndicatorStatus::OFF)
214 {
215 lbm_->releaseControl(changed_turn_signal, node_name_);
216 // we need to force set the <indicator, owner> mapping
217 // as series of requestControl may not be able to exactly achieve previous combination
218 lbm_->setIndicatorControllers(prev_owners_before_turn_);
219 // reset as the turn is finished
220 prev_owners_before_turn_.clear();
221 }
222}

229int LightBarManager::setIndicator(LightBarIndicator ind, IndicatorStatus ind_status, const std::string& requester_name)
230{
231 // Handle the msg/service translation
232 int response_code = 0;
233 if (static_cast<uint8_t>(ind) >= INDICATOR_COUNT || static_cast<uint8_t>(ind_status) > 1)
234 {
235 // Invalid indicator ID or ind_status
236 response_code = 1;
237 return response_code;
238 }
239
240 // Check if the requester has control of this light
241 std::string current_controller = lbm_->getIndicatorControllers()[ind];
242 if (requester_name == "" || current_controller != requester_name)
243 {
244 ROS_WARN_STREAM(requester_name << " failed to set the LightBarIndicator ID" << ind
245 << " as this was already controlled by " << current_controller);
246 response_code = 1;
247 return response_code;
248 }
249
250 std::vector<IndicatorStatus> light_status_proposed = lbm_->setIndicator(ind, ind_status, requester_name);
251 cav_msgs::LightBarStatus msg = lbm_->getLightBarStatusMsg(light_status_proposed);
252 cav_srvs::SetLights srv;
253 srv.request.set_state = msg;
254
255 // Try to send the request
256 if (lightbar_driver_client_.call(srv))
257 {
258 // if successful, update the local copy.
259 lbm_->light_status = light_status_proposed;
260 response_code = 0;
261 }
262 else
263 {
264 ROS_WARN_STREAM("In function: " << __FUNCTION__ << ": Failed to set lights. ROSservice call to the driver failed. ");
265 response_code = 2;
266 }
267 return response_code;

270bool LightBarManager::setIndicatorCallBack(cav_srvs::SetLightBarIndicatorRequest& req, cav_srvs::SetLightBarIndicatorResponse& res)
271{
272 LightBarIndicator indicator;
273 int response_code = 0;
274 if (req.cda_type.type != NULL)
275 indicator = lbm_->getIndicatorFromCDAType(static_cast<LightBarCDAType>(req.cda_type.type));
276 else
277 indicator = static_cast<LightBarIndicator>(req.indicator.indicator);
278
279 response_code = setIndicator(indicator, static_cast<IndicatorStatus>(req.state), req.requester_name);
280 res.status_code = response_code;
281 if (response_code != 0)
282 return false;
283
284 return true;

287void LightBarManager::init(std::string mode)
288{
289 ROS_INFO("Initalizing lightbar manager node...");
290
291 // Load the spin rate param to determine how fast to process messages
292 // Default rate 10.0 Hz
293 spin_rate_ = pnh_.param<double>("spin_rate_hz", 10.0);
294 std::string lightbar_driver_service_name= pnh_.param<std::string>("lightbar_driver_service_name", "set_lights");
295 std::string guidance_state_topic_name = pnh_.param<std::string>("guidance_state_topic_name", "state");
296 std::string turn_signal_topic_name = pnh_.param<std::string>("turn_signal_topic_name", "turn_signal_command");
297 // Init our ROS objects
298 request_control_server_= nh_.advertiseService("request_control", &LightBarManager::requestControlCallBack, this);
299 release_control_server_= nh_.advertiseService("release_control", &LightBarManager::releaseControlCallBack, this);
300 set_indicator_server_= nh_.advertiseService("set_indicator", &LightBarManager::setIndicatorCallBack, this);
301 indicator_control_publisher_ = nh_.advertise<cav_msgs::LightBarIndicatorControllers>("indicator_control", 5);
302 guidance_state_subscriber_ = nh_.subscribe(guidance_state_topic_name, 5, &LightBarManager::stateChangeCallBack, this);
303 turn_signal_subscriber_ = nh_.subscribe(turn_signal_topic_name, 5, &LightBarManager::turnSignalCallback, this);
304 lightbar_driver_client_ = nh_.serviceClient<cav_srvs::SetLights>(lightbar_driver_service_name);
305
306 // Load Conversion table, CDAType to Indicator mapping
307 std::map<std::string,std::string> cda_ind_map_raw;
308 pnh_.getParam("lightbar_cda_to_ind_table", cda_ind_map_raw);
309 lbm_->setIndicatorCDAMap(cda_ind_map_raw);
310
311 // Initialize indicator control map. Fills with supporting indicators with empty string name as owners.
312 lbm_->setIndicatorControllers();
313
314 // Initialize indicator representation of lightbar status to all OFF
315 for (int i =0; i < INDICATOR_COUNT; i++)
316 lbm_->light_status.push_back(OFF);
317
318 // Load lightbar priorities.
319 pnh_.getParam("lightbar_priorities", lbm_->control_priorities);
320
321 // Setup priorities for unit test
322 if (mode == "test")
323 setupUnitTest();
324
325 // Take control of green light
326 bool normal_operation = pnh_.param<bool>("normal_operation", true);
327 std::vector<LightBarIndicator> denied_list, greens = {GREEN_SOLID, GREEN_FLASH};
328 denied_list = lbm_->requestControl(greens, node_name_);
329 if (denied_list.size() != 0)
330 {
331 if (normal_operation)
332 {
333 ROS_WARN_STREAM("In fuction " << __FUNCTION__ << ", LightBarManager was not able to take control of all green indicators."
334 << ".\n Please check priority list rosparameter, and ensure " << node_name_ << " has the highest priority."
335 << ".\n If this is intended, pass false to normal_operation argument. Exiting...");
336 throw INVALID_LIGHTBAR_MANAGER_PRIORITY();
337 }
338 else
339 {
340 ROS_WARN_STREAM("In fuction " << __FUNCTION__ << ", LightBarManager was not able to take control of all green indicators."
341 << ".\n Resuming...");
342 }
343 }
344 return;
345}
