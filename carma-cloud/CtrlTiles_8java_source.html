157 @Override
158 protected void doGet(HttpServletRequest oRequest, HttpServletResponse oResponse)
159 throws ServletException, IOException
160 {
161 long lNow = TimeSource.getInstance().currentTimeMillis();
162 String[] sUriParts = oRequest.getRequestURI().split("/");
163 int nZ = Integer.parseInt(sUriParts[sUriParts.length - 3]);
164 int nX = Integer.parseInt(sUriParts[sUriParts.length - 2]);
165 int nY = Integer.parseInt(sUriParts[sUriParts.length - 1]);
166
167 double[] dBounds = TileUtil.getTileBounds(nZ, nX, nY);
168 double dXAdjust = 0.001;
169 double dYAdjust = 0.001;
170 int[] nTiles = new int[2];
171 Mercator.getInstance().metersToTile(dBounds[0] + dXAdjust, dBounds[3] - dYAdjust, g_nZoom, nTiles); // determine the correct tiles for the default zoom level
172 int nDefaultX = nTiles[0];
173 int nDefaultY = nTiles[1];
174 Path oIndexFile = Paths.get(String.format(g_sTdFileFormat, nDefaultX, g_nZoom, nDefaultX, nDefaultY) + ".ndx");
175 Path oTdFile = Paths.get(String.format(g_sTdFileFormat, nDefaultX, g_nZoom, nDefaultX, nDefaultY));
176 if (!Files.exists(oTdFile) || !Files.exists(oIndexFile))
177 return;
178
179 ArrayList<byte[]> yIdsToRender = new ArrayList();
180 byte[] yIdBuf = new byte[16];
181 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oIndexFile))))
182 {
183 while (oIn.available() > 0)
184 {
185 CtrlIndex oIndex = new CtrlIndex(oIn);
186 if ((oIndex.m_lStart >= lNow || oIndex.m_lEnd > lNow) && m_nZoomFilter[oIndex.m_nType] <= nZ) // everything valid now and in the future add to tile
187 yIdsToRender.add(oIndex.m_yId);
188 }
189 }
190 Introsort.usort(yIdsToRender, TrafCtrl.ID_COMP);
191 double[] dClipBounds = TileUtil.getClippingBounds(nZ, nX, nY, dBounds);
192
193 int nExtent = 4096;
194 VectorTile.Tile.Layer.Builder oLayerBuilder = VectorTile.Tile.Layer.newBuilder();
195 oLayerBuilder.setVersion(2);
196 oLayerBuilder.setExtent(nExtent);
197 VectorTile.Tile.Feature.Builder oFeatureBuilder = VectorTile.Tile.Feature.newBuilder();
198 VectorTile.Tile.Value.Builder oValueBuilder = VectorTile.Tile.Value.newBuilder();
199 int[] nCur = new int[2];
200 int[] nPointBuffer = Arrays.newIntArray(1024);
201 Area oPolyClip = TileUtil.getClippingArea(nZ, nX, nY, dClipBounds);
202
203 ArrayList<TdLayer> oLayers = new ArrayList();
204 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(oTdFile))))
205 {
206 while (oIn.available() > 0)
207 {
208 TdLayer oTemp = new TdLayer(oIn, false);
209 int nFeatures = oIn.readInt();
210 for (int nIndex = 0; nIndex < nFeatures; nIndex++)
211 {
212 oIn.read(yIdBuf);
213 int nBytesToSkip = oIn.readInt();
214 int nSearch = Collections.binarySearch(yIdsToRender, yIdBuf, TrafCtrl.ID_COMP);
215 if (nSearch < 0)
216 oIn.skipBytes(nBytesToSkip);
217 else
218 oTemp.add(new TdFeature(oIn, oTemp.m_oKeys.length * 2, yIdBuf));
219 }
220 int nIndex = Collections.binarySearch(oLayers, oTemp);
221 if (nIndex < 0)
222 {
223 oLayers.add(~nIndex, oTemp);
224 }
225 else
226 {
227 oLayers.get(nIndex).addAll(oTemp);
228 }
229 }
230 }
231
232 VectorTile.Tile.Builder oTile = VectorTile.Tile.newBuilder();
233 ArrayList<byte[]> oIds = new ArrayList();
234 for (TdLayer oLayer : oLayers)
235 {
236 oIds.clear();
237 oLayerBuilder.clearFeatures();
238 oLayerBuilder.clearKeys();
239 oLayerBuilder.clearValues();
240 oLayerBuilder.setName(oLayer.m_sName);
241 int nIdKeyIndex = oLayer.m_oKeys.length;
242 for (String sKey : oLayer.m_oKeys)
243 {
244 oLayerBuilder.addKeys(sKey);
245 }
246 oLayerBuilder.addKeys("sponge_id");
247
248 int nIdValueOffset = oLayer.m_oValues.length;
249 for (String sValue : oLayer.m_oValues)
250 {
251 oValueBuilder.setStringValue(sValue);
252 oLayerBuilder.addValues(oValueBuilder.build());
253 }
254 for (TdFeature oFeature : oLayer)
255 {
256 int nIndex = Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP);
257 if (nIndex < 0)
258 oIds.add(~nIndex, oFeature.m_yCtrlId);
259 }
260
261 for (byte[] yId : oIds)
262 {
263 oValueBuilder.setStringValue(TrafCtrl.getId(yId));
264 oLayerBuilder.addValues(oValueBuilder.build());
265 }
266
267 switch (oLayer.m_yGeoType)
268 {
269 case TdLayer.POINT:
270 {
271 oFeatureBuilder.setType(VectorTile.Tile.GeomType.POINT);
272 for (TdFeature oFeature : oLayer)
273 {
274 double[] dGeo = oFeature.m_dGeo;
275 if (!Geo.isInside(dGeo[1], dGeo[2], dClipBounds[3], dClipBounds[2], dClipBounds[1], dClipBounds[0], 0))
276 continue;
277
278 for (int nTag : oFeature.m_nTags)
279 oFeatureBuilder.addTags(nTag);
280 oFeatureBuilder.addTags(nIdKeyIndex);
281 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
282 TileUtil.addMercPointToFeature(oFeatureBuilder, nCur, dBounds, nExtent, dGeo[1], dGeo[2]);
283 if (oFeatureBuilder.getGeometryCount() > 0)
284 oLayerBuilder.addFeatures(oFeatureBuilder.build());
285 nCur[0] = nCur[1] = 0;
286 oFeatureBuilder.clearGeometry();
287 oFeatureBuilder.clearTags();
288 }
289 break;
290 }
291 case TdLayer.LINESTRING:
292 {
293 oFeatureBuilder.setType(VectorTile.Tile.GeomType.LINESTRING);
294 ArrayList<double[]> oClipped = new ArrayList();
295 for (TdFeature oFeature : oLayer)
296 {
297 double[] dGeo = oFeature.m_dGeo;
298 oClipped.clear();
299 TileUtil.clipLineString(dGeo, dClipBounds, oClipped);
300 if (oClipped.isEmpty())
301 continue;
302
303 for (int nTag : oFeature.m_nTags)
304 oFeatureBuilder.addTags(nTag);
305 oFeatureBuilder.addTags(nIdKeyIndex);
306 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
307 for (double[] dLine : oClipped)
308 {
309 nPointBuffer = TileUtil.addMercLinestring(oFeatureBuilder, nCur, dBounds, nExtent, dLine, nPointBuffer);
310 if (oFeatureBuilder.getGeometryCount() > 0)
311 oLayerBuilder.addFeatures(oFeatureBuilder.build());
312 nCur[0] = nCur[1] = 0;
313 oFeatureBuilder.clearGeometry();
314 oFeatureBuilder.clearTags();
315 }
316 }
317 break;
318 }
319 case TdLayer.POLYGON:
320 {
321 oFeatureBuilder.setType(VectorTile.Tile.GeomType.POLYGON);
322 for (TdFeature oFeature : oLayer)
323 {
324 double[] dGeo = oFeature.m_dGeo;
325 Area oArea = Geo.getArea(dGeo, 1);
326 if (oArea == null)
327 continue;
328 oArea.intersect(oPolyClip);
329 if (oArea.isEmpty())
330 continue;
331
332 for (int nTag : oFeature.m_nTags)
333 oFeatureBuilder.addTags(nTag);
334 oFeatureBuilder.addTags(nIdKeyIndex);
335 oFeatureBuilder.addTags(nIdValueOffset + Collections.binarySearch(oIds, oFeature.m_yCtrlId, TrafCtrl.ID_COMP));
336 nPointBuffer = TileUtil.addPolygon(oFeatureBuilder, nCur, dBounds, nExtent, oArea, nPointBuffer);
337 if (oFeatureBuilder.getGeometryCount() > 0)
338 oLayerBuilder.addFeatures(oFeatureBuilder.build());
339 nCur[0] = nCur[1] = 0;
340 oFeatureBuilder.clearGeometry();
341 oFeatureBuilder.clearTags();
342 }
343 break;
344 }
345 default:
346 {
347 oFeatureBuilder.setType(VectorTile.Tile.GeomType.UNKNOWN);
348 break;
349 }
350 }
351 if (oLayerBuilder.getFeaturesCount() > 0)
352 oTile.addLayers(oLayerBuilder.build());
353 }
354
355 oResponse.setContentType("application/x-protobuf");
356 if (oTile.getLayersCount() > 0)
357 oTile.build().writeTo(oResponse.getOutputStream());
358 }

464 String sReg = oReq.getParameter("reg");

469 bReg = true;

471 if (sVal != null)
472 {
473 double dVal = Double.parseDouble(sVal);
474 if (sUnits.length > 0)
475 dVal = Units.getInstance().convert(sUnits[1], sUnits[0], dVal);
476 nControlValue = (int)Math.round(dVal);

478 else
479 {
480 sVal = oReq.getParameter("value1");
481 if (sVal == null)

487 String sVal2 = oReq.getParameter("value2");

489 nControlValue <<= 16;

492 else
493 nControlValue = Integer.parseInt(sVal);
494 }
495 }
496

498 int nStartIndex = Integer.parseInt(oReq.getParameter("s")) * 4 + 1; // add one since we use the growable arrays with the insertion index at position 0
499 int nEndIndex = Integer.parseInt(oReq.getParameter("e")) * 4 + 1;
500 String sFile = g_sCtrlDir + sId + ".bin";
501 TrafCtrl oOriginalCtrl;

511 for (int nIndex = nStartIndex; nIndex <= nEndIndex;)
512 {
513 double dXc = dC[nIndex];
514 double dYc = dC[nIndex + 1];
515 dCenter = Arrays.addAndUpdate(dCenter, dXc, dYc);
516 double dW = Geo.distance(dXc, dYc, dNT[nIndex++], dNT[nIndex++]) * 2;
517 dCenter = Arrays.add(dCenter, dW);
518 }
519 CtrlLineArcs oCla = new CtrlLineArcs(-1, -1, -1, -1, XodrUtil.getLaneType("driving"), dCenter, 0.1);

522 ArrayList<TrafCtrl> oCtrls = new ArrayList();
523 oCtrls.add(oCtrl);

525 }

527 {
528 oEx.printStackTrace();
529 oRes.sendError(HttpServletResponse.SC_BAD_REQUEST);
530 }

570 sLabel = Text.truncate(sLabel, 63).trim();

595 String sId = oReq.getParameter("id");
596 String sFile = g_sCtrlDir + sId + ".bin";
597 TrafCtrl oOriginalCtrl;
598 try (DataInputStream oIn = new DataInputStream(new BufferedInputStream(FileUtil.newInputStream(Paths.get(sFile)))))
599 {
600 oOriginalCtrl = new TrafCtrl(oIn, true, true);
601 }
602 TrafCtrl oCtrlToWrite;
603 Collections.sort(nVtypes);

605 int nIndex = nVtypes.size();
606 while (nIndex-- > 0 && bSameVtypes)
