127 // Trigger timeout event if a timeout has occurred for the currently tracked ERV
128 if(has_tracked_erv_){

139 boost::optional<lanelet::BasicPoint2d> ApproachingEmergencyVehiclePlugin::getErvPositionInMap(const double& current_latitude, const double& current_longitude){
140 // Check if the map projection is available
141 if (!map_projector_) {
142 throw std::invalid_argument("Attempting to get ERV's current position in map before map projection was set");
143 }
144
145 // Build map projector
146 lanelet::projection::LocalFrameProjector projector(map_projector_.get().c_str());
147
148 // Create vector to hold ERV's projected current position
149 std::vector<lanelet::BasicPoint3d> erv_current_position_projected_vec;
150
151 // Add ERV's current location to the beginning of erv_current_position_projected_vec
152 lanelet::GPSPoint current_erv_location;
153 current_erv_location.lat = current_latitude;
154 current_erv_location.lon = current_longitude;
155
156 // Convert ERV's projected position to its position in the map frame
157 erv_current_position_projected_vec.emplace_back(projector.forward(current_erv_location));
158 auto erv_current_position_in_map_vec = lanelet::utils::transform(erv_current_position_projected_vec, [](auto a) { return lanelet::traits::to2D(a); });
159
160 // Conduct size check since only the first element is being returned
161 if(!erv_current_position_in_map_vec.empty()){
162 return erv_current_position_in_map_vec[0];
163 }
164 else{
165 return boost::optional<lanelet::BasicPoint2d>();
166 }

169 boost::optional<ErvInformation> ApproachingEmergencyVehiclePlugin::getErvInformationFromBsm(carma_v2x_msgs::msg::BSM::UniquePtr msg){
170 // Initialize ErvInformation object, which will be populated with information from this BSM if it is from an ERV
171 ErvInformation erv_information;
172
173 // Get vehicle_id from the BSM
174 std::stringstream ss;
175 for(size_t i = 0; i < msg->core_data.id.size(); ++i){
176 ss << std::setfill('0') << std::setw(2) << std::hex << (unsigned)msg->core_data.id.at(i);
177 }
178 erv_information.vehicle_id = ss.str();
179
180 // Get timestamp from BSM
181 erv_information.latest_bsm_timestamp = rclcpp::Time(msg->header.stamp, this->get_clock()->get_clock_type());
182
183 // Check whether vehicle's lights and sirens are active
184 bool has_active_lights_and_sirens = false;
185 if(msg->presence_vector & carma_v2x_msgs::msg::BSM::HAS_PART_II){
186
187 // Parse BSM Part II Content to determine whether the vehicle's lights and sirens are active
188 if(msg->part_ii.size() > 0){
189 for(size_t i = 0; i < msg->part_ii.size(); ++i){
190
191 // special_vehicle_extensions contain the status of the vehicle's lights and sirens
192 if(msg->part_ii[i].part_ii_id == carma_v2x_msgs::msg::BSMPartIIExtension::SPECIAL_VEHICLE_EXT){
193 carma_v2x_msgs::msg::SpecialVehicleExtensions special_vehicle_ext = msg->part_ii[i].special_vehicle_extensions;
194
195 if(special_vehicle_ext.presence_vector & carma_v2x_msgs::msg::SpecialVehicleExtensions::HAS_VEHICLE_ALERTS){
196 if(special_vehicle_ext.vehicle_alerts.siren_use.siren_in_use == j2735_v2x_msgs::msg::SirenInUse::IN_USE
197 && special_vehicle_ext.vehicle_alerts.lights_use.lightbar_in_use == j2735_v2x_msgs::msg::LightbarInUse::IN_USE){
198 has_active_lights_and_sirens = true;
199 }
200 }
201 }
202 }
203 }
204 }
205
206 if(!has_active_lights_and_sirens){
207 // BSM is not a valid ERV BSM since the lights and sirens are not both active; return an empty object
208 return boost::optional<ErvInformation>();
209 }
210
211 // Get vehicle's current speed from the BSM
212 if(msg->core_data.presence_vector & carma_v2x_msgs::msg::BSMCoreData::SPEED_AVAILABLE){
213 erv_information.current_speed = msg->core_data.speed;
214 }
215 else{
216 // BSM is not a valid ERV BSM since current speed is not included; return an empty object
217 return boost::optional<ErvInformation>();
218 }
219
220 // Get vehicle's current latitude from the BSM
221 if(msg->core_data.presence_vector & carma_v2x_msgs::msg::BSMCoreData::LATITUDE_AVAILABLE){
222 erv_information.current_latitude = msg->core_data.latitude;
223 }
224 else{
225 // BSM is not a valid ERV BSM since current latitude is not included; return an empty object
226 return boost::optional<ErvInformation>();
227 }
228
229 // Get vehicle's current longitude from the BSM
230 if(msg->core_data.presence_vector & carma_v2x_msgs::msg::BSMCoreData::LONGITUDE_AVAILABLE){
231 erv_information.current_longitude = msg->core_data.longitude;
232 }
233 else{
234 // BSM is not a valid ERV BSM since current longitude is not included; return an empty object
235 return boost::optional<ErvInformation>();
236 }
237
238 // Get vehicle's current position in the map frame from the BSM
239 boost::optional<lanelet::BasicPoint2d> erv_position_in_map = getErvPositionInMap(erv_information.current_latitude, erv_information.current_longitude);
240
241 if(erv_position_in_map){
242 erv_information.current_position_in_map = *erv_position_in_map;
243 }
244
245 // Get vehicle's route destination points from the BSM
246 std::vector<carma_v2x_msgs::msg::Position3D> erv_destination_points;
247 if(msg->presence_vector & carma_v2x_msgs::msg::BSM::HAS_REGIONAL){
248
249 // Parse BSM Regional Extensions to determine whether BSM includes the ERV's route destination points
250 if(msg->regional.size() > 0){
251 for(size_t i = 0; i < msg->regional.size(); ++i){
252 if(msg->regional[i].regional_extension_id == carma_v2x_msgs::msg::BSMRegionalExtension::ROUTE_DESTINATIONS){
253 erv_destination_points = msg->regional[i].route_destination_points;
254 break;
255 }
256 }
257 }
258 }
259
260 if(erv_destination_points.empty()){
261 // BSM is not a valid ERV BSM since it does not include destination points; return an empty object
262 return boost::optional<ErvInformation>();
263 }
264
265 // Generate ERV's route based on its current position and its destination points
266 lanelet::Optional<lanelet::routing::Route> erv_future_route = generateErvRoute(erv_information.current_latitude, erv_information.current_longitude, erv_destination_points);
267
268 if(!erv_future_route){
269 // ERV cannot be tracked since its route could not be generated; return an empty object
270 return boost::optional<ErvInformation>();
271 }
272
273 // Get intersecting lanelet between ERV's future route and ego vehicle's future shortest path
274 boost::optional<lanelet::ConstLanelet> intersecting_lanelet = getRouteIntersectingLanelet(erv_future_route.get(), wm_->getRoute()->shortestPath());
275
276 if(intersecting_lanelet){
277 erv_information.intersecting_lanelet = *intersecting_lanelet;
278 }
279 else{
280 // No intersecting lanelet between ERV and ego vehicle was found; return an empty object
281 return boost::optional<ErvInformation>();
282 }
283
284 // Get the time (seconds) until the ERV passes the ego vehicle
285 double seconds_until_passing = getSecondsUntilPassing(erv_future_route, erv_information.current_position_in_map, erv_information.current_speed, erv_information.intersecting_lanelet);
286
287 if(epsilon_ <= seconds_until_passing && seconds_until_passing <= config_.approaching_threshold){
288 RCLCPP_INFO_STREAM(rclcpp::get_logger(logger_name), "Detected approaching ERV; passing ego vehicle in " << seconds_until_passing << " seconds");
289 erv_information.seconds_until_passing = seconds_until_passing;
290 }
291 else{
292 RCLCPP_INFO_STREAM(rclcpp::get_logger(logger_name), "Detected non-approaching ERV; passing ego vehicle in " << seconds_until_passing << " seconds");
293
294 // ERV will not be tracked since it is not considered to be approaching the ego vehicle; return an empty object
295 return boost::optional<ErvInformation>();
296 }
297
298 // Determine whether ERV is currently in the rightmost lane
299 // Note: For 'lane index', 0 is rightmost lane, 1 is second rightmost, etc.; Only the current travel direction is considered
300 if(!erv_future_route->shortestPath().empty()){
301 lanelet::ConstLanelet erv_current_lanelet = erv_future_route->shortestPath()[0];
302 int lane_index = wm_->getMapRoutingGraph()->rights(erv_current_lanelet).size();
303
304 if(lane_index == 0){
305 erv_information.in_rightmost_lane = true;
306 }
307 else{
308 erv_information.in_rightmost_lane = false;
309 }
310 }
311
312 return erv_information;
313 }

321 lanelet::Optional<lanelet::routing::Route> ApproachingEmergencyVehiclePlugin::generateErvRoute(double current_latitude, double current_longitude,
322 std::vector<carma_v2x_msgs::msg::Position3D> erv_destination_points){
323 // Check if the map projection is available
324 if (!map_projector_) {
325 throw std::invalid_argument("Attempting to generate an ERV's route before map projection was set");
326 }
327
328 // Build map projector
329 lanelet::projection::LocalFrameProjector projector(map_projector_.get().c_str());
330
331 // Create vector to hold ERV's destination points
332 std::vector<lanelet::BasicPoint3d> erv_destination_points_projected;
333
334 // Add ERV's current location to the beginning of erv_destination_points
335 lanelet::GPSPoint current_erv_location;
336
337 current_erv_location.lat = current_latitude;
338 current_erv_location.lon = current_longitude;
339
340 erv_destination_points_projected.emplace_back(projector.forward(current_erv_location));
341
342 // Add ERV's future destination points to erv_destination_points
343 if(erv_destination_points.size() > 0){
344 for(size_t i = 0; i < erv_destination_points.size(); ++i){
345 carma_v2x_msgs::msg::Position3D position_3d_point = erv_destination_points[i];
346
347 lanelet::GPSPoint erv_destination_point;
348 erv_destination_point.lon = position_3d_point.longitude;
349 erv_destination_point.lat = position_3d_point.latitude;
350
351 if(position_3d_point.elevation_exists){
352 erv_destination_point.ele = position_3d_point.elevation;
353 }
354
355 erv_destination_points_projected.emplace_back(projector.forward(erv_destination_point));
356 }
357 }
358
359 // Convert ERV destination points to map frame
360 auto erv_destination_points_in_map = lanelet::utils::transform(erv_destination_points_projected, [](auto a) { return lanelet::traits::to2D(a); });
361
362 // Verify that ERV destination points are geometrically in the map
363 for(size_t i = 0; i < erv_destination_points_in_map.size(); ++i){
364 auto pt = erv_destination_points_in_map[i];
365
366 // Return empty route if a destination point is not contained within the map
367 if((wm_->getLaneletsFromPoint(erv_destination_points_in_map[i])).empty()){
368 RCLCPP_ERROR_STREAM(rclcpp::get_logger(logger_name), "ERV destination point " << i
369 << " is not contained in a lanelet map; x: " << pt.x() << " y: " << pt.y());
370
371 return lanelet::Optional<lanelet::routing::Route>();
372 }
373 }
374
375 // Obtain ERV's starting lanelet
376 auto starting_lanelet_vector = lanelet::geometry::findNearest(wm_->getMap()->laneletLayer, erv_destination_points_in_map.front(), 1);
377 if(starting_lanelet_vector.empty())
378 {
379 RCLCPP_ERROR_STREAM(rclcpp::get_logger(logger_name), "Found no lanelets in the map. ERV routing cannot be completed.");
380
381 // Return empty route
382 return lanelet::Optional<lanelet::routing::Route>();
383 }
384 auto starting_lanelet = lanelet::ConstLanelet(starting_lanelet_vector[0].second.constData());
385
386 // Obtain ERV's ending lanelet
387 auto ending_lanelet_vector = lanelet::geometry::findNearest(wm_->getMap()->laneletLayer, erv_destination_points_in_map.back(), 1);
388 auto ending_lanelet = lanelet::ConstLanelet(ending_lanelet_vector[0].second.constData());
389
390 // Obtain ERV's via lanelets
391 std::vector<lanelet::BasicPoint2d> via = std::vector<lanelet::BasicPoint2d>(erv_destination_points_in_map.begin() + 1, erv_destination_points_in_map.end() - 1);
392 lanelet::ConstLanelets via_lanelets_vector;
393 for(const auto& point : via){
394 auto lanelet_vector = lanelet::geometry::findNearest(wm_->getMap()->laneletLayer, point, 1);
395 via_lanelets_vector.emplace_back(lanelet::ConstLanelet(lanelet_vector[0].second.constData()));
396 }
397
398 // Generate the ERV's route
399 auto erv_route = wm_->getMapRoutingGraph()->getRouteVia(starting_lanelet, via_lanelets_vector, ending_lanelet);
400
401 return erv_route;
402 }

404 void ApproachingEmergencyVehiclePlugin::incomingBsmCallback(carma_v2x_msgs::msg::BSM::UniquePtr msg)
405 {
406
407 // If there is already an ERV approaching the ego vehicle, only process this BSM futher if enough time has passed since the previously processed BSM
408 if(has_tracked_erv_){
409
410 // Get the vehicle ID associated with the received BSM
411 std::stringstream ss;
412 for(size_t i = 0; i < msg->core_data.id.size(); ++i){
413 ss << std::setfill('0') << std::setw(2) << std::hex << (unsigned)msg->core_data.id.at(i);
414 }
415 std::string erv_vehicle_id = ss.str();
416
417 if(erv_vehicle_id == tracked_erv_.vehicle_id){
418 rclcpp::Time current_bsm_timestamp = rclcpp::Time(msg->header.stamp, this->get_clock()->get_clock_type());
419 double seconds_since_prev_bsm = (current_bsm_timestamp - tracked_erv_.latest_bsm_timestamp).seconds();
420
421 if(seconds_since_prev_bsm < (1.0 / config_.bsm_processing_frequency)){
422 // Do not process ERV's BSM further since not enough time has passed since its previously processed BSM
423 return;
424 }
425 }
426 else{
427 // Do not process BSM further since it is not for the currently tracked ERV
428 return;
429 }
430 }
431
432 // Get ErvInformation object with information from the BSM if it is from an active ERV that is approaching the ego vehicle
433 boost::optional<ErvInformation> erv_information = getErvInformationFromBsm(std::move(msg));
434 if(!erv_information){
435 // BSM is not from an active ERV that is approaching the ego vehicle
436 return;
437 }
438 else{
439 // Update tracked ERV information since this is an active ERV that is approaching the ego vehicle
440 tracked_erv_ = *erv_information;
441 tracked_erv_.latest_update_time = this->get_clock()->now();
442
443 if(!has_tracked_erv_){
444 has_tracked_erv_ = true;
445 }
446 }
447
448 return;
449 }

451 double ApproachingEmergencyVehiclePlugin::getSecondsUntilPassing(lanelet::Optional<lanelet::routing::Route>& erv_future_route, const lanelet::BasicPoint2d& erv_position_in_map,
452 const double& erv_current_speed, lanelet::ConstLanelet& intersecting_lanelet){
453
454 // Obtain ego vehicle and ERV distances to the end of the intersecting lanelet so neither vehicle will currently be past that point
455 lanelet::ConstLineString2d intersecting_centerline = lanelet::utils::to2D(intersecting_lanelet.centerline());
456 lanelet::BasicPoint2d intersecting_end_point = intersecting_centerline.back();
457
458 // Get ego vehicle's distance to the intersecting lanelet's centerline endpoint
459 double ego_dist_to_lanelet = wm_->routeTrackPos(intersecting_end_point).downtrack - latest_route_state_.down_track;
460
461 // Set erv_world_model_ route to the erv_future_route
462 lanelet::routing::Route route = std::move(*erv_future_route);
463 carma_wm::LaneletRoutePtr erv_future_route_ptr = std::make_shared<lanelet::routing::Route>(std::move(route));
464 erv_world_model_->setRoute(erv_future_route_ptr);
465
466 // Get downtrack of intersecting lanelet's centerline endpoint on ERV's route
467 double erv_dist_to_lanelet = erv_world_model_->routeTrackPos(intersecting_end_point).downtrack - erv_world_model_->routeTrackPos(erv_position_in_map).downtrack;
468
469 // Calculate seconds_until_passing and protect against division by zero
470 double delta_speed = erv_current_speed - current_speed_;
471 if(delta_speed == 0.0){
472 delta_speed = epsilon_;
473 }
474 double seconds_until_passing = (erv_dist_to_lanelet - ego_dist_to_lanelet) / delta_speed;
475
476 return seconds_until_passing;
477 }

479 boost::optional<lanelet::ConstLanelet> ApproachingEmergencyVehiclePlugin::getRouteIntersectingLanelet(const lanelet::routing::Route& erv_future_route,
480 const lanelet::routing::LaneletPath& ego_shortest_path){
481
482 // Get the ego vehicle's future shortest path lanelets
483 double ending_downtrack = wm_->getRouteEndTrackPos().downtrack;
484 std::vector<lanelet::ConstLanelet> ego_future_shortest_path = wm_->getLaneletsBetween(latest_route_state_.down_track, ending_downtrack);
485
486 if(ego_future_shortest_path.empty()){
487 RCLCPP_WARN_STREAM(rclcpp::get_logger(logger_name), "Remaining shortest path for ego vehicle not found; intersecting lanelet with ERV will not be computed");
488 return boost::optional<lanelet::ConstLanelet>();
489 }
490
491 // Loop through ego vehicle's future shortest path and find first lanelet that exists in ERV's future route
492 for(size_t i = 0; i < ego_future_shortest_path.size(); ++i){
493 if(erv_future_route.contains(ego_future_shortest_path[i])){
494 return boost::optional<lanelet::ConstLanelet>(ego_future_shortest_path[i]);
495 }
496 }
497
498 // No intersecting lanelet was found, return empty object
499 return boost::optional<lanelet::ConstLanelet>();
500 }
