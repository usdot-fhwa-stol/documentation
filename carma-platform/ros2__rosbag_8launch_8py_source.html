25# This function is used to generate a command to record a ROS 2 rosbag that excludes topics
26# topics as provided in the appropriate configuration file.
27def record_ros2_rosbag(context: LaunchContext, vehicle_config_param_file):
28
29 # Convert LaunchConfiguration object to its string representation
30 vehicle_config_param_file_string = context.perform_substitution(vehicle_config_param_file)
31
32 # Initialize string that will contain the regex for topics to exclude from the ROS 2 rosbag
33 exclude_topics_regex = ""
34
35 # Open vehicle config params file to process various rosbag settings
36 with open(vehicle_config_param_file_string, 'r') as f:
37
38 vehicle_config_params = yaml.safe_load(f)
39
40 if "use_ros2_rosbag" in vehicle_config_params:
41 if vehicle_config_params["use_ros2_rosbag"] == True:
42
43 if "exclude_default" in vehicle_config_params:
44 if (vehicle_config_params["exclude_default"] == True) and ("excluded_default_topics" in vehicle_config_params):
45 for topic in vehicle_config_params["excluded_default_topics"]:
46 exclude_topics_regex += str(topic) + "|"
47
48 if "exclude_lidar" in vehicle_config_params:
49 if (vehicle_config_params["exclude_lidar"] == True) and ("excluded_lidar_topics" in vehicle_config_params):
50 for topic in vehicle_config_params["excluded_lidar_topics"]:
51 exclude_topics_regex += str(topic) + "|"
52
53 if "exclude_camera" in vehicle_config_params:
54 if (vehicle_config_params["exclude_camera"] == True) and ("excluded_camera_topics" in vehicle_config_params):
55 for topic in vehicle_config_params["excluded_camera_topics"]:
56 exclude_topics_regex += str(topic) + "|"
57
58 if "exclude_can" in vehicle_config_params:
59 if (vehicle_config_params["exclude_can"] == True) and ("excluded_can_topics" in vehicle_config_params):
60 for topic in vehicle_config_params["excluded_can_topics"]:
61 exclude_topics_regex += str(topic) + "|"
62
63 proc = ExecuteProcess(
64 cmd=['ros2', 'bag', 'record', '-o', '/opt/carma/logs/rosbag2_' + str(datetime.now().strftime('%Y-%m-%d_%H%M%S')), '-a', '-x', exclude_topics_regex],
65 output='screen',
66 shell='true'
67 )
68
69 return [proc]
