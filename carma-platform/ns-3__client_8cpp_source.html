97void NS3Client::process(const std::shared_ptr<const std::vector<uint8_t>>& data)
98{
99 auto & entry = *data;
100 // Valid message should begin with 2 bytes message ID and 1 byte length.
101 for (size_t i = 0; i < entry.size() - 3; i++) { // leave 3 bytes after (for lsb of id, length byte 1, and either message body or length byte 2)
102 // Generate the 16-bit message id from two bytes, skip if it isn't a valid one
103 uint16_t msg_id = (static_cast<uint16_t>(entry[i]) << 8) | static_cast<uint16_t>(entry[i + 1]);
104
105 // Parse the length, check it doesn't run over
106 size_t len = 0;
107 size_t len_byte_1 = entry[i + 2];
108 int len_bytes = 0;
109 // length < 128 encoded by single byte with msb set to 0
110 if ((len_byte_1 & 0x80 ) == 0x00) {
111 len = static_cast<size_t>(len_byte_1);
112 len_bytes = 1;
113 // check for 0 length
114 if (len_byte_1 == 0x00) { continue; }
115 }
116 // length < 16384 encoded by 14 bits in 2 bytes (10xxxxxx xxxxxxxx)
117 else if ((len_byte_1 & 0x40) == 0x00) { //we know msb = 1, check that next bit is 0
118 if (i + 3 < entry.size()) {
119 size_t len_byte_2 = entry[i + 3];
120 len = ((len_byte_1 & 0x3f) << 8) | len_byte_2;
121 len_bytes = 2;
122 }
123 }
124 else {
125 // TODO lengths greater than 16383 (0x3FFF) are encoded by splitting up the message into discrete chunks, each with its own length
126 // marker. It doesn't look like we'll be receiving anything that long
127 std::cerr << "NS3Client::process() : received a message with length field longer than 16383." << std::endl;
128 continue;
129 }
130 if (len == -1) { continue; }
131 // If the length makes sense bsmPub(fits in the buffer), copy out the message bytes and pass to the Application class
132 if ((i + 1 + len + len_bytes) < entry.size()) {
133 // bool found_valid_msg = true;
134 size_t start_index = i;
135 size_t end_index = i + 2 + len + len_bytes; // includes 2 msgID bytes before message body
136 //this constructor has range [first, last) hence the + 1
137 std::vector<uint8_t> msg_vec(entry.begin() + start_index, entry.begin() + end_index);
138 onMessageReceived(msg_vec, msg_id);
139 break;
140 }
141 }
142}
