25 GuidanceControllerNode::GuidanceControllerNode(const rclcpp::NodeOptions &options)
26 : BaseSubsystemController(options)
27 {
28 // Don't automatically trigger state transitions from base class on configure
29 // In this class the managed nodes list first needs to be modified then the transition will be triggered manually
30 trigger_managed_nodes_configure_from_base_class_ = false;
31 config_.required_plugins = declare_parameter<std::vector<std::string>>("required_plugins", config_.required_plugins);
32 config_.auto_activated_plugins = declare_parameter<std::vector<std::string>>("auto_activated_plugins", config_.auto_activated_plugins);
33 config_.ros2_initial_plugins = declare_parameter<std::vector<std::string>>("ros2_initial_plugins", config_.ros2_initial_plugins);
34 }

36 cr2::CallbackReturn GuidanceControllerNode::handle_on_configure(const rclcpp_lifecycle::State &prev_state) {
37 auto base_return = BaseSubsystemController::handle_on_configure(prev_state);
38
39 if (base_return != cr2::CallbackReturn::SUCCESS) {
40 RCLCPP_ERROR(get_logger(), "Guidance Controller could not configure");
41 return base_return;
42 }
43
44 config_ = GuidanceControllerConfig();
45
46 auto base_managed_nodes = lifecycle_mgr_.get_managed_nodes();
47
48 std::string plugin_namespace = base_config_.subsystem_namespace + "/plugins/";
49
50
51 std::vector<std::string> guidance_plugin_nodes;
52 // Extract the nodes under the plugin namespaces (ie. /guidance/plugins/)
53 std::copy_if(base_managed_nodes.begin(), base_managed_nodes.end(),
54 std::back_inserter(guidance_plugin_nodes),
55 [plugin_namespace](const std::string& s) { return s.rfind(plugin_namespace, 0) == 0; });
56
57 std::vector<std::string> non_plugin_guidance_nodes = get_non_intersecting_set(base_managed_nodes, guidance_plugin_nodes);
58
59 lifecycle_mgr_.set_managed_nodes(non_plugin_guidance_nodes);
60
61 // Load required plugins and default enabled plugins
62 get_parameter<std::vector<std::string>>("required_plugins", config_.required_plugins);
63 get_parameter<std::vector<std::string>>("auto_activated_plugins", config_.auto_activated_plugins);
64 get_parameter<std::vector<std::string>>("ros2_initial_plugins", config_.ros2_initial_plugins);
65
66 RCLCPP_INFO_STREAM(get_logger(), "Config: " << config_);
67
68 // The core need is that plugins need to be managed separately from guidance nodes
69 auto plugin_lifecycle_manager = std::make_shared<ros2_lifecycle_manager::Ros2LifecycleManager>(
70 get_node_base_interface(), get_node_graph_interface(), get_node_logging_interface(), get_node_services_interface());
71
72 plugin_manager_ = std::make_shared<PluginManager>(
73 config_.required_plugins,
74 config_.auto_activated_plugins,
75 config_.ros2_initial_plugins,
76 plugin_lifecycle_manager,
77 [this](){ return get_current_state().id(); },
78 [this](auto node, auto ns){ return get_service_names_and_types_by_node(node, ns); },
79 std_msec(base_config_.service_timeout_ms), std_msec(base_config_.call_timeout_ms)
80 );
81
82 plugin_discovery_sub_ = create_subscription<carma_planning_msgs::msg::Plugin>(
83 "plugin_discovery", 50,
84 std::bind(&PluginManager::update_plugin_status, plugin_manager_, std::placeholders::_1));
85
86 get_registered_plugins_server_ = create_service<carma_planning_msgs::srv::PluginList>(
87 "plugins/get_registered_plugins",
88 std::bind(&PluginManager::get_registered_plugins, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
89
90 get_active_plugins_server_ = create_service<carma_planning_msgs::srv::PluginList>(
91 "plugins/get_active_plugins",
92 std::bind(&PluginManager::get_active_plugins, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
93
94 activate_plugin_server_ = create_service<carma_planning_msgs::srv::PluginActivation>(
95 "plugins/activate_plugin", // TODO check topic name
96 std::bind(&PluginManager::activate_plugin, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
97
98 get_strategic_plugins_by_capability_server_ = create_service<carma_planning_msgs::srv::GetPluginApi>(
99 "plugins/get_strategic_plugins_by_capability",
100 std::bind(&PluginManager::get_strategic_plugins_by_capability, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
101
102 get_tactical_plugins_by_capability_server_ = create_service<carma_planning_msgs::srv::GetPluginApi>(
103 "plugins/get_tactical_plugins_by_capability",
104 std::bind(&PluginManager::get_tactical_plugins_by_capability, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
105
106 get_control_plugins_by_capability_server_ = create_service<carma_planning_msgs::srv::GetPluginApi>(
107 "plugins/get_control_plugins_by_capability",
108 std::bind(&PluginManager::get_control_plugins_by_capability, plugin_manager_, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
109
110
111
112 // With all of our non-plugin managed nodes now being tracked we can execute their configure operations
113 bool success = lifecycle_mgr_.configure(std_msec(base_config_.service_timeout_ms), std_msec(base_config_.call_timeout_ms)).empty();
114
115 // Configure our plugins
116 try {
117 plugin_manager_->configure(); // Only checking required nodes. Other node failure tracked by activity status
118 } catch(const std::runtime_error& e) {
119 RCLCPP_ERROR_STREAM(get_logger(), "Error configuring plugin manager " << e.what());
120 success = false;
121 }
122
123
124
125 if (success)
126 {
127
128 RCLCPP_INFO_STREAM(get_logger(), "Subsystem able to configure");
129 return CallbackReturn::SUCCESS;
130 }
131 else
132 {
133
134 RCLCPP_INFO_STREAM(get_logger(), "Subsystem unable to configure");
135 return CallbackReturn::FAILURE;
136 }
137
138 }

140 cr2::CallbackReturn GuidanceControllerNode::handle_on_activate(const rclcpp_lifecycle::State &prev_state)
141 {
142 auto base_return = BaseSubsystemController::handle_on_activate(prev_state); // This will activate all nodes in the namespace TODO what about the plugins?

145 RCLCPP_ERROR(get_logger(), "Guidance Controller could not activate");

155 }
156
