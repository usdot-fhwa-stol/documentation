26 //Convert the SPAT j2735 message to cav_msgs
27 boost::optional<j2735_v2x_msgs::msg::SPAT> SPAT_Message::decode_spat_message(std::vector<uint8_t> &binary_array)
28 {
29 //Decode the binary message into SPAT message
30 j2735_v2x_msgs::msg::SPAT output;
31 //decode results - stored in binary array
32 asn_dec_rval_t rval;
33 MessageFrame_t *message = nullptr;
34
35 //copy from vector to array
36 size_t len = binary_array.size();
37
38 uint8_t buf[len];
39 std::copy(binary_array.begin(), binary_array.end(), buf);
40 //use asn1c lib to decode
41
42 rval = uper_decode(0, &asn_DEF_MessageFrame, (void **) &message, buf, len, 0, 0);
43 //if decode success
44 if (rval.code == RC_OK)
45 {
46 //1. Decode time stamp - Minute of the year
47 long int *minute_of_the_year = new long int;
48 bool is_time_stamp_exists;
49 if (message->value.choice.SPAT.timeStamp)
50 {
51 is_time_stamp_exists = true;
52 minute_of_the_year = message->value.choice.SPAT.timeStamp;
53 }
54 else
55 {
56 is_time_stamp_exists = false;
57 *minute_of_the_year = DEFAULT_MINUTE_OF_YEAR_;
58 // RCLCPP_DEBUG_STREAM( get_logger(), "Minute of the year value doesn't exist, set to Default");
59 }
60
61 output.time_stamp_exists = is_time_stamp_exists;
62 output.time_stamp = *minute_of_the_year;
63 //2. Decode name
64
65 size_t str_len = 0;
66 if(message->value.choice.SPAT.name){
67 std::string name = "";
68 str_len = message->value.choice.SPAT.name->size;
69 for (size_t i = 0; i < str_len; i++)
70 {
71 name += message->value.choice.SPAT.name->buf[i];
72 }
73 output.name_exists = true;
74 output.name = name;
75 }
76 else{
77 output.name_exists = false;
78 }
79
80 //3. Decode Intersection
81 j2735_v2x_msgs::msg::IntersectionStateList intersections;
82
83 //IntersectionStateList is an array of intersectionState - of size 1-32
84 for (size_t i = 0; i < message->value.choice.SPAT.intersections.list.count; i++)
85 {
86 if(!message->value.choice.SPAT.intersections.list.array[i]){
87 continue;
88 }
89 j2735_v2x_msgs::msg::IntersectionState intersection;
90 IntersectionState_t *state = new IntersectionState_t;
91 state = message->value.choice.SPAT.intersections.list.array[i];
92 //Decode intersection name
93 if (state->name)
94 {
95 intersection.name_exists = true;
96
97 for (size_t j = 0; j < state->name->size; j++)
98 {
99 intersection.name += state->name->buf[i];
100 }
101 }
102 else
103 {
104 intersection.name_exists = false;
105 intersection.name = DEFAULT_STRING_;
106 // RCLCPP_DEBUG_STREAM( get_logger(), "Intersection name doesn't exist, set to Default");
107 }
108 //Decode id
109 j2735_v2x_msgs::msg::IntersectionReferenceID id;
110 id.id = state->id.id;
111 if (state->id.region)
112 {
113 id.region_exists = true;
114 id.region = *state->id.region;
115 }
116 else
117 {
118 id.region_exists = false;
119 id.region = j2735_v2x_msgs::msg::IntersectionReferenceID::REGION_UNAVAILABLE;
120 // RCLCPP_DEBUG_STREAM( get_logger(), "Intersection ID doesn't exist, set to Region Unavailable");
121 }
122
123 intersection.id = id;
124
125 //Decode Msg count
126 intersection.revision = state->revision;
127
128 //Decode status
129 int bit_assigned = 0;
130 for(size_t t = 0;t< state->status.size;t++){
131 if(state->status.buf[i] == 1){
132 bit_assigned = t;
133 break;
134 }
135 }
136 intersection.status.intersection_status_object = bit_assigned;
137
138 //Minute of the year
139
140 bool moy_exists = false;
141 if (state->moy)
142 {
143 MinuteOfTheYear_t *moy = new MinuteOfTheYear_t;
144 moy = state->moy;
145 intersection.moy = *moy;
146 intersection.moy_exists = true;
147 }
148 else
149 {
150 intersection.moy_exists = false;
151 intersection.moy = j2735_v2x_msgs::msg::IntersectionState::MOY_INVALID;
152 // RCLCPP_DEBUG_STREAM( get_logger(), "Intersection moy doesn't exis, set to MOY_INVALID");
153 }
154 //Time stamp
155 if (state->timeStamp)
156 {
157 DSecond_t *time_stamp = new DSecond_t;
158 time_stamp = state->timeStamp;
159 intersection.time_stamp = *time_stamp;
160 intersection.time_stamp_exists = true;
161 }
162 else
163 {
164 intersection.time_stamp = j2735_v2x_msgs::msg::IntersectionState::TIME_STAMP_UNAVAILABLE;
165 intersection.time_stamp_exists = false;
166 // RCLCPP_DEBUG_STREAM( get_logger(), "Intersection time stamp doesn't exist, value set to TIME_STAMP_UNAVAILABLE");
167 }
168 //Enabled lanes list
169
170 bool enabled_lanes_exists = false;
171 if (state->enabledLanes)
172 {
173 enabled_lanes_exists = true;
174 j2735_v2x_msgs::msg::EnabledLaneList enabled_lanes_list;
175 for (size_t j = 0; j < state->enabledLanes->list.count; j++)
176 {
177 LaneID_t *enabled_lane_id = new LaneID_t;
178 enabled_lane_id = state->enabledLanes->list.array[j];
179 enabled_lanes_list.lane_id_list.push_back(*enabled_lane_id);
180 }
181 intersection.enabled_lanes = enabled_lanes_list;
182 }
183 else{
184 intersection.enabled_lanes_exists = false;
185 }
186 intersection.enabled_lanes_exists = enabled_lanes_exists;
187
188 //MovementList
189 j2735_v2x_msgs::msg::MovementList movement_states;
190 //movement states is an array of Movement states
191 for (size_t j = 0; j < state->states.list.count; j++)
192 {
193 if(!state->states.list.array[j]){
194 continue;
195 }
196 j2735_v2x_msgs::msg::MovementState movement_state;
197 movement_state.movement_name_exists = false;
198 if (state->states.list.array[j]->movementName)
199 {
200 movement_state.movement_name_exists = true;
201
202 size_t len = state->states.list.array[j]->movementName->size;
203 for (int k = 0; k < len; k++)
204 {
205 movement_state.movement_name += state->states.list.array[j]->movementName->buf[k];
206 }
207
208 }
209
210 //Signal Group ID
211 movement_state.signal_group = SIGNAL_GROUP_UNAVAILABLE_;
212 if (state->states.list.array[j]->signalGroup)
213 {
214 movement_state.signal_group = state->states.list.array[j]->signalGroup;
215 }
216
217 //State Time Speed Movement Event List
218 j2735_v2x_msgs::msg::MovementEventList movement_event_list;
219 for (int k = 0; k < state->states.list.array[j]->state_time_speed.list.count; k++)
220 {
221 if(!state->states.list.array[j]->state_time_speed.list.array[k]){
222 continue;
223 }
224 j2735_v2x_msgs::msg::MovementEvent movement_event;
225 //Decode movement event
226 //1. MovementPhaseState
227 if(state->states.list.array[j]->state_time_speed.list.array[k]->eventState){
228 movement_event.event_state.movement_phase_state = state->states.list.array[j]->state_time_speed.list.array[k]->eventState;
229 }
230 else{
231 movement_event.event_state.movement_phase_state = j2735_v2x_msgs::msg::MovementPhaseState::UNAVAILABLE;
232 // RCLCPP_DEBUG_STREAM( get_logger(), "Movement event - event state, value doesn't exist. Set to default UNAVAILABLE");
233 }
234
235 //2. TimeChangeDetails
236 movement_event.timing_exists = false;
237 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing)
238 {
239 movement_event.timing_exists = true;
240
241 j2735_v2x_msgs::msg::TimeChangeDetails timing;
242 timing.start_time_exists = false;
243 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing->startTime)
244 {
245 timing.start_time_exists = true;
246 DSRC_TimeMark_t *start_time = new DSRC_TimeMark_t;
247 start_time = state->states.list.array[j]->state_time_speed.list.array[k]->timing->startTime;
248 timing.start_time = *start_time;
249 }
250
251 timing.min_end_time = state->states.list.array[j]->state_time_speed.list.array[k]->timing->minEndTime;
252
253 timing.max_end_time_exists = false;
254 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing->maxEndTime)
255 {
256 timing.max_end_time_exists = true;
257 DSRC_TimeMark_t *end_time = new DSRC_TimeMark_t;
258 end_time = state->states.list.array[j]->state_time_speed.list.array[k]->timing->maxEndTime;
259 timing.max_end_time = *end_time;
260 }
261 timing.likely_time_exists = false;
262 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing->likelyTime)
263 {
264 timing.likely_time_exists = true;
265 DSRC_TimeMark_t *likely_time = new DSRC_TimeMark_t;
266 likely_time = state->states.list.array[j]->state_time_speed.list.array[k]->timing->likelyTime;
267 timing.likely_time = *likely_time;
268 }
269
270 timing.confidence_exists = false;
271 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing->confidence)
272 {
273 timing.confidence_exists = true;
274 TimeIntervalConfidence_t *confidence = new TimeIntervalConfidence_t;
275 confidence = state->states.list.array[j]->state_time_speed.list.array[k]->timing->confidence;
276 timing.confidence = *confidence;
277 }
278
279 timing.next_time_exists = false;
280 if (state->states.list.array[j]->state_time_speed.list.array[k]->timing->nextTime)
281 {
282 timing.next_time_exists = true;
283 DSRC_TimeMark_t *next_time = new DSRC_TimeMark_t;
284 next_time = state->states.list.array[j]->state_time_speed.list.array[k]->timing->nextTime;
285 timing.next_time = *next_time;
286 }
287 movement_event.timing = timing;
288 }
289 //3. Advisory Speed List
290 movement_event.speeds_exists = false;
291 if (state->states.list.array[j]->state_time_speed.list.array[k]->speeds)
292 {
293 movement_event.speeds_exists = true;
294
295 for (size_t l = 0; l < state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.count; l++)
296 {
297 j2735_v2x_msgs::msg::AdvisorySpeed advisory_speed;
298
299 advisory_speed.type.advisory_speed_type = state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->type;
300
301 advisory_speed.speed_exists = false;
302 if (state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->speed)
303 {
304 advisory_speed.speed_exists = true;
305 SpeedAdvice_t *speed_advice = new SpeedAdvice_t;
306 speed_advice = state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->speed;
307 advisory_speed.speed = *speed_advice;
308 }
309 else
310 {
311 advisory_speed.speed = j2735_v2x_msgs::msg::AdvisorySpeed::SPEED_UNAVAILABLE;
312 // RCLCPP_DEBUG_STREAM( get_logger(), "Advisory speed- speed doesn't exist, assigned default value speed_unavailable");
313 }
314
315 SpeedConfidence_t *confidence = new SpeedConfidence_t;
316 confidence = state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->confidence;
317 advisory_speed.confidence.speed_confidence = *confidence;
318
319 advisory_speed.distance_exists = false;
320 if (state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->distance)
321 {
322 advisory_speed.distance_exists = true;
323 ZoneLength_t *distance = new ZoneLength_t;
324 distance = state->states.list.array[j]->state_time_speed.list.array[k]->speeds->list.array[l]->distance;
325 advisory_speed.distance = *distance;
326 }
327 else
328 {
329 advisory_speed.distance = j2735_v2x_msgs::msg::AdvisorySpeed::DISTANCE_UNKNOWN;
330 // RCLCPP_DEBUG_STREAM( get_logger(), "Advisory speed - distance, value doesn't exist, set to default distance_unknown");
331 }
332
333 // RESTRICTION CLASS ID not DEFINED in incoming state message
334
335 movement_event.speeds.advisory_speed_list.push_back(advisory_speed);
336 }
337 }
338
339 movement_event_list.movement_event_list.push_back(movement_event);
340 }
341 movement_state.state_time_speed.movement_event_list = movement_event_list.movement_event_list;
342 movement_states.movement_list.push_back(movement_state);
343 }
344 intersection.states = movement_states;
345
346 //ManeuverAssistList
347 j2735_v2x_msgs::msg::ManeuverAssistList maneuver_assist_list;
348 bool maneuver_assist_list_exists = false;
349 if (state->maneuverAssistList)
350 {
351 maneuver_assist_list_exists = true;
352 }
353 intersection.maneuever_assist_list_exists = maneuver_assist_list_exists;
354 intersection.maneuever_assist_list = maneuver_assist_list;
355
356 output.intersections.intersection_state_list.push_back(intersection);
357 }
358
359 //4. Regional - not implemented yet
360
361 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
362 return boost::optional<j2735_v2x_msgs::msg::SPAT>(output);
363 }
364
365 // RCLCPP_WARN_STREAM( rclcpp::get_logger(), "SPAT Message decoding failed");
366 ASN_STRUCT_FREE(asn_DEF_MessageFrame, message);
367 return boost::optional<j2735_v2x_msgs::msg::SPAT>{};
368 }

370 boost::optional<std::vector<uint8_t>> SPAT_Message::encode_spat_message(const j2735_v2x_msgs::msg::SPAT &plainMessage)
371 {
372 //encode result placeholder
373 uint8_t buffer[2048] = {0};
374 size_t buffer_size = sizeof(buffer);
375 asn_enc_rval_t ec;
376 MessageFrame_t* message;
377 message = (MessageFrame_t*) calloc(1, sizeof(MessageFrame_t));
378
379 //if mem allocation fails
380 if(!message)
381 {
382 // RCLCPP_WARN_STREAM( rclcpp::get_logger(), "Cannot allocate mem for SPAT encoding");
383 return boost::optional<std::vector<uint8_t>>{};
384 }
385 //set message type to SPAT
386 message->messageId ;
387 message->value.present = MessageFrame__value_PR_SPAT;
388
389 SPAT* spat_msg;
390 spat_msg = (SPAT*) calloc(1, sizeof(SPAT));
391
392 //Encode timestamp
393 MinuteOfTheYear_t* timestamp = new MinuteOfTheYear_t;
394 if(plainMessage.time_stamp_exists){
395 *timestamp = plainMessage.time_stamp;
396 }
397 else{
398 // RCLCPP_DEBUG_STREAM( get_logger(), "Encoding, Assigning default timestamp");
399 *timestamp = DEFAULT_TIME_STAMP_;
400 }
401
402 message->value.choice.SPAT.timeStamp = timestamp;
403 //encode Descriptive Name
404 std::string name = DEFAULT_STRING_;
405 if(plainMessage.name_exists){
406 name = plainMessage.name;
407 uint8_t string_content[name.size()];
408 for(size_t i = 0; i < name.size(); i++){
409 string_content[i] = name[i];
410 }
411 message->value.choice.SPAT.name->buf = string_content;
412 message->value.choice.SPAT.name->size = name.size();
413 }
414 else{
415 // RCLCPP_DEBUG_STREAM( get_logger(), "Encoding, name doesn't exist");
416 }
417
418
419 //Encode Intersections
420 IntersectionStateList_t* intersectionStateList;
421 intersectionStateList = new IntersectionStateList_t;
422 for(size_t i = 0; i < plainMessage.intersections.intersection_state_list.size(); i++)
423 {
424 IntersectionState_t* intersectionState;
425 intersectionState = new IntersectionState_t;
426
427 if(plainMessage.intersections.intersection_state_list[i].name_exists){
428 size_t name_string_size = plainMessage.intersections.intersection_state_list[i].name.size();
429
430 uint8_t string_content_name[name_string_size];
431 for(size_t i = 0; i < name_string_size; i++){
432 string_content_name[i] = plainMessage.intersections.intersection_state_list[i].name[i];
433 }
434
435 intersectionState->name->buf = string_content_name;
436 intersectionState->name->size = name_string_size;
437 }
438 else{
439 // RCLCPP_DEBUG_STREAM( get_logger(), "Intersection state name doesn't exist for state "<< i);
440 }
441 //No else condition for name doesn't exist
442
443 //Intersection ID - bit string - convert from bit string to int16
444 intersectionState->id.id = plainMessage.intersections.intersection_state_list[i].id.id;
445
446 //Encode Revision
447 intersectionState->revision = plainMessage.intersections.intersection_state_list[i].revision;
448
449 //Encode Intersection Status
450 //Last 2 bits are reserved
451
452 uint8_t status_object[16] = {0};
453 int bit_to_set = plainMessage.intersections.intersection_state_list[i].status.intersection_status_object;
454 //Set the bit to 1
455 if(bit_to_set < 14){
456 status_object[15 - bit_to_set] = 1;
457 }
458
459 intersectionState->status.buf = status_object;
460 intersectionState->status.size = 16;
461 //Encode MinuteoftheYear
462 MinuteOfTheYear_t* minute_of_year = new MinuteOfTheYear_t;
463 if(plainMessage.intersections.intersection_state_list[i].moy_exists)
464 {
465 *minute_of_year = plainMessage.intersections.intersection_state_list[i].moy;
466 }
467 else{
468 *minute_of_year = plainMessage.intersections.intersection_state_list[i].MOY_INVALID;
469 }
470 intersectionState->moy = minute_of_year;
471
472 //Encode time stamp
473 DSecond_t* state_time_stamp = new DSecond_t;
474 if(plainMessage.intersections.intersection_state_list[i].time_stamp_exists){
475 *state_time_stamp = plainMessage.intersections.intersection_state_list[i].time_stamp;
476 }
477 else{
478 *state_time_stamp = plainMessage.intersections.intersection_state_list[i].TIME_STAMP_UNAVAILABLE;
479 }
480 intersectionState->timeStamp = state_time_stamp;
481
482 //Encode Enabled lanes
483 if(plainMessage.intersections.intersection_state_list[i].enabled_lanes_exists){
484 EnabledLaneList_t* enabled_lanes = new EnabledLaneList_t;
485 for(size_t j = 0; j < plainMessage.intersections.intersection_state_list[i].enabled_lanes.lane_id_list.size(); j++){
486 LaneID_t* lane_id = new LaneID_t;
487 *lane_id = plainMessage.intersections.intersection_state_list[i].enabled_lanes.lane_id_list[j];
488 asn_sequence_add(&enabled_lanes->list, lane_id);
489 }
490 intersectionState->enabledLanes = enabled_lanes;
491
492 }
493
494 //Movement List - List of movement states
495 //Encode Movement State
496 //1. movement name
497 MovementList_t* movementStateList = new MovementList_t;
498 //intersectionState->states.list
499 for(size_t j =0; j < plainMessage.intersections.intersection_state_list[i].states.movement_list.size(); j++){
500 MovementState_t* movementstate = new MovementState_t;
501
502 //Movement name
503 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].movement_name_exists){
504 size_t movement_name_size = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].movement_name.size();
505 uint8_t movement_name_array[movement_name_size];
506 for(size_t k = 0; k < movement_name_size; k++){
507 movement_name_array[k] = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].movement_name[k];
508 }
509 movementstate->movementName->buf = movement_name_array;
510 movementstate->movementName->size = movement_name_size;
511 }
512
513 //Signal group
514 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].signal_group){
515 movementstate->signalGroup = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].signal_group;
516 }
517 else{
518 movementstate->signalGroup = DEFAULT_SIGNAL_GROUP_;
519 }
520
521 //Encode State time speed
522 MovementEventList_t* movement_event_list = new MovementEventList_t;
523 for(size_t k = 0; k < plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list.size(); k++){
524
525 MovementEvent_t* movement_event = new MovementEvent_t;
526
527 movement_event->eventState = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].event_state.movement_phase_state;
528 //Time Change Details
529 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing_exists){
530
531 TimeChangeDetails_t* time_change_details = new TimeChangeDetails_t;
532 //start time
533 DSRC_TimeMark_t* state_start_time = new DSRC_TimeMark_t;
534 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.start_time_exists){
535 *state_start_time = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.start_time;
536 }
537 else{
538 *state_start_time = DEFAULT_TIME_MARK_;
539 }
540 time_change_details->startTime = state_start_time;
541 //min_end_time
542 DSRC_TimeMark_t* state_min_end_time = new DSRC_TimeMark_t;
543 *state_min_end_time = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.min_end_time;
544 time_change_details->minEndTime = *state_min_end_time;
545
546 //max end time
547 DSRC_TimeMark_t* state_max_end_time = new DSRC_TimeMark_t;
548 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.max_end_time_exists){
549 *state_max_end_time = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.max_end_time;
550 }
551 else{
552 *state_max_end_time = DEFAULT_TIME_MARK_;
553 }
554 time_change_details->maxEndTime = state_max_end_time;
555
556 //likely time
557 DSRC_TimeMark_t* state_likely_time = new DSRC_TimeMark_t;
558 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.likely_time_exists){
559 *state_likely_time = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.likely_time;
560 }
561 else{
562 *state_likely_time = DEFAULT_TIME_MARK_;
563 }
564 time_change_details->likelyTime = state_likely_time;
565
566 //confidence
567 TimeIntervalConfidence_t* state_confidence = new TimeIntervalConfidence_t;
568 if(plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.confidence_exists){
569 *state_confidence = plainMessage.intersections.intersection_state_list[i].states.movement_list[j].state_time_speed.movement_event_list[k].timing.confidence;
570 }
571 time_change_details->confidence = state_confidence;
572 //Else condition not defined
573
574 //Time Mark
