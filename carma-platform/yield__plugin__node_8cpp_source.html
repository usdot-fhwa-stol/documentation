52 carma_ros2_utils::CallbackReturn YieldPluginNode::on_configure_plugin()
53 {
54 config_ = YieldPluginConfig();
55
56 get_parameter<double>("acceleration_adjustment_factor", config_.acceleration_adjustment_factor);
57 get_parameter<double>("min_obstacle_speed_in_ms", config_.min_obstacle_speed_in_ms);
58 get_parameter<double>("on_route_vehicle_collision_horizon_in_s", config_.on_route_vehicle_collision_horizon_in_s);
59 get_parameter<double>("collision_check_radius_in_m", config_.collision_check_radius_in_m);
60 get_parameter<double>("yield_max_deceleration_in_ms2", config_.yield_max_deceleration_in_ms2);
61 get_parameter<double>("minimum_safety_gap_in_meters", config_.minimum_safety_gap_in_meters);
62 get_parameter<double>("max_stop_speed_in_ms", config_.max_stop_speed_in_ms);
63 get_parameter<bool>("enable_cooperative_behavior", config_.enable_cooperative_behavior);
64 get_parameter<bool>("always_accept_mobility_request", config_.always_accept_mobility_request);
65 get_parameter<int>("acceptable_passed_timesteps", config_.acceptable_passed_timesteps);
66 get_parameter<double>("intervehicle_collision_distance_in_m", config_.intervehicle_collision_distance_in_m);
67
68 get_parameter<double>("safety_collision_time_gap_in_s", config_.safety_collision_time_gap_in_s);
69 get_parameter<bool>("enable_adjustable_gap", config_.enable_adjustable_gap);
70 get_parameter<int>("acceptable_urgency", config_.acceptable_urgency);
71 get_parameter<double>("speed_moving_average_window_size", config_.speed_moving_average_window_size);
72 get_parameter<double>("min_obj_avoidance_plan_time_in_s", config_.min_obj_avoidance_plan_time_in_s);
73 get_parameter<double>("vehicle_length", config_.vehicle_length);
74 get_parameter<double>("vehicle_height", config_.vehicle_height);
75 get_parameter<double>("vehicle_width", config_.vehicle_width);
76 get_parameter<std::string>("vehicle_id", config_.vehicle_id);
77
78 RCLCPP_INFO_STREAM(get_logger(), "YieldPlugin Params: " << config_);
79
80 worker_ = std::make_shared<YieldPlugin>(shared_from_this(), get_world_model(), config_,
81 [this](auto msg) { mob_resp_pub_->publish(msg); },
82 [this](auto msg) { lc_status_pub_->publish(msg); });
83
84 // Publisher
85 mob_resp_pub_ = create_publisher<carma_v2x_msgs::msg::MobilityResponse>("outgoing_mobility_response", 1);
86 lc_status_pub_ = create_publisher<carma_planning_msgs::msg::LaneChangeStatus>("cooperative_lane_change_status", 10);
87
88 // Subscriber
89 mob_request_sub_ = create_subscription<carma_v2x_msgs::msg::MobilityRequest>("incoming_mobility_request", 10, std::bind(&YieldPlugin::mobilityrequest_cb,worker_.get(),std_ph::_1));
90 bsm_sub_ = create_subscription<carma_v2x_msgs::msg::BSM>("bsm_outbound", 1, std::bind(&YieldPlugin::bsm_cb,worker_.get(),std_ph::_1));
91 georeference_sub_ = create_subscription<std_msgs::msg::String>("georeference", 10,
92 [this](const std_msgs::msg::String::SharedPtr msg) {
93 worker_->set_georeference_string(msg->data);
94 });
95
96 // Return success if everything initialized successfully
97 external_objects_sub_ = create_subscription<carma_perception_msgs::msg::ExternalObjectList>("external_object_predictions", 20,
98 [this](const carma_perception_msgs::msg::ExternalObjectList::SharedPtr msg) {
99 worker_->set_external_objects(msg->objects);
100 });
101 // Return success if everything initialized successfully
102 return CallbackReturn::SUCCESS;
103 }
