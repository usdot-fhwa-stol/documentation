108 signal_phase_and_timing::movement_event tsc_state::get_following_event(const signal_phase_and_timing::movement_event& current_event,
109 uint64_t current_event_end_time, const signal_group_state& phase_state) const
110 {
111 signal_phase_and_timing::movement_event next_event;
112 switch (current_event.event_state){
113 case signal_phase_and_timing::movement_phase_state::protected_movement_allowed: //Green
114 // Create next movement - yellow
115 next_event.event_state = signal_phase_and_timing::movement_phase_state::protected_clearance;
116 next_event.timing.start_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time);
117 next_event.timing.min_end_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time + phase_state.yellow_duration);
118 break;
119
120 case signal_phase_and_timing::movement_phase_state::protected_clearance: //Yellow
121 // Create next movement - red
122 next_event.event_state = signal_phase_and_timing::movement_phase_state::stop_and_remain;
123 next_event.timing.start_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time);
124 next_event.timing.min_end_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time + phase_state.red_duration);
125 break;
126
127 case signal_phase_and_timing::movement_phase_state::stop_and_remain: //Red
128 // Create next movement - green
129 next_event.event_state = signal_phase_and_timing::movement_phase_state::protected_movement_allowed;
130 next_event.timing.start_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time);
131 next_event.timing.min_end_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time + phase_state.green_duration);
132 break;
133
134 default:
135 SPDLOG_ERROR("This movement phase is not supported. Movement phase type: {0}", int(current_event.event_state));
136 throw snmp_client_exception("Failed request for unsupported movement phase type: " + std::to_string(int(current_event.event_state)));
137 }
138 next_event.timing.max_end_time = next_event.timing.min_end_time;
139 return next_event;

142 void tsc_state::add_future_movement_events(std::shared_ptr<signal_phase_and_timing::spat> spat_ptr)
143 {
144 // Modify spat according to phase configuration
145 // Note: Only first intersection is populated
146 auto intersection_state = spat_ptr->get_intersection();
147 for (auto movement : intersection_state.states)
148 {
149 int signal_group_id = movement.signal_group;
150
151 // Assumption here is that the movement states have only one event, since this method adds future events to the list.
152 //Throw exception is list size is great than 1
153 if (movement.state_time_speed.size() > 1)
154 {
155 SPDLOG_ERROR("Event list has more than one events, not usable when adding future movement events. Associated with Signal Group: {0}", signal_group_id);
156 throw monitor_states_exception("Event list has more than one events, not usable when adding future movement events. Associated with Signal Group:" + std::to_string(signal_group_id));
157 }
158
159 // Get movement_state by reference
160 auto& current_movement = intersection_state.get_movement(signal_group_id);
161
162 // Get start time as epoch time
163 uint64_t start_time = movement.state_time_speed.front().timing.get_epoch_start_time();
164
165 // Check if signal_group_id is associated with a vehicle phase : Only vehicle phases mapped to signal_group_states
166 signal_group_state phase_state;
167 if(signal_group_state_map_.find(signal_group_id) == signal_group_state_map_.end()){
168 continue;
169 }
170
171 phase_state = signal_group_state_map_[signal_group_id];
172
173 uint64_t current_event_end_time_epoch = 0;
174 signal_phase_and_timing::movement_event current_event = current_movement.state_time_speed.front();
175
176 switch(current_event.event_state){
177 case signal_phase_and_timing::movement_phase_state::stop_and_remain : //Red
178 current_event_end_time_epoch = start_time + phase_state.red_duration;
179 break;
180
181 case signal_phase_and_timing::movement_phase_state::protected_movement_allowed : //Green
182 current_event_end_time_epoch = start_time + phase_state.green_duration;
183 break;
184
185 case signal_phase_and_timing::movement_phase_state::protected_clearance : //Yellow
186 current_event_end_time_epoch = start_time + phase_state.yellow_duration;
187 break;
188
189 default:
190 SPDLOG_DEBUG("This movement phase is not supported. Movement phase type: {0}", int(current_movement.state_time_speed.front().event_state));
191 throw monitor_states_exception("This movement phase is not supported. Movement phase type: " + std::to_string(int(current_movement.state_time_speed.front().event_state)));
192 }
193 // Update end_time for current_event
194 current_movement.state_time_speed.front().timing.min_end_time = convert_msepoch_to_hour_tenth_secs(current_event_end_time_epoch);
195 current_movement.state_time_speed.front().timing.max_end_time = current_movement.state_time_speed.front().timing.min_end_time;
196
197 for(int i = 0; i < required_following_movements_; ++i)
198 {
199 signal_phase_and_timing::movement_event next_event = get_following_event(current_event, current_event_end_time_epoch, phase_state);
200 current_event = next_event;
201 current_event_end_time_epoch = convert_hour_tenth_secs2epoch_ts(current_event.timing.min_end_time);
202 //Add events to list
203 current_movement.state_time_speed.push_back(next_event);
204 }
205
206 }
207 spat_ptr->set_intersection(intersection_state);
208

293 void tsc_state::map_phase_and_signalgroup(const std::vector<int>& phase_channels, bool is_source_vehicle_channel)
294 {
295 // According to NTCIP 1202 v03 documentation Signal Group ID in a SPAT message is the Channel Number from TSC
296
297 // Get phases associated with vehicle phase channels
298 request_type request_type = request_type::GET;
299 for(int channel : phase_channels)
300 {
301 snmp_response_obj phase_num;
302 phase_num.type = snmp_response_obj::response_type::INTEGER;
303
304 std::string control_source_parameter_oid = ntcip_oids::CHANNEL_CONTROL_SOURCE_PARAMETER + "." + std::to_string(channel);
305 snmp_client_worker_->process_snmp_request(control_source_parameter_oid, request_type, phase_num);
306
307 // According to NTCIP 1202 v03 returned value of 0 here would mean a phase is not associated with the channel
308 if(phase_num.val_int != 0)
309 {
310 if(is_source_vehicle_channel){
311 vehicle_phase_num_map_.insert(std::make_pair(phase_num.val_int, channel));
312 signal_group_phase_map_.insert(std::make_pair(channel, phase_num.val_int));
313 }
314 else{
315 ped_phase_num_map_.insert(std::make_pair(phase_num.val_int, channel));
316 }
317 SPDLOG_DEBUG("Found mapping between signal group: {0} and phase num: {1}", channel , phase_num.val_int );
318 }
319 }

403 int tsc_state::get_red_duration(int phase_num)
404 {
405 //red duration is calculated as red_clearance time for current phase + (green duration + yellow duration + red_clearance) for other phases in sequence
406
407 // Find signal state associated with phase num
408 int current_signal_group;
409 current_signal_group = get_signal_group_id(phase_num);
410 auto current_signal_group_state = signal_group_state_map_[current_signal_group];
411 // Only add clearance time for current phase
412 int red_duration = current_signal_group_state.red_clearance;
413
414 for(auto phase : current_signal_group_state.phase_seq)
415 {
416 // Find state params for each phase in seq
417 int seq_signal_group;
418 seq_signal_group = get_signal_group_id(phase);
419 auto seq_signal_group_state = signal_group_state_map_[seq_signal_group];
420 if(phase == phase_num)
421 {
422 continue;
423 }
424 else{
425 red_duration += seq_signal_group_state.green_duration + seq_signal_group_state.yellow_duration + seq_signal_group_state.red_clearance;
426 }
427 }
428
429 return red_duration;
430 }

502 uint16_t tsc_state::convert_msepoch_to_hour_tenth_secs(uint64_t epoch_time_ms) const{
503 auto system_time = std::chrono::system_clock::now();
504 auto duration = system_time.time_since_epoch();
505 auto hours_since_epoch = std::chrono::duration_cast<std::chrono::hours>(duration).count();
506 auto hour_tenth_secs = (epoch_time_ms - (hours_since_epoch * HOUR_TO_SECONDS_ * SECOND_TO_MILLISECONDS_))/100;
507 return static_cast<uint16_t>(hour_tenth_secs);
508 }

513 auto hours_since_epoch = std::chrono::duration_cast<std::chrono::hours>(duration).count();
514 auto epoch_start_time = hours_since_epoch * HOUR_TO_SECONDS_ * SECOND_TO_MILLISECONDS_ + hour_tenth_secs * 100;
515 return epoch_start_time;
