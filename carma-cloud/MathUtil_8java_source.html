23 public static void getIntersection(double dPx, double dPy, double dEnd1x, double dEnd1y, double dQx, double dQy, double dEnd2x, double dEnd2y, double[] dInter)
24 {
25 dInter[0] = Double.NaN;
26 dInter[1] = Double.NaN;
27 double dDeltaQPx = dQx - dPx;
28 double dDeltaQPy = dQy - dPy;
29 double dRx = dEnd1x - dPx;
30 double dRy = dEnd1y - dPy;
31 double dSx = dEnd2x - dQx;
32 double dSy = dEnd2y - dQy;
33 double dRCrossS = cross(dRx, dRy, dSx, dSy);
34 if (dRCrossS == 0)
35 return;
36 double dT = cross(dDeltaQPx, dDeltaQPy, dSx, dSy) / dRCrossS;
37 if (dT < 0 || dT > 1)
38 return;
39 double dU = cross(dDeltaQPx, dDeltaQPy, dRx, dRy) / dRCrossS;
40 if (dU < 0 || dU > 1)
41 return;
42 dInter[0] = dPx + dT * dRx;
43 dInter[1] = dPy + dT * dRy;
44 }

108 public static double normalizeRadians(double dRad)
109 {
110 if (dRad < 0)
111 return dRad + Math.ceil(-dRad / TWOPI) * TWOPI;
112 else if (dRad >= TWOPI)
113 return dRad - Math.floor(dRad / TWOPI) * TWOPI;
