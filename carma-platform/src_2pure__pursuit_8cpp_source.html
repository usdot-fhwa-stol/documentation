29 double PurePursuit::getLookaheadDist(const cav_msgs::TrajectoryPlanPoint& tp) const{
30 double x_diff = (tp.x - current_pose_.position.x);
31 double y_diff = (tp.y - current_pose_.position.y);
32 double dist = std::sqrt(x_diff * x_diff + y_diff * y_diff);
33 ROS_DEBUG_STREAM("calculated lookahead: " << dist);
34 return dist;
35 }

61 double PurePursuit::calculateKappa(const cav_msgs::TrajectoryPlanPoint& tp)
62 {
63 double lookahead = getLookaheadDist(tp);
64 ROS_DEBUG_STREAM("used lookahead: " << lookahead);
65 double alpha = getAlphaSin(tp, current_pose_);
66 ROS_DEBUG_STREAM("calculated alpha: " << alpha);
67 double kappa = 2 * (alpha)/(lookahead);
68 ROS_DEBUG_STREAM("calculated kappa: " << kappa);
69 return kappa;
70 }

72 void PurePursuit::calculateSteer(const cav_msgs::TrajectoryPlanPoint& tp)
73 {
74
75 double kappa = calculateKappa(tp);
76 ROS_DEBUG_STREAM("kappa pp: " << kappa);
77
78 double lookahead = getLookaheadDist(tp);
79 ROS_DEBUG_STREAM("lookahead pp: " << lookahead);
80
81
82 double error=kappa*lookahead*lookahead/2;
83 ROS_DEBUG_STREAM("error term pp: " << error);
84
85 // Integral term
86 _integral += error * config_.dt;
87 ROS_DEBUG_STREAM("Integral term pp: " << _integral);
88
89 if (_integral > config_.integratorMax_pp){
90 _integral = config_.integratorMax_pp;
91 }
92 else if (_integral < config_.integratorMin_pp){
93 _integral = config_.integratorMin_pp;
94 }
95 double Iout = config_.Ki_pp * _integral;
96 ROS_DEBUG_STREAM("Iout pp: " << Iout);
97 double steering = atan(config_.wheelBase * kappa)+Iout;
98
99
100 steering += config_.correctionAngle;
101 ROS_DEBUG_STREAM("calculated steering angle: " << steering);
102 double filtered_steering = lowPassfilter(config_.lowpassGain, prev_steering, steering);
103 ROS_DEBUG_STREAM("filtered steering: " << filtered_steering);
104 if (std::isnan(filtered_steering)) filtered_steering = prev_steering;
105 prev_steering = filtered_steering;
106 steering_angle_ = filtered_steering;
107
108 double ang_vel = velocity_ * kappa;
109 ROS_DEBUG_STREAM("calculated angular velocity: " << ang_vel);
110 double filtered_ang_vel = lowPassfilter(config_.lowpassGain, prev_ang_vel, ang_vel);
111 ROS_DEBUG_STREAM("filtered angular velocity: " << filtered_ang_vel);
112 prev_ang_vel = filtered_ang_vel;
113 if (std::isnan(filtered_ang_vel)) filtered_ang_vel = prev_ang_vel;
114 angular_velocity_ = filtered_ang_vel;
115 }

117 double PurePursuit::getAlphaSin(cav_msgs::TrajectoryPlanPoint tp, geometry_msgs::Pose current_pose)
118 {
119 tf::Transform inverse;
120 tf::poseMsgToTF(current_pose, inverse);
121 tf::Transform transform = inverse.inverse();
122
123 geometry_msgs::Point point_msg;
124 point_msg.x = tp.x;
125 point_msg.y = tp.y;
126 point_msg.z = current_pose.position.z;
127 tf::Point p;
128 pointMsgToTF(point_msg, p);
129 tf::Point tf_p = transform * p;
130 geometry_msgs::Point tf_point_msg;
131 pointTFToMsg(tf_p, tf_point_msg);
132 ROS_DEBUG_STREAM("relative latitude: " << tf_point_msg.y);
133 ROS_DEBUG_STREAM("relative longitude: " << tf_point_msg.x);
134 ROS_DEBUG_STREAM("relative z: " << tf_point_msg.z);
135 double vec_mag = std::sqrt(tf_point_msg.y*tf_point_msg.y + tf_point_msg.x*tf_point_msg.x + tf_point_msg.z*tf_point_msg.z);
136 ROS_DEBUG_STREAM("relative vector mag: " << vec_mag);
137 double sin_alpha = tf_point_msg.y/vec_mag;
138 ROS_DEBUG_STREAM("alpha sin from transform: " << sin_alpha);
139
140 double angle_tp_map = atan2(tp.y, tp.x); // angle of vector to tp point in map frame
141 ROS_DEBUG_STREAM("angle_tp_map: " << angle_tp_map);
142 tf::Quaternion quat;
143 tf::quaternionMsgToTF(current_pose.orientation, quat);
144 double roll, pitch, yaw;
145 tf::Matrix3x3(quat).getRPY(roll, pitch, yaw);
146 ROS_DEBUG_STREAM("yaw: " << yaw);
147 double alpha = angle_tp_map - yaw;
148 double sin_alpha2 = sin(alpha);
149 ROS_DEBUG_STREAM("alpha from orientation: " << alpha);
150 ROS_DEBUG_STREAM("alpha sin from orientation: " << sin_alpha2);
151
152 return sin_alpha;
153 }
