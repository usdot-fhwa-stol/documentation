29double CostofComfort::compute_cost(cav_msgs::ManeuverPlan plan) const
30{
31 double cost = 0.0;
32 int maneuver_size = plan.maneuvers.size();
33 for (auto it = plan.maneuvers.begin(); it != plan.maneuvers.end(); it++)
34 {
35 double average_acceleration = fabs((cost_utils::get_maneuver_start_speed(*it) - cost_utils::get_maneuver_end_speed(*it)) /
36 (cost_utils::get_maneuver_end_time(*it).toSec() - cost_utils::get_maneuver_start_time(*it).toSec()));
37 cost += average_acceleration;
38
39 // If there is a lane change, add 1.0 as the cost
40 if (cost_utils::get_maneuver_starting_lane_id(*it).compare(cost_utils::get_maneuver_ending_lane_id(*it)) != 0)
41 {
42 cost += 1.0;
43 }
44 }
45
46 // Normalize the cost to 0-1
47 cost = cost / ((fabs(max_deceleration_) + 1.0) * maneuver_size);
48
49 return cost;
50}
