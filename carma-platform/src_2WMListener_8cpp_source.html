25WMListener::WMListener(bool multi_thread) : worker_(std::unique_ptr<WMListenerWorker>(new WMListenerWorker)), multi_threaded_(multi_thread)
26{
27
28 ROS_DEBUG_STREAM("WMListener: Creating world model listener");
29
30 if (multi_threaded_)
31 {
32 ROS_DEBUG_STREAM("WMListener: Using multi-threaded subscription");
33 nh_.setCallbackQueue(&async_queue_);
34 }
35 map_update_sub_= nh_.subscribe("map_update", 200, &WMListener::mapUpdateCallback, this);
36 map_sub_ = nh_.subscribe("semantic_map", 2, &WMListenerWorker::mapCallback, worker_.get());
37 route_sub_ = nh_.subscribe("route", 1, &WMListenerWorker::routeCallback, worker_.get());
38 roadway_objects_sub_ = nh_.subscribe("roadway_objects", 1, &WMListenerWorker::roadwayObjectListCallback, worker_.get());
39 traffic_spat_sub_ = nh_.subscribe("incoming_spat", 20, &WMListenerWorker::incomingSpatCallback, worker_.get());
40
41 double cL;
42 nh2_.getParam("/config_speed_limit", cL);
43 setConfigSpeedLimit(cL);
44
45 std::string participant;
46 nh2_.getParam("/vehicle_participant_type", participant);
47 worker_->setVehicleParticipationType(participant);
48
49
50 // Set up AsyncSpinner for multi-threaded use case
51 if (multi_threaded_)
52 {
53 wm_spinner_ = std::unique_ptr<ros::AsyncSpinner>(new ros::AsyncSpinner(1, &async_queue_));
54 wm_spinner_->start();
55 }
56}
