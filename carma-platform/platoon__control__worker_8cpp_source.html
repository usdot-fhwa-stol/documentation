48 void PlatoonControlWorker::generateSpeed(const cav_msgs::TrajectoryPlanPoint& point)
49 {
50 double speed_cmd = 0;
51
52 if (!last_cmd_set_)
53 {
54 // last speed command for smooth speed transition
55 lastCmdSpeed = currentSpeed;
56 last_cmd_set_ = true;
57 }
58
59 PlatoonLeaderInfo leader = platoon_leader;
60 if(leader.staticId != "" && leader.staticId != ctrl_config_.vehicleID)
61 {
62 double controllerOutput = 0.0;
63
64
65 double leaderCurrentPosition = leader.vehiclePosition;
66 ROS_DEBUG_STREAM("The current leader position is " << leaderCurrentPosition);
67
68 double desiredHostPosition = leaderCurrentPosition - desired_gap_;
69 ROS_DEBUG_STREAM("desiredHostPosition = " << desiredHostPosition);
70
71 double hostVehiclePosition = leaderCurrentPosition - actual_gap_;
72 ROS_DEBUG_STREAM("hostVehiclePosition = " << hostVehiclePosition);
73
74 controllerOutput = pid_ctrl_.calculate(desiredHostPosition, hostVehiclePosition);
75
76 double adjSpeedCmd = controllerOutput + leader.commandSpeed;
77 ROS_DEBUG_STREAM("Adjusted Speed Cmd = " << adjSpeedCmd << "; Controller Output = " << controllerOutput
78 << "; Leader CmdSpeed= " << leader.commandSpeed << "; Adjustment Cap " << ctrl_config_.adjustmentCap);
79 // After we get a adjSpeedCmd, we apply three filters on it if the filter is enabled
80 // First: we do not allow the difference between command speed of the host vehicle and the leader's commandSpeed higher than adjustmentCap
81
82 speed_cmd = adjSpeedCmd;
83 ROS_DEBUG_STREAM("A speed command is generated from command generator: " << speed_cmd << " m/s");
84
85 if(enableMaxAdjustmentFilter)
86 {
87 if(speed_cmd > leader.commandSpeed + ctrl_config_.adjustmentCap) {
88 speed_cmd = leader.commandSpeed + ctrl_config_.adjustmentCap;
89 } else if(speed_cmd < leader.commandSpeed - ctrl_config_.adjustmentCap) {
90 speed_cmd = leader.commandSpeed - ctrl_config_.adjustmentCap;
91 }
92 ROS_DEBUG_STREAM("The adjusted cmd speed after max adjustment cap is " << speed_cmd << " m/s");
93 }
94
95 }
96
97 else if (leader.staticId == ctrl_config_.vehicleID)
98 {
99 ROS_DEBUG_STREAM("Host vehicle is the leader");
100 speed_cmd = currentSpeed;
101
102 if(enableMaxAdjustmentFilter)
103 {
104 if(speed_cmd > ctrl_config_.adjustmentCap)
105 {
106 speed_cmd = ctrl_config_.adjustmentCap;
107 }
108
109 ROS_DEBUG_STREAM("The adjusted leader cmd speed after max adjustment cap is " << speed_cmd << " m/s");
110 }
111
112 pid_ctrl_.reset();
113 }
114
115 else
116 {
117 // If there is no leader available, the vehicle will stop. This means there is a mis-communication between platooning strategic and control plugins.
118 ROS_DEBUG_STREAM("There is no leader available");
119 speed_cmd = 0.0;
120 pid_ctrl_.reset();
121 }
122
123
124
125 // Third: we allow do not a large gap between two consecutive speed commands
126 if(enableMaxAccelFilter) {
127
128 double max = lastCmdSpeed + (ctrl_config_.maxAccel * (ctrl_config_.cmdTmestamp / 1000.0));
129 double min = lastCmdSpeed - (ctrl_config_.maxAccel * (ctrl_config_.cmdTmestamp / 1000.0));
130 if(speed_cmd > max) {
131 speed_cmd = max;
132 } else if (speed_cmd < min) {
133 speed_cmd = min;
134 }
135 lastCmdSpeed = speed_cmd;
136 ROS_DEBUG_STREAM("The speed command after max accel cap is: " << speed_cmd << " m/s");
137 }
138
139 speedCmd_ = speed_cmd;
140
141 lastCmdSpeed = speedCmd_;
142
143 }
