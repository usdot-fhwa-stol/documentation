27auto transform_from_map_to_utm(
28 carma_cooperative_perception_interfaces::msg::DetectionList detection_list,
29 const std::string & map_origin) -> carma_cooperative_perception_interfaces::msg::DetectionList
30{
31 gsl::owner<PJ_CONTEXT *> context = proj_context_create();
32 proj_log_level(context, PJ_LOG_NONE);
33
34 if (context == nullptr) {
35 const std::string error_string{proj_errno_string(proj_context_errno(context))};
36 throw std::invalid_argument("Could not create PROJ context: " + error_string + '.');
37 }
38
39 gsl::owner<PJ *> map_transformation = proj_create(context, map_origin.c_str());
40
41 if (map_transformation == nullptr) {
42 const std::string error_string{proj_errno_string(proj_context_errno(context))};
43 throw std::invalid_argument("Could not create PROJ transform: " + error_string + '.');
44 }
45
46 std::vector<carma_cooperative_perception_interfaces::msg::Detection> new_detections;
47 for (auto detection : detection_list.detections) {
48 // Coordinate order is easting (meters), northing (meters)
49 const auto position_planar{
50 proj_coord(detection.pose.pose.position.x, detection.pose.pose.position.y, 0, 0)};
51 const auto proj_inverse{proj_trans(map_transformation, PJ_DIRECTION::PJ_INV, position_planar)};
52 const Wgs84Coordinate position_wgs84{
53 units::angle::radian_t{proj_inverse.lp.phi}, units::angle::radian_t{proj_inverse.lp.lam},
54 units::length::meter_t{detection.pose.pose.position.z}};
55
56 const auto utm_zone{calculate_utm_zone(position_wgs84)};
57 const auto position_utm{project_to_utm(position_wgs84)};
58
59 detection.header.frame_id = to_string(utm_zone);
60 detection.pose.pose.position.x = remove_units(position_utm.easting);
61 detection.pose.pose.position.y = remove_units(position_utm.northing);
62
63 new_detections.push_back(std::move(detection));
64 }
65
66 std::swap(detection_list.detections, new_detections);
67
68 proj_destroy(map_transformation);
69 proj_context_destroy(context);
70
71 return detection_list;
72}

82auto ExternalObjectListToDetectionListNode::handle_on_configure(
83 const rclcpp_lifecycle::State & /* previous_state */) -> carma_ros2_utils::CallbackReturn
84{
85 publisher_ = create_publisher<output_msg_type>("output/detections", 1);
86
87 external_objects_subscription_ = create_subscription<input_msg_type>(
88 "input/external_objects", 1, [this](input_msg_shared_pointer msg_ptr) {
89 const auto current_state{this->get_current_state().label()};
90
91 if (current_state == "active") {
92 publish_as_detection_list(*msg_ptr);
93 } else {
94 RCLCPP_WARN(
95 this->get_logger(),
96 "Trying to receive message on the topic '%s', but the containing node is not activated. "
97 "Current node state: '%s'",
98 this->external_objects_subscription_->get_topic_name(), current_state.c_str());
99 }
100 });
101
102 georeference_subscription_ = create_subscription<std_msgs::msg::String>(
103 "input/georeference", 1, [this](std_msgs::msg::String::SharedPtr msg_ptr) {
104 const auto current_state{this->get_current_state().label()};
105
106 if (current_state == "active") {
107 update_georeference(*msg_ptr);
108 } else {
109 RCLCPP_WARN(
110 this->get_logger(),
111 "Trying to receive message on the topic '%s', but the containing node is not activated. "
112 "Current node state: '%s'",
113 this->georeference_subscription_->get_topic_name(), current_state.c_str());
114 }
115 });
116
117 declare_parameter(
118 "small_vehicle_motion_model",
119 carma_cooperative_perception_interfaces::msg::Detection::MOTION_MODEL_CTRV);
120
121 declare_parameter(
122 "large_vehicle_motion_model",
123 carma_cooperative_perception_interfaces::msg::Detection::MOTION_MODEL_CTRV);
124
125 declare_parameter(
126 "motorcycle_motion_model",
127 carma_cooperative_perception_interfaces::msg::Detection::MOTION_MODEL_CTRA);
128
129 declare_parameter(
130 "pedestrian_motion_model",
131 carma_cooperative_perception_interfaces::msg::Detection::MOTION_MODEL_CV);
132
133 declare_parameter(
134 "unknown_motion_model",
135 carma_cooperative_perception_interfaces::msg::Detection::MOTION_MODEL_CV);
136
137 on_set_parameters_callback_ =
138 add_on_set_parameters_callback([this](const std::vector<rclcpp::Parameter> & parameters) {
139 rcl_interfaces::msg::SetParametersResult result;
140 result.successful = true;
141 result.reason = "success";
142
143 for (const auto & parameter : parameters) {
144 if (parameter.get_name() == "small_vehicle_motion_model") {
145 this->motion_model_mapping_.small_vehicle_model =
146 static_cast<std::uint8_t>(parameter.as_int());
147 } else if (parameter.get_name() == "large_vehicle_motion_model") {
148 this->motion_model_mapping_.large_vehicle_model =
149 static_cast<std::uint8_t>(parameter.as_int());
150 } else if (parameter.get_name() == "motorcycle_motion_model") {
151 this->motion_model_mapping_.motorcycle_model =
152 static_cast<std::uint8_t>(parameter.as_int());
153 } else if (parameter.get_name() == "pedestrian_motion_model") {
154 this->motion_model_mapping_.pedestrian_model =
155 static_cast<std::uint8_t>(parameter.as_int());
156 } else if (parameter.get_name() == "unknown_motion_model") {
157 this->motion_model_mapping_.unknown_model = static_cast<std::uint8_t>(parameter.as_int());
158 } else {
159 result.successful = false;
160 result.reason = "Unexpected parameter name '" + parameter.get_name() + '\'';
161 }
162 }
163
164 return result;
165 });
166
167 return carma_ros2_utils::CallbackReturn::SUCCESS;
168}

190auto ExternalObjectListToDetectionListNode::publish_as_detection_list(
191 const input_msg_type & msg) const -> void
192{
193 try {
194 const auto detection_list{transform_from_map_to_utm(
195 to_detection_list_msg(msg, motion_model_mapping_), map_georeference_)};
196
197 publisher_->publish(detection_list);
198 } catch (const std::invalid_argument & e) {
199 RCLCPP_ERROR(
200 this->get_logger(), "Could not convert external object list to detection list: %s", e.what());
201 }
202}
