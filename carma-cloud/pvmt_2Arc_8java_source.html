31 @Override
32 public void addPoints(Road oRoad, double dMaxStep, boolean bIncludeEnd, Proj oProj, double[] dProjPt)
33 {
34 try
35 {
36 int nLimit;
37 double dEndDistOnRoad = m_dS + m_dL;
38 ArrayDeque<LaneSection> oDeque = new ArrayDeque();
39 oRoad.getLaneSections(oDeque, m_dS, dEndDistOnRoad);
40 double dR = 1.0 / m_dC;
41 double dH = m_dX + dR * Math.cos(m_dHdg + Mercator.PI_OVER_TWO);
42 double dK = m_dY + dR * Math.sin(m_dHdg + Mercator.PI_OVER_TWO);
43 double dInitTheta = m_dHdg + 3.0 * Mercator.PI_OVER_TWO;
44 double dTotalAngle = 0;
45 while (!oDeque.isEmpty())
46 {
47 LaneSection oSection = oDeque.removeFirst();
48 double dStart = Math.max(m_dS, oSection.m_dS);
49 double dEnd = oDeque.isEmpty() ? dEndDistOnRoad : oDeque.getFirst().m_dS;
50 double dLength = dEnd - dStart;
51 if (dMaxStep > dLength)
52 nLimit = 1;
53 else
54 nLimit = (int)(dLength / dMaxStep) + 1;
55
56 double dTheta = dLength * m_dC;
57
58 oRoad.m_dLaneZero = Arrays.ensureCapacity(oRoad.m_dLaneZero, nLimit * 2);
59 oRoad.m_dTrack = Arrays.ensureCapacity(oRoad.m_dTrack, nLimit * 2);
60 double dAngleStep = dTheta / nLimit;
61 double dLengthStep = dLength / nLimit;
62 if (bIncludeEnd) // add last point for last geometry
63 ++nLimit;
64 for (int i = 0; i < nLimit; i++)
65 {
66 double dAngle = dInitTheta + dTotalAngle;
67 dTotalAngle += dAngleStep;
68 double dX = dH + dR * Math.cos(dAngle);
69 double dY = dK + dR * Math.sin(dAngle);
70 oRoad.m_dTrack = addProjectedPoint(dX, dY, oRoad.m_dTrack, oProj, dProjPt);
71 oRoad.m_dTrack = Arrays.add(oRoad.m_dTrack, 0);
72
73 double dLengthAlongRoad = dLengthStep * i + dStart;
74 LaneOffset oLaneOffset = LaneOffset.getLaneOffset(dLengthAlongRoad, oRoad.m_oLaneOffsets);
75 double dHdg = m_dHdg + dTotalAngle;
76 double dT = MathUtil.cubic(dLengthAlongRoad - oLaneOffset.m_dS, oLaneOffset.m_dA, oLaneOffset.m_dB, oLaneOffset.m_dC, oLaneOffset.m_dD);
77 double dXLaneZero = dX - Math.sin(dHdg) * dT;
78 double dYLaneZero = dY + Math.cos(dHdg) * dT;
79 oRoad.m_dLaneZero = addProjectedPoint(dXLaneZero, dYLaneZero, oRoad.m_dLaneZero, oProj, dProjPt);
80 oRoad.m_dLaneZero = Arrays.add(oRoad.m_dLaneZero, 0);
81
82 addLanePoints(oRoad, dHdg, dXLaneZero, dYLaneZero, dLengthAlongRoad, oSection, oProj, dProjPt);
83 }
84 dTotalAngle -= dAngleStep;
85 }
86 }
87 catch (Exception oEx)
88 {
89 oEx.printStackTrace();
90 }
91 }

107 private void addLanePoints(LaneSection oLaneSection, ArrayList<Lane> oLanes, double dTangent, double dXLaneZero, double dYLaneZero, double dLengthAlongRoad, Proj oProj, double[] dProjPt, Road oRoad, double dH, double dK, double dR, double dAngle)

133 double dProjW = Geo.distance(dXProjPrevLane, dYProjPrevLane, dProjPt[0], dProjPt[1]);
134 dXProjPrevLane = dProjPt[0];
135 dYProjPrevLane = dProjPt[1];

137 double dYPath = (dYPrevLane + dYNextLane) / 2;

139 if (this instanceof Arc)

141 oRoad.m_dPerps.add(new double[]{dPx, dPy, dXProjPrevLane, dYProjPrevLane});
