33carma_v2x_msgs::msg::MobilityOperation TrafficIncidentWorker::mobilityMessageGenerator(
34 const gps_msgs::msg::GPSFix & pinpoint_msg)
35{
36 carma_v2x_msgs::msg::MobilityOperation traffic_mobility_msg;
37
38 traffic_mobility_msg.m_header.timestamp = pinpoint_msg.header.stamp.sec * 1000;
39 traffic_mobility_msg.m_header.sender_id = sender_id_;
40
41 traffic_mobility_msg.strategy = USE_CASE_NAME_;
42
43 traffic_mobility_msg.strategy_params =
44 "lat:" + doubleToString(pinpoint_msg.latitude) + "," +
45 "lon:" + doubleToString(pinpoint_msg.longitude) + "," +
46 "downtrack:" + anytypeToString(down_track_) + "," + "uptrack:" + anytypeToString(up_track_) +
47 "," + "min_gap:" + anytypeToString(min_gap_) + "," +
48 "advisory_speed:" + anytypeToString(advisory_speed_) + "," + "event_reason:" + event_reason_ +
49 "," + "event_type:" + event_type_;
50
51 return traffic_mobility_msg;
52}
