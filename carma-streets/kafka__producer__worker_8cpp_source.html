80 void kafka_producer_worker::send(const std::string &msg)
81 {
82
83 if (!_run)
84 return;
85
86 if (msg.empty())
87 {
88 _producer->poll(0);
89 return;
90 }
91
92 // produce messages
93 while (true)
94 {
95 RdKafka::ErrorCode resp = _producer->produce(_topic,
96 _partition,
97 RdKafka::Producer::RK_MSG_COPY,
98 const_cast<char *>(msg.c_str()),
99 msg.size(),
100 NULL,
101 NULL);
102 if (resp != RdKafka::ERR_NO_ERROR)
103 {
104 SPDLOG_CRITICAL(" {0} Produce failed: {1} ", _producer->name(), RdKafka::err2str(resp));
105 if (resp == RdKafka::ERR__QUEUE_FULL)
106 {
107 /* If the internal queue is full, wait for
108 * messages to be delivered and then retry.
109 * The internal queue represents both
110 * messages to be sent and messages that have
111 * been sent or failed, awaiting their
112 * delivery report callback to be called.
113 *
114 * The internal queue is limited by the
115 * configuration property
116 * queue.buffering.max.messages */
117 _producer->poll(1000 /*block for max 1000ms*/);
118 continue;
119 }
120 }
121 else
122 {
123 SPDLOG_TRACE(" {0} Produced message ( {1} bytes ) , message content: {2}", _producer->name(), msg.size(), msg.c_str());
124 }
125
126 // break the loop regardless of sucessfully sent or failed
127 break;
128 }
129
130 /* A producer application should continually serve
131 * the delivery report queue by calling poll()
132 * at frequent intervals.
133 * Either put the poll call in your main loop, or in a
134 * dedicated thread, or call it after every produce() call.
135 * Just make sure that poll() is still called
136 * during periods where you are not producing any messages
137 * to make sure previously produced messages have their
138 * delivery report callback served (and any other callbacks
139 * you register). */
140 _producer->poll(0);
141 }

143 void kafka_producer_worker::stop()
144 {
145 /* Wait for final messages to be delivered or fail.
146 * flush() is an abstraction over poll() which
147 * waits for all messages to be delivered. */
148 _run = false;
149 SPDLOG_CRITICAL("Stopping producer client.. ");
150 SPDLOG_CRITICAL("Flushing final messages... ");
151 try
152 {
153 if (_producer)
154 {
155 _producer->flush(10 * 1000 /* wait for max 10 seconds */);
156
157 if (_producer->outq_len() > 0)
158 SPDLOG_INFO(" {0} {1} message(s) were not delivered ", _producer->name(), _producer->outq_len());
159 }
160 }
161 catch (const std::runtime_error &e)
162 {
163 SPDLOG_CRITICAL("Error encountered flushing producer : {0}", e.what());
