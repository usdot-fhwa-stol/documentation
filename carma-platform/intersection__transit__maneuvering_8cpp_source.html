143 bool IntersectionTransitManeuvering::plan_trajectory_cb(cav_srvs::PlanTrajectoryRequest& req, cav_srvs::PlanTrajectoryResponse& resp)
144 {
145
146 ros::WallTime start_time = ros::WallTime::now(); // Start timeing the execution time for planning so it can be logged
147
148 std::vector<cav_msgs::Maneuver> maneuver_plan;
149
150 auto related_maneuvers = resp.related_maneuvers;
151
152 ROS_DEBUG_STREAM("Starting planning for maneuver index: " << req.maneuver_index_to_plan);
153 ROS_DEBUG_STREAM("req.maneuver_plan.maneuvers.size(): " << req.maneuver_plan.maneuvers.size());
154 for(size_t i = req.maneuver_index_to_plan; i < req.maneuver_plan.maneuvers.size(); i++)
155 {
156 ROS_DEBUG("Looping");
157 if(req.maneuver_plan.maneuvers[i].type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_STRAIGHT ||
158 req.maneuver_plan.maneuvers[i].type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_LEFT_TURN ||
159 req.maneuver_plan.maneuvers[i].type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_RIGHT_TURN )
160 {
161 ROS_DEBUG_STREAM("Found valid maneuver for planning at index: " << i);
162 maneuver_plan.push_back(req.maneuver_plan.maneuvers[i]);
163 related_maneuvers.push_back(i);
164 }
165 else
166 {
167 break;
168 }
169 }
170
171
172 converted_maneuvers_ = convert_maneuver_plan(maneuver_plan);
173 cav_srvs::PlanTrajectoryRequest new_req;
174
175
176 for(auto i : converted_maneuvers_)
177 {
178 new_req.maneuver_plan.maneuvers.push_back(i);
179 }
180
181 new_req.header = req.header;
182 new_req.vehicle_state = req.vehicle_state;
183 new_req.initial_trajectory_plan = req.initial_trajectory_plan;
184
185
186 ROS_DEBUG_STREAM("About to call client for service: " << object_->getTopic());
187 ROS_DEBUG_STREAM("Client service actually exists: " << object_->exists());
188
189
190 if(object_->call(new_req,resp))//Since we're using an interface for this process, the call() functionality will come from somewhere else
191 {
192 ROS_DEBUG_STREAM("Call Successful");
193 } else {
194 ROS_DEBUG_STREAM("Failed to call service");
195 return false;
196 }
197
198
199 resp.related_maneuvers = related_maneuvers; // Set the related maneuvers using the origional maneuver indexs not those sent to inlane-cruising
200 for (auto maneuver : related_maneuvers) {
201 resp.maneuver_status.push_back(cav_srvs::PlanTrajectory::Response::MANEUVER_IN_PROGRESS);
202 }
203
204 ros::WallTime end_time = ros::WallTime::now();
205
206 ros::WallDuration duration = end_time - start_time;
207 ROS_DEBUG_STREAM("ExecutionTime: " << duration.toSec());
208 return true;
209 }

211 std::vector<cav_msgs::Maneuver> IntersectionTransitManeuvering::convert_maneuver_plan(const std::vector<cav_msgs::Maneuver>& maneuvers)
212 {
213 if (maneuvers.size() == 0)
214 {
215 throw std::invalid_argument("No maneuvers to convert");
216 }
217
218 std::vector<cav_msgs::Maneuver> new_maneuver_plan;
219 cav_msgs::Maneuver new_maneuver;
220 new_maneuver.type = cav_msgs::Maneuver::LANE_FOLLOWING; //All of the converted maneuvers will be of type LANE_FOLLOWING
221 ROS_DEBUG_STREAM("Input Maneuver Type = "<< static_cast<int>(maneuvers.front().type));
222 for(const auto& maneuver : maneuvers)
223 {
224 /*Throw exception if the manuever type does not match INTERSECTION_TRANSIT*/
225 if ( maneuver.type != cav_msgs::Maneuver::INTERSECTION_TRANSIT_STRAIGHT &&
226 maneuver.type != cav_msgs::Maneuver::INTERSECTION_TRANSIT_LEFT_TURN &&
227 maneuver.type != cav_msgs::Maneuver::INTERSECTION_TRANSIT_RIGHT_TURN)
228 {
229 throw std::invalid_argument("Intersection transit maneuvering does not support this maneuver type");
230 }
231
232 /*Convert IT Straight*/
233 if (maneuver.type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_STRAIGHT)
234 {
235 ROS_DEBUG_STREAM("Converting INTERSECTION_TRANSIT_STRAIGHT");
236
237 new_maneuver.lane_following_maneuver.parameters.maneuver_id = maneuver.intersection_transit_straight_maneuver.parameters.maneuver_id;
238 new_maneuver.lane_following_maneuver.parameters.planning_strategic_plugin = maneuver.intersection_transit_straight_maneuver.parameters.planning_strategic_plugin;
239 new_maneuver.lane_following_maneuver.parameters.planning_tactical_plugin = maneuver.intersection_transit_straight_maneuver.parameters.planning_tactical_plugin;
240 new_maneuver.lane_following_maneuver.parameters.negotiation_type = maneuver.intersection_transit_straight_maneuver.parameters.negotiation_type;
241 new_maneuver.lane_following_maneuver.parameters.presence_vector = cav_msgs::ManeuverParameters::HAS_TACTICAL_PLUGIN;
242
243 new_maneuver.lane_following_maneuver.start_dist = maneuver.intersection_transit_straight_maneuver.start_dist;
244 new_maneuver.lane_following_maneuver.start_speed = maneuver.intersection_transit_straight_maneuver.start_speed;
245 new_maneuver.lane_following_maneuver.start_time = maneuver.intersection_transit_straight_maneuver.start_time;
246
247 new_maneuver.lane_following_maneuver.end_dist = maneuver.intersection_transit_straight_maneuver.end_dist;
248 new_maneuver.lane_following_maneuver.end_speed = maneuver.intersection_transit_straight_maneuver.end_speed;
249 new_maneuver.lane_following_maneuver.end_time = maneuver.intersection_transit_straight_maneuver.end_time;
250
251
252 new_maneuver.lane_following_maneuver.lane_ids = { maneuver.intersection_transit_straight_maneuver.starting_lane_id };
253
254 new_maneuver_plan.push_back(new_maneuver);
255 }
256
257 /*Convert IT LEFT TURN*/
258 if (maneuver.type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_LEFT_TURN)
259 {
260 ROS_DEBUG_STREAM("Converting INTERSECTION_TRANSIT_LEFT_TURN");
261
262 new_maneuver.lane_following_maneuver.parameters.maneuver_id = maneuver.intersection_transit_left_turn_maneuver.parameters.maneuver_id;
263 new_maneuver.lane_following_maneuver.parameters.planning_strategic_plugin = maneuver.intersection_transit_left_turn_maneuver.parameters.planning_strategic_plugin;
264 new_maneuver.lane_following_maneuver.parameters.planning_tactical_plugin = maneuver.intersection_transit_left_turn_maneuver.parameters.planning_tactical_plugin;
265 new_maneuver.lane_following_maneuver.parameters.negotiation_type = maneuver.intersection_transit_left_turn_maneuver.parameters.negotiation_type;
266 new_maneuver.lane_following_maneuver.parameters.presence_vector = cav_msgs::ManeuverParameters::HAS_TACTICAL_PLUGIN;
267
268 new_maneuver.lane_following_maneuver.start_dist = maneuver.intersection_transit_left_turn_maneuver.start_dist;
269 new_maneuver.lane_following_maneuver.start_speed = maneuver.intersection_transit_left_turn_maneuver.start_speed;
270 new_maneuver.lane_following_maneuver.start_time = maneuver.intersection_transit_left_turn_maneuver.start_time;
271
272 new_maneuver.lane_following_maneuver.end_dist = maneuver.intersection_transit_left_turn_maneuver.end_dist;
273 new_maneuver.lane_following_maneuver.end_speed = maneuver.intersection_transit_left_turn_maneuver.end_speed;
274 new_maneuver.lane_following_maneuver.end_time = maneuver.intersection_transit_left_turn_maneuver.end_time;
275
276 new_maneuver.lane_following_maneuver.lane_ids = { maneuver.intersection_transit_left_turn_maneuver.starting_lane_id };
277
278 new_maneuver_plan.push_back(new_maneuver);
279 }
280
281 /*Convert IT RIGHT TURN*/
282 if (maneuver.type == cav_msgs::Maneuver::INTERSECTION_TRANSIT_RIGHT_TURN)
283 {
284
285 ROS_DEBUG_STREAM("Converting INTERSECTION_TRANSIT_RIGHT_TURN");
286
287 new_maneuver.lane_following_maneuver.parameters.maneuver_id = maneuver.intersection_transit_right_turn_maneuver.parameters.maneuver_id;
288 new_maneuver.lane_following_maneuver.parameters.planning_strategic_plugin = maneuver.intersection_transit_right_turn_maneuver.parameters.planning_strategic_plugin;
289 new_maneuver.lane_following_maneuver.parameters.planning_tactical_plugin = maneuver.intersection_transit_right_turn_maneuver.parameters.planning_tactical_plugin;
290 new_maneuver.lane_following_maneuver.parameters.negotiation_type = maneuver.intersection_transit_right_turn_maneuver.parameters.negotiation_type;
291 new_maneuver.lane_following_maneuver.parameters.presence_vector = cav_msgs::ManeuverParameters::HAS_TACTICAL_PLUGIN;
292
293 new_maneuver.lane_following_maneuver.start_dist = maneuver.intersection_transit_right_turn_maneuver.start_dist;
294 new_maneuver.lane_following_maneuver.start_speed = maneuver.intersection_transit_right_turn_maneuver.start_speed;
295 new_maneuver.lane_following_maneuver.start_time = maneuver.intersection_transit_right_turn_maneuver.start_time;
296
297 new_maneuver.lane_following_maneuver.end_dist = maneuver.intersection_transit_right_turn_maneuver.end_dist;
298 new_maneuver.lane_following_maneuver.end_speed = maneuver.intersection_transit_right_turn_maneuver.end_speed;
299 new_maneuver.lane_following_maneuver.end_time = maneuver.intersection_transit_right_turn_maneuver.end_time;
300
301 new_maneuver.lane_following_maneuver.lane_ids = { maneuver.intersection_transit_right_turn_maneuver.starting_lane_id };
302
303 new_maneuver_plan.push_back(new_maneuver);
304 }
305
306 ROS_DEBUG_STREAM("Original Maneuver : " << maneuver << std::endl
307 << "Converted Maneuver: " << new_maneuver);
308
309 }//end for-loop
310
311 return new_maneuver_plan;
312
