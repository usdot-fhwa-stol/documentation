18 std::string to_json(const sensor_data_sharing_msg &msg) {
19 rapidjson::Document doc;
20 rapidjson::Value sdsm_json(rapidjson::kObjectType);
21 sdsm_json.AddMember("msg_cnt", msg._msg_count, doc.GetAllocator());
22 sdsm_json.AddMember("source_id", msg._source_id, doc.GetAllocator());
23 sdsm_json.AddMember("equipment_type", static_cast<int>(msg._equipment_type), doc.GetAllocator());
24 // Construct SDSM Time Stamp JSON Object
25 auto time_stamp_json = create_timestamp(msg._time_stamp, doc.GetAllocator());
26 sdsm_json.AddMember("sdsm_time_stamp", time_stamp_json, doc.GetAllocator());
27 // Construct reference position JSON Object
28 auto position_3d_json = create_position_3d( msg._ref_positon, doc.GetAllocator() );
29 sdsm_json.AddMember("ref_pos", position_3d_json, doc.GetAllocator());
30 sdsm_json.AddMember("ref_pos_xy_conf", create_positional_accuracy( msg._ref_position_confidence, doc.GetAllocator()), doc.GetAllocator());
31 if ( msg._ref_position_elevation_confidence.has_value() )
32 sdsm_json.AddMember("ref_pos_el_conf", static_cast<unsigned int >(msg._ref_position_elevation_confidence.value()), doc.GetAllocator());
33 // Construct object list
34 sdsm_json.AddMember("objects", create_detected_object_list(msg._objects, doc.GetAllocator()), doc.GetAllocator());
35
36 rapidjson::StringBuffer buffer;
37 rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
38 sdsm_json.Accept(writer);
39
40 return buffer.GetString();
41 }
